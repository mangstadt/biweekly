088/**
089 * <p>
090 * Reads xCals (XML-encoded iCalendar objects) in a streaming fashion.
091 * </p>
092 * <p>
093 * <b>Example:</b>
094 * </p>
095 * 
096 * <pre class="brush:java">
097 * File file = new File("icals.xml");
098 * XCalReader reader = null;
099 * try {
100 *   reader = new XCalReader(file);
101 *   ICalendar ical;
102 *   while ((ical = reader.readNext()) != null) {
103 *     //...
104 *   }
105 * } finally {
106 *   if (reader != null) reader.close();
107 * }
108 * </pre>
109 * @author Michael Angstadt
110 * @see <a href="http://tools.ietf.org/html/rfc6321">RFC 6321</a>
111 */
112public class XCalReader extends StreamReader {
113        private final Source source;
114        private final Closeable stream;
115
116        private volatile ICalendar readICal;
117        private volatile TransformerException thrown;
118
119        private final ReadThread thread = new ReadThread();
120        private final Object lock = new Object();
121        private final BlockingQueue<Object> readerBlock = new ArrayBlockingQueue<Object>(1);
122        private final BlockingQueue<Object> threadBlock = new ArrayBlockingQueue<Object>(1);
123
124        /**
125         * @param str the string to read from
126         */
127        public XCalReader(String str) {
128                this(new StringReader(str));
129        }
130
131        /**
132         * @param in the input stream to read from
133         */
134        public XCalReader(InputStream in) {
135                source = new StreamSource(in);
136                stream = in;
137        }

139        /**
140         * @param file the file to read from
141         * @throws FileNotFoundException if the file doesn't exist
142         */
143        public XCalReader(File file) throws FileNotFoundException {
144                this(new BufferedInputStream(new FileInputStream(file)));
145        }

148         * @param reader the reader to read from

151                source = new StreamSource(reader);

156         * @param node the DOM node to read from

159                source = new DOMSource(node);

163        @Override
164        protected ICalendar _readNext() throws IOException {
165                readICal = null;
166                warnings.clear();
167                context = new ParseContext();
168                thrown = null;
169
170                if (!thread.started) {
171                        thread.start();
172                } else {
173                        if (thread.finished || thread.closed) {
174                                return null;
175                        }
176
177                        try {
178                                threadBlock.put(lock);
179                        } catch (InterruptedException e) {
180                                return null;
181                        }
182                }
183
184                //wait until thread reads xCard
185                try {
186                        readerBlock.take();
187                } catch (InterruptedException e) {
188                        return null;
189                }
190
191                if (thrown != null) {
192                        throw new IOException(thrown);
193                }
194
195                return readICal;
196        }
197
198        private class ReadThread extends Thread {
199                private final SAXResult result;
200                private final Transformer transformer;
201                private volatile boolean finished = false, started = false, closed = false;
202
203                public ReadThread() {
204                        setName(getClass().getSimpleName());
205
206                        //create the transformer
207                        try {
208                                TransformerFactory factory = TransformerFactory.newInstance();
209                                XmlUtils.applyXXEProtection(factory);
210
211                                transformer = factory.newTransformer();
212                        } catch (TransformerConfigurationException e) {
213                                //shouldn't be thrown because it's a simple configuration
214                                throw new RuntimeException(e);
215                        }
216
217                        //prevent error messages from being printed to stderr
218                        transformer.setErrorListener(new NoOpErrorListener());
219
220                        result = new SAXResult(new ContentHandlerImpl());
221                }
222
223                @Override
224                public void run() {
225                        started = true;
226
227                        try {
228                                transformer.transform(source, result);
229                        } catch (TransformerException e) {
230                                if (!thread.closed) {
231                                        thrown = e;
232                                }
233                        } finally {
234                                finished = true;
235                                try {
236                                        readerBlock.put(lock);
237                                } catch (InterruptedException e) {
238                                        //ignore
239                                }
240                        }
241                }
242        }
243
244        private class ContentHandlerImpl extends DefaultHandler {
245                private final Document DOC = XmlUtils.createDocument();
246                private final XCalStructure structure = new XCalStructure();
247                private final StringBuilder characterBuffer = new StringBuilder();
248                private final LinkedList<ICalComponent> componentStack = new LinkedList<ICalComponent>();
249
250                private Element propertyElement, parent;
251                private QName paramName;
252                private ICalComponent curComponent;
253                private ICalParameters parameters;
254
255                @Override
256                public void characters(char[] buffer, int start, int length) throws SAXException {
257                        characterBuffer.append(buffer, start, length);
258                }
259
260                @Override
261                public void startElement(String namespace, String localName, String qName, Attributes attributes) throws SAXException {
262                        QName qname = new QName(namespace, localName);
263                        String textContent = emptyCharacterBuffer();
264
265                        if (structure.isEmpty()) {
266                                //<icalendar>
267                                if (ICALENDAR.equals(qname)) {
268                                        structure.push(ElementType.icalendar);
269                                }
270                                return;
271                        }
272
273                        ElementType parentType = structure.peek();
274                        ElementType typeToPush = null;
275                        if (parentType != null) {
276                                switch (parentType) {
277
278                                case icalendar:
279                                        //<vcalendar>
280                                        if (VCALENDAR.equals(qname)) {
281                                                ICalComponentScribe<? extends ICalComponent> scribe = index.getComponentScribe(localName, ICalVersion.V2_0);
282                                                ICalComponent component = scribe.emptyInstance();
283
284                                                curComponent = component;
285                                                readICal = (ICalendar) component;
286                                                typeToPush = ElementType.component;
287                                        }
288                                        break;
289
290                                case component:
291                                        if (PROPERTIES.equals(qname)) {
292                                                //<properties>
293                                                typeToPush = ElementType.properties;
294                                        } else if (COMPONENTS.equals(qname)) {
295                                                //<components>
296                                                componentStack.add(curComponent);
297                                                curComponent = null;
298
299                                                typeToPush = ElementType.components;
300                                        }
301                                        break;
302
303                                case components:
304                                        //start component element
305                                        if (XCAL_NS.equals(namespace)) {
306                                                ICalComponentScribe<? extends ICalComponent> scribe = index.getComponentScribe(localName, ICalVersion.V2_0);
307                                                curComponent = scribe.emptyInstance();
308
309                                                ICalComponent parent = componentStack.getLast();
310                                                parent.addComponent(curComponent);
311
312                                                typeToPush = ElementType.component;
313                                        }
314                                        break;
315
316                                case properties:
317                                        //start property element
318                                        propertyElement = createElement(namespace, localName, attributes);
319                                        parameters = new ICalParameters();
320                                        parent = propertyElement;
321                                        typeToPush = ElementType.property;
322                                        break;
323
324                                case property:
325                                        //<parameters>
326                                        if (PARAMETERS.equals(qname)) {
327                                                typeToPush = ElementType.parameters;
328                                        }
329                                        break;
330
331                                case parameters:
332                                        //inside of <parameters>
333                                        if (XCAL_NS.equals(namespace)) {
334                                                paramName = qname;
335                                                typeToPush = ElementType.parameter;
336                                        }
337                                        break;
338
339                                case parameter:
340                                        //inside of a parameter element
341                                        if (XCAL_NS.equals(namespace)) {
342                                                typeToPush = ElementType.parameterValue;
343                                        }
344                                        break;
345                                case parameterValue:
346                                        //should never have child elements
347                                        break;
348                                }
349                        }
350
351                        //append element to property element
352                        if (propertyElement != null && typeToPush != ElementType.property && typeToPush != ElementType.parameters && !structure.isUnderParameters()) {
353                                if (textContent.length() > 0) {
354                                        parent.appendChild(DOC.createTextNode(textContent));
355                                }
356
357                                Element element = createElement(namespace, localName, attributes);
358                                parent.appendChild(element);
359                                parent = element;
360                        }
361
362                        structure.push(typeToPush);
363                }
364
365                @Override
366                public void endElement(String namespace, String localName, String qName) throws SAXException {
367                        String textContent = emptyCharacterBuffer();
368
369                        if (structure.isEmpty()) {
370                                //no <icalendar> elements were read yet
371                                return;
372                        }
373
374                        ElementType type = structure.pop();
375                        if (type == null && (propertyElement == null || structure.isUnderParameters())) {
376                                //it's a non-xCal element
377                                return;
378                        }
379
380                        if (type != null) {
381                                switch (type) {
382                                case parameterValue:
383                                        parameters.put(paramName.getLocalPart(), textContent);
384                                        break;
385
386                                case parameter:
387                                        //do nothing
388                                        break;
389
390                                case parameters:
391                                        //do nothing
392                                        break;
393
394                                case property:
395                                        context.getWarnings().clear();
396                                        context.setPropertyName(localName);
397
398                                        propertyElement.appendChild(DOC.createTextNode(textContent));
399
400                                        //unmarshal property and add to parent component
401                                        QName propertyQName = new QName(propertyElement.getNamespaceURI(), propertyElement.getLocalName());
402                                        ICalPropertyScribe<? extends ICalProperty> scribe = index.getPropertyScribe(propertyQName);
403                                        try {
404                                                ICalProperty property = scribe.parseXml(propertyElement, parameters, context);
405                                                if (property instanceof Version && curComponent instanceof ICalendar) {
406                                                        Version versionProp = (Version) property;
407                                                        ICalVersion version = versionProp.toICalVersion();
408                                                        if (version != null) {
409                                                                ICalendar ical = (ICalendar) curComponent;
410                                                                ical.setVersion(version);
411                                                                context.setVersion(version);
412
413                                                                propertyElement = null;
414                                                                break;
415                                                        }
416                                                }
417
418                                                curComponent.addProperty(property);
419                                                warnings.addAll(context.getWarnings());
420                                        } catch (SkipMeException e) {
421                                                //@formatter:off
422                                                warnings.add(new ParseWarning.Builder(context)
423                                                        .message(0, e.getMessage())
424                                                        .build()
425                                                );
426                                                //@formatter:on
427                                        } catch (CannotParseException e) {
428                                                //@formatter:off
429                                                warnings.add(new ParseWarning.Builder(context)
430                                                        .message(e)
431                                                        .build()
432                                                );
433                                                //@formatter:on
434
435                                                scribe = index.getPropertyScribe(Xml.class);
436                                                ICalProperty property = scribe.parseXml(propertyElement, parameters, context);
437                                                curComponent.addProperty(property);
438                                        }
439
440                                        propertyElement = null;
441                                        break;
442
443                                case component:
444                                        curComponent = null;
445
446                                        //</vcalendar>
447                                        if (VCALENDAR.getNamespaceURI().equals(namespace) && VCALENDAR.getLocalPart().equals(localName)) {
448                                                //wait for readNext() to be called again
449                                                try {
450                                                        readerBlock.put(lock);
451                                                        threadBlock.take();
452                                                } catch (InterruptedException e) {
453                                                        throw new SAXException(e);
454                                                }
455                                                return;
456                                        }
457                                        break;
458
459                                case properties:
460                                        break;
461
462                                case components:
463                                        curComponent = componentStack.removeLast();
464                                        break;
465
466                                case icalendar:
467                                        break;
468                                }
469                        }
470
471                        //append element to property element
472                        if (propertyElement != null && type != ElementType.property && type != ElementType.parameters && !structure.isUnderParameters()) {
473                                if (textContent.length() > 0) {
474                                        parent.appendChild(DOC.createTextNode(textContent));
475                                }
476                                parent = (Element) parent.getParentNode();
477                        }
478                }
479
480                private String emptyCharacterBuffer() {
481                        String textContent = characterBuffer.toString();
482                        characterBuffer.setLength(0);
483                        return textContent;
484                }
485
486                private Element createElement(String namespace, String localName, Attributes attributes) {
487                        Element element = DOC.createElementNS(namespace, localName);
488                        applyAttributesTo(element, attributes);
489                        return element;
490                }
491
492                private void applyAttributesTo(Element element, Attributes attributes) {
493                        for (int i = 0; i < attributes.getLength(); i++) {
494                                String qname = attributes.getQName(i);
495                                if (qname.startsWith("xmlns:")) {
496                                        continue;
497                                }
498
499                                String name = attributes.getLocalName(i);
500                                String value = attributes.getValue(i);
501                                element.setAttribute(name, value);
502                        }
503                }
504        }

506        private enum ElementType {
507                /*
508                 * Note: A value is missing for "vcalendar" because it is treated as a
509                 * "component".
510                 * 
511                 * Note: These enum values are in lower-case to make them stand out from
512                 * the "XCalQNames" variable names, many of which are identically named.
513                 */
514                icalendar, components, properties, component, property, parameters, parameter, parameterValue;
515        }
516
517        /**
518         * <p>
519         * Keeps track of the structure of an xCal XML document.
520         * </p>
521         * 
522         * <p>
523         * Note that this class is here because you can't just do QName comparisons
524         * on a one-by-one basis. The location of an XML element within the XML
525         * document is important too. It's possible for two elements to have the
526         * same QName, but be treated differently depending on their location (e.g.
527         * the {@code <duration>} property has a {@code <duration>} data type)
528         * </p>
529         */
530        private static class XCalStructure {
531                private final List<ElementType> stack = new ArrayList<ElementType>();
532
533                /**
534                 * Pops the top element type off the stack.
535                 * @return the element type or null if the stack is empty
536                 */
537                public ElementType pop() {
538                        return isEmpty() ? null : stack.remove(stack.size() - 1);
539                }
540
541                /**
542                 * Looks at the top element type.
543                 * @return the top element type or null if the stack is empty
544                 */
545                public ElementType peek() {
546                        return isEmpty() ? null : stack.get(stack.size() - 1);
547                }
548
549                /**
550                 * Adds an element type to the stack.
551                 * @param type the type to add or null if the XML element is not an xCal
552                 * element
553                 */
554                public void push(ElementType type) {
555                        stack.add(type);
556                }
557
558                /**
559                 * Determines if the leaf node is under a {@code <parameters>} element.
560                 * @return true if it is, false if not
561                 */
562                public boolean isUnderParameters() {
563                        //get the first non-null type
564                        ElementType nonNull = null;
565                        for (int i = stack.size() - 1; i >= 0; i--) {
566                                ElementType type = stack.get(i);
567                                if (type != null) {
568                                        nonNull = type;
569                                        break;
570                                }
571                        }
572
573                        //@formatter:off
574                        return
575                        nonNull == ElementType.parameters ||
576                        nonNull == ElementType.parameter ||
577                        nonNull == ElementType.parameterValue;
578                        //@formatter:on
579                }
580
581                /**
582                 * Determines if the stack is empty
583                 * @return true if the stack is empty, false if not
584                 */
585                public boolean isEmpty() {
586                        return stack.isEmpty();
587                }
588        }

590        /**

593        private static class NoOpErrorListener implements ErrorListener {
594                public void error(TransformerException e) {
595                        //do nothing
596                }
597
598                public void fatalError(TransformerException e) {
599                        //do nothing
600                }
601
602                public void warning(TransformerException e) {
603                        //do nothing
604                }
605        }

607        /**
608         * Closes the underlying input stream.
609         */
610        public void close() throws IOException {
611                if (thread.isAlive()) {
612                        thread.closed = true;
613                        thread.interrupt();
614                }
615
616                if (stream != null) {
617                        stream.close();
618                }
619        }
