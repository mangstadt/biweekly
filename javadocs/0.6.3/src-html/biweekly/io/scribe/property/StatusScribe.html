032/**
033 * Marshals {@link Status} properties.
034 * @author Michael Angstadt
035 */
036public class StatusScribe extends TextPropertyScribe<Status> {
037        public StatusScribe() {
038                super(Status.class, "STATUS");
039        }
040
041        @Override
042        protected String _writeText(Status property, WriteContext context) {
043                if (context.getVersion() == ICalVersion.V1_0 && property.isNeedsAction()) {
044                        //vCal doesn't have a hyphen in the value
045                        return "NEEDS ACTION";
046                }
047                return super._writeText(property, context);
048        }
049
050        @Override
051        protected Status newInstance(String value, ICalVersion version) {
052                if (version == ICalVersion.V1_0 && "NEEDS ACTION".equalsIgnoreCase(value)) {
053                        //vCal doesn't have a hyphen in the value
054                        return Status.needsAction();
055                }
056                return new Status(value);
057        }
058}
