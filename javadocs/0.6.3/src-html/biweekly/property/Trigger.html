039/**
040 * <p>
041 * Defines when to trigger an alarm.
042 * </p>
043 * <p>
044 * <b>Code sample:</b>
045 * </p>
046 * 
047 * <pre class="brush:java">
048 * //15 minutes before the start time
049 * Duration duration = Duration.builder().prior(true).minutes(15).build();
050 * Trigger trigger = new Trigger(duration, Related.START);
051 * VAlarm alarm = VAlarm.display(trigger, "Meeting in 15 minutes");
052 * </pre>
053 * @author Michael Angstadt
054 * @see <a href="http://tools.ietf.org/html/rfc5545#page-133">RFC 5545
055 * p.133-6</a>
056 * @see <a href="http://tools.ietf.org/html/rfc2445#page-127">RFC 2445
057 * p.127-9</a>
058 */
059public class Trigger extends ICalProperty {
060        private Duration duration;
061        private Date date;
062
063        /**
064         * Creates a trigger property.
065         * @param duration the relative time
066         * @param related the date-time field that the duration is relative to
067         */
068        public Trigger(Duration duration, Related related) {
069                setDuration(duration, related);
070        }
071
072        /**
073         * Creates a trigger property.
074         * @param date the date-time the alarm will trigger.
075         */
076        public Trigger(Date date) {
077                setDate(date);
078        }
079
080        /**
081         * Copy constructor.
082         * @param original the property to make a copy of
083         */
084        public Trigger(Trigger original) {
085                super(original);
086                date = (original.date == null) ? null : new Date(original.date.getTime());
087                duration = original.duration;
088        }
089
090        /**
091         * Gets the relative time at which the alarm will trigger.
092         * @return the relative time or null if an absolute time is set
093         */
094        public Duration getDuration() {
095                return duration;
096        }
097
098        /**
099         * Sets a relative time at which the alarm will trigger.
100         * @param duration the relative time
101         * @param related the date-time field that the duration is relative to
102         */
103        public void setDuration(Duration duration, Related related) {
104                this.date = null;
105                this.duration = duration;
106                setRelated(related);
107        }
108
109        /**
110         * Gets the date-time that the alarm will trigger.
111         * @return the date-time or null if a relative duration is set
112         */
113        public Date getDate() {
114                return date;
115        }
116
117        /**
118         * Sets the date-time that the alarm will trigger.
119         * @param date the date-time the alarm will trigger.
120         */
121        public void setDate(Date date) {
122                this.date = date;
123                this.duration = null;
124                setRelated(null);
125        }
126
127        /**
128         * Gets the date-time field that the duration is relative to.
129         * @return the field or null if not set
130         * @see <a href="http://tools.ietf.org/html/rfc5545#page-24">RFC 5545
131         * p.24</a>
132         */
133        public Related getRelated() {
134                return parameters.getRelated();
135        }
136
137        /**
138         * Sets the date-time field that the duration is relative to.
139         * @param related the field or null to remove
140         * @see <a href="http://tools.ietf.org/html/rfc5545#page-24">RFC 5545
141         * p.24</a>
142         */
143        public void setRelated(Related related) {
144                parameters.setRelated(related);
145        }
146
147        @Override
148        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
149                if (duration == null && date == null) {
150                        warnings.add(new ValidationWarning(33));
151                }
152
153                Related related = getRelated();
154                if (duration != null && related == null) {
155                        warnings.add(new ValidationWarning(10));
156                }
157        }
158
159        @Override
160        protected Map<String, Object> toStringValues() {
161                Map<String, Object> values = new LinkedHashMap<String, Object>();
162                values.put("duration", duration);
163                values.put("date", date);
164                return values;
165        }
166
167        @Override
168        public Trigger copy() {
169                return new Trigger(this);
170        }
171
172        @Override
173        public int hashCode() {
174                final int prime = 31;
175                int result = super.hashCode();
176                result = prime * result + ((date == null) ? 0 : date.hashCode());
177                result = prime * result + ((duration == null) ? 0 : duration.hashCode());
178                return result;
179        }
180
181        @Override
182        public boolean equals(Object obj) {
183                if (this == obj) return true;
184                if (!super.equals(obj)) return false;
185                Trigger other = (Trigger) obj;
186                if (date == null) {
187                        if (other.date != null) return false;
188                } else if (!date.equals(other.date)) return false;
189                if (duration == null) {
190                        if (other.duration != null) return false;
191                } else if (!duration.equals(other.duration)) return false;
192                return true;
193        }
194}
