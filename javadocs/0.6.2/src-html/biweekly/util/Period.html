030/**
031 * A period of time.
032 * @author Michael Angstadt
033 */
034public final class Period {
035        /*
036         * Note: The getter methods must not make copies of the date objects they
037         * return! The date objects in this class must be directly referenced in
038         * order to support timezones.
039         * 
040         * Yes, this means that this class is not fully immutable. Date objects can
041         * be modified by calling "setTime()", but biweekly makes use of this method
042         * in order to convert dates into their proper timezones.
043         */
044        private final ICalDate startDate;
045        private final ICalDate endDate;
046        private final Duration duration;
047
048        /**
049         * Creates a new time period.
050         * @param startDate the start date
051         * @param endDate the end date
052         */
053        public Period(Date startDate, Date endDate) {
054                //@formatter:off
055                this(
056                        (startDate == null) ? null : new ICalDate(startDate),
057                        (endDate == null) ? null : new ICalDate(endDate)
058                );
059                //@formatter:on
060        }
061
062        /**
063         * Creates a new time period.
064         * @param startDate the start date
065         * @param endDate the end date
066         */
067        public Period(ICalDate startDate, ICalDate endDate) {
068                this.startDate = startDate;
069                this.endDate = endDate;
070                duration = null;
071        }
072
073        /**
074         * Creates a new time period.
075         * @param startDate the start date
076         * @param duration the length of time after the start date
077         */
078        public Period(Date startDate, Duration duration) {
079                //@formatter:off
080                this(
081                        (startDate == null) ? null : new ICalDate(startDate),
082                        duration
083                );
084                //@formatter:on
085        }
086
087        /**
088         * Creates a new time period.
089         * @param startDate the start date
090         * @param duration the length of time after the start date
091         */
092        public Period(ICalDate startDate, Duration duration) {
093                this.startDate = startDate;
094                this.duration = duration;
095                endDate = null;
096        }
097
098        /**
099         * Copies an existing time period.
100         * @param period the period to copy
101         */
102        public Period(Period period) {
103                this.startDate = (period.startDate == null) ? null : new ICalDate(period.startDate);
104                this.endDate = (period.endDate == null) ? null : new ICalDate(period.endDate);
105                this.duration = period.duration;
106        }
107
108        /**
109         * Gets the start date.
110         * @return the start date
111         */
112        public ICalDate getStartDate() {
113                return startDate;
114        }
115
116        /**
117         * Gets the end date. This will be null if a duration was defined.
118         * @return the end date or null if not set
119         */
120        public ICalDate getEndDate() {
121                return endDate;
122        }
123
124        /**
125         * Gets the length of time after the start date. This will be null if an end
126         * date was defined.
127         * @return the duration or null if not set
128         */
129        public Duration getDuration() {
130                return duration;
131        }
132
133        @Override
134        public int hashCode() {
135                final int prime = 31;
136                int result = 1;
137                result = prime * result + ((duration == null) ? 0 : duration.hashCode());
138                result = prime * result + ((endDate == null) ? 0 : endDate.hashCode());
139                result = prime * result + ((startDate == null) ? 0 : startDate.hashCode());
140                return result;
141        }
142
143        @Override
144        public boolean equals(Object obj) {
145                if (this == obj) return true;
146                if (obj == null) return false;
147                if (getClass() != obj.getClass()) return false;
148                Period other = (Period) obj;
149                if (duration == null) {
150                        if (other.duration != null) return false;
151                } else if (!duration.equals(other.duration)) return false;
152                if (endDate == null) {
153                        if (other.endDate != null) return false;
154                } else if (!endDate.equals(other.endDate)) return false;
155                if (startDate == null) {
156                        if (other.startDate != null) return false;
157                } else if (!startDate.equals(other.startDate)) return false;
158                return true;
159        }
160}
