064/**
065 * <p>
066 * Calculates the occurrences of an individual RRULE definition or groups of
067 * RRULEs, RDATEs, EXRULEs, and EXDATEs.
068 * </p>
069 * <p>
070 * <b>Glossary</b>
071 * </p>
072 * <ul>
073 * <li>Period - year|month|day|...</li>
074 * <li>Day of the week - an int in [0,6]</li>
075 * <li>Day of the year - zero indexed in [0,365]</li>
076 * <li>Day of the month - 1 indexed in [1,31]</li>
077 * <li>Month - 1 indexed integer in [1,12]</li>
078 * </ul>
079 * <p>
080 * <b>Abstractions</b>
081 * </p>
082 * <ul>
083 * <li>Generator - a function corresponding to an RRULE part that takes a date
084 * and returns a later (year or month or day depending on its period) within the
085 * next larger period. A generator ignores all periods in its input smaller than
086 * its period.</li>
087 * <li>Filter - a function that returns true iff the given date matches the
088 * subrule.</li>
089 * <li>Condition - returns true if the given date is past the end of the
090 * recurrence.</li>
091 * </ul>
092 * <p>
093 * All the functions that represent rule parts are stateful.
094 * </p>
095 * @author mikesamuel+svn@gmail.com (Mike Samuel)
096 * @author Michael Angstadt
097 */
098public class RecurrenceIteratorFactory {
099        /**
100         * Creates a recurrence iterator from an RDATE or EXDATE list.
101         * @param dates the list of dates
102         * @return the iterator
103         */
104        public static RecurrenceIterator createRecurrenceIterator(Collection<? extends DateValue> dates) {
105                DateValue[] datesArray = dates.toArray(new DateValue[0]);
106                return new RDateIteratorImpl(datesArray);
107        }

109        /**
110         * Creates a recurrence iterable from an RRULE.
111         * @param rrule the recurrence rule
112         * @param dtStart the start date of the series
113         * @param tzid the timezone that the start date is in, as well as the
114         * timezone to iterate in
115         * @return the iterable
116         */
117        public static RecurrenceIterable createRecurrenceIterable(final Recurrence rrule, final DateValue dtStart, final TimeZone tzid) {
118                return new RecurrenceIterable() {
119                        public RecurrenceIterator iterator() {
120                                return createRecurrenceIterator(rrule, dtStart, tzid);
121                        }
122                };
123        }

125        /**
126         * Creates a recurrence iterator from an RRULE.
127         * @param rrule the recurrence rule
128         * @param dtStart the start date of the series
129         * @param tzid the timezone that the start date is in, as well as the
130         * timezone to iterate in
131         * @return the iterator
132         */
133        public static RecurrenceIterator createRecurrenceIterator(Recurrence rrule, DateValue dtStart, TimeZone tzid) {
134                Frequency freq = rrule.getFrequency();
135
136                /*
137                 * If the given RRULE is malformed and does not have a frequency
138                 * specified, default to "yearly".
139                 */
140                if (freq == null) {
141                        freq = Frequency.YEARLY;
142                }
143
144                DayOfWeek wkst = rrule.getWorkweekStarts();
145
146                ICalDate until = rrule.getUntil();
147
148                DateValue untilUtc = (until == null) ? null : Google2445Utils.convert(until, tzid);
149
150                int count = toInt(rrule.getCount());
151                int interval = toInt(rrule.getInterval());
152                ByDay[] byDay = rrule.getByDay().toArray(new ByDay[0]);
153                int[] byMonth = toIntArray(rrule.getByMonth());
154                int[] byMonthDay = toIntArray(rrule.getByMonthDay());
155                int[] byWeekNo = toIntArray(rrule.getByWeekNo());
156                int[] byYearDay = toIntArray(rrule.getByYearDay());
157                int[] bySetPos = toIntArray(rrule.getBySetPos());
158                int[] byHour = toIntArray(rrule.getByHour());
159                int[] byMinute = toIntArray(rrule.getByMinute());
160                int[] bySecond = toIntArray(rrule.getBySecond());
161
162                if (interval <= 0) {
163                        interval = 1;
164                }
165
166                if (wkst == null) {
167                        wkst = DayOfWeek.MONDAY;
168                }
169
170                //optimize out BYSETPOS where possible
171                if (bySetPos.length > 0) {
172                        switch (freq) {
173                        case HOURLY:
174                                if (byHour.length > 0 && byMinute.length <= 1 && bySecond.length <= 1) {
175                                        byHour = filterBySetPos(byHour, bySetPos);
176                                }
177
178                                /*
179                                 * Handling bySetPos for rules that are more frequent than daily
180                                 * tends to lead to large amounts of processor being used before
181                                 * other work limiting features can kick in since there many
182                                 * seconds between dtStart and where the year limit kicks in.
183                                 * There are no known use cases for the use of bySetPos with
184                                 * hourly minutely and secondly rules so we just ignore it.
185                                 */
186                                bySetPos = NO_INTS;
187                                break;
188                        case MINUTELY:
189                                if (byMinute.length > 0 && bySecond.length <= 1) {
190                                        byMinute = filterBySetPos(byMinute, bySetPos);
191                                }
192                                //see bySetPos handling comment above
193                                bySetPos = NO_INTS;
194                                break;
195                        case SECONDLY:
196                                if (bySecond.length > 0) {
197                                        bySecond = filterBySetPos(bySecond, bySetPos);
198                                }
199                                //see bySetPos handling comment above
200                                bySetPos = NO_INTS;
201                                break;
202                        default:
203                        }
204                }
205
206                DateValue start = dtStart;
207                if (bySetPos.length > 0) {
208                        /*
209                         * Roll back until the beginning of the period to make sure that any
210                         * positive indices are indexed properly. The actual iterator
211                         * implementation is responsible for anything < dtStart.
212                         */
213                        switch (freq) {
214                        case YEARLY:
215                                start = (dtStart instanceof TimeValue) ? new DateTimeValueImpl(start.year(), 1, 1, 0, 0, 0) : new DateValueImpl(start.year(), 1, 1);
216                                break;
217                        case MONTHLY:
218                                start = (dtStart instanceof TimeValue) ? new DateTimeValueImpl(start.year(), start.month(), 1, 0, 0, 0) : new DateValueImpl(start.year(), start.month(), 1);
219                                break;
220                        case WEEKLY:
221                                int d = (7 + wkst.ordinal() - TimeUtils.dayOfWeek(dtStart).getCalendarConstant()) % 7;
222                                start = TimeUtils.add(dtStart, new DateValueImpl(0, 0, -d));
223                                break;
224                        default:
225                                break;
226                        }
227                }
228
229                /*
230                 * Recurrences are implemented as a sequence of periodic generators.
231                 * First a year is generated, and then months, and within months, days.
232                 */
233                ThrottledGenerator yearGenerator = Generators.serialYearGenerator(freq == Frequency.YEARLY ? interval : 1, dtStart);
234                Generator monthGenerator = null;
235                Generator dayGenerator = null;
236                Generator secondGenerator = null;
237                Generator minuteGenerator = null;
238                Generator hourGenerator = null;
239
240                /*
241                 * When multiple generators are specified for a period, they act as a
242                 * union operator. We could have multiple generators (say, for day) and
243                 * then run each and merge the results, but some generators are more
244                 * efficient than others. So to avoid generating 53 Sundays and throwing
245                 * away all but 1 for RRULE:FREQ=YEARLY;BYDAY=TU;BYWEEKNO=1, we
246                 * reimplement some of the more prolific generators as filters.
247                 */
248                // TODO(msamuel): don't need a list here
249                List<Predicate<? super DateValue>> filters = new ArrayList<Predicate<? super DateValue>>();
250
251                switch (freq) {
252                case SECONDLY:
253                        if (bySecond.length == 0 || interval != 1) {
254                                secondGenerator = Generators.serialSecondGenerator(interval, dtStart);
255                                if (bySecond.length > 0) {
256                                        filters.add(Filters.bySecondFilter(bySecond));
257                                }
258                        }
259                        break;
260                case MINUTELY:
261                        if (byMinute.length == 0 || interval != 1) {
262                                minuteGenerator = Generators.serialMinuteGenerator(interval, dtStart);
263                                if (byMinute.length > 0) {
264                                        filters.add(Filters.byMinuteFilter(byMinute));
265                                }
266                        }
267                        break;
268                case HOURLY:
269                        if (byHour.length == 0 || interval != 1) {
270                                hourGenerator = Generators.serialHourGenerator(interval, dtStart);
271                                if (byHour.length > 0) {
272                                        filters.add(Filters.byHourFilter(bySecond));
273                                }
274                        }
275                        break;
276                case DAILY:
277                        break;
278                case WEEKLY:
279                        /*
280                         * Week is not considered a period because a week may span multiple
281                         * months and/or years. There are no week generators, so a filter is
282                         * used to make sure that FREQ=WEEKLY;INTERVAL=2 only generates
283                         * dates within the proper week.
284                         */
285                        if (byDay.length > 0) {
286                                dayGenerator = Generators.byDayGenerator(byDay, false, start);
287                                byDay = NO_DAYS;
288                                if (interval > 1) {
289                                        filters.add(Filters.weekIntervalFilter(interval, wkst, dtStart));
290                                }
291                        } else {
292                                dayGenerator = Generators.serialDayGenerator(interval * 7, dtStart);
293                        }
294                        break;
295                case YEARLY:
296                        if (byYearDay.length > 0) {
297                                /*
298                                 * The BYYEARDAY rule part specifies a COMMA separated list of
299                                 * days of the year. Valid values are 1 to 366 or -366 to -1.
300                                 * For example, -1 represents the last day of the year (December
301                                 * 31st) and -306 represents the 306th to the last day of the
302                                 * year (March 1st).
303                                 */
304                                dayGenerator = Generators.byYearDayGenerator(byYearDay, start);
305                                break;
306                        }
307                        // $FALL-THROUGH$
308                case MONTHLY:
309                        if (byMonthDay.length > 0) {
310                                /*
311                                 * The BYMONTHDAY rule part specifies a COMMA separated list of
312                                 * days of the month. Valid values are 1 to 31 or -31 to -1. For
313                                 * example, -10 represents the tenth to the last day of the
314                                 * month.
315                                 */
316                                dayGenerator = Generators.byMonthDayGenerator(byMonthDay, start);
317                                byMonthDay = NO_INTS;
318                        } else if (byWeekNo.length > 0 && Frequency.YEARLY == freq) {
319                                /*
320                                 * The BYWEEKNO rule part specifies a COMMA separated list of
321                                 * ordinals specifying weeks of the year. This rule part is only
322                                 * valid for YEARLY rules.
323                                 */
324                                dayGenerator = Generators.byWeekNoGenerator(byWeekNo, wkst, start);
325                                byWeekNo = NO_INTS;
326                        } else if (byDay.length > 0) {
327                                /*
328                                 * Each BYDAY value can also be preceded by a positive (n) or
329                                 * negative (-n) integer. If present, this indicates the nth
330                                 * occurrence of the specific day within the MONTHLY or YEARLY
331                                 * RRULE. For example, within a MONTHLY rule, +1MO (or simply
332                                 * 1MO) represents the first Monday within the month, whereas
333                                 * -1MO represents the last Monday of the month. If an integer
334                                 * modifier is not present, it means all days of this type
335                                 * within the specified frequency. For example, within a MONTHLY
336                                 * rule, MO represents all Mondays within the month.
337                                 */
338                                dayGenerator = Generators.byDayGenerator(byDay, Frequency.YEARLY == freq && byMonth.length == 0, start);
339                                byDay = NO_DAYS;
340                        } else {
341                                if (Frequency.YEARLY == freq) {
342                                        monthGenerator = Generators.byMonthGenerator(new int[] { dtStart.month() }, start);
343                                }
344                                dayGenerator = Generators.byMonthDayGenerator(new int[] { dtStart.day() }, start);
345                        }
346                        break;
347                }
348
349                if (secondGenerator == null) {
350                        secondGenerator = Generators.bySecondGenerator(bySecond, start);
351                }
352                if (minuteGenerator == null) {
353                        if (byMinute.length == 0 && freq.compareTo(Frequency.MINUTELY) < 0) {
354                                minuteGenerator = Generators.serialMinuteGenerator(1, dtStart);
355                        } else {
356                                minuteGenerator = Generators.byMinuteGenerator(byMinute, start);
357                        }
358                }
359                if (hourGenerator == null) {
360                        if (byHour.length == 0 && freq.compareTo(Frequency.HOURLY) < 0) {
361                                hourGenerator = Generators.serialHourGenerator(1, dtStart);
362                        } else {
363                                hourGenerator = Generators.byHourGenerator(byHour, start);
364                        }
365                }
366
367                if (dayGenerator == null) {
368                        boolean dailyOrMoreOften = freq.compareTo(Frequency.DAILY) <= 0;
369                        if (byMonthDay.length > 0) {
370                                dayGenerator = Generators.byMonthDayGenerator(byMonthDay, start);
371                                byMonthDay = NO_INTS;
372                        } else if (byDay.length > 0) {
373                                dayGenerator = Generators.byDayGenerator(byDay, Frequency.YEARLY == freq, start);
374                                byDay = NO_DAYS;
375                        } else if (dailyOrMoreOften) {
376                                dayGenerator = Generators.serialDayGenerator(Frequency.DAILY == freq ? interval : 1, dtStart);
377                        } else {
378                                dayGenerator = Generators.byMonthDayGenerator(new int[] { dtStart.day() }, start);
379                        }
380                }
381
382                if (byDay.length > 0) {
383                        filters.add(Filters.byDayFilter(byDay, Frequency.YEARLY == freq, wkst));
384                        byDay = NO_DAYS;
385                }
386
387                if (byMonthDay.length > 0) {
388                        filters.add(Filters.byMonthDayFilter(byMonthDay));
389                }
390
391                //generator inference common to all periods
392                if (byMonth.length > 0) {
393                        monthGenerator = Generators.byMonthGenerator(byMonth, start);
394                } else if (monthGenerator == null) {
395                        monthGenerator = Generators.serialMonthGenerator(freq == Frequency.MONTHLY ? interval : 1, dtStart);
396                }
397
398                /*
399                 * The condition tells the iterator when to halt. The condition is
400                 * exclusive, so the date that triggers it will not be included.
401                 */
402                Predicate<DateValue> condition;
403                boolean canShortcutAdvance = true;
404                if (count != 0) {
405                        condition = Conditions.countCondition(count);
406
407                        /*
408                         * We can't shortcut because the countCondition must see every
409                         * generated instance.
410                         * 
411                         * TODO(msamuel): If count is large, we might try predicting the end
412                         * date so that we can convert the COUNT condition to an UNTIL
413                         * condition.
414                         */
415                        canShortcutAdvance = false;
416                } else if (untilUtc != null) {
417                        if ((untilUtc instanceof TimeValue) != (dtStart instanceof TimeValue)) {
418                                // TODO(msamuel): warn
419                                if (dtStart instanceof TimeValue) {
420                                        untilUtc = TimeUtils.dayStart(untilUtc);
421                                } else {
422                                        untilUtc = TimeUtils.toDateValue(untilUtc);
423                                }
424                        }
425                        condition = Conditions.untilCondition(untilUtc);
426                } else {
427                        condition = Predicates.<DateValue> alwaysTrue();
428                }
429
430                //combine filters into a single function
431                Predicate<? super DateValue> filter;
432                switch (filters.size()) {
433                case 0:
434                        filter = Predicates.<DateValue> alwaysTrue();
435                        break;
436                case 1:
437                        filter = filters.get(0);
438                        break;
439                default:
440                        filter = Predicates.and(filters);
441                        break;
442                }
443
444                Generator instanceGenerator;
445                if (bySetPos.length > 0) {
446                        instanceGenerator = InstanceGenerators.bySetPosInstanceGenerator(bySetPos, freq, wkst, filter, yearGenerator, monthGenerator, dayGenerator, hourGenerator, minuteGenerator, secondGenerator);
447                } else {
448                        instanceGenerator = InstanceGenerators.serialInstanceGenerator(filter, yearGenerator, monthGenerator, dayGenerator, hourGenerator, minuteGenerator, secondGenerator);
449                }
450
451                return new RRuleIteratorImpl(dtStart, tzid, condition, instanceGenerator, yearGenerator, monthGenerator, dayGenerator, hourGenerator, minuteGenerator, secondGenerator, canShortcutAdvance);
452        }

454        /**
455         * Generates a recurrence iterator that iterates over the union of the given
456         * recurrence iterators.
457         * @param first the first recurrence iterator
458         * @param rest the other recurrence iterators
459         * @return the union iterator
460         */
461        public static RecurrenceIterator join(RecurrenceIterator first, RecurrenceIterator... rest) {
462                List<RecurrenceIterator> all = new ArrayList<RecurrenceIterator>();
463                all.add(first);
464                all.addAll(Arrays.asList(rest));
465                return new CompoundIteratorImpl(all, Collections.<RecurrenceIterator> emptyList());
466        }

468        /**
469         * <p>
470         * Generates a recurrence iterator that iterates over all the dates in a
471         * {@link RecurrenceIterator}, excluding those dates found in another
472         * {@link RecurrenceIterator}.
473         * </p>
474         * <p>
475         * Exclusions trump inclusions, and {@link DateValue dates} and
476         * {@link DateTimeValue date-times} never match one another.
477         * </p>
478         * @param included the dates to include
479         * @param excluded the dates to exclude
480         * @return the resultant iterator
481         */
482        public static RecurrenceIterator except(RecurrenceIterator included, RecurrenceIterator excluded) {
483                return new CompoundIteratorImpl(Collections.<RecurrenceIterator> singleton(included), Collections.<RecurrenceIterator> singleton(excluded));
484        }

486        /**
487         * <p>
488         * Creates an optimized version of an array based on the given BYSETPOS
489         * array.
490         * </p>
491         * <p>
492         * For example, given the array <code>BYMONTH=2,3,4,5</code> and a BYSETPOS
493         * of <code>BYSETPOS=1,-1</code>, this method will return
494         * <code>BYMONTH=2,5</code>.
495         * </p>
496         * @param members the array to optimize
497         * @param bySetPos the BYSETPOS array
498         * @return the optimized array
499         */
500        private static int[] filterBySetPos(int[] members, int[] bySetPos) {
501                members = Util.uniquify(members);
502                IntSet iset = new IntSet();
503                for (int pos : bySetPos) {
504                        if (pos == 0) {
505                                continue;
506                        }
507                        if (pos < 0) {
508                                pos += members.length;
509                        } else {
510                                --pos; // Zero-index.
511                        }
512                        if (pos >= 0 && pos < members.length) {
513                                iset.add(members[pos]);
514                        }
515                }
516                return iset.toIntArray();
517        }

519        /**
520         * Converts an {@link Integer} list to an int array. Null values are
521         * converted to zero.
522         * @param list the {@link Integer} list
523         * @return the int array
524         */
525        private static int[] toIntArray(List<Integer> list) {
526                int[] array = new int[list.size()];
527                int i = 0;
528                for (Integer intObj : list) {
529                        array[i++] = toInt(intObj);
530                }
531                return array;
532        }
533
534        private static int toInt(Integer integer) {
535                return (integer == null) ? 0 : integer;
536        }

538        private static final int[] NO_INTS = new int[0];
539        private static final ByDay[] NO_DAYS = new ByDay[0];
