041/**
042 * Marshals {@link EmailAlarm} properties.
043 * @author Michael Angstadt
044 */
045public class EmailAlarmScribe extends VCalAlarmPropertyScribe<EmailAlarm> {
046        public EmailAlarmScribe() {
047                super(EmailAlarm.class, "MALARM");
048        }
049
050        @Override
051        protected ICalDataType _defaultDataType(ICalVersion version) {
052                return ICalDataType.TEXT;
053        }
054
055        @Override
056        protected List<String> writeData(EmailAlarm property) {
057                String email = property.getEmail();
058                String note = property.getNote();
059                if (email == null && note == null) {
060                        return Arrays.asList();
061                }
062
063                List<String> dataValues = new ArrayList<String>(2);
064                dataValues.add((email == null) ? "" : email);
065                dataValues.add((note == null) ? "" : note);
066                return dataValues;
067        }
068
069        @Override
070        protected EmailAlarm create(ICalDataType dataType, SemiStructuredValueIterator it) {
071                String email = it.next();
072                String note = it.next();
073
074                EmailAlarm property = new EmailAlarm(email);
075                property.setNote(note);
076                return property;
077        }
078
079        @Override
080        protected void toVAlarm(VAlarm valarm, EmailAlarm property) {
081                String email = property.getEmail();
082                if (email != null) {
083                        valarm.addAttendee(new Attendee(null, email));
084                }
085                valarm.setDescription(property.getNote());
086        }
087
088        @Override
089        protected Action action() {
090                return Action.email();
091        }
092}
