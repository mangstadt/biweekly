039/**
040 * <p>
041 * Defines a list of dates or time periods that help define a recurrence rule.
042 * It must contain either dates or time periods. It cannot contain a combination
043 * of both.
044 * </p>
045 * <p>
046 * <b>Code sample:</b>
047 * </p>
048 * 
049 * <pre class="brush:java">
050 * VEvent event = new VEvent();
051 * 
052 * //date-time values
053 * Date datetime = ...
054 * RecurrenceDates rdate = new RecurrenceDates();
055 * rdate.getDates().add(new ICalDate(datetime, true));
056 * event.addRecurrenceDates(rdate);
057 * 
058 * //date values
059 * Date date = ...
060 * RecurrenceDates rdate = new RecurrenceDates();
061 * rdate.getDates().add(new ICalDate(date, false));
062 * event.addRecurrenceDates(rdate);
063 * 
064 * //periods
065 * Period period = ...
066 * rdate = new RecurrenceDates();
067 * rdate.getPeriods().add(period);
068 * event.addRecurrenceDates(rdate);
069 * </pre>
070 * @author Michael Angstadt
071 * @see <a href="http://tools.ietf.org/html/rfc5545#page-120">RFC 5545
072 * p.120-2</a>
073 * @see <a href="http://tools.ietf.org/html/rfc2445#page-115">RFC 2445
074 * p.115-7</a>
075 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.34</a>
076 */
077public class RecurrenceDates extends ICalProperty {
078        private final List<ICalDate> dates;
079        private final List<Period> periods;
080
081        public RecurrenceDates() {
082                dates = new ArrayList<ICalDate>();
083                periods = new ArrayList<Period>();
084        }
085
086        /**
087         * Copy constructor.
088         * @param original the property to make a copy of
089         */
090        public RecurrenceDates(RecurrenceDates original) {
091                super(original);
092
093                dates = new ArrayList<ICalDate>(original.dates.size());
094                for (ICalDate date : original.dates) {
095                        dates.add(new ICalDate(date));
096                }
097
098                periods = new ArrayList<Period>(original.periods.size());
099                for (Period period : original.periods) {
100                        periods.add(new Period(period));
101                }
102        }
103
104        /**
105         * Gets the list that stores this property's recurrence dates.
106         * @return the dates (this list is mutable)
107         */
108        public List<ICalDate> getDates() {
109                return dates;
110        }
111
112        /**
113         * Gets the list that stores this property's time periods.
114         * @return the time periods (this list is mutable)
115         */
116        public List<Period> getPeriods() {
117                return periods;
118        }
119
120        @Override
121        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
122                if (dates.isEmpty() && periods.isEmpty()) {
123                        //no value
124                        warnings.add(new ValidationWarning(26));
125                }
126
127                if (!dates.isEmpty() && !periods.isEmpty()) {
128                        //can't mix dates and periods
129                        warnings.add(new ValidationWarning(49));
130                }
131
132                if (version == ICalVersion.V1_0 && !periods.isEmpty()) {
133                        //1.0 doesn't support periods
134                        warnings.add(new ValidationWarning(51));
135                }
136
137                if (!dates.isEmpty()) {
138                        //can't mix date and date-time values
139                        boolean hasTime = dates.get(0).hasTime();
140                        for (ICalDate date : dates.subList(1, dates.size())) {
141                                if (date.hasTime() != hasTime) {
142                                        warnings.add(new ValidationWarning(50));
143                                        break;
144                                }
145                        }
146                }
147        }
148
149        @Override
150        protected Map<String, Object> toStringValues() {
151                Map<String, Object> values = new LinkedHashMap<String, Object>();
152                values.put("dates", dates);
153                values.put("periods", periods);
154                return values;
155        }
156
157        @Override
158        public RecurrenceDates copy() {
159                return new RecurrenceDates(this);
160        }
161
162        @Override
163        public int hashCode() {
164                final int prime = 31;
165                int result = super.hashCode();
166                result = prime * result + dates.hashCode();
167                result = prime * result + periods.hashCode();
168                return result;
169        }
170
171        @Override
172        public boolean equals(Object obj) {
173                if (this == obj) return true;
174                if (!super.equals(obj)) return false;
175                RecurrenceDates other = (RecurrenceDates) obj;
176                if (!dates.equals(other.dates)) return false;
177                if (!periods.equals(other.periods)) return false;
178                return true;
179        }
180}
