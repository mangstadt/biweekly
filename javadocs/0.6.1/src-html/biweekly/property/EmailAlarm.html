032/**
033 * Defines an alarm that sends an email when triggered. It is recommended that
034 * the {@link VAlarm} component be used to create alarms.
035 * @author Michael Angstadt
036 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.32</a>
037 * @see VAlarm#email
038 */
039public class EmailAlarm extends VCalAlarmProperty {
040        private String email, note;
041
042        public EmailAlarm(String email) {
043                this.email = email;
044        }
045
046        /**
047         * Copy constructor.
048         * @param original the property to make a copy of
049         */
050        public EmailAlarm(EmailAlarm original) {
051                super(original);
052                email = original.email;
053                note = original.note;
054        }
055
056        /**
057         * Gets the email address.
058         * @return the email address
059         */
060        public String getEmail() {
061                return email;
062        }
063
064        /**
065         * Sets the email address.
066         * @param email the email address
067         */
068        public void setEmail(String email) {
069                this.email = email;
070        }
071
072        /**
073         * Gets the note to send.
074         * @return the note
075         */
076        public String getNote() {
077                return note;
078        }
079
080        /**
081         * Sets the note to send
082         * @param note the note
083         */
084        public void setNote(String note) {
085                this.note = note;
086        }
087
088        @Override
089        protected Map<String, Object> toStringValues() {
090                Map<String, Object> values = super.toStringValues();
091                values.put("email", email);
092                values.put("note", note);
093                return values;
094        }
095
096        @Override
097        public EmailAlarm copy() {
098                return new EmailAlarm(this);
099        }
100
101        @Override
102        public int hashCode() {
103                final int prime = 31;
104                int result = super.hashCode();
105                result = prime * result + ((email == null) ? 0 : email.hashCode());
106                result = prime * result + ((note == null) ? 0 : note.hashCode());
107                return result;
108        }
109
110        @Override
111        public boolean equals(Object obj) {
112                if (this == obj) return true;
113                if (!super.equals(obj)) return false;
114                EmailAlarm other = (EmailAlarm) obj;
115                if (email == null) {
116                        if (other.email != null) return false;
117                } else if (!email.equals(other.email)) return false;
118                if (note == null) {
119                        if (other.note != null) return false;
120                } else if (!note.equals(other.note)) return false;
121                return true;
122        }
123}
