040/**
041 * Marshals {@link Attendee} properties.
042 * @author Michael Angstadt
043 */
044public class AttendeeScribe extends ICalPropertyScribe<Attendee> {
045        public AttendeeScribe() {
046                super(Attendee.class, "ATTENDEE");
047        }
048
049        @Override
050        protected ICalDataType _defaultDataType(ICalVersion version) {
051                switch (version) {
052                case V1_0:
053                        return null;
054                default:
055                        return ICalDataType.CAL_ADDRESS;
056                }
057        }
058
059        @Override
060        protected ICalDataType _dataType(Attendee property, ICalVersion version) {
061                if (version == ICalVersion.V1_0 && property.getUri() != null) {
062                        return ICalDataType.URL;
063                }
064                return defaultDataType(version);
065        }
066
067        @Override
068        protected ICalParameters _prepareParameters(Attendee property, WriteContext context) {
069                /*
070                 * Note: Parameter values are assigned using "put()" instead of the
071                 * appropriate "setter" methods so that any existing parameter values
072                 * are not overwritten.
073                 */
074
075                ICalParameters copy = new ICalParameters(property.getParameters());
076
077                //RSVP parameter
078                //1.0 - Uses the values "YES" and "NO"
079                //2.0 - Uses the values "TRUE" and "FALSE"
080                Boolean rsvp = property.getRsvp();
081                if (rsvp != null) {
082                        String value;
083                        switch (context.getVersion()) {
084                        case V1_0:
085                                value = rsvp ? "YES" : "NO";
086                                break;
087
088                        default:
089                                value = rsvp ? "TRUE" : "FALSE";
090                                break;
091                        }
092
093                        copy.put(ICalParameters.RSVP, value);
094                }
095
096                //ROLE and EXPECT parameters
097                //1.0 - Uses ROLE and EXPECT
098                //2.0 - Uses only ROLE
099                Role role = property.getRole();
100                ParticipationLevel level = property.getParticipationLevel();
101                switch (context.getVersion()) {
102                case V1_0:
103                        if (role != null) {
104                                copy.put(ICalParameters.ROLE, role.getValue());
105                        }
106                        if (level != null) {
107                                copy.put(ICalParameters.EXPECT, level.getValue(context.getVersion()));
108                        }
109                        break;
110
111                default:
112                        String value = null;
113                        if (role == Role.CHAIR) {
114                                value = role.getValue();
115                        } else if (level != null) {
116                                value = level.getValue(context.getVersion());
117                        } else if (role != null) {
118                                value = role.getValue();
119                        }
120
121                        if (value != null) {
122                                copy.put(ICalParameters.ROLE, value);
123                        }
124                        break;
125                }
126
127                //PARTSTAT vs STATUS
128                //1.0 - Calls the parameter "STATUS"
129                //2.0 - Calls the parameter "PARTSTAT"
130                ParticipationStatus partStat = property.getParticipationStatus();
131                if (partStat != null) {
132                        String paramName;
133                        String paramValue;
134
135                        switch (context.getVersion()) {
136                        case V1_0:
137                                paramName = ICalParameters.STATUS;
138                                paramValue = (partStat == ParticipationStatus.NEEDS_ACTION) ? "NEEDS ACTION" : partStat.getValue();
139                                break;
140
141                        default:
142                                paramName = ICalParameters.PARTSTAT;
143                                paramValue = partStat.getValue();
144                                break;
145                        }
146
147                        copy.put(paramName, paramValue);
148                }
149
150                //CN parameter
151                String name = property.getCommonName();
152                if (name != null && context.getVersion() != ICalVersion.V1_0) {
153                        copy.put(ICalParameters.CN, name);
154                }
155
156                //EMAIL parameter
157                String uri = property.getUri();
158                String email = property.getEmail();
159                if (uri != null && email != null && context.getVersion() != ICalVersion.V1_0) {
160                        copy.put(ICalParameters.EMAIL, email);
161                }
162
163                return copy;
164        }
165
166        @Override
167        protected Attendee _parseText(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
168                String uri = null, name = null, email = null;
169                Boolean rsvp = null;
170                Role role = null;
171                ParticipationLevel participationLevel = null;
172                ParticipationStatus participationStatus = null;
173
174                switch (context.getVersion()) {
175                case V1_0:
176                        Iterator<String> it = parameters.get(ICalParameters.RSVP).iterator();
177                        while (it.hasNext()) {
178                                String rsvpStr = it.next();
179
180                                if ("YES".equalsIgnoreCase(rsvpStr)) {
181                                        rsvp = Boolean.TRUE;
182                                        it.remove();
183                                        break;
184                                }
185
186                                if ("NO".equalsIgnoreCase(rsvpStr)) {
187                                        rsvp = Boolean.FALSE;
188                                        it.remove();
189                                        break;
190                                }
191                        }
192
193                        String roleStr = parameters.first(ICalParameters.ROLE);
194                        if (roleStr != null) {
195                                role = Role.get(roleStr);
196                                parameters.remove(ICalParameters.ROLE, roleStr);
197                        }
198
199                        String expectStr = parameters.getExpect();
200                        if (expectStr != null) {
201                                participationLevel = ParticipationLevel.get(expectStr);
202                                parameters.remove(ICalParameters.EXPECT, expectStr);
203                        }
204
205                        String statusStr = parameters.getStatus();
206                        if (statusStr != null) {
207                                participationStatus = ParticipationStatus.get(statusStr);
208                                parameters.remove(ICalParameters.STATUS, statusStr);
209                        }
210
211                        int bracketStart = value.lastIndexOf('<');
212                        int bracketEnd = value.lastIndexOf('>');
213                        if (bracketStart >= 0 && bracketEnd >= 0 && bracketStart < bracketEnd) {
214                                name = value.substring(0, bracketStart).trim();
215                                email = value.substring(bracketStart + 1, bracketEnd).trim();
216                        } else if (dataType == ICalDataType.URL) {
217                                uri = value;
218                        } else {
219                                email = value;
220                        }
221
222                        break;
223
224                default:
225                        it = parameters.get(ICalParameters.RSVP).iterator();
226                        while (it.hasNext()) {
227                                String rsvpStr = it.next();
228
229                                if ("TRUE".equalsIgnoreCase(rsvpStr)) {
230                                        rsvp = Boolean.TRUE;
231                                        it.remove();
232                                        break;
233                                }
234
235                                if ("FALSE".equalsIgnoreCase(rsvpStr)) {
236                                        rsvp = Boolean.FALSE;
237                                        it.remove();
238                                        break;
239                                }
240                        }
241
242                        roleStr = parameters.first(ICalParameters.ROLE);
243                        if (roleStr != null) {
244                                if (roleStr.equalsIgnoreCase(Role.CHAIR.getValue())) {
245                                        role = Role.CHAIR;
246                                } else {
247                                        ParticipationLevel l = ParticipationLevel.find(roleStr);
248                                        if (l == null) {
249                                                role = Role.get(roleStr);
250                                        } else {
251                                                participationLevel = l;
252                                        }
253                                }
254                                parameters.remove(ICalParameters.ROLE, roleStr);
255                        }
256
257                        String participationStatusStr = parameters.getParticipationStatus();
258                        if (participationStatusStr != null) {
259                                participationStatus = ParticipationStatus.get(participationStatusStr);
260                                parameters.remove(ICalParameters.PARTSTAT, participationStatusStr);
261                        }
262
263                        name = parameters.getCommonName();
264                        if (name != null) {
265                                parameters.remove(ICalParameters.CN, name);
266                        }
267
268                        email = parameters.getEmail();
269                        if (email == null) {
270                                int colon = value.indexOf(':');
271                                if (colon == 6) {
272                                        String scheme = value.substring(0, colon);
273                                        if (scheme.equalsIgnoreCase("mailto")) {
274                                                email = value.substring(colon + 1);
275                                        } else {
276                                                uri = value;
277                                        }
278                                } else {
279                                        uri = value;
280                                }
281                        } else {
282                                uri = value;
283                                parameters.remove(ICalParameters.EMAIL, email);
284                        }
285
286                        break;
287                }
288
289                Attendee attendee = new Attendee(name, email, uri);
290                attendee.setParticipationStatus(participationStatus);
291                attendee.setParticipationLevel(participationLevel);
292                attendee.setRole(role);
293                attendee.setRsvp(rsvp);
294
295                return attendee;
296        }
297
298        @Override
299        protected String _writeText(Attendee property, WriteContext context) {
300                String uri = property.getUri();
301                if (uri != null) {
302                        return uri;
303                }
304
305                String name = property.getCommonName();
306                String email = property.getEmail();
307                switch (context.getVersion()) {
308                case V1_0:
309                        if (name != null && email != null) {
310                                return escape(name + " <" + email + ">");
311                        }
312                        if (email != null) {
313                                return escape(email);
314                        }
315
316                        break;
317
318                default:
319                        if (email != null) {
320                                return "mailto:" + email;
321                        }
322                        break;
323                }
324
325                return "";
326        }
