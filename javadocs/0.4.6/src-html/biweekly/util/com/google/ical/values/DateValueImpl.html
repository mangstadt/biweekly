018package biweekly.util.com.google.ical.values;
019
020/**
021 * A calendar date.
022 *
023 * @author Neal Gafter
024 */
025public class DateValueImpl implements DateValue {
026  private final int year, month, day;
027
028  public DateValueImpl(int year, int month, int day) {
029    this.year = year;
030    this.month = month;
031    this.day = day;
032  }
033
034  public int year() {
035    return year;
036  }
037
038  public int month() {
039    return month;
040  }
041
042  public int day() {
043    return day;
044  }
045
046  @Override
047  public String toString() {
048    return String.format("%04d%02d%02d", year, month, day);
049  }
050
051  public final int compareTo(DateValue other) {
052    int n0 = this.day() +               // 5 bits
053             (this.month() << 5) +      // 4 bits
054             (this.year() << 9);
055    int n1 = other.day() +
056             (other.month() << 5) +
057             (other.year() << 9);
058    if (n0 != n1) return n0 - n1;
059    if (!(this instanceof TimeValue))
060      return (other instanceof TimeValue) ? -1 : 0;
061
062    TimeValue self = (TimeValue) this;
063    if (!(other instanceof TimeValue)) return 1;
064    TimeValue othr = (TimeValue) other;
065    int m0 = self.second() +            // 6 bits
066             (self.minute() << 6) +     // 6 bits
067             (self.hour() << 12);
068    int m1 = othr.second() +
069             (othr.minute() << 6) +
070             (othr.hour() << 12);
071    return m0 - m1;
072  }
073
074  @Override
075  public boolean equals(Object o) {
076    if (!(o instanceof DateValue)) { return false; }
077
078    return 0 == this.compareTo((DateValue) o);
079  }
080
081  @Override
082  public int hashCode() {
083    return (this.year << 9) + (this.month << 5) + this.day;
084  }
085}
