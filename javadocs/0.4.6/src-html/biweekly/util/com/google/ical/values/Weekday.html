015package biweekly.util.com.google.ical.values;
016
017import biweekly.util.com.google.ical.util.TimeUtils;
018
019/**
020 * days of the week enum.  Names correspond to RFC2445 literals.
021 *
022 * @author mikesamuel+svn@gmail.com (Mike Samuel)
023 */
024public enum Weekday {
025  SU(0),
026  MO(1),
027  TU(2),
028  WE(3),
029  TH(4),
030  FR(5),
031  SA(6),
032  ;
033
034  /**
035   * agrees with values returned by the javascript builtin Date.getDay and the
036   * corresponding ical.js function/method, but is one less than the Java
037   * Calendar's <code>DAY_OF_WEEK</code> values.
038   */
039  public final int jsDayNum;
040  /**
041   * agrees with the java weekday values as found in java.util.Calendar.
042   */
043  public final int javaDayNum;
044
045  Weekday(int wDayNum) {
046    this.jsDayNum = wDayNum;
047    this.javaDayNum = 1 + wDayNum;
048  }
049
050  private static Weekday[] VALUES = new Weekday[7];
051
052  static {
053    System.arraycopy(values(), 0, VALUES, 0, 7);
054  }
055
056  public static Weekday valueOf(DateValue dv) {
057    int dayIndex =
058      TimeUtils.fixedFromGregorian(dv.year(), dv.month(), dv.day()) % 7;
059    if (dayIndex < 0) { dayIndex += 7; }
060    return VALUES[dayIndex];
061  }
062
063  public static Weekday firstDayOfWeekInMonth(int year, int month) {
064    int result = TimeUtils.fixedFromGregorian(year, month, 1) % 7;
065    return VALUES[(result >= 0) ? result : result + 7];
066  }
067
068  public Weekday successor() {
069    return VALUES[(ordinal() + 1) % 7];
070  }
071
072  public Weekday predecessor() {
073    return VALUES[(ordinal() - 1 + 7) % 7];
074  }
075
076}
