033/**
034 * <p>
035 * Specifies the type of <a href="http://tools.ietf.org/html/rfc5546">iTIP</a>
036 * request that the iCalendar object represents. If the iCalendar object is just
037 * being used as a container to hold calendar information, then this property
038 * does not need to be defined.
039 * </p>
040 * <p>
041 * If the iCalendar object is defined as a MIME message entity, this property
042 * MUST be set to the value of the "Content-Type" header's "method" parameter,
043 * if present.
044 * </p>
045 * <p>
046 * <b>Code sample:</b>
047 * </p>
048 * 
049 * <pre class="brush:java">
050 * ICalendar ical = new ICalendar();
051 * 
052 * Method method = Method.request();
053 * ical.setMethod(method);
054 * </pre>
055 * @author Michael Angstadt
056 * @see <a href="http://tools.ietf.org/html/rfc5546">RFC 5546</a>
057 * @see <a href="http://tools.ietf.org/html/rfc5545#page-77">RFC 5545 p.77-8</a>
058 * @see <a href="http://tools.ietf.org/html/rfc2445#page-74">RFC 2445 p.74-5</a>
059 */
060public class Method extends EnumProperty {
061        private static final String ADD = "ADD";
062        private static final String CANCEL = "CANCEL";
063        private static final String COUNTER = "COUNTER";
064        private static final String DECLINECOUNTER = "DECLINECOUNTER";
065        private static final String PUBLISH = "PUBLISH";
066        private static final String REFRESH = "REFRESH";
067        private static final String REPLY = "REPLY";
068        private static final String REQUEST = "REQUEST";
069
070        /**
071         * Creates a new method property.
072         * @param value the property value
073         */
074        public Method(String value) {
075                super(value);
076        }
077
078        /**
079         * Copy constructor.
080         * @param original the property to make a copy of
081         */
082        public Method(Method original) {
083                super(original);
084        }
085
086        /**
087         * Constructs a METHOD property whose value is "ADD".
088         * @return the property
089         */
090        public static Method add() {
091                return create(ADD);
092        }
093
094        /**
095         * Determines if this property's value is "ADD".
096         * @return true if the value is "ADD", false if not
097         */
098        public boolean isAdd() {
099                return is(ADD);
100        }
101
102        /**
103         * Constructs a METHOD property whose value is "CANCEL".
104         * @return the property
105         */
106        public static Method cancel() {
107                return create(CANCEL);
108        }
109
110        /**
111         * Determines if this property's value is "CANCEL".
112         * @return true if the value is "CANCEL", false if not
113         */
114        public boolean isCancel() {
115                return is(CANCEL);
116        }
117
118        /**
119         * Constructs a METHOD property whose value is "COUNTER".
120         * @return the property
121         */
122        public static Method counter() {
123                return create(COUNTER);
124        }
125
126        /**
127         * Determines if this property's value is "COUNTER".
128         * @return true if the value is "COUNTER", false if not
129         */
130        public boolean isCounter() {
131                return is(COUNTER);
132        }
133
134        /**
135         * Constructs a METHOD property whose value is "DECLINECOUNTER".
136         * @return the property
137         */
138        public static Method declineCounter() {
139                return create(DECLINECOUNTER);
140        }
141
142        /**
143         * Determines if this property's value is "DECLINECOUNTER".
144         * @return true if the value is "DECLINECOUNTER", false if not
145         */
146        public boolean isDeclineCounter() {
147                return is(DECLINECOUNTER);
148        }
149
150        /**
151         * Constructs a METHOD property whose value is "PUBLISH".
152         * @return the property
153         */
154        public static Method publish() {
155                return create(PUBLISH);
156        }
157

159         * Determines if this property's value is "PUBLISH".

162        public boolean isPublish() {
163                return is(PUBLISH);
164        }

167         * Constructs a METHOD property whose value is "REFRESH".
168         * @return the property

170        public static Method refresh() {
171                return create(REFRESH);
172        }
173
174        /**
175         * Determines if this property's value is "REFRESH".
176         * @return true if the value is "REFRESH", false if not
177         */
178        public boolean isRefresh() {
179                return is(REFRESH);
180        }
181
182        /**
183         * Constructs a METHOD property whose value is "REPLY".
184         * @return the property
185         */
186        public static Method reply() {
187                return create(REPLY);
188        }
189
190        /**
191         * Determines if this property's value is "REPLY".
192         * @return true if the value is "REPLY", false if not
193         */
194        public boolean isReply() {
195                return is(REPLY);
196        }

199         * Constructs a METHOD property whose value is "REQUEST".
200         * @return the property
201         */
202        public static Method request() {
203                return create(REQUEST);
204        }

207         * Determines if this property's value is "REQUEST".
208         * @return true if the value is "REQUEST", false if not

211                return is(REQUEST);
212        }
213
214        private static Method create(String value) {
215                return new Method(value);
216        }
217
218        @Override
219        protected Collection<String> getStandardValues(ICalVersion version) {
220                return Arrays.asList(ADD, CANCEL, COUNTER, DECLINECOUNTER, PUBLISH, REFRESH, REPLY, REQUEST);
221        }
222
223        @Override
224        public Method copy() {
225                return new Method(this);
226        }
