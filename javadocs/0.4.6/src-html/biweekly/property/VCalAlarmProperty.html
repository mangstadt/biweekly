035/**
036 * <p>
037 * Defines an alarm property that is part of the vCalendar (1.0) standard (such
038 * as {@link AudioAlarm}).
039 * </p>
040 * <p>
041 * Classes that extend this class are used internally by this library for
042 * parsing purposes. If you are creating a new iCalendar object and need to
043 * define an alarm, it is recommended that you use the {@link VAlarm} component
044 * to create a new alarm.
045 * </p>
046 * @author Michael Angstadt
047 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0</a>
048 */
049public class VCalAlarmProperty extends ICalProperty {
050        protected Date start;
051        protected Duration snooze;
052        protected Integer repeat;
053
054        public VCalAlarmProperty() {
055                //empty
056        }
057
058        /**
059         * Copy constructor.
060         * @param original the property to make a copy of
061         */
062        public VCalAlarmProperty(VCalAlarmProperty original) {
063                super(original);
064                start = new Date(original.start.getTime());
065                snooze = original.snooze;
066                repeat = original.repeat;
067        }
068
069        public Date getStart() {
070                return start;
071        }
072
073        public void setStart(Date start) {
074                this.start = start;
075        }
076
077        public Duration getSnooze() {
078                return snooze;
079        }
080
081        public void setSnooze(Duration snooze) {
082                this.snooze = snooze;
083        }
084
085        public Integer getRepeat() {
086                return repeat;
087        }
088
089        public void setRepeat(Integer repeat) {
090                this.repeat = repeat;
091        }
092
093        @Override
094        protected Map<String, Object> toStringValues() {
095                Map<String, Object> values = new LinkedHashMap<String, Object>();
096                values.put("start", start);
097                values.put("snooze", snooze);
098                values.put("repeat", repeat);
099                return values;
100        }
101
102        @Override
103        public int hashCode() {
104                final int prime = 31;
105                int result = super.hashCode();
106                result = prime * result + ((repeat == null) ? 0 : repeat.hashCode());
107                result = prime * result + ((snooze == null) ? 0 : snooze.hashCode());
108                result = prime * result + ((start == null) ? 0 : start.hashCode());
109                return result;
110        }
111
112        @Override
113        public boolean equals(Object obj) {
114                if (this == obj) return true;
115                if (!super.equals(obj)) return false;
116                VCalAlarmProperty other = (VCalAlarmProperty) obj;
117                if (repeat == null) {
118                        if (other.repeat != null) return false;
119                } else if (!repeat.equals(other.repeat)) return false;
120                if (snooze == null) {
121                        if (other.snooze != null) return false;
122                } else if (!snooze.equals(other.snooze)) return false;
123                if (start == null) {
124                        if (other.start != null) return false;
125                } else if (!start.equals(other.start)) return false;
126                return true;
127        }
128}
