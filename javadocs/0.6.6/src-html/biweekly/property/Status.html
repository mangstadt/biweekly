037/**
038 * <p>
039 * Defines the status of the component that this property belongs to, such as a
040 * to-do task being "completed".
041 * </p>
042 * 
043 * <p>
044 * <b>Code sample (creating):</b>
045 * </p>
046 * 
047 * <pre class="brush:java">
048 * VTodo todo = new VTodo();
049 * 
050 * Status status = Status.completed();
051 * todo.setStatus(status);
052 * </pre>
053 * 
054 * <p>
055 * <b>Code sample (retrieving):</b>
056 * </p>
057 * 
058 * <pre class="brush:java">
059 * ICalendar ical = ...
060 * for (VTodo todo : ical.getTodos()) {
061 *   Status status = todo.getStatus();
062 *   if (action.isCompleted()) {
063 *     //...
064 *   } else if (action.isDraft()) {
065 *     //...
066 *   }
067 *   //etc.
068 * }
069 * </pre>
070 * @author Michael Angstadt
071 * @see <a href="http://tools.ietf.org/html/rfc5545#page-92">RFC 5545 p.92-3</a>
072 * @see <a href="http://tools.ietf.org/html/rfc2445#page-88">RFC 2445 p.88-9</a>
073 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.35-6</a>
074 */

313
314        @Override
315        protected Collection<String> getStandardValues(ICalVersion version) {
316                switch (version) {
317                case V1_0:
318                        return Arrays.asList(ACCEPTED, COMPLETED, CONFIRMED, DECLINED, DELEGATED, NEEDS_ACTION, SENT, TENTATIVE);
319                default:
320                        return Arrays.asList(CANCELLED, COMPLETED, CONFIRMED, DRAFT, FINAL, IN_PROGRESS, NEEDS_ACTION, TENTATIVE);
321                }
322        }
323
324        @Override
325        protected Collection<ICalVersion> getValueSupportedVersions() {
326                if (value == null) {
327                        return Collections.emptyList();
328                }
329
330                if (isCompleted() || isConfirmed() || isNeedsAction() || isTentative()) {
331                        return Arrays.asList(ICalVersion.values());
332                }
333                if (isCancelled() || isDraft() || isFinal() || isInProgress()) {
334                        return Arrays.asList(ICalVersion.V2_0_DEPRECATED, ICalVersion.V2_0);
335                }
336                if (isAccepted() || isDeclined() || isDelegated() || isSent()) {
337                        return Collections.singletonList(ICalVersion.V1_0);
338                }
339
340                return Collections.emptyList();
341        }
342
343        @Override
344        public Status copy() {
345                return new Status(this);
346        }
