030/**
031 * Gets Java {@link TimeZone} objects that correspond with TZID parameters that
032 * contain global timezone IDs (as opposed to IDs that correspond with a
033 * VTIMEZONE component).
034 * @author Michael Angstadt
035 * @see <a href="https://tools.ietf.org/html/rfc5545#section-3.8.3.1">RFC 5545
036 * section 3.8.3.1</a>
037 * @see <a href="https://tools.ietf.org/html/rfc2445#section-4.2.19">RFC 2445
038 * section 4.2.19</a>
039 */
040public interface GlobalTimezoneIdResolver {
041        /**
042         * Returns an appropriate Java {@link TimeZone} object that corresponds to
043         * the given global ID.
044         * @param globalId the global ID (the value of the TZID parameter, without
045         * the forward slash prefix)
046         * @return the corresponding {@link TimeZone} object or null if the global
047         * ID is not recognized
048         */
049        TimeZone resolve(String globalId);
050}
