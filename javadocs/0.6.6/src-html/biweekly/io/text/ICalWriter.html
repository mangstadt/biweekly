061/**
062 * <p>
063 * Writes {@link ICalendar} objects to a plain-text iCalendar data stream.
064 * </p>
065 * <p>
066 * <b>Example:</b>
067 * </p>
068 * 
069 * <pre class="brush:java">
070 * ICalendar ical1 = ...
071 * ICalendar ical2 = ...
072 * File file = new File("icals.ics");
073 * ICalWriter writer = null;
074 * try {
075 *   writer = new ICalWriter(file, ICalVersion.V2_0);
076 *   writer.write(ical1);
077 *   writer.write(ical2);
078 * } finally {
079 *   if (writer != null) writer.close();
080 * }
081 * </pre>
082 * 
083 * <p>
084 * <b>Changing the line folding settings:</b>
085 * </p>
086 * 
087 * <pre class="brush:java">
088 * ICalWriter writer = new ICalWriter(...);
089 * 
090 * //disable line folding
091 * writer.getVObjectWriter().getFoldedLineWriter().setLineLength(null);
092 * 
093 * //set line length (defaults to 75)
094 * writer.getVObjectWriter().getFoldedLineWriter().setLineLength(50);
095 * 
096 * //change folded line indent string (defaults to one space character)
097 * writer.getVObjectWriter().getFoldedLineWriter().setIndent("\t");
098 *
099 * </pre>
100 * @author Michael Angstadt
101 * @see <a href="http://www.imc.org/pdi/pdiproddev.html">1.0 specs</a>
102 * @see <a href="https://tools.ietf.org/html/rfc2445">RFC 2445</a>
103 * @see <a href="http://tools.ietf.org/html/rfc5545">RFC 5545</a>
104 */
105public class ICalWriter extends StreamWriter implements Flushable {
106        private final VObjectWriter writer;
107        private ICalVersion targetVersion;
108
109        /**
110         * Creates a new iCalendar writer.
111         * @param out the output stream to write to
112         * @param targetVersion the iCalendar version to adhere to
113         */
114        public ICalWriter(OutputStream out, ICalVersion targetVersion) {
115                this((targetVersion == ICalVersion.V1_0) ? new OutputStreamWriter(out) : new Utf8Writer(out), targetVersion);
116        }
117
118        /**
119         * Creates a new iCalendar writer.
120         * @param file the file to write to
121         * @param targetVersion the iCalendar version to adhere to
122         * @throws IOException if the file cannot be written to
123         */
124        public ICalWriter(File file, ICalVersion targetVersion) throws IOException {
125                this(file, false, targetVersion);
126        }
127
128        /**
129         * Creates a new iCalendar writer.
130         * @param file the file to write to
131         * @param targetVersion the iCalendar version to adhere to
132         * @param append true to append to the end of the file, false to overwrite
133         * it
134         * @throws IOException if the file cannot be written to
135         */
136        public ICalWriter(File file, boolean append, ICalVersion targetVersion) throws IOException {
137                this((targetVersion == ICalVersion.V1_0) ? new FileWriter(file, append) : new Utf8Writer(file, append), targetVersion);
138        }
139
140        /**
141         * Creates a new iCalendar writer.
142         * @param writer the writer to write to
143         * @param targetVersion the iCalendar version to adhere to
144         */
145        public ICalWriter(Writer writer, ICalVersion targetVersion) {
146                this.writer = new VObjectWriter(writer, targetVersion.getSyntaxStyle());
147                this.targetVersion = targetVersion;
148        }
149
150        /**
151         * Gets the writer object that is used internally to write to the output
152         * stream.
153         * @return the raw writer
154         */
155        public VObjectWriter getVObjectWriter() {
156                return writer;
157        }
158
159        /**
160         * Gets the version that the written iCalendar objects will adhere to.
161         * @return the iCalendar version
162         */
163        @Override
164        public ICalVersion getTargetVersion() {
165                return targetVersion;
166        }
167
168        /**
169         * Sets the version that the written iCalendar objects will adhere to.
170         * @param targetVersion the iCalendar version
171         */
172        public void setTargetVersion(ICalVersion targetVersion) {
173                this.targetVersion = targetVersion;
174                writer.setSyntaxStyle(targetVersion.getSyntaxStyle());
175        }
176
177        /**
178         * <p>
179         * Gets whether the writer will apply circumflex accent encoding on
180         * parameter values (disabled by default). This escaping mechanism allows
181         * for newlines and double quotes to be included in parameter values. It is
182         * only supported by iCalendar version 2.0.
183         * </p>
184         * @return true if circumflex accent encoding is enabled, false if not
185         * @see VObjectWriter#isCaretEncodingEnabled()
186         */
187        public boolean isCaretEncodingEnabled() {
188                return writer.isCaretEncodingEnabled();
189        }
190
191        /**
192         * <p>
193         * Sets whether the writer will apply circumflex accent encoding on
194         * parameter values (disabled by default). This escaping mechanism allows
195         * for newlines and double quotes to be included in parameter values. It is
196         * only supported by iCalendar version 2.0.
197         * </p>
198         * <p>
199         * Note that this encoding mechanism is defined separately from the
200         * iCalendar specification and may not be supported by the consumer of the
201         * iCalendar object.
202         * </p>
203         * @param enable true to use circumflex accent encoding, false not to
204         * @see VObjectWriter#setCaretEncodingEnabled(boolean)
205         */
206        public void setCaretEncodingEnabled(boolean enable) {
207                writer.setCaretEncodingEnabled(enable);
208        }
209
210        @Override
211        protected void _write(ICalendar ical) throws IOException {
212                writeComponent(ical, null);
213        }
214
215        /**
216         * Writes a component to the data stream.
217         * @param component the component to write
218         * @param parent the parent component
219         * @throws IOException if there's a problem writing to the data stream
220         */
221        @SuppressWarnings({ "rawtypes", "unchecked" })
222        private void writeComponent(ICalComponent component, ICalComponent parent) throws IOException {
223                boolean inICalendar = component instanceof ICalendar;
224                boolean inVCalRoot = inICalendar && getTargetVersion() == ICalVersion.V1_0;
225                boolean inICalRoot = inICalendar && getTargetVersion() != ICalVersion.V1_0;
226
227                ICalComponentScribe componentScribe = index.getComponentScribe(component);
228                try {
229                        componentScribe.checkForDataModelConversions(component, parent, getTargetVersion());
230                } catch (DataModelConversionException e) {
231                        for (ICalComponent c : e.getComponents()) {
232                                writeComponent(c, parent);
233                        }
234                        for (ICalProperty p : e.getProperties()) {
235                                writeProperty(p);
236                        }
237                        return;
238                }
239
240                writer.writeBeginComponent(componentScribe.getComponentName());
241
242                List propertyObjs = componentScribe.getProperties(component);
243                if (inICalendar && component.getProperty(Version.class) == null) {
244                        propertyObjs.add(0, new Version(getTargetVersion()));
245                }
246
247                for (Object propertyObj : propertyObjs) {
248                        context.setParent(component); //set parent here incase a scribe resets the parent
249                        ICalProperty property = (ICalProperty) propertyObj;
250                        writeProperty(property);
251                }
252
253                List subComponents = componentScribe.getComponents(component);
254                if (inICalRoot) {
255                        //add the VTIMEZONE components
256                        Collection<VTimezone> timezones = getTimezoneComponents();
257                        for (VTimezone timezone : timezones) {
258                                if (!subComponents.contains(timezone)) {
259                                        subComponents.add(0, timezone);
260                                }
261                        }
262                }
263
264                for (Object subComponentObj : subComponents) {
265                        ICalComponent subComponent = (ICalComponent) subComponentObj;
266                        writeComponent(subComponent, component);
267                }
268
269                if (inVCalRoot) {
270                        Collection<VTimezone> timezones = getTimezoneComponents();
271                        if (!timezones.isEmpty()) {
272                                VTimezone timezone = timezones.iterator().next();
273                                VCalTimezoneProperties props = convert(timezone, context.getDates());
274
275                                Timezone tz = props.getTz();
276                                if (tz != null) {
277                                        writeProperty(tz);
278                                }
279                                for (Daylight daylight : props.getDaylights()) {
280                                        writeProperty(daylight);
281                                }
282                        }
283                }
284
285                writer.writeEndComponent(componentScribe.getComponentName());
286        }
287
288        @SuppressWarnings({ "rawtypes", "unchecked" })
289        private void writeProperty(ICalProperty property) throws IOException {
290                ICalPropertyScribe scribe = index.getPropertyScribe(property);
291
292                //marshal property
293                String value;
294                try {
295                        value = scribe.writeText(property, context);
296                } catch (SkipMeException e) {
297                        return;
298                } catch (DataModelConversionException e) {
299                        for (ICalComponent c : e.getComponents()) {
300                                writeComponent(c, context.getParent());
301                        }
302                        for (ICalProperty p : e.getProperties()) {
303                                writeProperty(p);
304                        }
305                        return;
306                }
307
308                //get parameters
309                ICalParameters parameters = scribe.prepareParameters(property, context);
310
311                /*
312                 * Set the property's data type.
313                 * 
314                 * Only add a VALUE parameter if the data type is: (1) not "unknown" (2)
315                 * different from the property's default data type
316                 */
317                ICalDataType dataType = scribe.dataType(property, targetVersion);
318                if (dataType != null && dataType != scribe.defaultDataType(targetVersion)) {
319                        parameters = new ICalParameters(parameters);
320                        parameters.setValue(dataType);
321                }
322
323                //get the property name
324                String propertyName = scribe.getPropertyName(getTargetVersion());
325
326                //write property to data stream
327                writer.writeProperty(null, propertyName, new VObjectParameters(parameters.getMap()), value);
328        }
329
330        /**
331         * Flushes the output stream.
332         * @throws IOException if there's a problem flushing the output stream
333         */
334        public void flush() throws IOException {
335                writer.flush();
336        }
337
338        /**
339         * Closes the output stream.
340         * @throws IOException if there's a problem closing the output stream
341         */
342        public void close() throws IOException {
343                writer.close();
344        }
