045/**
046 * Chainer class for writing traditional, plain-text iCalendar objects.
047 * @see Biweekly#write(Collection)
048 * @see Biweekly#write(ICalendar...)
049 * @author Michael Angstadt
050 */
051public class ChainingTextWriter extends ChainingWriter<ChainingTextWriter> {
052        private ICalVersion version;
053        private boolean caretEncoding = false;
054        private boolean foldLines = true;
055
056        /**
057         * @param icals the iCalendar objects to write
058         */
059        public ChainingTextWriter(Collection<ICalendar> icals) {
060                super(icals);
061        }
062
063        /**
064         * <p>
065         * Sets the version that all the iCalendar objects will be marshalled to.
066         * The version that is attached to each individual {@link ICalendar} object
067         * will be ignored.
068         * </p>
069         * <p>
070         * If no version is passed into this method, the writer will look at the
071         * version attached to each individual {@link ICalendar} object and marshal
072         * it to that version. And if a {@link ICalendar} object has no version
073         * attached to it, then it will be marshalled to version 2.0.
074         * </p>
075         * @param version the version to marshal the iCalendar objects to
076         * @return this
077         */
078        public ChainingTextWriter version(ICalVersion version) {
079                this.version = version;
080                return this;
081        }
082
083        /**
084         * Sets whether the writer will use circumflex accent encoding for parameter
085         * values (disabled by default). This only applies to version 2.0 iCalendar
086         * objects.
087         * @param enable true to use circumflex accent encoding, false not to
088         * @return this
089         * @see ICalWriter#setCaretEncodingEnabled(boolean)
090         * @see <a href="http://tools.ietf.org/html/rfc6868">RFC 6868</a>
091         */
092        public ChainingTextWriter caretEncoding(boolean enable) {
093                this.caretEncoding = enable;
094                return this;
095        }
096
097        /**
098         * <p>
099         * Sets whether to fold long lines. Line folding is when long lines are
100         * split up into multiple lines. No data is lost or changed when a line is
101         * folded.
102         * </p>
103         * <p>
104         * Line folding is enabled by default. If the iCalendar consumer is not
105         * parsing your iCalendar objects properly, disabling line folding may help.
106         * </p>
107         * @param foldLines true to enable line folding, false to disable it
108         * (defaults to true)
109         * @return this
110         */
111        public ChainingTextWriter foldLines(boolean foldLines) {
112                this.foldLines = foldLines;
113                return this;
114        }
115
116        @Override
117        public ChainingTextWriter tz(TimeZone defaultTimeZone, boolean outlookCompatible) {
118                return super.tz(defaultTimeZone, outlookCompatible);
119        }
120
121        @Override
122        public ChainingTextWriter register(ICalPropertyScribe<? extends ICalProperty> scribe) {
123                return super.register(scribe);
124        }
125
126        @Override
127        public ChainingTextWriter register(ICalComponentScribe<? extends ICalComponent> scribe) {
128                return super.register(scribe);
129        }
130
131        /**
132         * Writes the iCalendar objects to a string.
133         * @return the iCalendar string
134         */
135        public String go() {
136                StringWriter sw = new StringWriter();
137                try {
138                        go(sw);
139                } catch (IOException e) {
140                        //should never be thrown because we're writing to a string
141                        throw new RuntimeException(e);
142                }
143                return sw.toString();
144        }
145
146        /**
147         * Writes the iCalendar objects to an output stream.
148         * @param out the output stream to write to
149         * @throws IOException if there's a problem writing to the output stream
150         */
151        public void go(OutputStream out) throws IOException {
152                go(new ICalWriter(out, getICalWriterConstructorVersion()));
153        }
154
155        /**
156         * Writes the iCalendar objects to a file. If the file exists, it will be
157         * overwritten.
158         * @param file the file to write to
159         * @throws IOException if there's a problem writing to the file
160         */
161        public void go(File file) throws IOException {
162                go(file, false);
163        }
164
165        /**
166         * Writes the iCalendar objects to a file.
167         * @param file the file to write to
168         * @param append true to append onto the end of the file, false to overwrite
169         * it
170         * @throws IOException if there's a problem writing to the file
171         */
172        public void go(File file, boolean append) throws IOException {
173                ICalWriter writer = new ICalWriter(file, append, getICalWriterConstructorVersion());
174                try {
175                        go(writer);
176                } finally {
177                        writer.close();
178                }
179        }
180
181        /**
182         * Writes the iCalendar objects to a writer.
183         * @param writer the writer to write to
184         * @throws IOException if there's a problem writing to the writer
185         */
186        public void go(Writer writer) throws IOException {
187                go(new ICalWriter(writer, getICalWriterConstructorVersion()));
188        }
189
190        private void go(ICalWriter writer) throws IOException {
191                writer.setCaretEncodingEnabled(caretEncoding);
192                if (!foldLines) {
193                        writer.getVObjectWriter().getFoldedLineWriter().setLineLength(null);
194                }
195                if (defaultTimeZone != null) {
196                        writer.setGlobalTimezone(defaultTimeZone);
197                }
198                if (index != null) {
199                        writer.setScribeIndex(index);
200                }
201
202                for (ICalendar ical : icals) {
203                        if (version == null) {
204                                //use the version that's assigned to each individual iCalendar object
205                                ICalVersion icalVersion = ical.getVersion();
206                                if (icalVersion == null) {
207                                        icalVersion = ICalVersion.V2_0;
208                                }
209                                writer.setTargetVersion(icalVersion);
210                        }
211                        writer.write(ical);
212                        writer.flush();
213                }
214        }
215
216        /**
217         * <p>
218         * Gets the {@link ICalVersion} object to pass into the {@link ICalWriter}
219         * constructor. The constructor does not allow a null version, so this
220         * method ensures that a non-null version is passed in.
221         * </p>
222         * <p>
223         * If the user hasn't chosen a version, the version that is passed into the
224         * constructor doesn't matter. This is because the writer's target version
225         * is reset every time an iCalendar object is written (see the
226         * {@link #go(ICalWriter)} method).
227         * </p>
228         * @return the version to pass into the constructor
229         */
230        private ICalVersion getICalWriterConstructorVersion() {
231                return (version == null) ? ICalVersion.V2_0 : version;
232        }
233}
