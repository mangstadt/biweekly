032/**
033 * <p>
034 * Contains information used for accessing a conference system.
035 * </p>
036 * <p>
037 * <b>Code sample:</b>
038 * </p>
039 * 
040 * <pre class="brush:java">
041 * VEvent event = new VEvent();
042 * 
043 * Conference conference = new Conference(&quot;tel:+1-412-555-0123,,,654321&quot;);
044 * conference.setLabel(&quot;Audio conference, access code=77869&quot;);
045 * conference.getFeatures().add(Feature.AUDIO);
046 * conference.getFeatures().add(Feature.PHONE);
047 * event.addConference(conference);
048 * </pre>
049 * @author Michael Angstadt
050 * @see <a
051 * href="http://tools.ietf.org/html/draft-ietf-calext-extensions-01#page-11">draft-ietf-calext-extensions-01
052 * p.11</a>
053 */
054public class Conference extends ICalProperty {
055        private String uri, text;
056
057        /**
058         * Creates a conference property.
059         * @param uri the uri
060         */
061        public Conference(String uri) {
062                this.uri = uri;
063        }
064
065        /**
066         * Copy constructor.
067         * @param original the property to make a copy of
068         */
069        public Conference(Conference original) {
070                super(original);
071                uri = original.uri;
072                text = original.text;
073        }
074
075        /**
076         * Gets the URI value of this property.
077         * @return the URI value or null if not set
078         */
079        public String getUri() {
080                return uri;
081        }
082
083        /**
084         * Sets the value of this property to a URI.
085         * @param uri the URI
086         */
087        public void setUri(String uri) {
088                this.uri = uri;
089                text = null;
090        }
091
092        /**
093         * Gets the plain text value of this property.
094         * @return the plain text value or null if not set
095         */
096        public String getText() {
097                return text;
098        }
099
100        /**
101         * Sets the value of this property to a plain text value.
102         * @param text the plain text value
103         */
104        public void setText(String text) {
105                this.text = text;
106                uri = null;
107        }
108
109        /**
110         * Gets the list that holds the features that this conference supports (for
111         * example, audio and video).
112         * @return the features (this list is mutable)
113         */
114        public List<Feature> getFeatures() {
115                return parameters.getFeatures();
116        }
117
118        @Override
119        public String getLabel() {
120                return super.getLabel();
121        }
122
123        @Override
124        public void setLabel(String label) {
125                super.setLabel(label);
126        }
127
128        @Override
129        public Conference copy() {
130                return new Conference(this);
131        }
132}
