054/**
055 * Parses iCalendar objects from a data stream.
056 * @author Michael Angstadt
057 */
058public abstract class StreamReader implements Closeable {
059        protected final ParseWarnings warnings = new ParseWarnings();
060        protected ScribeIndex index = new ScribeIndex();
061        protected ParseContext context;
062
063        /**
064         * <p>
065         * Registers an experimental property scribe. Can also be used to override
066         * the scribe of a standard property (such as DTSTART). Calling this method
067         * is the same as calling:
068         * </p>
069         * <p>
070         * {@code getScribeIndex().register(scribe)}.
071         * </p>
072         * @param scribe the scribe to register
073         */
074        public void registerScribe(ICalPropertyScribe<? extends ICalProperty> scribe) {
075                index.register(scribe);
076        }
077
078        /**
079         * <p>
080         * Registers an experimental component scribe. Can also be used to override
081         * the scribe of a standard component (such as VEVENT). Calling this method
082         * is the same as calling:
083         * </p>
084         * <p>
085         * {@code getScribeIndex().register(scribe)}.
086         * </p>
087         * @param scribe the scribe to register
088         */
089        public void registerScribe(ICalComponentScribe<? extends ICalComponent> scribe) {
090                index.register(scribe);
091        }
092
093        /**
094         * Gets the object that manages the component/property scribes.
095         * @return the scribe index
096         */
097        public ScribeIndex getScribeIndex() {
098                return index;
099        }
100
101        /**
102         * Sets the object that manages the component/property scribes.
103         * @param index the scribe index
104         */
105        public void setScribeIndex(ScribeIndex index) {
106                this.index = index;
107        }
108
109        /**
110         * Gets the warnings from the last iCalendar object that was read. This list
111         * is reset every time a new iCalendar object is read.
112         * @return the warnings or empty list if there were no warnings
113         */
114        public List<String> getWarnings() {
115                return warnings.copy();
116        }
117
118        /**
119         * Reads all iCalendar objects from the data stream.
120         * @return the iCalendar objects
121         * @throws IOException if there's a problem reading from the stream
122         */
123        public List<ICalendar> readAll() throws IOException {
124                List<ICalendar> icals = new ArrayList<ICalendar>();
125                ICalendar ical;
126                while ((ical = readNext()) != null) {
127                        icals.add(ical);
128                }
129                return icals;
130        }
131
132        /**
133         * Reads the next iCalendar object from the data stream.
134         * @return the next iCalendar object or null if there are no more
135         * @throws IOException if there's a problem reading from the stream
136         */
137        public ICalendar readNext() throws IOException {
138                warnings.clear();
139                context = new ParseContext();
140                ICalendar ical = _readNext();
141                if (ical == null) {
142                        return null;
143                }
144
145                ical.setVersion(context.getVersion());
146                handleTimezones(ical);
147                return ical;
148        }
149
150        /**
151         * Reads the next iCalendar object from the data stream.
152         * @return the next iCalendar object or null if there are no more
153         * @throws IOException if there's a problem reading from the stream
154         */
155        protected abstract ICalendar _readNext() throws IOException;
156
157        private void handleTimezones(ICalendar ical) {
158                TimezoneInfo tzinfo = ical.getTimezoneInfo();
159
160                //convert vCalendar DAYLIGHT and TZ properties to a VTIMEZONE component
161                TimezoneAssignment vcalTimezone = extractVCalTimezone(ical);
162
163                //assign a TimeZone object to each VTIMEZONE component.
164                Iterator<VTimezone> it = ical.getComponents(VTimezone.class).iterator();
165                while (it.hasNext()) {
166                        VTimezone component = it.next();
167
168                        //make sure the component has an ID
169                        String id = ValuedProperty.getValue(component.getTimezoneId());
170                        if (id == null || id.trim().length() == 0) {
171                                //note: do not remove invalid VTIMEZONE components from the ICalendar object
172                                warnings.add(null, null, 39);
173                                continue;
174                        }
175
176                        TimeZone timezone = new ICalTimeZone(component);
177                        tzinfo.getTimezones().add(new TimezoneAssignment(timezone, component));
178
179                        //remove the component from the ICalendar object
180                        it.remove();
181                }
182
183                if (vcalTimezone != null) {
184                        //vCal: parse floating dates according to the DAYLIGHT and TZ properties (which were converted to a VTIMEZONE component)
185                        for (TimezonedDate timezonedDate : context.getFloatingDates()) {
186                                ICalDate date = timezonedDate.getDate();
187
188                                //parse its raw date components under its real timezone
189                                Date realDate = date.getRawComponents().toDate(vcalTimezone.getTimeZone());
190
191                                //update the ICalDate object with the new timestamp
192                                date.setTime(realDate.getTime());
193                        }
194                } else {
195                        //iCal: treat floating dates as floating dates
196                        for (TimezonedDate timezonedDate : context.getFloatingDates()) {
197                                tzinfo.setFloating(timezonedDate.getProperty(), true);
198                        }
199                }
200
201                for (Map.Entry<String, List<TimezonedDate>> entry : context.getTimezonedDates()) {
202                        String tzid = entry.getKey();
203
204                        TimezoneAssignment assignment;
205                        boolean solidus = tzid.startsWith("/");
206                        if (solidus) {
207                                //treat the TZID parameter value as an Olsen timezone ID
208                                String globalId = tzid.substring(1);
209                                TimeZone timezone = ICalDateFormat.parseTimeZoneId(globalId);
210                                if (timezone == null) {
211                                        //timezone could not be determined
212                                        warnings.add(null, null, 38, tzid);
213                                        continue;
214                                }
215                                assignment = new TimezoneAssignment(timezone, globalId);
216                                tzinfo.getTimezones().add(assignment);
217                        } else {
218                                assignment = tzinfo.getTimezoneById(tzid);
219                                if (assignment == null) {
220                                        //A VTIMEZONE component couldn't found
221                                        //so treat the TZID parameter value as an Olsen timezone ID
222                                        TimeZone timezone = ICalDateFormat.parseTimeZoneId(tzid);
223                                        if (timezone == null) {
224                                                //timezone could not be determined
225                                                warnings.add(null, null, 38, tzid);
226                                                continue;
227                                        }
228                                        assignment = new TimezoneAssignment(timezone, tzid);
229                                        tzinfo.getTimezones().add(assignment);
230
231                                        warnings.add(null, null, 37, tzid);
232                                }
233                        }
234
235                        Calendar cal = Calendar.getInstance(assignment.getTimeZone());
236                        List<TimezonedDate> timezonedDates = entry.getValue();
237                        for (TimezonedDate timezonedDate : timezonedDates) {
238                                //assign the property to the timezone
239                                ICalProperty property = timezonedDate.getProperty();
240                                tzinfo.setTimezone(property, assignment);
241
242                                ICalDate date = timezonedDate.getDate();
243
244                                //parse its raw date components under its real timezone
245                                Date realDate = date.getRawComponents().toDate(cal);
246
247                                //update the Date object with the new timestamp
248                                date.setTime(realDate.getTime());
249
250                                //remove the TZID parameter
251                                property.getParameters().setTimezoneId(null);
252                        }
253                }
254        }
255
256        private TimezoneAssignment extractVCalTimezone(ICalendar ical) {
257                List<Daylight> daylights = ical.removeProperties(Daylight.class);
258                List<Timezone> timezones = ical.removeProperties(Timezone.class);
259
260                Timezone timezone = timezones.isEmpty() ? null : timezones.get(0);
261                VTimezone vcalComponent = convert(daylights, timezone);
262                if (vcalComponent == null) {
263                        return null;
264                }
265
266                TimeZone icalTimezone = new ICalTimeZone(vcalComponent);
267                TimezoneInfo tzinfo = ical.getTimezoneInfo();
268                TimezoneAssignment assignment = new TimezoneAssignment(icalTimezone, vcalComponent);
269                tzinfo.setDefaultTimezone(assignment);
270
271                return assignment;
272        }
