053/**
054 * <p>
055 * Writes {@link ICalendar} objects to a JSON data stream (jCal).
056 * </p>
057 * <p>
058 * <b>Example:</b>
059 * </p>
060 * 
061 * <pre class="brush:java">
062 * ICalendar ical1 = ...
063 * ICalendar ical2 = ...
064 * File file = new File("icals.json");
065 * JCalWriter writer = null;
066 * try {
067 *   writer = new JCalWriter(file);
068 *   writer.write(ical1);
069 *   writer.write(ical2);
070 * } finally {
071 *   if (writer != null) writer.close();
072 * }
073 * </pre>
074 * @author Michael Angstadt
075 * @see <a href="http://tools.ietf.org/html/rfc7265">RFC 7265</a>
076 */
077public class JCalWriter extends StreamWriter implements Flushable {
078        private final JCalRawWriter writer;
079        private final ICalVersion targetVersion = ICalVersion.V2_0;
080
081        /**
082         * @param out the output stream to write to (UTF-8 encoding will be used)
083         */
084        public JCalWriter(OutputStream out) {
085                this(utf8Writer(out));
086        }
087
088        /**
089         * @param out the output stream to write to (UTF-8 encoding will be used)
090         * @param wrapInArray true to wrap all iCalendar objects in a parent array,
091         * false not to (useful when writing more than one iCalendar object)
092         */
093        public JCalWriter(OutputStream out, boolean wrapInArray) {
094                this(utf8Writer(out), wrapInArray);
095        }
096
097        /**
098         * @param file the file to write to (UTF-8 encoding will be used)
099         * @throws IOException if the file cannot be written to
100         */
101        public JCalWriter(File file) throws IOException {
102                this(utf8Writer(file));
103        }
104
105        /**
106         * @param file the file to write to (UTF-8 encoding will be used)
107         * @param wrapInArray true to wrap all iCalendar objects in a parent array,
108         * false not to (useful when writing more than one iCalendar object)
109         * @throws IOException if the file cannot be written to
110         */
111        public JCalWriter(File file, boolean wrapInArray) throws IOException {
112                this(utf8Writer(file), wrapInArray);
113        }
114
115        /**
116         * @param writer the writer to write to
117         */
118        public JCalWriter(Writer writer) {
119                this(writer, false);
120        }
121
122        /**
123         * @param writer the writer to write to
124         * @param wrapInArray true to wrap all iCalendar objects in a parent array,
125         * false not to (useful when writing more than one iCalendar object)
126         */
127        public JCalWriter(Writer writer, boolean wrapInArray) {
128                this.writer = new JCalRawWriter(writer, wrapInArray);
129        }
130
131        /**
132         * @param generator the generator to write to
133         */
134        public JCalWriter(JsonGenerator generator) {
135                this.writer = new JCalRawWriter(generator);
136        }
137
138        /**
139         * Gets whether or not the JSON will be pretty-printed.
140         * @return true if it will be pretty-printed, false if not (defaults to
141         * false)
142         */
143        public boolean isPrettyPrint() {
144                return writer.isPrettyPrint();
145        }
146
147        /**
148         * Sets whether or not to pretty-print the JSON.
149         * @param prettyPrint true to pretty-print it, false not to (defaults to
150         * false)
151         */
152        public void setPrettyPrint(boolean prettyPrint) {
153                writer.setPrettyPrint(prettyPrint);
154        }
155
156        @Override
157        protected void _write(ICalendar ical) throws IOException {
158                writeComponent(ical);
159        }
160
161        @Override
162        protected ICalVersion getTargetVersion() {
163                return targetVersion;
164        }
165
166        /**
167         * Writes a component to the data stream.
168         * @param component the component to write
169         * @throws IllegalArgumentException if the scribe class for a component or
170         * property object cannot be found (only happens when an experimental
171         * property/component scribe is not registered with the
172         * {@code registerScribe} method.)
173         * @throws IOException if there's a problem writing to the data stream
174         */
175        @SuppressWarnings({ "rawtypes", "unchecked" })
176        private void writeComponent(ICalComponent component) throws IOException {
177                ICalComponentScribe componentScribe = index.getComponentScribe(component);
178                writer.writeStartComponent(componentScribe.getComponentName().toLowerCase());
179
180                List propertyObjs = componentScribe.getProperties(component);
181                if (component instanceof ICalendar && component.getProperty(Version.class) == null) {
182                        propertyObjs.add(0, new Version(targetVersion));
183                }
184
185                //write properties
186                for (Object propertyObj : propertyObjs) {
187                        context.setParent(component); //set parent here incase a scribe resets the parent
188                        ICalProperty property = (ICalProperty) propertyObj;
189                        ICalPropertyScribe propertyScribe = index.getPropertyScribe(property);
190
191                        //marshal property
192                        ICalParameters parameters;
193                        JCalValue value;
194                        try {
195                                parameters = propertyScribe.prepareParameters(property, context);
196                                value = propertyScribe.writeJson(property, context);
197                        } catch (SkipMeException e) {
198                                continue;
199                        }
200
201                        //write property
202                        String propertyName = propertyScribe.getPropertyName().toLowerCase();
203                        ICalDataType dataType = propertyScribe.dataType(property, targetVersion);
204                        writer.writeProperty(propertyName, parameters, dataType, value);
205                }
206
207                //write sub-components
208                List subComponents = componentScribe.getComponents(component);
209                if (component instanceof ICalendar) {
210                        //add the VTIMEZONE components that were auto-generated by TimezoneOptions
211                        Collection<VTimezone> tzs = getTimezoneComponents();
212                        for (VTimezone tz : tzs) {
213                                if (!subComponents.contains(tz)) {
214                                        subComponents.add(0, tz);
215                                }
216                        }
217                }
218                for (Object subComponentObj : subComponents) {
219                        ICalComponent subComponent = (ICalComponent) subComponentObj;
220                        writeComponent(subComponent);
221                }
222
223                writer.writeEndComponent();
224        }
225
226        /**
227         * Flushes the stream.
228         * @throws IOException if there's a problem flushing the stream
229         */
230        public void flush() throws IOException {
231                writer.flush();
232        }
233
234        /**
235         * Finishes writing the JSON document and closes the underlying
236         * {@link Writer} object.
237         * @throws IOException if there's a problem closing the stream
238         */
239        public void close() throws IOException {
240                writer.close();
241        }
242
243        /**
244         * Finishes writing the JSON document so that it is syntactically correct.
245         * No more iCalendar objects can be written once this method is called.
246         * @throws IOException if there's a problem writing to the data stream
247         */
248        public void closeJsonStream() throws IOException {
249                writer.closeJsonStream();
250        }
251}
