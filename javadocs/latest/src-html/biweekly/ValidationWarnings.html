038/**
039 * <p>
040 * Holds the validation warnings of an iCalendar object.
041 * </p>
042 * <p>
043 * <b>Examples:</b>
044 * </p>
045 * 
046 * <pre class="brush:java">
047 * //validate an iCalendar object
048 * ValidationWarnings warnings = ical.validate();
049 * 
050 * //print all warnings to a string:
051 * System.out.println(warnings.toString());
052 * //sample output:
053 * //[ICalendar]: ProductId is not set (it is a required property).
054 * //[ICalendar &gt; VEvent &gt; DateStart]: DateStart must come before DateEnd.
055 * //[ICalendar &gt; VEvent &gt; VAlarm]: The trigger must specify which date field its duration is relative to.
056 * 
057 * //iterate over each warnings group
058 * //this gives you access to the property/component object and its parent components
059 * for (WarningsGroup group : warnings) {
060 * ICalProperty prop = group.getProperty();
061 *   if (prop == null) {
062 *     //then it was a component that caused the warnings
063 *     ICalComponent comp = group.getComponent();
064 *   }
065 * 
066 *   //get parent components
067 *   List&lt;ICalComponent&gt; hierarchy = group.getComponentHierarchy();
068 * 
069 *   //get warning messages
070 *   List&lt;String&gt; messages = group.getMessages();
071 * }
072 * 
073 * //you can also get the warnings of specific properties/components
074 * List&lt;WarningsGroup&gt; dtstartWarnings = warnings.getByProperty(DateStart.class);
075 * List&lt;WarningsGroup&gt; veventWarnings = warnings.getByComponent(VEvent.class);
076 * </pre>
077 * @author Michael Angstadt
078 * @see ICalendar#validate(ICalVersion)
079 */
080public class ValidationWarnings implements Iterable<WarningsGroup> {
081        private final List<WarningsGroup> warnings;
082
083        /**
084         * Creates a new validation warnings list.
085         * @param warnings the validation warnings
086         */
087        public ValidationWarnings(List<WarningsGroup> warnings) {
088                this.warnings = warnings;
089        }
090
091        /**
092         * Gets all validation warnings of a given property.
093         * @param propertyClass the property (e.g. {@code DateStart.class})
094         * @return the validation warnings
095         */
096        public List<WarningsGroup> getByProperty(Class<? extends ICalProperty> propertyClass) {
097                List<WarningsGroup> warnings = new ArrayList<WarningsGroup>();
098                for (WarningsGroup group : this.warnings) {
099                        ICalProperty property = group.getProperty();
100                        if (property == null) {
101                                continue;
102                        }
103
104                        if (propertyClass == property.getClass()) {
105                                warnings.add(group);
106                        }
107                }
108                return warnings;
109        }
110
111        /**
112         * Gets all validation warnings of a given component.
113         * @param componentClass the component (e.g. {@code VEvent.class})
114         * @return the validation warnings
115         */
116        public List<WarningsGroup> getByComponent(Class<? extends ICalComponent> componentClass) {
117                List<WarningsGroup> warnings = new ArrayList<WarningsGroup>();
118                for (WarningsGroup group : this.warnings) {
119                        ICalComponent component = group.getComponent();
120                        if (component == null) {
121                                continue;
122                        }
123
124                        if (componentClass == component.getClass()) {
125                                warnings.add(group);
126                        }
127                }
128                return warnings;
129        }
130
131        /**
132         * Gets all the validation warnings.
133         * @return the validation warnings
134         */
135        public List<WarningsGroup> getWarnings() {
136                return warnings;
137        }
138
139        /**
140         * Determines whether there are any validation warnings.
141         * @return true if there are none, false if there are one or more
142         */
143        public boolean isEmpty() {
144                return warnings.isEmpty();
145        }
146
147        /**
148         * <p>
149         * Outputs all validation warnings as a newline-delimited string. For
150         * example:
151         * </p>
152         * 
153         * <pre>
154         * [ICalendar]: ProductId is not set (it is a required property).
155         * [ICalendar &gt; VEvent &gt; DateStart]: DateStart must come before DateEnd.
156         * [ICalendar &gt; VEvent &gt; VAlarm]: The trigger must specify which date field its duration is relative to.
157         * </pre>
158         */
159        @Override
160        public String toString() {
161                return StringUtils.join(warnings, StringUtils.NEWLINE);
162        }
163
164        /**
165         * Iterates over each warning group (same as calling
166         * {@code getWarnings().iterator()}).
167         * @return the iterator
168         */
169        public Iterator<WarningsGroup> iterator() {
170                return warnings.iterator();
171        }
172
173        /**
174         * Holds the validation warnings of a property or component.
175         * @author Michael Angstadt
176         */
177        public static class WarningsGroup {
178                private final ICalProperty property;
179                private final ICalComponent component;
180                private final List<ICalComponent> componentHierarchy;
181                private final List<ValidationWarning> warnings;
182
183                /**
184                 * Creates a new set of validation warnings for a property.
185                 * @param property the property that caused the warnings
186                 * @param componentHierarchy the hierarchy of components that the
187                 * property belongs to
188                 * @param warning the warnings
189                 */
190                public WarningsGroup(ICalProperty property, List<ICalComponent> componentHierarchy, List<ValidationWarning> warning) {
191                        this(null, property, componentHierarchy, warning);
192                }
193
194                /**
195                 * Creates a new set of validation warnings for a component.
196                 * @param component the component that caused the warnings
197                 * @param componentHierarchy the hierarchy of components that the
198                 * component belongs to
199                 * @param warning the warnings
200                 */
201                public WarningsGroup(ICalComponent component, List<ICalComponent> componentHierarchy, List<ValidationWarning> warning) {
202                        this(component, null, componentHierarchy, warning);
203                }
204
205                private WarningsGroup(ICalComponent component, ICalProperty property, List<ICalComponent> componentHierarchy, List<ValidationWarning> warning) {
206                        this.component = component;
207                        this.property = property;
208                        this.componentHierarchy = componentHierarchy;
209                        this.warnings = warning;
210                }
211
212                /**
213                 * Gets the property object that caused the validation warnings.
214                 * @return the property object or null if a component caused the
215                 * warnings.
216                 */
217                public ICalProperty getProperty() {
218                        return property;
219                }
220
221                /**
222                 * Gets the component object that caused the validation warnings.
223                 * @return the component object or null if a property caused the
224                 * warnings.
225                 */
226                public ICalComponent getComponent() {
227                        return component;
228                }
229
230                /**
231                 * Gets the hierarchy of components that the property or component
232                 * belongs to.
233                 * @return the component hierarchy
234                 */
235                public List<ICalComponent> getComponentHierarchy() {
236                        return componentHierarchy;
237                }
238
239                /**
240                 * Gets the warnings that belong to the property or component.
241                 * @return the warnings
242                 */
243                public List<ValidationWarning> getWarnings() {
244                        return warnings;
245                }
246
247                /**
248                 * <p>
249                 * Outputs each message in this warnings group as a newline-delimited
250                 * string. Each line includes the component hierarchy and the name of
251                 * the property/component. For example:
252                 * </p>
253                 * 
254                 * <pre>
255                 * [ICalendar &gt; VEvent &gt; VAlarm]: Email alarms must have at least one attendee.
256                 * [ICalendar &gt; VEvent &gt; VAlarm]: The trigger must specify which date field its duration is relative to.
257                 * </pre>
258                 */
259                @Override
260                public String toString() {
261                        final String prefix = "[" + buildPath() + "]: ";
262                        return StringUtils.join(warnings, StringUtils.NEWLINE, new JoinCallback<ValidationWarning>() {
263                                public void handle(StringBuilder sb, ValidationWarning warning) {
264                                        sb.append(prefix).append(warning);
265                                }
266                        });
267                }
268
269                private String buildPath() {
270                        StringBuilder sb = new StringBuilder();
271
272                        if (!componentHierarchy.isEmpty()) {
273                                String delimitor = " > ";
274
275                                StringUtils.join(componentHierarchy, delimitor, sb, new JoinCallback<ICalComponent>() {
276                                        public void handle(StringBuilder sb, ICalComponent component) {
277                                                sb.append(component.getClass().getSimpleName());
278                                        }
279                                });
280                                sb.append(delimitor);
281                        }
282
283                        Object obj = (property == null) ? component : property;
284                        sb.append(obj.getClass().getSimpleName());
285
286                        return sb.toString();
287                }
288        }
289}
