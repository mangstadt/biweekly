053/**
054 * Contains utility methods related to the google-rfc-2445 project.
055 * @author Michael Angstadt
056 * @see <a href="https://code.google.com/p/google-rfc-2445/">google-rfc-2445</a>
057 */
058public final class Google2445Utils {
059        /**
060         * <p>
061         * Converts an {@link ICalDate} object to a google-rfc-2445
062         * {@link DateValue} object using the {@link DateTimeComponents raw date
063         * components} of the {@link ICalDate}.
064         * </p>
065         * <p>
066         * If the {@link ICalDate} object does not have raw date components, then
067         * the returned {@link DateValue} object will represent the {@link ICalDate}
068         * in the local timezone.
069         * </p>
070         * @param date the date object
071         * @return the google-rfc-2445 object
072         */
073        public static DateValue convertFromRawComponents(ICalDate date) {
074                DateTimeComponents raw = date.getRawComponents();
075                if (raw == null) {
076                        raw = new DateTimeComponents(date);
077                }
078
079                return convert(raw);
080        }
081
082        /**
083         * Converts a {@link DateTimeComponents} object to a google-rfc-2445
084         * {@link DateValue} object.
085         * @param components the date time components object
086         * @return the google-rfc-2445 object
087         */
088        public static DateValue convert(DateTimeComponents components) {
089                if (components.hasTime()) {
090                        //@formatter:off
091                        return new DateTimeValueImpl(
092                                components.getYear(),
093                                components.getMonth(),
094                                components.getDate(),
095                                components.getHour(),
096                                components.getMinute(),
097                                components.getSecond()
098                        );
099                        //@formatter:on
100                }
101
102                //@formatter:off
103                return new DateValueImpl(
104                        components.getYear(),
105                        components.getMonth(),
106                        components.getDate()
107                );
108                //@formatter:on
109        }
110
111        /**
112         * Converts an {@link ICalDate} object to a google-rfc-2445
113         * {@link DateValue} object.
114         * @param date the date object
115         * @param timezone the timezone the returned object will be in
116         * @return the google-rfc-2445 object
117         */
118        public static DateValue convert(ICalDate date, TimeZone timezone) {
119                Calendar c = Calendar.getInstance(timezone);
120                c.setTime(date);
121
122                /*
123                 * Do not create a "DateValueImpl" object if "date.hasTime() == false".
124                 * This interferes with any recurrence iterators the object is passed
125                 * into.
126                 * 
127                 * See: https://github.com/mangstadt/biweekly/issues/47
128                 */
129
130                //@formatter:off
131                return new DateTimeValueImpl(
132                        c.get(Calendar.YEAR),
133                        c.get(Calendar.MONTH)+1,
134                        c.get(Calendar.DATE),
135                        c.get(Calendar.HOUR_OF_DAY),
136                        c.get(Calendar.MINUTE),
137                        c.get(Calendar.SECOND)
138                );
139                //@formatter:on
140        }
141
142        /**
143         * Converts an {@link ICalDate} object to a google-rfc-2445
144         * {@link DateValue} object. The returned object will be in UTC.
145         * @param date the date object
146         * @return the google-rfc-2445 object (in UTC)
147         */
148        public static DateValue convertUtc(ICalDate date) {
149                return convert(date, utc());
150        }
151
152        /**
153         * Converts a google-rfc-2445 {@link DateValue} object to a {@link ICalDate}
154         * object.
155         * @param date the date value object
156         * @param timezone the timezone the date value is in
157         * @return the converted object
158         */
159        public static ICalDate convert(DateValue date, TimeZone timezone) {
160                Calendar c = Calendar.getInstance(timezone);
161                c.clear();
162                c.set(Calendar.YEAR, date.year());
163                c.set(Calendar.MONTH, date.month() - 1);
164                c.set(Calendar.DATE, date.day());
165
166                boolean hasTime = (date instanceof DateTimeValue);
167                if (hasTime) {
168                        DateTimeValue dateTime = (DateTimeValue) date;
169                        c.set(Calendar.HOUR_OF_DAY, dateTime.hour());
170                        c.set(Calendar.MINUTE, dateTime.minute());
171                        c.set(Calendar.SECOND, dateTime.second());
172                }
173
174                return new ICalDate(c.getTime(), hasTime);
175        }

177        /**
178         * Converts a google-rfc-2445 {@link DateValue} object to a {@link ICalDate}
179         * object. It is assumed that the given {@link DateValue} object is in UTC.
180         * @param date the date value object (in UTC)
181         * @return the converted object
182         */
183        public static ICalDate convertUtc(DateValue date) {
184                return convert(date, utc());
185        }

187        /**
188         * Creates a recurrence iterator based on the given recurrence rule.
189         * @param recurrence the recurrence rule
190         * @param start the start date
191         * @param timezone the timezone to iterate in. This is needed in order to
192         * account for when the iterator passes over a daylight savings boundary.
193         * @return the recurrence iterator
194         */
195        public static RecurrenceIterator createRecurrenceIterator(Recurrence recurrence, ICalDate start, TimeZone timezone) {
196                DateValue startValue = convert(start, timezone);
197                return RecurrenceIteratorFactory.createRecurrenceIterator(recurrence, startValue, timezone);
198        }

201         * Creates a recurrence iterator based on the given recurrence rule.
202         * @param recurrence the recurrence rule
203         * @param start the start date
204         * @param timezone the timezone to iterate in. This is needed in order to
205         * account for when the iterator passes over a daylight savings boundary.
206         * @return the recurrence iterator
207         */
208        public static RecurrenceIterable createRecurrenceIterable(Recurrence recurrence, ICalDate start, TimeZone timezone) {
209                DateValue startValue = convert(start, timezone);
210                return RecurrenceIteratorFactory.createRecurrenceIterable(recurrence, startValue, timezone);
211        }

213        /**
214         * <p>
215         * Creates an iterator that computes the dates defined by the
216         * {@link RecurrenceRule} and {@link RecurrenceDates} properties (if
217         * present), and excludes those dates which are defined by the
218         * {@link ExceptionRule} and {@link ExceptionDates} properties (if present).
219         * </p>
220         * <p>
221         * In order for {@link RecurrenceRule} and {@link ExceptionRule} properties
222         * to be included in this iterator, a {@link DateStart} property must be
223         * defined.
224         * </p>
225         * <p>
226         * {@link Period} values in {@link RecurrenceDates} properties are not
227         * supported and are ignored.
228         * </p>
229         * @param component the component
230         * @param timezone the timezone to iterate in. This is needed in order to
231         * adjust for when the iterator passes over a daylight savings boundary.
232         * This parameter is ignored if the start date of the given component does
233         * not have a time component.
234         * @return the iterator
235         */
236        public static DateIterator getDateIterator(ICalComponent component, TimeZone timezone) {
237                DateStart dtstart = component.getProperty(DateStart.class);
238                ICalDate start = ValuedProperty.getValue(dtstart);
239
240                /*
241                 * If the start date is just a date and does not have a time component,
242                 * then the default timezone must be used, because this is the timezone
243                 * biweekly used to parse the date.
244                 */
245                if (start != null && !start.hasTime()) {
246                        timezone = TimeZone.getDefault();
247                }
248
249                /////////////INCLUDE/////////////
250
251                List<RecurrenceIterator> include = new ArrayList<RecurrenceIterator>();
252
253                if (start != null) {
254                        for (RecurrenceRule rrule : component.getProperties(RecurrenceRule.class)) {
255                                Recurrence recurrence = ValuedProperty.getValue(rrule);
256                                if (recurrence != null) {
257                                        include.add(createRecurrenceIterator(recurrence, start, timezone));
258                                }
259                        }
260                }
261
262                List<ICalDate> allDates = new ArrayList<ICalDate>();
263                for (RecurrenceDates rdate : component.getProperties(RecurrenceDates.class)) {
264                        allDates.addAll(rdate.getDates());
265                }
266                if (!allDates.isEmpty()) {
267                        include.add(new ICalDateRecurrenceIterator(allDates));
268                }
269
270                if (include.isEmpty()) {
271                        if (start == null) {
272                                return new EmptyDateIterator();
273                        }
274                        include.add(new ICalDateRecurrenceIterator(Collections.singletonList(start)));
275                }
276
277                /////////////EXCLUDE/////////////
278
279                List<RecurrenceIterator> exclude = new ArrayList<RecurrenceIterator>();
280
281                if (start != null) {
282                        for (ExceptionRule exrule : component.getProperties(ExceptionRule.class)) {
283                                Recurrence recurrence = ValuedProperty.getValue(exrule);
284                                if (recurrence != null) {
285                                        exclude.add(createRecurrenceIterator(recurrence, start, timezone));
286                                }
287                        }
288                }
289
290                allDates = new ArrayList<ICalDate>();
291                for (ExceptionDates exdate : component.getProperties(ExceptionDates.class)) {
292                        allDates.addAll(exdate.getValues());
293                }
294                if (!allDates.isEmpty()) {
295                        exclude.add(new ICalDateRecurrenceIterator(allDates));
296                }
297
298                /////////////JOIN/////////////
299
300                RecurrenceIterator includeJoined = join(include);
301                if (exclude.isEmpty()) {
302                        return DateIteratorFactory.createDateIterator(includeJoined);
303                }
304
305                RecurrenceIterator excludeJoined = join(exclude);
306                RecurrenceIterator iterator = RecurrenceIteratorFactory.except(includeJoined, excludeJoined);
307                return DateIteratorFactory.createDateIterator(iterator);
308        }
309
310        /**
311         * Creates a single {@link RecurrenceIterator} that is a union of the given
312         * iterators.
313         * @param iterators the iterators
314         * @return the union of the given iterators
315         * @see RecurrenceIteratorFactory#join
316         */
317        private static RecurrenceIterator join(List<RecurrenceIterator> iterators) {
318                if (iterators.size() == 1) {
319                        return iterators.get(0);
320                }
321                RecurrenceIterator first = iterators.get(0);
322                List<RecurrenceIterator> rest = iterators.subList(1, iterators.size());
323                return RecurrenceIteratorFactory.join(first, rest.toArray(new RecurrenceIterator[0]));
324        }

331                return TimeZone.getTimeZone("UTC");

334        /**
335         * A {@link DateIterator} with nothing in it.
336         */
337        public static class EmptyDateIterator implements DateIterator {
338                public boolean hasNext() {
339                        return false;
340                }
341
342                public Date next() {
343                        throw new NoSuchElementException();
344                }
345
346                public void remove() {
347                        throw new UnsupportedOperationException();
348                }
349
350                public void advanceTo(Date newStartUtc) {
351                        //empty
352                }
353        }
354
355        private static class ICalDateRecurrenceIterator implements RecurrenceIterator {
356                private final List<ICalDate> dates;
357                private int index = 0;
358
359                /*
360                 * Note: I don't think a timezone needs to be passed in here, because it
361                 * appears that the DateValue objects are expected to be in UTC (judging
362                 * by the parameter name in the "advanceTo" method).
363                 */
364                public ICalDateRecurrenceIterator(List<ICalDate> dates) {
365                        this.dates = new ArrayList<ICalDate>(dates);
366                        Collections.sort(this.dates);
367                }
368
369                public boolean hasNext() {
370                        return index < dates.size();
371                }
372
373                public DateValue next() {
374                        ICalDate next = dates.get(index++);
375                        return convertUtc(next);
376                }
377
378                public void advanceTo(DateValue newStartUtc) {
379                        ICalDate newStart = convertUtc(newStartUtc);
380                        while (index < dates.size() && newStart.compareTo(dates.get(index)) > 0) {
381                                index++;
382                        }
383                }
384
385                public void remove() {
386                        throw new UnsupportedOperationException();
387                }
388        }
