033/**
034 * <p>
035 * Represents a period of time (for example, "2 hours and 30 minutes").
036 * </p>
037 * <p>
038 * This class is immutable. Use the {@link #builder} method to construct a new
039 * instance, or the {@link #parse} method to parse a duration string.
040 * </p>
041 * 
042 * <p>
043 * <b>Examples:</b>
044 * </p>
045 * 
046 * <pre class="brush:java">
047 * Duration duration = Duration.builder().hours(2).minutes(30).build();
048 * Duration duration = Duration.parse("PT2H30M");
049 * 
050 * //add a duration value to a Date
051 * Date start = ...
052 * Date end = duration.add(start);
053 * </pre>
054 * @author Michael Angstadt
055 */
056public final class Duration {
057        private final Integer weeks, days, hours, minutes, seconds;
058        private final boolean prior;
059
060        private Duration(Builder b) {
061                weeks = b.weeks;
062                days = b.days;
063                hours = b.hours;
064                minutes = b.minutes;
065                seconds = b.seconds;
066                prior = b.prior;
067        }
068
069        /**
070         * Parses a duration string.
071         * @param value the duration string (e.g. "P30DT10H")
072         * @return the parsed duration
073         * @throws IllegalArgumentException if the duration string is invalid
074         */
075        public static Duration parse(String value) {
076                /*
077                 * Implementation note: Regular expressions are not used to improve
078                 * performance.
079                 */
080
081                if (value.isEmpty()) {
082                        throw parseError(value);
083                }
084
085                int index = 0;
086                char first = value.charAt(index);
087                boolean prior = (first == '-');
088                if (first == '-' || first == '+') {
089                        index++;
090                }
091
092                if (index >= value.length() || value.charAt(index) != 'P') {
093                        throw parseError(value);
094                }
095
096                Builder builder = new Builder();
097                builder.prior(prior);
098
099                StringBuilder buffer = new StringBuilder();
100                for (int i = index + 1; i < value.length(); i++) {
101                        char c = value.charAt(i);
102
103                        if (c == 'T') {
104                                /*
105                                 * A "T" character is supposed to immediately precede the time
106                                 * component value(s). It is required by the syntax, but not
107                                 * really necessary. Ignore it.
108                                 */
109                                continue;
110                        }
111
112                        if (c >= '0' && c <= '9') {
113                                buffer.append(c);
114                                continue;
115                        }
116
117                        if (buffer.length() == 0) {
118                                throw parseError(value);
119                        }
120
121                        Integer num = Integer.valueOf(buffer.toString());
122                        buffer.setLength(0);
123
124                        switch (c) {
125                        case 'W':
126                                builder.weeks(num);
127                                break;
128                        case 'D':
129                                builder.days(num);
130                                break;
131                        case 'H':
132                                builder.hours(num);
133                                break;
134                        case 'M':
135                                builder.minutes(num);
136                                break;
137                        case 'S':
138                                builder.seconds(num);
139                                break;
140                        default:
141                                throw parseError(value);
142                        }
143                }
144
145                return builder.build();
146        }
147
148        private static IllegalArgumentException parseError(String value) {
149                return Messages.INSTANCE.getIllegalArgumentException(20, value);
150        }

152        /**
153         * Builds a duration based on the difference between two dates.
154         * @param start the start date
155         * @param end the end date
156         * @return the duration
157         */
158        public static Duration diff(Date start, Date end) {
159                return fromMillis(end.getTime() - start.getTime());
160        }

162        /**
163         * Builds a duration from a number of milliseconds.
164         * @param milliseconds the number of milliseconds
165         * @return the duration
166         */
167        public static Duration fromMillis(long milliseconds) {
168                Duration.Builder builder = builder();
169
170                if (milliseconds < 0) {
171                        builder.prior(true);
172                        milliseconds *= -1;
173                }
174
175                int seconds = (int) (milliseconds / 1000);
176
177                int weeks = seconds / (60 * 60 * 24 * 7);
178                if (weeks > 0) {
179                        builder.weeks(weeks);
180                }
181                seconds %= 60 * 60 * 24 * 7;
182
183                int days = seconds / (60 * 60 * 24);
184                if (days > 0) {
185                        builder.days(days);
186                }
187                seconds %= 60 * 60 * 24;
188
189                int hours = seconds / (60 * 60);
190                if (hours > 0) {
191                        builder.hours(hours);
192                }
193                seconds %= 60 * 60;
194
195                int minutes = seconds / (60);
196                if (minutes > 0) {
197                        builder.minutes(minutes);
198                }
199                seconds %= 60;
200
201                if (seconds > 0) {
202                        builder.seconds(seconds);
203                }
204
205                return builder.build();
206        }

264        /**
265         * Adds this duration value to a {@link Date} object.
266         * @param date the date to add to
267         * @return the new date value
268         */
269        public Date add(Date date) {
270                Calendar c = Calendar.getInstance();
271                c.setTime(date);
272
273                if (weeks != null) {
274                        int weeks = this.weeks * (prior ? -1 : 1);
275                        c.add(Calendar.DATE, weeks * 7);
276                }
277                if (days != null) {
278                        int days = this.days * (prior ? -1 : 1);
279                        c.add(Calendar.DATE, days);
280                }
281                if (hours != null) {
282                        int hours = this.hours * (prior ? -1 : 1);
283                        c.add(Calendar.HOUR_OF_DAY, hours);
284                }
285                if (minutes != null) {
286                        int minutes = this.minutes * (prior ? -1 : 1);
287                        c.add(Calendar.MINUTE, minutes);
288                }
289                if (seconds != null) {
290                        int seconds = this.seconds * (prior ? -1 : 1);
291                        c.add(Calendar.SECOND, seconds);
292                }
293
294                return c.getTime();
295        }

297        /**
298         * Converts the duration value to milliseconds.
299         * @return the duration value in milliseconds (will be negative if
300         * {@link #isPrior} is true)
301         */
302        public long toMillis() {
303                long totalSeconds = 0;
304
305                if (weeks != null) {
306                        totalSeconds += 60L * 60 * 24 * 7 * weeks;
307                }
308                if (days != null) {
309                        totalSeconds += 60L * 60 * 24 * days;
310                }
311                if (hours != null) {
312                        totalSeconds += 60L * 60 * hours;
313                }
314                if (minutes != null) {
315                        totalSeconds += 60L * minutes;
316                }
317                if (seconds != null) {
318                        totalSeconds += seconds;
319                }
320                if (prior) {
321                        totalSeconds *= -1;
322                }
323
324                return totalSeconds * 1000;
325        }

327        /**
328         * Determines if any time components are present.
329         * @return true if the duration has at least one time component, false if
330         * not
331         */
332        public boolean hasTime() {
333                return hours != null || minutes != null || seconds != null;
334        }

336        @Override
337        public int hashCode() {
338                final int prime = 31;
339                int result = 1;
340                result = prime * result + ((days == null) ? 0 : days.hashCode());
341                result = prime * result + ((hours == null) ? 0 : hours.hashCode());
342                result = prime * result + ((minutes == null) ? 0 : minutes.hashCode());
343                result = prime * result + (prior ? 1231 : 1237);
344                result = prime * result + ((seconds == null) ? 0 : seconds.hashCode());
345                result = prime * result + ((weeks == null) ? 0 : weeks.hashCode());
346                return result;
347        }
348
349        @Override
350        public boolean equals(Object obj) {
351                if (this == obj) return true;
352                if (obj == null) return false;
353                if (getClass() != obj.getClass()) return false;
354                Duration other = (Duration) obj;
355                if (days == null) {
356                        if (other.days != null) return false;
357                } else if (!days.equals(other.days)) return false;
358                if (hours == null) {
359                        if (other.hours != null) return false;
360                } else if (!hours.equals(other.hours)) return false;
361                if (minutes == null) {
362                        if (other.minutes != null) return false;
363                } else if (!minutes.equals(other.minutes)) return false;
364                if (prior != other.prior) return false;
365                if (seconds == null) {
366                        if (other.seconds != null) return false;
367                } else if (!seconds.equals(other.seconds)) return false;
368                if (weeks == null) {
369                        if (other.weeks != null) return false;
370                } else if (!weeks.equals(other.weeks)) return false;
371                return true;
372        }

374        /**
375         * Converts the duration to its string representation.
376         * @return the string representation (e.g. "P4DT1H" for "4 days and 1 hour")
377         */
378        @Override
379        public String toString() {
380                StringBuilder sb = new StringBuilder();
381
382                if (prior) {
383                        sb.append('-');
384                }
385                sb.append('P');
386
387                if (weeks != null) {
388                        sb.append(weeks).append('W');
389                }
390
391                if (days != null) {
392                        sb.append(days).append('D');
393                }
394
395                if (hasTime()) {
396                        sb.append('T');
397
398                        if (hours != null) {
399                                sb.append(hours).append('H');
400                        }
401
402                        if (minutes != null) {
403                                sb.append(minutes).append('M');
404                        }
405
406                        if (seconds != null) {
407                                sb.append(seconds).append('S');
408                        }
409                }
410
411                return sb.toString();
412        }
