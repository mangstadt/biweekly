015package biweekly.util.com.google.ical.values;
016
017import java.text.ParseException;
018import java.util.Map;
019import java.util.TimeZone;
020
021/**
022 * a list of dates, as from an RDATE or EXDATE ical property.
023 *
024 * <p>See RFC 2445 sections 4.8.5.1 and 4.8.5.3.
025 *
026 * @author mikesamuel+svn@gmail.com (Mike Samuel)
027 */
028public class RDateList extends AbstractIcalObject {
029
030  private TimeZone tzid;
031  private DateValue[] datesUtc;
032  private IcalValueType valueType;
033
034  public RDateList(String icalString, TimeZone tzid) throws ParseException {
035    setTzid(tzid);
036    parse(icalString, RRuleSchema.instance());
037  }
038
039  public RDateList(TimeZone tzid) {
040    setTzid(tzid);
041    setName("RDATE");
042    datesUtc = new DateValue[0];
043  }
044
045  public TimeZone getTzid() { return this.tzid; }
046  public void setTzid(TimeZone tzid) {
047    assert null != tzid;
048    this.tzid = tzid;
049  }
050
051  public DateValue[] getDatesUtc() {
052    return null != this.datesUtc ? this.datesUtc.clone() : null;
053  }
054  public void setDatesUtc(DateValue[] datesUtc) {
055    this.datesUtc = datesUtc.clone();
056    if (datesUtc.length > 0) {
057      setValueType((datesUtc[0] instanceof TimeValue)
058                   ? IcalValueType.DATE_TIME
059                   : IcalValueType.DATE);
060    }
061  }
062
063  /**
064   * the type of the values contained by this list as reported by the ical
065   * "VALUE" parameter, typically DATE or DATE-TIME.
066   */
067  public IcalValueType getValueType() {
068    return valueType;
069  }
070
071  public void setValueType(IcalValueType valueType) {
072    this.valueType = valueType;
073  }
074
075  /** returns a String containing ical content lines. */
076  public String toIcal() {
077    StringBuilder buf = new StringBuilder();
078    buf.append(this.getName().toUpperCase());
079    buf.append(";TZID=\"").append(tzid.getID()).append('"');
080    buf.append(";VALUE=").append(valueType.toIcal());
081    if (hasExtParams()) {
082      for (Map.Entry<String, String> param : getExtParams().entrySet()) {
083        String k = param.getKey(),
084               v = param.getValue();
085        if (ICAL_SPECIALS.matcher(v).find()) {
086          v = "\"" + v + "\"";
087        }
088        buf.append(';').append(k).append('=').append(v);
089      }
090    }
091    buf.append(':');
092    for (int i = 0; i < datesUtc.length; ++i) {
093      if (0 != i) { buf.append(','); }
094      DateValue v = datesUtc[i];
095      buf.append(v);
096      if (v instanceof TimeValue) { buf.append('Z'); }
097    }
098    return buf.toString();
099  }
100
101}
