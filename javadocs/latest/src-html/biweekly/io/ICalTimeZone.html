067/**
068 * A timezone that is based on an iCalendar {@link VTimezone} component. This
069 * class is not thread safe.
070 * @author Michael Angstadt
071 */
072@SuppressWarnings("serial")
073public class ICalTimeZone extends TimeZone {
074        private final VTimezone component;
075        private final Map<Observance, List<DateValue>> observanceDateCache;
076        final List<Observance> sortedObservances;
077        private final int rawOffset;
078        private final TimeZone utc = TimeZone.getTimeZone("UTC");
079        private final Calendar utcCalendar = Calendar.getInstance(utc);

081        /**
082         * Creates a new timezone based on an iCalendar VTIMEZONE component.
083         * @param component the VTIMEZONE component to wrap
084         */
085        public ICalTimeZone(VTimezone component) {
086                this.component = component;
087
088                int numObservances = component.getStandardTimes().size() + component.getDaylightSavingsTime().size();
089                observanceDateCache = new IdentityHashMap<Observance, List<DateValue>>(numObservances);
090
091                sortedObservances = calculateSortedObservances();
092
093                rawOffset = calculateRawOffset();
094
095                String id = getValue(component.getTimezoneId());
096                if (id != null) {
097                        setID(id);
098                }
099        }

101        /**
102         * Builds a list of all the observances in the VTIMEZONE component, sorted
103         * by DTSTART.
104         * @return the sorted observances
105         */
106        private List<Observance> calculateSortedObservances() {
107                List<DaylightSavingsTime> daylights = component.getDaylightSavingsTime();
108                List<StandardTime> standards = component.getStandardTimes();
109
110                int numObservances = standards.size() + daylights.size();
111                List<Observance> sortedObservances = new ArrayList<Observance>(numObservances);
112
113                sortedObservances.addAll(standards);
114                sortedObservances.addAll(daylights);
115
116                Collections.sort(sortedObservances, new Comparator<Observance>() {
117                        public int compare(Observance left, Observance right) {
118                                ICalDate startLeft = getValue(left.getDateStart());
119                                ICalDate startRight = getValue(right.getDateStart());
120                                if (startLeft == null && startRight == null) {
121                                        return 0;
122                                }
123                                if (startLeft == null) {
124                                        return -1;
125                                }
126                                if (startRight == null) {
127                                        return 1;
128                                }
129
130                                return startLeft.getRawComponents().compareTo(startRight.getRawComponents());
131                        }
132                });
133
134                return Collections.unmodifiableList(sortedObservances);
135        }

137        @Override
138        public String getDisplayName(boolean daylight, int style, Locale locale) {
139                ListIterator<Observance> it = sortedObservances.listIterator(sortedObservances.size());
140                while (it.hasPrevious()) {
141                        Observance observance = it.previous();
142
143                        if (daylight && observance instanceof DaylightSavingsTime) {
144                                List<TimezoneName> names = observance.getTimezoneNames();
145                                if (!names.isEmpty()) {
146                                        String name = names.get(0).getValue();
147                                        if (name != null) {
148                                                return name;
149                                        }
150                                }
151                        }
152
153                        if (!daylight && observance instanceof StandardTime) {
154                                List<TimezoneName> names = observance.getTimezoneNames();
155                                if (!names.isEmpty()) {
156                                        String name = names.get(0).getValue();
157                                        if (name != null) {
158                                                return name;
159                                        }
160                                }
161                        }
162                }
163
164                return super.getDisplayName(daylight, style, locale);
165        }

167        @Override
168        public int getOffset(int era, int year, int month, int day, int dayOfWeek, int millis) {
169                int hour = millis / 1000 / 60 / 60;
170                millis -= hour * 1000 * 60 * 60;
171                int minute = millis / 1000 / 60;
172                millis -= minute * 1000 * 60;
173                int second = millis / 1000;
174
175                Observance observance = getObservance(year, month + 1, day, hour, minute, second);
176                if (observance == null) {
177                        /*
178                         * Find the first observance that has a DTSTART property and a
179                         * TZOFFSETFROM property.
180                         */
181                        for (Observance obs : sortedObservances) {
182                                ICalDate dateStart = getValue(obs.getDateStart());
183                                if (dateStart == null) {
184                                        continue;
185                                }
186
187                                UtcOffset offsetFrom = getValue(obs.getTimezoneOffsetFrom());
188                                if (offsetFrom == null) {
189                                        continue;
190                                }
191
192                                return (int) offsetFrom.getMillis();
193                        }
194                        return 0;
195                }
196
197                UtcOffset offsetTo = getValue(observance.getTimezoneOffsetTo());
198                return (offsetTo == null) ? 0 : (int) offsetTo.getMillis();
199        }

206        private int calculateRawOffset() {
207                Observance observance = getObservance(new Date());
208                if (observance == null) {
209                        //return the offset of the first STANDARD component
210                        for (Observance obs : sortedObservances) {
211                                if (!(obs instanceof StandardTime)) {
212                                        continue;
213                                }
214
215                                UtcOffset offsetTo = getValue(obs.getTimezoneOffsetTo());
216                                if (offsetTo == null) {
217                                        continue;
218                                }
219
220                                return (int) offsetTo.getMillis();
221                        }
222                        return 0;
223                }
224
225                UtcOffset offset = getValue((observance instanceof StandardTime) ? observance.getTimezoneOffsetTo() : observance.getTimezoneOffsetFrom());
226                return (offset == null) ? 0 : (int) offset.getMillis();
227        }

229        @Override
230        public boolean inDaylightTime(Date date) {
231                if (!useDaylightTime()) {
232                        return false;
233                }
234
235                Observance observance = getObservance(date);
236                return (observance == null) ? false : (observance instanceof DaylightSavingsTime);
237        }

248        @Override
249        public boolean useDaylightTime() {
250                for (Observance observance : sortedObservances) {
251                        if (observance instanceof DaylightSavingsTime) {
252                                return true;
253                        }
254                }
255                return false;
256        }

259         * Gets the timezone information of a date.
260         * @param date the date
261         * @return the timezone information
262         */
263        public Boundary getObservanceBoundary(Date date) {
264                utcCalendar.setTime(date);
265                int year = utcCalendar.get(Calendar.YEAR);
266                int month = utcCalendar.get(Calendar.MONTH) + 1;
267                int day = utcCalendar.get(Calendar.DATE);
268                int hour = utcCalendar.get(Calendar.HOUR);
269                int minute = utcCalendar.get(Calendar.MINUTE);
270                int second = utcCalendar.get(Calendar.SECOND);
271
272                return getObservanceBoundary(year, month, day, hour, minute, second);
273        }

315        /**
316         * Gets the observance information of a date.
317         * @param year the year
318         * @param month the month (1-12)
319         * @param day the day of the month
320         * @param hour the hour
321         * @param minute the minute
322         * @param second the second
323         * @return the observance information or null if none was found
324         */
325        private Boundary getObservanceBoundary(int year, int month, int day, int hour, int minute, int second) {
326                if (sortedObservances.isEmpty()) {
327                        return null;
328                }
329
330                DateValue givenTime = new DateTimeValueImpl(year, month, day, hour, minute, second);
331                int closestIndex = -1;
332                Observance closest = null;
333                DateValue closestValue = null;
334                for (int i = 0; i < sortedObservances.size(); i++) {
335                        Observance observance = sortedObservances.get(i);
336
337                        //skip observances that start after the given time
338                        ICalDate dtstart = getValue(observance.getDateStart());
339                        if (dtstart != null) {
340                                DateValue dtstartValue = convertFromRawComponents(dtstart);
341                                if (dtstartValue.compareTo(givenTime) > 0) {
342                                        continue;
343                                }
344                        }
345
346                        DateValue dateValue = getObservanceDateClosestToTheGivenDate(observance, givenTime, false);
347                        if (dateValue != null && (closestValue == null || closestValue.compareTo(dateValue) < 0)) {
348                                closestValue = dateValue;
349                                closest = observance;
350                                closestIndex = i;
351                        }
352                }
353
354                Observance observanceIn = closest;
355                DateValue observanceInStart = closestValue;
356                Observance observanceAfter = null;
357                DateValue observanceAfterStart = null;
358                if (closestIndex < sortedObservances.size() - 1) {
359                        observanceAfter = sortedObservances.get(closestIndex + 1);
360                        observanceAfterStart = getObservanceDateClosestToTheGivenDate(observanceAfter, givenTime, true);
361                }
362
363                /*
364                 * If any of the DTSTART properties are missing their time components,
365                 * then observanceInStart/observanceAfterStart could be a DateValue
366                 * object. If so, convert it to a DateTimeValue object (see Issue 77).
367                 */
368                if (observanceInStart != null && !(observanceInStart instanceof DateTimeValue)) {
369                        observanceInStart = new DTBuilder(observanceInStart).toDateTime();
370                }
371                if (observanceAfterStart != null && !(observanceAfterStart instanceof DateTimeValue)) {
372                        observanceAfterStart = new DTBuilder(observanceAfterStart).toDateTime();
373                }
374
375                return new Boundary((DateTimeValue) observanceInStart, observanceIn, (DateTimeValue) observanceAfterStart, observanceAfter);
376        }

378        /**
379         * Iterates through each of the timezone boundary dates defined by the given
380         * observance and finds the date that comes closest to the given date.
381         * @param observance the observance
382         * @param givenDate the given date
383         * @param after true to return the closest date <b>greater than</b> the
384         * given date, false to return the closest date <b>less than or equal to</b>
385         * the given date.
386         * @return the closest date
387         */
388        private DateValue getObservanceDateClosestToTheGivenDate(Observance observance, DateValue givenDate, boolean after) {
389                List<DateValue> dateCache = observanceDateCache.get(observance);
390                if (dateCache == null) {
391                        dateCache = new ArrayList<DateValue>();
392                        observanceDateCache.put(observance, dateCache);
393                }
394
395                if (dateCache.isEmpty()) {
396                        DateValue prev = null, cur = null;
397                        boolean stopped = false;
398                        RecurrenceIterator it = createIterator(observance);
399                        while (it.hasNext()) {
400                                cur = it.next();
401                                dateCache.add(cur);
402
403                                if (givenDate.compareTo(cur) < 0) {
404                                        //stop if we have passed the givenTime
405                                        stopped = true;
406                                        break;
407                                }
408
409                                prev = cur;
410                        }
411                        return after ? (stopped ? cur : null) : prev;
412                }
413
414                DateValue last = dateCache.get(dateCache.size() - 1);
415                int comparison = last.compareTo(givenDate);
416                if ((after && comparison <= 0) || comparison < 0) {
417                        RecurrenceIterator it = createIterator(observance);
418
419                        /*
420                         * The "advanceTo()" method skips all dates that are less than the
421                         * given date. I would have thought that we would have to call
422                         * "next()" once because we want it to skip the date that is equal
423                         * to the "last" date. But this causes all the unit tests to fail,
424                         * so I guess not.
425                         */
426                        it.advanceTo(last);
427                        //it.next();
428
429                        DateValue prev = null, cur = null;
430                        boolean stopped = false;
431                        while (it.hasNext()) {
432                                cur = it.next();
433                                dateCache.add(cur);
434
435                                if (givenDate.compareTo(cur) < 0) {
436                                        //stop if we have passed the givenTime
437                                        stopped = true;
438                                        break;
439                                }
440
441                                prev = cur;
442                        }
443                        return after ? (stopped ? cur : null) : prev;
444                }
445
446                /*
447                 * The date is somewhere in the cached list, so find it.
448                 * 
449                 * Note: Read the "binarySearch" method Javadoc carefully for an
450                 * explanation of its return value.
451                 */
452                int index = Collections.binarySearch(dateCache, givenDate);
453
454                if (index < 0) {
455                        /*
456                         * The index where the date would be if it was inside the list.
457                         */
458                        index = (index * -1) - 1;
459
460                        if (after) {
461                                /*
462                                 * This is where the date would be if it was inside the list, so
463                                 * we want to return the date value that's currently at that
464                                 * position.
465                                 */
466                                int afterIndex = index;
467
468                                return (afterIndex < dateCache.size()) ? dateCache.get(afterIndex) : null;
469                        }
470
471                        int beforeIndex = index - 1;
472                        if (beforeIndex < 0) {
473                                return null;
474                        }
475                        if (beforeIndex >= dateCache.size()) {
476                                return dateCache.get(dateCache.size() - 1);
477                        }
478                        return dateCache.get(beforeIndex);
479                }
480
481                /*
482                 * An exact match was found.
483                 */
484                if (after) {
485                        int afterIndex = index + 1; //remember: the date must be >
486                        return (afterIndex < dateCache.size()) ? dateCache.get(afterIndex) : null;
487                }
488                return dateCache.get(index); //remember: the date must be <=
489        }

491        /**
492         * Creates an iterator which iterates over each of the dates in an
493         * observance.
494         * @param observance the observance
495         * @return the iterator
496         */
497        RecurrenceIterator createIterator(Observance observance) {
498                List<RecurrenceIterator> inclusions = new ArrayList<RecurrenceIterator>();
499                List<RecurrenceIterator> exclusions = new ArrayList<RecurrenceIterator>();
500
501                ICalDate dtstart = getValue(observance.getDateStart());
502                if (dtstart != null) {
503                        DateValue dtstartValue = convertFromRawComponents(dtstart);
504
505                        //add DTSTART property
506                        inclusions.add(new DateValueRecurrenceIterator(Arrays.asList(dtstartValue)));
507
508                        //add RRULE properties
509                        for (RecurrenceRule rrule : observance.getProperties(RecurrenceRule.class)) {
510                                Recurrence recur = rrule.getValue();
511                                if (recur != null) {
512                                        inclusions.add(RecurrenceIteratorFactory.createRecurrenceIterator(recur, dtstartValue, utc));
513                                }
514                        }
515
516                        //add EXRULE properties
517                        for (ExceptionRule exrule : observance.getProperties(ExceptionRule.class)) {
518                                Recurrence recur = exrule.getValue();
519                                if (recur != null) {
520                                        exclusions.add(RecurrenceIteratorFactory.createRecurrenceIterator(recur, dtstartValue, utc));
521                                }
522                        }
523                }
524
525                //add RDATE properties
526                List<ICalDate> rdates = new ArrayList<ICalDate>();
527                for (RecurrenceDates rdate : observance.getRecurrenceDates()) {
528                        rdates.addAll(rdate.getDates());
529                }
530                Collections.sort(rdates);
531                inclusions.add(new DateRecurrenceIterator(rdates));
532
533                //add EXDATE properties
534                List<ICalDate> exdates = new ArrayList<ICalDate>();
535                for (ExceptionDates exdate : observance.getProperties(ExceptionDates.class)) {
536                        exdates.addAll(exdate.getValues());
537                }
538                Collections.sort(exdates);
539                exclusions.add(new DateRecurrenceIterator(exdates));
540
541                RecurrenceIterator included = join(inclusions);
542                if (exclusions.isEmpty()) {
543                        return included;
544                }
545
546                RecurrenceIterator excluded = join(exclusions);
547                return RecurrenceIteratorFactory.except(included, excluded);
548        }

550        private static RecurrenceIterator join(List<RecurrenceIterator> iterators) {
551                if (iterators.isEmpty()) {
552                        return new EmptyRecurrenceIterator();
553                }
554
555                RecurrenceIterator first = iterators.get(0);
556                if (iterators.size() == 1) {
557                        return first;
558                }
559
560                List<RecurrenceIterator> theRest = iterators.subList(1, iterators.size());
561                return RecurrenceIteratorFactory.join(first, theRest.toArray(new RecurrenceIterator[0]));
562        }

616        /**
617         * A recurrence iterator that wraps an {@link Iterator}.
618         */
619        private static abstract class IteratorWrapper<T> implements RecurrenceIterator {
620                protected final Iterator<T> it;
621                private DateValue next;
622
623                public IteratorWrapper(Iterator<T> it) {
624                        this.it = it;
625                }
626
627                public DateValue next() {
628                        if (next != null) {
629                                DateValue value = next;
630                                next = null;
631                                return value;
632                        }
633                        return toDateValue(it.next());
634                }
635
636                public boolean hasNext() {
637                        return next != null || it.hasNext();
638                }
639
640                public void advanceTo(DateValue newStartUtc) {
641                        if (this.next != null && this.next.compareTo(newStartUtc) >= 0) {
642                                return;
643                        }
644
645                        while (it.hasNext()) {
646                                DateValue next = toDateValue(it.next());
647                                if (next.compareTo(newStartUtc) >= 0) {
648                                        this.next = next;
649                                        break;
650                                }
651                        }
652                }
653
654                public void remove() {
655                        //RecurrenceIterator does not support this method
656                        throw new UnsupportedOperationException();
657                }
658
659                protected abstract DateValue toDateValue(T next);
660        }

662        /**
663         * Holds the timezone observance information of a particular date.
664         */
665        public static class Boundary {
666                private final DateTimeValue observanceInStart, observanceAfterStart;
667                private final Observance observanceIn, observanceAfter;
668
669                public Boundary(DateTimeValue observanceInStart, Observance observanceIn, DateTimeValue observanceAfterStart, Observance observanceAfter) {
670                        this.observanceInStart = observanceInStart;
671                        this.observanceAfterStart = observanceAfterStart;
672                        this.observanceIn = observanceIn;
673                        this.observanceAfter = observanceAfter;
674                }
675
676                /**
677                 * Gets start time of the observance that the date resides in.
678                 * @return the time
679                 */
680                public DateTimeValue getObservanceInStart() {
681                        return observanceInStart;
682                }
683
684                /**
685                 * Gets the start time the observance that comes after the observance
686                 * that the date resides in.
687                 * @return the time
688                 */
689                public DateTimeValue getObservanceAfterStart() {
690                        return observanceAfterStart;
691                }
692
693                /**
694                 * Gets the observance that the date resides in.
695                 * @return the observance
696                 */
697                public Observance getObservanceIn() {
698                        return observanceIn;
699                }
700
701                /**
702                 * Gets the observance that comes after the observance that the date
703                 * resides in.
704                 * @return the observance
705                 */
706                public Observance getObservanceAfter() {
707                        return observanceAfter;
708                }
709
710                @Override
711                public String toString() {
712                        return "Boundary [observanceInStart=" + observanceInStart + ", observanceAfterStart=" + observanceAfterStart + ", observanceIn=" + observanceIn + ", observanceAfter=" + observanceAfter + "]";
713                }
714        }
