424         * <p>
425         * Unmarshals a property from an XML document (xCal).
426         * </p>
427         * <p>
428         * This method should be overridden by child classes that wish to support
429         * xCal. The default implementation of this method will find the first child

435         * </p>
436         * @param element the property's XML element
437         * @param parameters the parsed parameters. These parameters will be
438         * assigned to the property object once this method returns. Therefore, do

452                return _parseText(value, dataType, parameters, context);

455        /**
456         * <p>
457         * Unmarshals a property from a JSON data stream (jCal).
458         * </p>
459         * <p>
460         * This method should be overridden by child classes that wish to support
461         * jCal. The default implementation of this method will convert the jCal
462         * property value to a string and pass it into the {@link #_parseText}
463         * method.
464         * </p>
465         * 
466         * <hr>
467         * 
468         * <p>
469         * The following paragraphs describe the way in which this method's default
470         * implementation converts a jCal value to a string:
471         * </p>
472         * <p>
473         * If the jCal value consists of a single, non-array, non-object value, then
474         * the value is converted to a string. Special characters (backslashes,
475         * commas, and semicolons) are escaped in order to simulate what the value
476         * might look like in a plain-text iCalendar object.<br>
477         * <code>["x-foo", {}, "text", "the;value"] --&gt; "the\;value"</code><br>
478         * <code>["x-foo", {}, "text", 2] --&gt; "2"</code>
479         * </p>
480         * <p>
481         * If the jCal value consists of multiple, non-array, non-object values,
482         * then all the values are appended together in a single string, separated
483         * by commas. Special characters (backslashes, commas, and semicolons) are
484         * escaped for each value in order to prevent commas from being treated as
485         * delimiters, and to simulate what the value might look like in a
486         * plain-text iCalendar object.<br>
487         * <code>["x-foo", {}, "text", "one", "two,three"] --&gt;
488         * "one,two\,three"</code>
489         * </p>
490         * <p>
491         * If the jCal value is a single array, then this array is treated as a
492         * "structured value", and converted its plain-text representation. Special
493         * characters (backslashes, commas, and semicolons) are escaped for each
494         * value in order to prevent commas and semicolons from being treated as
495         * delimiters.<br>
496         * <code>["x-foo", {}, "text", ["one", ["two", "three"], "four;five"]]
497         * --&gt; "one;two,three;four\;five"</code>
498         * </p>
499         * <p>
500         * If the jCal value starts with a JSON object, then the object is converted
501         * to a format identical to the one used in the RRULE and EXRULE properties.
502         * Special characters (backslashes, commas, semicolons, and equal signs) are
503         * escaped for each value in order to preserve the syntax of the string
504         * value.<br>
505         * <code>["x-foo", {}, "text", {"one": 1, "two": [2, 2.5]}] --&gt; "ONE=1;TWO=2,2.5"</code>
506         * </p>
507         * <p>
508         * For all other cases, behavior is undefined.
509         * </p>
510         * @param value the property's JSON value
511         * @param dataType the data type
512         * @param parameters the parsed parameters. These parameters will be
513         * assigned to the property object once this method returns. Therefore, do
514         * not assign any parameters to the property object itself whilst inside of
515         * this method, or else they will be overwritten.
516         * @param context the context
517         * @return the unmarshalled property object
518         * @throws CannotParseException if the scribe could not parse the property's
519         * value
520         * @throws SkipMeException if the property should not be added to the final
521         * {@link ICalendar} object
522         */
523        protected T _parseJson(JCalValue value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
524                String valueStr = jcalValueToString(value);
525                return _parseText(valueStr, dataType, parameters, context);
526        }

528        /**
529         * Converts a jCal value to its plain-text format representation.
530         * @param value the jCal value
531         * @return the plain-text format representation (for example, "1,2,3" for a
532         * list of values)
533         */
534        private static String jcalValueToString(JCalValue value) {
535                List<JsonValue> values = value.getValues();
536                if (values.size() > 1) {
537                        List<String> multi = value.asMulti();
538                        if (!multi.isEmpty()) {
539                                return VObjectPropertyValues.writeList(multi);
540                        }
541                }
542
543                if (!values.isEmpty() && values.get(0).getArray() != null) {
544                        List<List<String>> structured = value.asStructured();
545                        if (!structured.isEmpty()) {
546                                return VObjectPropertyValues.writeStructured(structured, true);
547                        }
548                }
549
550                if (values.get(0).getObject() != null) {
551                        ListMultimap<String, String> object = value.asObject();
552                        if (!object.isEmpty()) {
553                                return VObjectPropertyValues.writeMultimap(object.getMap());
554                        }
555                }
556
557                return VObjectPropertyValues.escape(value.asSingle());
558        }

581        protected static class DateParser {
582                private String value;
583                private Boolean hasTime;
584
585                /**
586                 * Creates a new date writer object.
587                 * @param value the date string to parse
588                 */
589                public DateParser(String value) {
590                        this.value = value;
591                }
592
593                /**
594                 * Forces the value to be parsed as a date-time or date value.
595                 * @param hasTime true to parsed as a date-time value, false to parse as
596                 * a date value, null to parse as whatever value it is (defaults to
597                 * null)
598                 * @return this
599                 */
600                public DateParser hasTime(Boolean hasTime) {
601                        this.hasTime = hasTime;
602                        return this;
603                }
604
605                /**
606                 * Parses the date string.
607                 * @return the parsed date
608                 * @throws IllegalArgumentException if the date string is invalid
609                 */
610                public ICalDate parse() {
611                        DateTimeComponents components = DateTimeComponents.parse(value, hasTime);
612                        Date date = components.toDate();
613                        boolean hasTime = components.hasTime();
614
615                        return new ICalDate(date, components, hasTime);
616                }

641        protected static DateWriter date(ICalDate date, ICalProperty property, WriteContext context) {
642                boolean floating;
643                TimeZone tz;
644                TimezoneAssignment globalTz = context.getGlobalTimezone();
645                if (globalTz == null) {
646                        TimezoneInfo tzinfo = context.getTimezoneInfo();
647                        floating = tzinfo.isFloating(property);
648                        TimezoneAssignment assignment = tzinfo.getTimezoneToWriteIn(property);
649                        tz = (assignment == null) ? null : assignment.getTimeZone();
650                } else {
651                        floating = false;
652                        tz = globalTz.getTimeZone();
653                }
654
655                context.addDate(date, floating, tz);
656                return date(date).tz(floating, tz);
657        }

664                private TimeZone timezone;
665                private boolean observance = false;
666                private boolean extended = false;
667                private boolean utc = false;
668
669                /**
670                 * Creates a new date writer object.
671                 * @param date the date to format
672                 */
673                public DateWriter(ICalDate date) {
674                        this.date = date;
675                }
676
677                /**
678                 * Sets whether the property is within an observance or not.
679                 * @param observance true if it's in an observance, false if not
680                 * (defaults to false)
681                 * @return this
682                 */
683                public DateWriter observance(boolean observance) {
684                        this.observance = observance;
685                        return this;
686                }
687
688                /**
689                 * Sets whether to write the value in UTC or not.
690                 * @param utc true to write in UTC, false not to
691                 * @return this
692                 */
693                public DateWriter utc(boolean utc) {
694                        this.utc = utc;
695                        return this;
696                }
697
698                /**
699                 * Sets the timezone.
700                 * @param floating true to use floating time, false not to
701                 * @param timezone the timezone
702                 * @return this
703                 */
704                public DateWriter tz(boolean floating, TimeZone timezone) {
705                        if (floating) {
706                                timezone = TimeZone.getDefault();
707                        }
708                        this.timezone = timezone;
709                        return this;
710                }
711
712                /**
713                 * Sets whether to use extended format or basic.
714                 * @param extended true to use extended format, false to use basic
715                 * (defaults to "false")
716                 * @return this
717                 */
718                public DateWriter extended(boolean extended) {
719                        this.extended = extended;
720                        return this;
721                }
722
723                /**
724                 * Creates the date string.
725                 * @return the date string
726                 */
727                public String write() {
728                        if (date == null) {
729                                return "";
730                        }
731
732                        if (observance) {
733                                return writeObservance();
734                        }
735
736                        if (utc) {
737                                return writeUtc();
738                        }
739
740                        return date.hasTime() ? writeDateTime() : writeDate();
741                }
742
743                private String writeObservance() {
744                        DateTimeComponents components = date.getRawComponents();
745                        if (components == null) {
746                                ICalDateFormat format = extended ? ICalDateFormat.DATE_TIME_EXTENDED_WITHOUT_TZ : ICalDateFormat.DATE_TIME_BASIC_WITHOUT_TZ;
747                                return format.format(date);
748                        }
749
750                        return components.toString(true, extended);
751                }
752
753                private String writeUtc() {
754                        ICalDateFormat format = extended ? ICalDateFormat.UTC_TIME_EXTENDED : ICalDateFormat.UTC_TIME_BASIC;
755                        return format.format(date);
756                }
757
758                private String writeDateTime() {
759                        ICalDateFormat format;
760                        if (timezone == null) {
761                                format = extended ? ICalDateFormat.UTC_TIME_EXTENDED : ICalDateFormat.UTC_TIME_BASIC;
762                        } else {
763                                format = extended ? ICalDateFormat.DATE_TIME_EXTENDED_WITHOUT_TZ : ICalDateFormat.DATE_TIME_BASIC_WITHOUT_TZ;
764                        }
765
766                        return format.format(date, timezone);
767                }
768
769                private String writeDate() {
770                        ICalDateFormat format = extended ? ICalDateFormat.DATE_EXTENDED : ICalDateFormat.DATE_BASIC;
771                        return format.format(date);
772                }
773        }

775        /**
776         * Adds a TZID parameter to a property's parameter list if necessary.
777         * @param property the property
778         * @param hasTime true if the property value has a time component, false if
779         * not
780         * @param context the write context
781         * @return the property's new set of parameters
782         */
783        protected static ICalParameters handleTzidParameter(ICalProperty property, boolean hasTime, WriteContext context) {
784                ICalParameters parameters = property.getParameters();
785
786                //date values don't have timezones
787                if (!hasTime) {
788                        return parameters;
789                }
790
791                //vCal doesn't use the TZID parameter
792                if (context.getVersion() == ICalVersion.V1_0) {
793                        return parameters;
794                }
795
796                //floating values don't have timezones
797                TimezoneInfo tzinfo = context.getTimezoneInfo();
798                boolean floating = tzinfo.isFloating(property);
799                if (floating) {
800                        return parameters;
801                }
802
803                TimezoneAssignment tz;
804                TimezoneAssignment globalTz = context.getGlobalTimezone();
805                if (globalTz == null) {
806                        tz = tzinfo.getTimezoneToWriteIn(property);
807                        if (tz == null) {
808                                //write in UTC
809                                return parameters;
810                        }
811                } else {
812                        tz = globalTz;
813                }
814
815                String tzid = null;
816                VTimezone component = tz.getComponent();
817                String globalId = tz.getGlobalId();
818                if (component != null) {
819                        tzid = ValuedProperty.getValue(component.getTimezoneId());
820                } else if (globalId != null) {
821                        tzid = '/' + globalId;
822                }
823
824                if (tzid == null) {
825                        //should never happen
826                        tzid = tz.getTimeZone().getID();
827                }
828
829                parameters = new ICalParameters(parameters);
830                parameters.setTimezoneId(tzid);
831                return parameters;
832        }

834        /**
835         * Creates a {@link CannotParseException}, indicating that the XML elements
836         * that the parser expected to find are missing from the property's XML
837         * element.
838         * @param dataTypes the expected data types (null for "unknown")
839         * @return the exception
840         */
841        protected static CannotParseException missingXmlElements(ICalDataType... dataTypes) {
842                String[] elements = new String[dataTypes.length];
843                for (int i = 0; i < dataTypes.length; i++) {
844                        ICalDataType dataType = dataTypes[i];
845                        elements[i] = (dataType == null) ? "unknown" : dataType.getName().toLowerCase();
846                }
847                return missingXmlElements(elements);
848        }

851         * Creates a {@link CannotParseException}, indicating that the XML elements
852         * that the parser expected to find are missing from property's XML element.
853         * @param elements the names of the expected XML elements
854         * @return the exception
855         */
856        protected static CannotParseException missingXmlElements(String... elements) {
857                return new CannotParseException(23, Arrays.toString(elements));
858        }
