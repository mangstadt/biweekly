088/**
089 * <p>
090 * Represents an XML document that contains iCalendar objects ("xCal" standard).
091 * This class can be used to read and write xCal documents.
092 * </p>
093 * <p>
094 * <b>Examples:</b>
095 * </p>
096 * 
097 * <pre class="brush:java">
098 * String xml =
099 * "&lt;?xml version=\"1.0\" encoding=\"utf-8\" ?&gt;" +
100 * "&lt;icalendar xmlns=\"urn:ietf:params:xml:ns:icalendar-2.0\"&gt;" +
101 *   "&lt;vcalendar&gt;" +
102 *     "&lt;properties&gt;" +
103 *       "&lt;prodid&gt;&lt;text&gt;-//Example Inc.//Example Client//EN&lt;/text&gt;&lt;/prodid&gt;" +
104 *       "&lt;version&gt;&lt;text&gt;2.0&lt;/text&gt;&lt;/version&gt;" +
105 *     "&lt;/properties&gt;" +
106 *     "&lt;components&gt;" +
107 *       "&lt;vevent&gt;" +
108 *         "&lt;properties&gt;" +
109 *           "&lt;dtstart&gt;&lt;date-time&gt;2013-06-27T13:00:00Z&lt;/date-time&gt;&lt;/dtstart&gt;" +
110 *           "&lt;dtend&gt;&lt;date-time&gt;2013-06-27T15:00:00Z&lt;/date-time&gt;&lt;/dtend&gt;" +
111 *           "&lt;summary&gt;&lt;text&gt;Team Meeting&lt;/text&gt;&lt;/summary&gt;" +
112 *         "&lt;/properties&gt;" +
113 *       "&lt;/vevent&gt;" +
114 *     "&lt;/components&gt;" +
115 *   "&lt;/vcalendar&gt;" +
116 * "&lt;/icalendar&gt;";
117 *     
118 * //parsing an existing xCal document
119 * XCalDocument xcal = new XCalDocument(xml);
120 * List&lt;ICalendar&gt; icals = xcal.getICalendars();
121 * 
122 * //creating an empty xCal document
123 * XCalDocument xcal = new XCalDocument();
124 * 
125 * //ICalendar objects can be added at any time
126 * ICalendar ical = new ICalendar();
127 * xcal.addICalendar(ical);
128 * 
129 * //retrieving the raw XML DOM
130 * Document document = xcal.getDocument();
131 * 
132 * //call one of the "write()" methods to output the xCal document
133 * File file = new File("meeting.xml");
134 * xcal.write(file);
135 * </pre>
136 * @author Michael Angstadt
137 * @see <a href="http://tools.ietf.org/html/rfc6321">RFC 6321</a>
138 */

196        /**
197         * Wraps an existing XML DOM object.
198         * @param document the XML DOM that contains the xCal document
199         */
200        public XCalDocument(Document document) {
201                this.document = document;
202
203                XPath xpath = XPathFactory.newInstance().newXPath();
204                xpath.setNamespaceContext(nsContext);
205
206                try {
207                        //find the <icalendar> element
208                        String prefix = nsContext.getPrefix();
209                        icalendarRootElement = (Element) xpath.evaluate("//" + prefix + ":" + ICALENDAR.getLocalPart(), document, XPathConstants.NODE);
210                } catch (XPathExpressionException e) {
211                        //never thrown, xpath expression is hard coded
212                }
213        }

222        }

464        /**
465         * Writes the xCal document to a writer.
466         * @param writer the writer
467         * @param outputProperties properties to assign to the JAXP transformer (see
468         * {@link Transformer#setOutputProperty})
469         * @throws TransformerException if there's a problem writing to the writer
470         */
471        public void write(Writer writer, Map<String, String> outputProperties) throws TransformerException {
472                Transformer transformer;
473                try {
474                        transformer = TransformerFactory.newInstance().newTransformer();
475                } catch (TransformerConfigurationException e) {
476                        //should never be thrown because we're not doing anything fancy with the configuration
477                        throw new RuntimeException(e);
478                } catch (TransformerFactoryConfigurationError e) {
479                        //should never be thrown because we're not doing anything fancy with the configuration
480                        throw new RuntimeException(e);
481                }
482
483                /*
484                 * Using Transformer#setOutputProperties(Properties) doesn't work for
485                 * some reason for setting the number of indentation spaces.
486                 */
487                for (Map.Entry<String, String> entry : outputProperties.entrySet()) {
488                        String key = entry.getKey();
489                        String value = entry.getValue();
490                        transformer.setOutputProperty(key, value);
491                }
492
493                DOMSource source = new DOMSource(document);
494                StreamResult result = new StreamResult(writer);
495                transformer.transform(source, result);
496        }

503        private class XCalDocumentStreamReader extends StreamReader {
504                private final Iterator<Element> vcalendarElements = getVCalendarElements().iterator();
505
506                @Override
507                protected ICalendar _readNext() throws IOException {
508                        if (!vcalendarElements.hasNext()) {
509                                return null;
510                        }
511
512                        context.setVersion(ICalVersion.V2_0);
513                        Element vcalendarElement = vcalendarElements.next();
514                        return parseICal(vcalendarElement);
515                }
516
517                private ICalendar parseICal(Element icalElement) {
518                        ICalComponent root = parseComponent(icalElement);
519                        if (root instanceof ICalendar) {
520                                return (ICalendar) root;
521                        }
522
523                        //shouldn't happen, since only <vcalendar> elements are passed into this method
524                        ICalendar ical = icalMarshaller.emptyInstance();
525                        ical.addComponent(root);
526                        return ical;
527                }
528
529                private ICalComponent parseComponent(Element componentElement) {
530                        //create the component object
531                        ICalComponentScribe<? extends ICalComponent> scribe = index.getComponentScribe(componentElement.getLocalName(), ICalVersion.V2_0);
532                        ICalComponent component = scribe.emptyInstance();
533                        boolean isICalendar = component instanceof ICalendar;
534
535                        //parse properties
536                        for (Element propertyWrapperElement : getChildElements(componentElement, PROPERTIES)) { //there should be only one <properties> element, but parse them all incase there are more
537                                for (Element propertyElement : XmlUtils.toElementList(propertyWrapperElement.getChildNodes())) {
538                                        ICalProperty property = parseProperty(propertyElement);
539                                        if (property == null) {
540                                                continue;
541                                        }
542
543                                        //set "ICalendar.version" if the value of the VERSION property is recognized
544                                        //otherwise, unmarshal VERSION like a normal property
545                                        if (isICalendar && property instanceof Version) {
546                                                Version version = (Version) property;
547                                                ICalVersion icalVersion = version.toICalVersion();
548                                                if (icalVersion != null) {
549                                                        context.setVersion(icalVersion);
550                                                        continue;
551                                                }
552                                        }
553
554                                        component.addProperty(property);
555                                }
556                        }
557
558                        //parse sub-components
559                        for (Element componentWrapperElement : getChildElements(componentElement, COMPONENTS)) { //there should be only one <components> element, but parse them all incase there are more
560                                for (Element subComponentElement : XmlUtils.toElementList(componentWrapperElement.getChildNodes())) {
561                                        if (!XCAL_NS.equals(subComponentElement.getNamespaceURI())) {
562                                                continue;
563                                        }
564
565                                        ICalComponent subComponent = parseComponent(subComponentElement);
566                                        component.addComponent(subComponent);
567                                }
568                        }
569
570                        return component;
571                }
572
573                private ICalProperty parseProperty(Element propertyElement) {
574                        ICalParameters parameters = parseParameters(propertyElement);
575                        String propertyName = propertyElement.getLocalName();
576                        QName qname = new QName(propertyElement.getNamespaceURI(), propertyName);
577
578                        context.getWarnings().clear();
579                        context.setPropertyName(propertyName);
580                        ICalPropertyScribe<? extends ICalProperty> scribe = index.getPropertyScribe(qname);
581                        try {
582                                ICalProperty property = scribe.parseXml(propertyElement, parameters, context);
583                                warnings.addAll(context.getWarnings());
584                                return property;
585                        } catch (SkipMeException e) {
586                                //@formatter:off
587                                warnings.add(new ParseWarning.Builder(context)
588                                        .message(0, e.getMessage())
589                                        .build()
590                                );
591                                //@formatter:on
592                                return null;
593                        } catch (CannotParseException e) {
594                                //@formatter:off
595                                warnings.add(new ParseWarning.Builder(context)
596                                        .message(e)
597                                        .build()
598                                );
599                                //@formatter:on
600
601                                scribe = index.getPropertyScribe(Xml.class);
602                                return scribe.parseXml(propertyElement, parameters, context);
603                        }
604                }
605
606                private ICalParameters parseParameters(Element propertyElement) {
607                        ICalParameters parameters = new ICalParameters();
608
609                        for (Element parametersElement : getChildElements(propertyElement, PARAMETERS)) { //there should be only one <parameters> element, but parse them all incase there are more
610                                List<Element> paramElements = XmlUtils.toElementList(parametersElement.getChildNodes());
611                                for (Element paramElement : paramElements) {
612                                        if (!XCAL_NS.equals(paramElement.getNamespaceURI())) {
613                                                continue;
614                                        }
615
616                                        String name = paramElement.getLocalName().toUpperCase();
617                                        List<Element> valueElements = XmlUtils.toElementList(paramElement.getChildNodes());
618                                        if (valueElements.isEmpty()) {
619                                                //this should never be true if the xCal follows the specs
620                                                String value = paramElement.getTextContent();
621                                                parameters.put(name, value);
622                                                continue;
623                                        }
624
625                                        for (Element valueElement : valueElements) {
626                                                if (!XCAL_NS.equals(valueElement.getNamespaceURI())) {
627                                                        continue;
628                                                }
629
630                                                String value = valueElement.getTextContent();
631                                                parameters.put(name, value);
632                                        }
633                                }
634                        }
635
636                        return parameters;
637                }

639                private List<Element> getVCalendarElements() {
640                        return (icalendarRootElement == null) ? Collections.<Element> emptyList() : getChildElements(icalendarRootElement, VCALENDAR);
641                }
642
643                private List<Element> getChildElements(Element parent, QName qname) {
644                        List<Element> elements = new ArrayList<Element>();
645                        for (Element child : XmlUtils.toElementList(parent.getChildNodes())) {
646                                QName childQName = new QName(child.getNamespaceURI(), child.getLocalName());
647                                if (qname.equals(childQName)) {
648                                        elements.add(child);
649                                }
650                        }
651                        return elements;
652                }

659        public class XCalDocumentStreamWriter extends XCalWriterBase {
660                @Override
661                public void write(ICalendar ical) {
662                        try {
663                                super.write(ical);
664                        } catch (IOException e) {
665                                //won't be thrown because we're writing to DOM
666                        }
667                }
668
669                @Override
670                protected void _write(ICalendar ical) {
671                        Element element = buildComponentElement(ical);
672
673                        if (icalendarRootElement == null) {
674                                icalendarRootElement = buildElement(ICALENDAR);
675                                Element documentRoot = document.getDocumentElement();
676                                if (documentRoot == null) {
677                                        document.appendChild(icalendarRootElement);
678                                } else {
679                                        documentRoot.appendChild(icalendarRootElement);
680                                }
681                        }
682                        icalendarRootElement.appendChild(element);
683                }
684
685                @SuppressWarnings({ "rawtypes", "unchecked" })
686                private Element buildComponentElement(ICalComponent component) {
687                        ICalComponentScribe componentScribe = index.getComponentScribe(component);
688                        Element componentElement = buildElement(componentScribe.getComponentName().toLowerCase());
689
690                        Element propertiesWrapperElement = buildElement(PROPERTIES);
691                        List propertyObjs = componentScribe.getProperties(component);
692                        if (component instanceof ICalendar && component.getProperty(Version.class) == null) {
693                                //add a version property
694                                propertyObjs.add(0, new Version(targetVersion));
695                        }
696
697                        for (Object propertyObj : propertyObjs) {
698                                context.setParent(component); //set parent here incase a scribe resets the parent
699                                ICalProperty property = (ICalProperty) propertyObj;
700
701                                //create property element
702                                Element propertyElement = buildPropertyElement(property);
703                                if (propertyElement != null) {
704                                        propertiesWrapperElement.appendChild(propertyElement);
705                                }
706                        }
707                        if (propertiesWrapperElement.hasChildNodes()) {
708                                componentElement.appendChild(propertiesWrapperElement);
709                        }
710
711                        List subComponents = componentScribe.getComponents(component);
712                        if (component instanceof ICalendar) {
713                                //add the VTIMEZONE components that were auto-generated by TimezoneOptions
714                                Collection<VTimezone> tzs = getTimezoneComponents();
715                                for (VTimezone tz : tzs) {
716                                        if (!subComponents.contains(tz)) {
717                                                subComponents.add(0, tz);
718                                        }
719                                }
720                        }
721                        Element componentsWrapperElement = buildElement(COMPONENTS);
722                        for (Object subComponentObj : subComponents) {
723                                ICalComponent subComponent = (ICalComponent) subComponentObj;
724                                Element subComponentElement = buildComponentElement(subComponent);
725                                if (subComponentElement != null) {
726                                        componentsWrapperElement.appendChild(subComponentElement);
727                                }
728                        }
729                        if (componentsWrapperElement.hasChildNodes()) {
730                                componentElement.appendChild(componentsWrapperElement);
731                        }
732
733                        return componentElement;
734                }
735
736                @SuppressWarnings({ "rawtypes", "unchecked" })
737                private Element buildPropertyElement(ICalProperty property) {
738                        Element propertyElement;
739                        ICalPropertyScribe scribe = index.getPropertyScribe(property);
740
741                        if (property instanceof Xml) {
742                                Xml xml = (Xml) property;
743
744                                Document value = xml.getValue();
745                                if (value == null) {
746                                        return null;
747                                }
748
749                                //import the XML element into the xCal DOM
750                                propertyElement = value.getDocumentElement();
751                                propertyElement = (Element) document.importNode(propertyElement, true);
752                        } else {
753                                propertyElement = buildElement(scribe.getQName());
754
755                                //marshal value
756                                try {
757                                        scribe.writeXml(property, propertyElement, context);
758                                } catch (SkipMeException e) {
759                                        return null;
760                                }
761                        }
762
763                        //build parameters
764                        ICalParameters parameters = scribe.prepareParameters(property, context);
765                        if (!parameters.isEmpty()) {
766                                Element parametersElement = buildParametersElement(parameters);
767                                propertyElement.insertBefore(parametersElement, propertyElement.getFirstChild());
768                        }
769
770                        return propertyElement;
771                }
772
773                private Element buildParametersElement(ICalParameters parameters) {
774                        Element parametersWrapperElement = buildElement(PARAMETERS);
775
776                        for (Map.Entry<String, List<String>> parameter : parameters) {
777                                String name = parameter.getKey().toLowerCase();
778                                ICalDataType dataType = parameterDataTypes.get(name);
779                                String dataTypeStr = (dataType == null) ? "unknown" : dataType.getName().toLowerCase();
780
781                                Element parameterElement = buildAndAppendElement(name, parametersWrapperElement);
782                                for (String parameterValue : parameter.getValue()) {
783                                        Element parameterValueElement = buildAndAppendElement(dataTypeStr, parameterElement);
784                                        parameterValueElement.setTextContent(parameterValue);
785                                }
786                        }
787
788                        return parametersWrapperElement;
789                }
790
791                private Element buildElement(String localName) {
792                        return buildElement(new QName(XCAL_NS, localName));
793                }
794
795                private Element buildElement(QName qname) {
796                        return document.createElementNS(qname.getNamespaceURI(), qname.getLocalPart());
797                }
798
799                private Element buildAndAppendElement(String localName, Element parent) {
800                        return buildAndAppendElement(new QName(XCAL_NS, localName), parent);
801                }
802
803                private Element buildAndAppendElement(QName qname, Element parent) {
804                        Element child = document.createElementNS(qname.getNamespaceURI(), qname.getLocalPart());
805                        parent.appendChild(child);
806                        return child;
807                }
808
809                public void close() {
810                        //do nothing
811                }
812        }
