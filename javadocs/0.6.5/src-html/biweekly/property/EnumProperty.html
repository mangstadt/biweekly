037/**
038 * Represents a property that has a defined set of acceptable values (for
039 * example, the {@link Action} property).
040 * @author Michael Angstadt
041 */
042public abstract class EnumProperty extends TextProperty {
043        /**
044         * Creates an enum property.
045         * @param value the property value
046         */
047        public EnumProperty(String value) {
048                super(value);
049        }
050
051        /**
052         * Copy constructor.
053         * @param original the property to make a copy of
054         */
055        public EnumProperty(EnumProperty original) {
056                super(original);
057        }
058
059        /**
060         * Compares the property's value with a given string (case-insensitive).
061         * @param value the string
062         * @return true if it's equal, false if not
063         */
064        protected boolean is(String value) {
065                return value.equalsIgnoreCase(this.value);
066        }
067
068        /**
069         * Gets the list of acceptable values for this property.
070         * @param version the version
071         * @return the list of acceptable values
072         */
073        protected abstract Collection<String> getStandardValues(ICalVersion version);
074
075        /**
076         * Gets the iCalendar versions that this property's value is supported in.
077         * Meant to be overridden by the child class.
078         * @return the supported versions
079         */
080        protected Collection<ICalVersion> getValueSupportedVersions() {
081                return (value == null) ? Collections.<ICalVersion> emptyList() : Arrays.asList(ICalVersion.values());
082        }
083
084        @Override
085        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
086                super.validate(components, version, warnings);
087                if (value == null) {
088                        return;
089                }
090
091                Collection<ICalVersion> supportedVersions = getValueSupportedVersions();
092                if (supportedVersions.isEmpty()) {
093                        //it's a non-standard value
094                        warnings.add(new ValidationWarning(28, value, getStandardValues(version)));
095                        return;
096                }
097
098                boolean supported = supportedVersions.contains(version);
099                if (!supported) {
100                        warnings.add(new ValidationWarning(46, value, supportedVersions));
101                }
102        }
103
104        @Override
105        public int hashCode() {
106                final int prime = 31;
107                int result = 1;
108                result = prime * result + parameters.hashCode();
109                result = prime * result + ((value == null) ? 0 : value.toLowerCase().hashCode());
110                return result;
111        }
112
113        @Override
114        public boolean equals(Object obj) {
115                if (this == obj) return true;
116                if (obj == null) return false;
117                if (getClass() != obj.getClass()) return false;
118                EnumProperty other = (EnumProperty) obj;
119                if (!parameters.equals(other.parameters)) return false;
120                if (value == null) {
121                        if (other.value != null) return false;
122                } else if (!value.equalsIgnoreCase(other.value)) return false;
123                return true;
124        }
125}
