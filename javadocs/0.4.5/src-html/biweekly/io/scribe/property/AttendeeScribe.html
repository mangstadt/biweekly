038/**
039 * Marshals {@link Attendee} properties.
040 * @author Michael Angstadt
041 */
042public class AttendeeScribe extends ICalPropertyScribe<Attendee> {
043        public AttendeeScribe() {
044                super(Attendee.class, "ATTENDEE");
045        }
046
047        @Override
048        protected ICalDataType _defaultDataType(ICalVersion version) {
049                switch (version) {
050                case V1_0:
051                        return null;
052                default:
053                        return ICalDataType.CAL_ADDRESS;
054                }
055        }
056
057        @Override
058        protected ICalDataType _dataType(Attendee property, ICalVersion version) {
059                if (version == ICalVersion.V1_0 && property.getUri() != null) {
060                        return ICalDataType.URL;
061                }
062                return defaultDataType(version);
063        }
064
065        @Override
066        protected ICalParameters _prepareParameters(Attendee property, WriteContext context) {
067                ICalParameters copy = new ICalParameters(property.getParameters());
068
069                //RSVP parameter
070                //1.0 - Uses the values "YES" and "NO"
071                //2.0 - Uses the values "TRUE" and "FALSE"
072                Boolean rsvp = property.getRsvp();
073                if (rsvp != null) {
074                        String value = null;
075                        switch (context.getVersion()) {
076                        case V1_0:
077                                value = rsvp ? "YES" : "NO";
078                                break;
079
080                        default:
081                                value = rsvp ? "TRUE" : "FALSE";
082                                break;
083                        }
084
085                        copy.put(ICalParameters.RSVP, value);
086                }
087
088                //ROLE and EXPECT parameters
089                //1.0 - Uses ROLE and EXPECT
090                //2.0 - Uses only ROLE
091                Role role = property.getRole();
092                ParticipationLevel level = property.getParticipationLevel();
093                switch (context.getVersion()) {
094                case V1_0:
095                        if (role != null) {
096                                copy.put("ROLE", role.getValue());
097                        }
098                        if (level != null) {
099                                copy.put("EXPECT", level.getValue(context.getVersion()));
100                        }
101                        break;
102
103                default:
104                        String value = null;
105                        if (role == Role.CHAIR) {
106                                value = role.getValue();
107                        } else if (level != null) {
108                                value = level.getValue(context.getVersion());
109                        } else if (role != null) {
110                                value = role.getValue();
111                        }
112
113                        if (value != null) {
114                                copy.put("ROLE", value);
115                        }
116                        break;
117                }
118
119                //PARTSTAT vs STATUS
120                //1.0 - Calls the parameter "STATUS"
121                //2.0 - Calls the parameter "PARTSTAT"
122                ParticipationStatus partStat = property.getParticipationStatus();
123                if (partStat != null) {
124                        String paramName;
125                        String paramValue;
126
127                        switch (context.getVersion()) {
128                        case V1_0:
129                                paramName = "STATUS";
130                                paramValue = (partStat == ParticipationStatus.NEEDS_ACTION) ? "NEEDS ACTION" : partStat.getValue();
131                                break;
132
133                        default:
134                                paramName = "PARTSTAT";
135                                paramValue = partStat.getValue();
136                                break;
137                        }
138
139                        copy.put(paramName, paramValue);
140                }
141
142                //CN parameter
143                String name = property.getCommonName();
144                if (name != null && context.getVersion() != ICalVersion.V1_0) {
145                        copy.put(ICalParameters.CN, name);
146                }
147
148                return copy;
149        }
150
151        @Override
152        protected Attendee _parseText(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
153                String uri = null, name = null, email = null;
154
155                Boolean rsvp = null;
156                String rsvpStr = parameters.first(ICalParameters.RSVP);
157
158                Role role = null;
159                String roleStr = parameters.first(ICalParameters.ROLE);
160
161                ParticipationLevel level = null;
162                ParticipationStatus status = null;
163
164                switch (context.getVersion()) {
165                case V1_0:
166                        if (rsvpStr != null) {
167                                if ("YES".equalsIgnoreCase(rsvpStr)) {
168                                        rsvp = Boolean.TRUE;
169                                } else if ("NO".equalsIgnoreCase(rsvpStr)) {
170                                        rsvp = Boolean.FALSE;
171                                }
172                        }
173
174                        if (roleStr != null) {
175                                role = Role.get(roleStr);
176                        }
177
178                        String expect = parameters.first("EXPECT");
179                        if (expect != null) {
180                                parameters.remove("EXPECT", expect);
181                                level = ParticipationLevel.get(expect);
182                        }
183
184                        String statusStr = parameters.first("STATUS");
185                        if (statusStr != null) {
186                                parameters.remove("STATUS", statusStr);
187                                status = ParticipationStatus.get(statusStr);
188                        }
189
190                        int bracketStart = value.lastIndexOf('<');
191                        int bracketEnd = value.lastIndexOf('>');
192                        if (bracketStart >= 0 && bracketEnd >= 0 && bracketStart < bracketEnd) {
193                                name = value.substring(0, bracketStart).trim();
194                                email = value.substring(bracketStart + 1, bracketEnd).trim();
195                        } else if (dataType == ICalDataType.URL) {
196                                uri = value;
197                        } else {
198                                email = value;
199                        }
200
201                        break;
202
203                default:
204                        if (rsvpStr != null) {
205                                if ("TRUE".equalsIgnoreCase(rsvpStr)) {
206                                        rsvp = Boolean.TRUE;
207                                } else if ("FALSE".equalsIgnoreCase(rsvpStr)) {
208                                        rsvp = Boolean.FALSE;
209                                }
210                        }
211
212                        if (roleStr != null) {
213                                if (roleStr.equalsIgnoreCase(Role.CHAIR.getValue())) {
214                                        role = Role.CHAIR;
215                                } else {
216                                        ParticipationLevel l = ParticipationLevel.find(roleStr);
217                                        if (l == null) {
218                                                role = Role.get(roleStr);
219                                        } else {
220                                                level = l;
221                                        }
222                                }
223                        }
224
225                        String partStat = parameters.first("PARTSTAT");
226                        if (partStat != null) {
227                                parameters.remove("PARTSTAT", partStat);
228                                status = ParticipationStatus.get(partStat);
229                        }
230
231                        name = parameters.first(ICalParameters.CN);
232                        if (name != null) {
233                                parameters.remove(ICalParameters.CN, name);
234                        }
235
236                        int colon = value.indexOf(':');
237                        if (colon == 6) {
238                                String scheme = value.substring(0, colon);
239                                if (scheme.equalsIgnoreCase("mailto")) {
240                                        email = value.substring(colon + 1);
241                                } else {
242                                        uri = value;
243                                }
244                        } else {
245                                uri = value;
246                        }
247
248                        break;
249                }
250
251                if (rsvp != null) {
252                        parameters.removeAll(ICalParameters.RSVP);
253                }
254                if (roleStr != null) {
255                        parameters.remove(ICalParameters.ROLE, roleStr);
256                }
257
258                Attendee attendee = new Attendee(name, email);
259                attendee.setParticipationStatus(status);
260                attendee.setParticipationLevel(level);
261                attendee.setRole(role);
262                attendee.setRsvp(rsvp);
263                attendee.setUri(uri);
264
265                return attendee;
266        }
267
268        @Override
269        protected String _writeText(Attendee property, WriteContext context) {
270                String uri = property.getUri();
271                if (uri != null) {
272                        return uri;
273                }
274
275                String name = property.getCommonName();
276                String email = property.getEmail();
277                switch (context.getVersion()) {
278                case V1_0:
279                        if (name != null && email != null) {
280                                return escape(name + " <" + email + ">");
281                        }
282                        if (email != null) {
283                                return escape(email);
284                        }
285
286                        break;
287
288                default:
289                        if (email != null) {
290                                return "mailto:" + email;
291                        }
292                        break;
293                }
294
295                return "";
296        }
