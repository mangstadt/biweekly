054/**
055 * Chainer class for writing xCal (XML-encoded iCalendar objects).
056 * @see Biweekly#writeXml(Collection)
057 * @see Biweekly#writeXml(ICalendar...)
058 * @author Michael Angstadt
059 */
060public class ChainingXmlWriter extends ChainingWriter<ChainingXmlWriter> {
061        private final XCalOutputProperties outputProperties = new XCalOutputProperties();
062        private final Map<String, ICalDataType> parameterDataTypes = new HashMap<String, ICalDataType>(0);
063
064        /**
065         * @param icals the iCValendar objects to write
066         */
067        public ChainingXmlWriter(Collection<ICalendar> icals) {
068                super(icals);
069        }
070
071        /**
072         * Sets the number of indent spaces to use for pretty-printing. If not set,
073         * then the XML will not be pretty-printed.
074         * @param indent the number of spaces in the indent string or "null" not to
075         * pretty-print (disabled by default)
076         * @return this
077         */
078        public ChainingXmlWriter indent(Integer indent) {
079                outputProperties.setIndent(indent);
080                return this;
081        }
082
083        /**
084         * Sets the XML version to use. Note that many JDKs only support 1.0
085         * natively. For XML 1.1 support, add a JAXP library like <a href=
086         * "http://search.maven.org/#search%7Cgav%7C1%7Cg%3A%22xalan%22%20AND%20a%3A%22xalan%22"
087         * >xalan</a> to your project.
088         * @param xmlVersion the XML version (defaults to "1.0")
089         * @return this
090         */
091        public ChainingXmlWriter xmlVersion(String xmlVersion) {
092                outputProperties.setXmlVersion(xmlVersion);
093                return this;
094        }
095
096        /**
097         * Assigns an output property to the JAXP transformer (see
098         * {@link Transformer#setOutputProperty}).
099         * @param name the property name
100         * @param value the property value
101         * @return this
102         */
103        public ChainingXmlWriter outputProperty(String name, String value) {
104                outputProperties.put(name, value);
105                return this;
106        }
107
108        /**
109         * Assigns all of the given output properties to the JAXP transformer (see
110         * {@link Transformer#setOutputProperty}).
111         * @param outputProperties the properties
112         * @return this
113         */
114        public ChainingXmlWriter outputProperties(Map<String, String> outputProperties) {
115                this.outputProperties.putAll(outputProperties);
116                return this;
117        }
118
119        @Override
120        public ChainingXmlWriter tz(TimeZone defaultTimeZone) {
121                return super.tz(defaultTimeZone);
122        }
123
124        @Override
125        public ChainingXmlWriter register(ICalPropertyScribe<? extends ICalProperty> scribe) {
126                return super.register(scribe);
127        }
128
129        @Override
130        public ChainingXmlWriter register(ICalComponentScribe<? extends ICalComponent> scribe) {
131                return super.register(scribe);
132        }
133
134        /**
135         * Registers the data type of a non-standard parameter. Non-standard
136         * parameters use the "unknown" data type by default.
137         * @param parameterName the parameter name (e.g. "x-foo")
138         * @param dataType the data type
139         * @return this
140         */
141        public ChainingXmlWriter register(String parameterName, ICalDataType dataType) {
142                parameterDataTypes.put(parameterName, dataType);
143                return this;
144        }
145
146        /**
147         * Writes the iCalendar objects to a string.
148         * @return the XML document
149         */
150        public String go() {
151                return createXCalDocument().write(outputProperties);
152        }
153
154        /**
155         * Writes the iCalendar objects to an output stream.
156         * @param out the output stream to write to
157         * @throws TransformerException if there's a problem writing to the output
158         * stream
159         */
160        public void go(OutputStream out) throws TransformerException {
161                createXCalDocument().write(out, outputProperties);
162        }
163
164        /**
165         * Writes the iCalendar objects to a file.
166         * @param file the file to write to
167         * @throws IOException if the file can't be opened
168         * @throws TransformerException if there's a problem writing to the file
169         */
170        public void go(File file) throws IOException, TransformerException {
171                createXCalDocument().write(file, outputProperties);
172        }
173
174        /**
175         * Writes the iCalendar objects to a writer.
176         * @param writer the writer to write to
177         * @throws TransformerException if there's a problem writing to the writer
178         */
179        public void go(Writer writer) throws TransformerException {
180                createXCalDocument().write(writer, outputProperties);
181        }
182
183        /**
184         * Generates an XML document object model (DOM) containing the iCalendar
185         * objects.
186         * @return the DOM
187         */
188        public Document dom() {
189                return createXCalDocument().getDocument();
190        }
191
192        private XCalDocument createXCalDocument() {
193                XCalDocument document = new XCalDocument();
194
195                XCalDocumentStreamWriter writer = document.writer();
196                if (defaultTimeZone != null) {
197                        TimezoneInfo tzinfo = new TimezoneInfo();
198                        tzinfo.setDefaultTimeZone(defaultTimeZone);
199                        writer.setTimezoneInfo(tzinfo);
200                }
201                for (Map.Entry<String, ICalDataType> entry : parameterDataTypes.entrySet()) {
202                        String parameterName = entry.getKey();
203                        ICalDataType dataType = entry.getValue();
204                        writer.registerParameterDataType(parameterName, dataType);
205                }
206                if (index != null) {
207                        writer.setScribeIndex(index);
208                }
209
210                for (ICalendar ical : icals) {
211                        writer.write(ical);
212                }
213
214                return document;
215        }
216}
