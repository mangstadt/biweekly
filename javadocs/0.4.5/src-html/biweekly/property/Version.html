037/**
038 * <p>
039 * Defines the min/max iCalendar versions a consumer must support in order to
040 * successfully parse the iCalendar object.
041 * </p>
042 * <p>
043 * <b>Code sample:</b>
044 * 
045 * <pre class="brush:java">
046 * ICalendar ical = new ICalendar();
047 * 
048 * //all ICalendar objects are given a VERSION property on creation
049 * ical.getVersion(); //&quot;2.0&quot;
050 * 
051 * //get the default iCal version
052 * Version version = Version.v2_0();
053 * ical.setVersion(version);
054 * </pre>
055 * 
056 * </p>
057 * @author Michael Angstadt
058 * @see <a href="http://tools.ietf.org/html/rfc5545#page-79">RFC 5545
059 * p.79-80</a>
060 * @see <a href="http://tools.ietf.org/html/rfc2445#page-76">RFC 2445 p.76-7</a>
061 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.24</a>
062 */
063public class Version extends ICalProperty {
064        public static final VersionNumber VCAL = new VersionNumber(ICalVersion.V1_0.getVersion());
065        public static final VersionNumber ICAL = new VersionNumber(ICalVersion.V2_0.getVersion());
066
067        private VersionNumber minVersion, maxVersion;
068
069        /**
070         * Creates a new version property.
071         * @param version the version that a consumer must support in order to
072         * successfully parse the iCalendar object
073         */
074        public Version(ICalVersion version) {
075                this((version == null) ? null : version.getVersion());
076        }
077
078        /**
079         * Creates a new version property.
080         * @param version the version that a consumer must support in order to
081         * successfully parse the iCalendar object
082         * @throws IllegalArgumentException if the version string is invalid
083         */
084        public Version(String version) {
085                this(null, version);
086        }
087
088        /**
089         * Creates a new version property.
090         * @param minVersion the minimum version that a consumer must support in
091         * order to successfully parse the iCalendar object
092         * @param maxVersion the maximum version that a consumer must support in
093         * order to successfully parse the iCalendar object
094         * @throws IllegalArgumentException if one of the versions strings are
095         * invalid
096         */
097        public Version(String minVersion, String maxVersion) {
098                this((minVersion == null) ? null : new VersionNumber(minVersion), (maxVersion == null) ? null : new VersionNumber(maxVersion));
099        }
100
101        private Version(VersionNumber minVersion, VersionNumber maxVersion) {
102                this.minVersion = minVersion;
103                this.maxVersion = maxVersion;
104        }
105
106        /**
107         * Copy constructor.
108         * @param original the property to make a copy of
109         */
110        public Version(Version original) {
111                super(original);
112                minVersion = original.minVersion;
113                maxVersion = original.maxVersion;
114        }
115
116        /**
117         * Creates a version property that is set to the older vCalendar version
118         * (1.0).
119         * @return the property instance
120         */
121        public static Version v1_0() {
122                return new Version(ICalVersion.V1_0);
123        }
124
125        /**
126         * Creates a version property that is set to the latest iCalendar version
127         * (2.0).
128         * @return the property instance
129         */
130        public static Version v2_0() {
131                return new Version(ICalVersion.V2_0);
132        }
133
134        /**
135         * Determines if this property is set to the older vCalendar version.
136         * @return true if the version is "1.0", false if not
137         */
138        public boolean isV1_0() {
139                return VCAL.equals(maxVersion);
140        }
141
142        /**
143         * Determines if this property is set to the latest iCalendar version.
144         * @return true if the version is "2.0", false if not
145         */
146        public boolean isV2_0() {
147                return ICAL.equals(maxVersion);
148        }
149
150        /**
151         * Gets the minimum version that a consumer must support in order to
152         * successfully parse the iCalendar object.
153         * @return the minimum version or null if not set
154         */
155        public VersionNumber getMinVersion() {
156                return minVersion;
157        }
158
159        /**
160         * Sets the minimum version that a consumer must support in order to
161         * successfully parse the iCalendar object.
162         * @param minVersion the minimum version or null to remove
163         */
164        public void setMinVersion(VersionNumber minVersion) {
165                this.minVersion = minVersion;
166        }
167
168        /**
169         * Gets the maximum version that a consumer must support in order to
170         * successfully parse the iCalendar object.
171         * @return the maximum version or null if not set
172         */
173        public VersionNumber getMaxVersion() {
174                return maxVersion;
175        }
176
177        /**
178         * Sets the maximum version that a consumer must support in order to
179         * successfully parse the iCalendar object.
180         * @param maxVersion the maximum version (this field is <b>required</b>)
181         */
182        public void setMaxVersion(VersionNumber maxVersion) {
183                this.maxVersion = maxVersion;
184        }
185
186        /**
187         * Converts this property's value to an {@link ICalVersion} enum.
188         * @return the {@link ICalVersion} enum or null if it couldn't be converted
189         */
190        public ICalVersion toICalVersion() {
191                if (minVersion == null && maxVersion != null) {
192                        return ICalVersion.get(maxVersion.toString());
193                }
194                return null;
195        }
196
197        @Override
198        protected void validate(List<ICalComponent> components, ICalVersion version, List<Warning> warnings) {
199                if (maxVersion == null) {
200                        warnings.add(Warning.validate(35));
201                }
202        }
203
204        @Override
205        protected Map<String, Object> toStringValues() {
206                Map<String, Object> values = new LinkedHashMap<String, Object>();
207                values.put("minVersion", minVersion);
208                values.put("maxVersion", maxVersion);
209                return values;
210        }
211
212        @Override
213        public Version copy() {
214                return new Version(this);
215        }
216
217        @Override
218        public int hashCode() {
219                final int prime = 31;
220                int result = super.hashCode();
221                result = prime * result + ((maxVersion == null) ? 0 : maxVersion.hashCode());
222                result = prime * result + ((minVersion == null) ? 0 : minVersion.hashCode());
223                return result;
224        }
225
226        @Override
227        public boolean equals(Object obj) {
228                if (this == obj) return true;
229                if (!super.equals(obj)) return false;
230                Version other = (Version) obj;
231                if (maxVersion == null) {
232                        if (other.maxVersion != null) return false;
233                } else if (!maxVersion.equals(other.maxVersion)) return false;
234                if (minVersion == null) {
235                        if (other.minVersion != null) return false;
236                } else if (!minVersion.equals(other.minVersion)) return false;
237                return true;
238        }
239}
