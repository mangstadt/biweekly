053/**
054 * <p>
055 * Defines a collection of time ranges that describe when a person is available
056 * and unavailable.
057 * </p>
058 * <p>
059 * <b>Examples:</b>
060 * </p>
061 * 
062 * <pre class="brush:java">
063 * VFreeBusy freebusy = new VFreeBusy();
064 * 
065 * Date start = ...
066 * Date end = ...
067 * freebusy.addFreeBusy(FreeBusyType.FREE, start, end);
068 * 
069 * start = ...
070 * Duration duration = Duration.builder().hours(2).build();
071 * freebusy.addFreeBusy(FreeBusyType.BUSY, start, duration);
072 * </pre>
073 * @author Michael Angstadt
074 * @see <a href="http://tools.ietf.org/html/rfc5545#page-59">RFC 5545
075 * p.59-62</a>
076 * @see <a href="http://tools.ietf.org/html/rfc2445#page-58">RFC 2445
077 * p.58-60</a>
078 */

080 * Note: References to the vCal 1.0 spec are omitted from the property

538        /**

540         * example, "free" between 1pm-3pm). This method will look for an existing
541         * property that has the given {@link FreeBusyType} and add the time period
542         * to it, or create a new property is one cannot be found.
543         * @param type the availability type (e.g. "free" or "busy")
544         * @param start the start date-time
545         * @param end the end date-time
546         * @return the property that was created/modified

553                FreeBusy found = findByFreeBusyType(type);
554                found.getValues().add(new Period(start, end));
555                return found;

567         * @see <a href="http://tools.ietf.org/html/rfc5545#page-100">RFC 5545
568         * p.100-1</a>
569         * @see <a href="http://tools.ietf.org/html/rfc2445#page-95">RFC 2445
570         * p.95-6</a>
571         */
572        public FreeBusy addFreeBusy(FreeBusyType type, Date start, Duration duration) {
573                FreeBusy found = findByFreeBusyType(type);
574                found.getValues().add(new Period(start, duration));
575                return found;
576        }
577
578        private FreeBusy findByFreeBusyType(FreeBusyType type) {
579                for (FreeBusy freeBusy : getFreeBusy()) {
580                        if (freeBusy.getType() == type) {
581                                return freeBusy;
582                        }
583                }
584
585                FreeBusy freeBusy = new FreeBusy();
586                freeBusy.setType(type);
587                addFreeBusy(freeBusy);
588                return freeBusy;
589        }

619        @SuppressWarnings("unchecked")
620        @Override
621        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
622                if (version == ICalVersion.V1_0) {
623                        warnings.add(new ValidationWarning(48, version));
624                }
625
626                checkRequiredCardinality(warnings, Uid.class, DateTimeStamp.class);
627                checkOptionalCardinality(warnings, Contact.class, DateStart.class, DateEnd.class, Organizer.class, Url.class);
628
629                ICalDate dateStart = getValue(getDateStart());
630                ICalDate dateEnd = getValue(getDateEnd());
631
632                //DTSTART is required if DTEND exists
633                if (dateEnd != null && dateStart == null) {
634                        warnings.add(new ValidationWarning(15));
635                }
636
637                //DTSTART and DTEND must contain a time component
638                if (dateStart != null && !dateStart.hasTime()) {
639                        warnings.add(new ValidationWarning(20, DateStart.class.getSimpleName()));
640                }
641                if (dateEnd != null && !dateEnd.hasTime()) {
642                        warnings.add(new ValidationWarning(20, DateEnd.class.getSimpleName()));
643                }
644
645                //DTSTART must come before DTEND
646                if (dateStart != null && dateEnd != null && dateStart.compareTo(dateEnd) >= 0) {
647                        warnings.add(new ValidationWarning(16));
648                }
649        }
