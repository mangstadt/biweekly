1221        @SuppressWarnings("unchecked")
1222        @Override
1223        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
1224                if (version == ICalVersion.V1_0) {
1225                        warnings.add(new ValidationWarning(48, version));
1226                }
1227
1228                checkRequiredCardinality(warnings, Uid.class, DateTimeStamp.class);
1229                checkOptionalCardinality(warnings, Classification.class, Created.class, DateStart.class, LastModified.class, Organizer.class, RecurrenceId.class, Sequence.class, Status.class, Summary.class, Url.class, Color.class);
1230                checkStatus(warnings, Status.draft(), Status.final_(), Status.cancelled());
1231
1232                //DTSTART and RECURRENCE-ID must have the same data type
1233                ICalDate recurrenceId = getValue(getRecurrenceId());
1234                ICalDate dateStart = getValue(getDateStart());
1235                if (recurrenceId != null && dateStart != null && dateStart.hasTime() != recurrenceId.hasTime()) {
1236                        warnings.add(new ValidationWarning(19));
1237                }
1238
1239                //BYHOUR, BYMINUTE, and BYSECOND cannot be specified in RRULE if DTSTART's data type is "date"
1240                //RFC 5545 p. 167
1241                Recurrence rrule = getValue(getRecurrenceRule());
1242                if (dateStart != null && rrule != null) {
1243                        if (!dateStart.hasTime() && (!rrule.getByHour().isEmpty() || !rrule.getByMinute().isEmpty() || !rrule.getBySecond().isEmpty())) {
1244                                warnings.add(new ValidationWarning(5));
1245                        }
1246                }
1247
1248                //there *should* be only 1 instance of RRULE
1249                //RFC 5545 p. 167
1250                if (getProperties(RecurrenceRule.class).size() > 1) {
1251                        warnings.add(new ValidationWarning(6));
1252                }
1253        }
