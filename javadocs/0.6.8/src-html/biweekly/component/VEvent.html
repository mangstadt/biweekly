078/**
079 * <p>
080 * Defines a scheduled activity, such as a two hour meeting.
081 * </p>
082 * <p>
083 * <b>Examples:</b>
084 * </p>
085 * 
086 * <pre class="brush:java">
087 * VEvent event = new VEvent();
088 * Date start = ...
089 * event.setDateStart(start);
090 * Date end = ...
091 * event.setDateEnd(end);
092 * event.setSummary("Team Meeting");
093 * event.setLocation("Room 21C");
094 * event.setCreated(new Date());
095 * event.setRecurrenceRule(new Recurrence.Builder(Frequency.WEEKLY).build());
096 * </pre>
097 * @author Michael Angstadt
098 * @see <a href="http://tools.ietf.org/html/rfc5545#page-52">RFC 5545 p.52-5</a>

1648        @SuppressWarnings("unchecked")
1649        @Override
1650        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
1651                if (version != ICalVersion.V1_0) {
1652                        checkRequiredCardinality(warnings, Uid.class, DateTimeStamp.class);
1653                        checkOptionalCardinality(warnings, Classification.class, Created.class, Description.class, Geo.class, LastModified.class, Location.class, Organizer.class, Priority.class, Status.class, Summary.class, Transparency.class, Url.class, RecurrenceId.class);
1654                }
1655
1656                checkOptionalCardinality(warnings, Color.class);
1657
1658                Status[] validStatuses = getValidStatuses(version);
1659                checkStatus(warnings, validStatuses);
1660
1661                ICalDate dateStart = getValue(getDateStart());
1662                ICalDate dateEnd = getValue(getDateEnd());
1663
1664                //DTSTART is always required, unless there is a METHOD property at the iCal root
1665                ICalComponent ical = components.get(0);
1666                if (version != ICalVersion.V1_0 && dateStart == null && ical.getProperty(Method.class) == null) {
1667                        warnings.add(new ValidationWarning(14));
1668                }
1669
1670                //DTSTART is required if DTEND exists
1671                if (dateEnd != null && dateStart == null) {
1672                        warnings.add(new ValidationWarning(15));
1673                }
1674
1675                if (dateStart != null && dateEnd != null) {
1676                        //DTSTART must come before DTEND
1677                        if (dateStart.compareTo(dateEnd) > 0) {
1678                                warnings.add(new ValidationWarning(16));
1679                        }
1680
1681                        //DTSTART and DTEND must have the same data type
1682                        if (dateStart.hasTime() != dateEnd.hasTime()) {
1683                                warnings.add(new ValidationWarning(17));
1684                        }
1685                }
1686
1687                //DTEND and DURATION cannot both exist
1688                if (dateEnd != null && getDuration() != null) {
1689                        warnings.add(new ValidationWarning(18));
1690                }
1691
1692                //DTSTART and RECURRENCE-ID must have the same data type
1693                ICalDate recurrenceId = getValue(getRecurrenceId());
1694                if (recurrenceId != null && dateStart != null && dateStart.hasTime() != recurrenceId.hasTime()) {
1695                        warnings.add(new ValidationWarning(19));
1696                }
1697
1698                //BYHOUR, BYMINUTE, and BYSECOND cannot be specified in RRULE if DTSTART's data type is "date"
1699                //RFC 5545 p. 167
1700                Recurrence rrule = getValue(getRecurrenceRule());
1701                if (dateStart != null && rrule != null) {
1702                        if (!dateStart.hasTime() && (!rrule.getByHour().isEmpty() || !rrule.getByMinute().isEmpty() || !rrule.getBySecond().isEmpty())) {
1703                                warnings.add(new ValidationWarning(5));
1704                        }
1705                }
1706
1707                //there *should* be only 1 instance of RRULE
1708                //RFC 5545 p. 167
1709                if (getProperties(RecurrenceRule.class).size() > 1) {
1710                        warnings.add(new ValidationWarning(6));
1711                }
1712        }
