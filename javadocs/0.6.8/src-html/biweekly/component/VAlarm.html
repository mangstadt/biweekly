047/**
048 * <p>
049 * Defines a reminder for an event or to-do task. This class contains static
050 * factory methods to aid in the construction of valid alarms.
051 * </p>
052 * <p>
053 * <b>Examples:</b>
054 * </p>
055 * 
056 * <pre class="brush:java">
057 * //audio alarm
058 * Trigger trigger = ...
059 * Attachment sound = ...
060 * VAlarm audio = VAlarm.audio(trigger, sound);
061 * 
062 * //display alarm
063 * Trigger trigger = ...
064 * String message = "Meeting at 1pm";
065 * VAlarm display = VAlarm.display(trigger, message);
066 * 
067 * //email alarm
068 * Trigger trigger = ...
069 * String subject = "Reminder: Meeting at 1pm";
070 * String body = "Team,\n\nThe team meeting scheduled for 1pm is about to start.  Snacks will be served!\n\nThanks,\nJohn";
071 * List&lt;String&gt; to = Arrays.asList("janedoe@example.com", "bobsmith@example.com");
072 * VAlarm email = VAlarm.email(trigger, subject, body, to);
073 * </pre>
074 * @author Michael Angstadt
075 * @see <a href="http://tools.ietf.org/html/rfc5545#page-71">RFC 5545 p.71-6</a>
076 * @see <a href="http://tools.ietf.org/html/rfc2445#page-67">RFC 2445
077 * p.67-73</a>
078 */

162        /**
163         * Creates an email alarm.
164         * @param trigger the trigger
165         * @param subject the email subject
166         * @param body the email body
167         * @param recipients the email address(es) to send the alert to
168         * @return the alarm
169         */
170        public static VAlarm email(Trigger trigger, String subject, String body, List<String> recipients) {
171                VAlarm alarm = new VAlarm(Action.email(), trigger);
172                alarm.setSummary(subject);
173                alarm.setDescription(body);
174                for (String recipient : recipients) {
175                        alarm.addAttendee(new Attendee(null, recipient));
176                }
177                return alarm;
178        }

518        @SuppressWarnings("unchecked")
519        @Override
520        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
521                checkRequiredCardinality(warnings, Action.class, Trigger.class);
522
523                validateAction(warnings);
524                validateTrigger(components, warnings);
525        }
526
527        @SuppressWarnings("unchecked")
528        private void validateAction(List<ValidationWarning> warnings) {
529                Action action = getAction();
530                if (action == null) {
531                        return;
532                }
533
534                //AUDIO alarms should not have more than 1 attachment
535                if (action.isAudio()) {
536                        if (getAttachments().size() > 1) {
537                                warnings.add(new ValidationWarning(7));
538                        }
539                }
540
541                //DESCRIPTION is required for DISPLAY alarms 
542                if (action.isDisplay()) {
543                        checkRequiredCardinality(warnings, Description.class);
544                }
545
546                if (action.isEmail()) {
547                        //SUMMARY and DESCRIPTION is required for EMAIL alarms
548                        checkRequiredCardinality(warnings, Summary.class, Description.class);
549
550                        //EMAIL alarms must have at least 1 ATTENDEE
551                        if (getAttendees().isEmpty()) {
552                                warnings.add(new ValidationWarning(8));
553                        }
554                } else {
555                        //only EMAIL alarms can have ATTENDEEs
556                        if (!getAttendees().isEmpty()) {
557                                warnings.add(new ValidationWarning(9));
558                        }
559                }
560
561                if (action.isProcedure()) {
562                        checkRequiredCardinality(warnings, Description.class);
563                }
564        }
565
566        private void validateTrigger(List<ICalComponent> components, List<ValidationWarning> warnings) {
567                Trigger trigger = getTrigger();
568                if (trigger == null) {
569                        return;
570                }
571
572                Related related = trigger.getRelated();
573                if (related == null) {
574                        return;
575                }
576
577                ICalComponent parent = components.get(components.size() - 1);
578
579                //if the TRIGGER is relative to DTSTART, confirm that DTSTART exists
580                if (related == Related.START && parent.getProperty(DateStart.class) == null) {
581                        warnings.add(new ValidationWarning(11));
582                }
583
584                //if the TRIGGER is relative to DTEND, confirm that DTEND (or DUE) exists
585                if (related == Related.END) {
586                        boolean noEndDate = false;
587
588                        if (parent instanceof VEvent) {
589                                noEndDate = (parent.getProperty(DateEnd.class) == null && (parent.getProperty(DateStart.class) == null || parent.getProperty(DurationProperty.class) == null));
590                        } else if (parent instanceof VTodo) {
591                                noEndDate = (parent.getProperty(DateDue.class) == null && (parent.getProperty(DateStart.class) == null || parent.getProperty(DurationProperty.class) == null));
592                        }
593
594                        if (noEndDate) {
595                                warnings.add(new ValidationWarning(12));
596                        }
597                }
598        }
