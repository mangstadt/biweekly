054/**
055 * <p>
056 * Contains static chaining factory methods for reading/writing iCalendar
057 * objects.
058 * </p>
059 * 
060 * <p>
061 * <b>Writing an iCalendar object</b>
062 * </p>
063 * 
064 * <pre class="brush:java">
065 * ICalendar ical = new ICalendar();
066 * 
067 * //string
068 * String icalString = Biweekly.write(ical).go();
069 * 
070 * //file
071 * File file = new File("meeting.ics");
072 * Biweekly.write(ical).go(file);
073 * 
074 * //output stream
075 * OutputStream out = ...
076 * Biweekly.write(ical).go(out);
077 * out.close();
078 * 
079 * //writer (should be configured to use UTF-8 encoding)
080 * Writer writer = ...
081 * Biweekly.write(ical).go(writer);
082 * writer.close();
083 * </pre>
084 * 
085 * <p>
086 * <b>Writing multiple iCalendar objects</b>
087 * </p>
088 * 
089 * <pre class="brush:java">
090 * ICalendar ical1 = new ICalendar();
091 * ICalendar ical2 = new ICalendar();
092 * 
093 * String icalString = Biweekly.write(ical1, ical2).go();
094 * </pre>
095 * 
096 * <p>
097 * <b>Writing an XML-encoded iCalendar object (xCal)</b>
098 * </p>
099 * 
100 * <pre class="brush:java">
101 * //Call writeXml() instead of write()
102 * ICalendar ical = new ICalendar();
103 * String xml = Biweekly.writeXml(ical).indent(2).go();
104 * </pre>
105 * 
106 * <p>
107 * <b>Writing a JSON-encoded iCalendar object (jCal)</b>
108 * </p>
109 * 
110 * <pre class="brush:java">
111 * //Call writeJson() instead of write()
112 * ICalendar ical = new ICalendar();
113 * String json = Biweekly.writeJson(ical).go();
114 * </pre>
115 * 
116 * <p>
117 * <b>Reading an iCalendar object</b>
118 * </p>
119 * 
120 * <pre class="brush:java">
121 * ICalendar ical;
122 * 
123 * //string
124 * String icalStr = ...
125 * ical = Biweekly.parse(icalStr).first();
126 * 
127 * //file
128 * File file = new File("meeting.ics");
129 * ical = Biweekly.parse(file).first();
130 * 
131 * //input stream
132 * InputStream in = ...
133 * ical = Biweekly.parse(in).first();
134 * in.close();  
135 * 
136 * //reader (should be configured to read UTF-8)
137 * Reader reader = ...
138 * ical = Biweekly.parse(reader).first();
139 * reader.close();
140 * </pre>
141 * 
142 * <p>
143 * <b>Reading multiple iCalendar objects</b>
144 * </p>
145 * 
146 * <pre class="brush:java">
147 * String icalStr = ...
148 * List&lt;ICalendar&gt; icals = Biweekly.parse(icalStr).all();
149 * </pre>
150 * 
151 * <p>
152 * <b>Reading an XML-encoded iCalendar object (xCal)</b>
153 * </p>
154 * 
155 * <pre class="brush:java">
156 * //Call parseXml() instead of parse()
157 * String xml = ...
158 * ICalendar ical = Biweekly.parseXml(xml).first();
159 * </pre>
160 * 
161 * <p>
162 * <b>Reading a JSON-encoded iCalendar object (Cal)</b>
163 * </p>
164 * 
165 * <pre class="brush:java">
166 * //Call parseJson() instead of parse()
167 * String json = ...
168 * ICalendar ical = Biweekly.parseJson(json).first();
169 * </pre>
170 * 
171 * <p>
172 * <b>Retrieving parser warnings</b>
173 * </p>
174 * 
175 * <pre class="brush:java">
176 * String icalStr = ...
177 * List&lt;List&lt;String&gt;&gt; warnings = new ArrayList&lt;List&lt;String&gt;&gt;();
178 * List&lt;ICalendar&gt; icals = Biweekly.parse(icalStr).warnings(warnings).all();
179 * int i = 0;
180 * for (List&lt;String&gt; icalWarnings : warnings) {
181 *   System.out.println("iCal #" + (i++) + " warnings:");
182 *   for (String warning : icalWarnings) {
183 *     System.out.println(warning);
184 *   }
185 * }
186 * </pre>
187 * 
188 * <p>
189 * The methods in this class make use of the following classes. These classes
190 * can be used if greater control over the read/write operation is required:
191 * </p>
192 * 
193 * <table class="simpleTable">
194 * <caption>Classes used by this class</caption>
195 * <tr>
196 * <th></th>
197 * <th>Classes</th>
198 * <th>Supports<br>
199 * streaming?</th>
200 * </tr>
201 * <tr>
202 * <th>Text</th>
203 * <td>{@link ICalReader} / {@link ICalWriter}</td>
204 * <td>yes</td>
205 * </tr>
206 * <tr>
207 * <th>XML</th>
208 * <td>{@link XCalDocument}</td>
209 * <td>no</td>
210 * </tr>
211 * <tr>
212 * <th>JSON</th>
213 * <td>{@link JCalReader} / {@link JCalWriter}</td>
214 * <td>yes</td>
215 * </tr>
216 * </table>
217 * @author Michael Angstadt
218 */

236         * The project webpage.
237         */
238        public static final String URL;

240        static {
241                InputStream in = null;
242                try {
243                        in = Biweekly.class.getResourceAsStream("biweekly.properties");
244                        Properties props = new Properties();
245                        props.load(in);
246
247                        VERSION = props.getProperty("version");
248                        GROUP_ID = props.getProperty("groupId");
249                        ARTIFACT_ID = props.getProperty("artifactId");
250                        URL = props.getProperty("url");
251                } catch (IOException e) {
252                        throw new RuntimeException(e);
253                } finally {
254                        IOUtils.closeQuietly(in);
255                }
256        }
