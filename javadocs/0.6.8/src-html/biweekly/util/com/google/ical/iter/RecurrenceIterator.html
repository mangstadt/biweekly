040package biweekly.util.com.google.ical.iter;
041
042import java.util.Iterator;
043
044import biweekly.util.com.google.ical.values.DateValue;
045
046/**
047 * Iterates over a series of dates in a recurrence rule in ascending order.
048 * @author mikesamuel+svn@gmail.com (Mike Samuel)
049 * @author Michael Angstadt
050 */
051public interface RecurrenceIterator extends Iterator<DateValue> {
052        /**
053         * Determines if there are more dates in the series.
054         * @return true if there are more dates, false if not
055         */
056        boolean hasNext();
057
058        /**
059         * Returns the next date in the series. If {@link #hasNext()} returns
060         * {@code false}, then this method's behavior is undefined.
061         * @return the next date (in UTC; will be strictly later than any date
062         * previously returned by this iterator)
063         */
064        DateValue next();
065
066        /**
067         * Skips all dates in the series that come before the given date, so that
068         * the next call to {@link #next} will return a date on or after the given
069         * date (assuming the recurrence includes such a date).
070         * @param newStartUtc the date to advance to (in UTC)
071         */
072        void advanceTo(DateValue newStartUtc);
073
074        /**
075         * Implementors of this interface are not expected to implement this method.
076         * @throws UnsupportedOperationException always
077         */
078        void remove();
079}
