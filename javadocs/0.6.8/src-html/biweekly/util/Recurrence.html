043/**
044 * <p>
045 * Represents a recurrence rule value.
046 * </p>
047 * <p>
048 * This class is immutable. Use the inner class {@link Builder} to construct a
049 * new instance.
050 * </p>
051 * <p>
052 * <b>Code sample:</b>
053 * </p>
054 * 
055 * <pre class="brush:java">
056 * Recurrence rrule = new Recurrence.Builder(Frequency.WEEKLY).interval(2).build();
057 * Recurrence copy = new Recurrence.Builder(rrule).interval(3).build();
058 * </pre>
059 * @author Michael Angstadt
060 * @see <a href="http://tools.ietf.org/html/rfc5545#page-38">RFC 5545
061 * p.38-45</a>
062 */
063public final class Recurrence {
064        private final Frequency frequency;
065        private final Integer interval;
066        private final Integer count;
067        private final ICalDate until;
068        private final List<Integer> bySecond;
069        private final List<Integer> byMinute;
070        private final List<Integer> byHour;
071        private final List<Integer> byMonthDay;
072        private final List<Integer> byYearDay;
073        private final List<Integer> byWeekNo;
074        private final List<Integer> byMonth;
075        private final List<Integer> bySetPos;
076        private final List<ByDay> byDay;
077        private final DayOfWeek workweekStarts;
078        private final Map<String, List<String>> xrules;

080        private Recurrence(Builder builder) {
081                frequency = builder.frequency;
082                interval = builder.interval;
083                count = builder.count;
084                until = builder.until;
085                bySecond = Collections.unmodifiableList(builder.bySecond);
086                byMinute = Collections.unmodifiableList(builder.byMinute);
087                byHour = Collections.unmodifiableList(builder.byHour);
088                byMonthDay = Collections.unmodifiableList(builder.byMonthDay);
089                byYearDay = Collections.unmodifiableList(builder.byYearDay);
090                byWeekNo = Collections.unmodifiableList(builder.byWeekNo);
091                byMonth = Collections.unmodifiableList(builder.byMonth);
092                bySetPos = Collections.unmodifiableList(builder.bySetPos);
093                byDay = Collections.unmodifiableList(builder.byDay);
094                workweekStarts = builder.workweekStarts;
095                xrules = Collections.unmodifiableMap(builder.xrules.getMap());
096        }

220         * Creates an iterator that computes the dates defined by this recurrence.

235         * Creates an iterator that computes the dates defined by this recurrence.
236         * @param startDate the date that the recurrence starts (typically, the
237         * value of the {@link DateStart} property)

250        @Override
251        public int hashCode() {
252                final int prime = 31;
253                int result = 1;
254                result = prime * result + byDay.hashCode();
255                result = prime * result + byHour.hashCode();
256                result = prime * result + byMinute.hashCode();
257                result = prime * result + byMonth.hashCode();
258                result = prime * result + byMonthDay.hashCode();
259                result = prime * result + bySecond.hashCode();
260                result = prime * result + bySetPos.hashCode();
261                result = prime * result + byWeekNo.hashCode();
262                result = prime * result + byYearDay.hashCode();
263                result = prime * result + ((count == null) ? 0 : count.hashCode());
264                result = prime * result + xrules.hashCode();
265                result = prime * result + ((frequency == null) ? 0 : frequency.hashCode());
266                result = prime * result + ((interval == null) ? 0 : interval.hashCode());
267                result = prime * result + ((until == null) ? 0 : until.hashCode());
268                result = prime * result + ((workweekStarts == null) ? 0 : workweekStarts.hashCode());
269                return result;
270        }

272        @Override
273        public boolean equals(Object obj) {
274                if (this == obj) return true;
275                if (obj == null) return false;
276                if (getClass() != obj.getClass()) return false;
277
278                Recurrence other = (Recurrence) obj;
279                if (!byDay.equals(other.byDay)) return false;
280                if (!byHour.equals(other.byHour)) return false;
281                if (!byMinute.equals(other.byMinute)) return false;
282                if (!byMonth.equals(other.byMonth)) return false;
283                if (!byMonthDay.equals(other.byMonthDay)) return false;
284                if (!bySecond.equals(other.bySecond)) return false;
285                if (!bySetPos.equals(other.bySetPos)) return false;
286                if (!byWeekNo.equals(other.byWeekNo)) return false;
287                if (!byYearDay.equals(other.byYearDay)) return false;
288                if (count == null) {
289                        if (other.count != null) return false;
290                } else if (!count.equals(other.count)) return false;
291                if (!xrules.equals(other.xrules)) return false;
292                if (frequency != other.frequency) return false;
293                if (interval == null) {
294                        if (other.interval != null) return false;
295                } else if (!interval.equals(other.interval)) return false;
296                if (until == null) {
297                        if (other.until != null) return false;
298                } else if (!until.equals(other.until)) return false;
299                if (workweekStarts != other.workweekStarts) return false;
300                return true;
301        }

342                /**
343                 * Constructs a new builder
344                 * @param recur the recurrence object to copy from
345                 */
346                public Builder(Recurrence recur) {
347                        frequency = recur.frequency;
348                        interval = recur.interval;
349                        count = recur.count;
350                        until = recur.until;
351                        bySecond = new ArrayList<Integer>(recur.bySecond);
352                        byMinute = new ArrayList<Integer>(recur.byMinute);
353                        byHour = new ArrayList<Integer>(recur.byHour);
354                        byDay = new ArrayList<ByDay>(recur.byDay);
355                        byMonthDay = new ArrayList<Integer>(recur.byMonthDay);
356                        byYearDay = new ArrayList<Integer>(recur.byYearDay);
357                        byWeekNo = new ArrayList<Integer>(recur.byWeekNo);
358                        byMonth = new ArrayList<Integer>(recur.byMonth);
359                        bySetPos = new ArrayList<Integer>(recur.bySetPos);
360                        workweekStarts = recur.workweekStarts;
361
362                        Map<String, List<String>> map = new HashMap<String, List<String>>(recur.xrules);
363                        xrules = new ListMultimap<String, String>(map);
364                }
