033/**
034 * <p>
035 * Records the original value of the {@link DateStart} property if a recurrence
036 * instance has been modified. It is used in conjunction with the {@link Uid}
037 * and {@link Sequence} properties to uniquely identify a recurrence instance.
038 * </p>
039 * <p>
040 * <b>Code sample:</b>
041 * </p>
042 * 
043 * <pre class="brush:java">
044 * VEvent event = new VEvent();
045 * 
046 * //date-time value
047 * Date datetime = ...
048 * RecurrenceId recurrenceId = new RecurrenceId(datetime);
049 * event.setRecurrenceId(recurrenceId);
050 * 
051 * //date value
052 * Date date = ...
053 * RecurrenceId recurrenceId = new RecurrenceId(date, false);
054 * event.setRecurrenceId(recurrenceId);
055 * </pre>
056 * @author Michael Angstadt
057 * @see <a href="http://tools.ietf.org/html/rfc5545#page-112">RFC 5545
058 * p.112-4</a>
059 * @see <a href="http://tools.ietf.org/html/rfc2445#page-107">RFC 2445
060 * p.107-9</a>
061 */
062public class RecurrenceId extends DateOrDateTimeProperty {
063        /**
064         * Creates a recurrence ID property.
065         * @param originalStartDate the original start date
066         */
067        public RecurrenceId(ICalDate originalStartDate) {
068                super(originalStartDate);
069        }
070
071        /**
072         * Creates a recurrence ID property.
073         * @param originalStartDate the original start date
074         */
075        public RecurrenceId(Date originalStartDate) {
076                super(originalStartDate);
077        }
078
079        /**
080         * Creates a recurrence ID property.
081         * @param originalStartDate the original start date
082         * @param hasTime true to include the time component of the date, false not
083         * to
084         */
085        public RecurrenceId(Date originalStartDate, boolean hasTime) {
086                super(originalStartDate, hasTime);
087        }
088
089        /**
090         * Copy constructor.
091         * @param original the property to make a copy of
092         */
093        public RecurrenceId(RecurrenceId original) {
094                super(original);
095        }
096
097        /**
098         * Gets the effective range of recurrence instances from the instance
099         * specified by this property.
100         * @return the range or null if not set
101         * @see <a href="http://tools.ietf.org/html/rfc5545#page-23">RFC 5545
102         * p.23-4</a>
103         */
104        public Range getRange() {
105                return parameters.getRange();
106        }
107
108        /**
109         * Sets the effective range of recurrence instances from the instance
110         * specified by this property.
111         * @param range the range or null to remove
112         * @see <a href="http://tools.ietf.org/html/rfc5545#page-23">RFC 5545
113         * p.23-4</a>
114         */
115        public void setRange(Range range) {
116                parameters.setRange(range);
117        }
118
119        @Override
120        public RecurrenceId copy() {
121                return new RecurrenceId(this);
122        }
123}
