037/**
038 * <p>
039 * Defines an file attachment (such as an image or document) that is associated
040 * with the component to which it belongs.
041 * </p>
042 * 
043 * <p>
044 * <b>Code sample:</b>
045 * </p>
046 * 
047 * <pre class="brush:java">
048 * VEvent event = new VEvent();
049 * 
050 * //from a byte array
051 * byte[] data = ...
052 * Attachment attach = new Attachment("image/png", data);
053 * event.addAttachment(attach);
054 * 
055 * //from a file 
056 * File file = new File("image.png");
057 * attach = new Attachment("image/png", file);
058 * event.addAttachment(attach);
059 * 
060 * //referencing a URL
061 * attach = new Attachment("image/png", "http://example.com/image.png");
062 * event.addAttachment(attach);
063 * </pre>
064 * @author Michael Angstadt
065 * @see <a href="http://tools.ietf.org/html/rfc5545#page-80">RFC 5545 p.80-1</a>
066 * @see <a href="http://tools.ietf.org/html/rfc2445#page-77">RFC 2445 p.77-8</a>
067 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.25</a>
068 */
069public class Attachment extends BinaryProperty {
070        private String contentId;
071
072        /**
073         * Creates a new attachment.
074         * @param formatType the content-type of the data (e.g. "image/png")
075         * @param file the file to attach
076         * @throws IOException if there's a problem reading from the file
077         */
078        public Attachment(String formatType, File file) throws IOException {
079                super(file);
080                setFormatType(formatType);
081        }
082
083        /**
084         * Creates a new attachment.
085         * @param formatType the content-type of the data (e.g. "image/png")
086         * @param data the binary data
087         */
088        public Attachment(String formatType, byte[] data) {
089                super(data);
090                setFormatType(formatType);
091        }
092
093        /**
094         * Creates a new attachment.
095         * @param formatType the content-type of the data (e.g. "image/png")
096         * @param uri a URL pointing to the resource (e.g.
097         * "http://example.com/image.png")
098         */
099        public Attachment(String formatType, String uri) {
100                super(uri);
101                setFormatType(formatType);
102        }
103
104        /**
105         * Copy constructor.
106         * @param original the property to make a copy of
107         */
108        public Attachment(Attachment original) {
109                super(original);
110                contentId = original.contentId;
111        }
112
113        @Override
114        public void setData(byte[] data) {
115                super.setData(data);
116                contentId = null;
117        }
118
119        @Override
120        public void setUri(String uri) {
121                super.setUri(uri);
122                contentId = null;
123        }
124
125        /**
126         * Sets the content ID.
127         * @return the content ID or null if not set
128         */
129        public String getContentId() {
130                return contentId;
131        }
132
133        /**
134         * Sets the content ID.
135         * @param contentId the content ID
136         */
137        public void setContentId(String contentId) {
138                this.contentId = contentId;
139                uri = null;
140                data = null;
141        }
142
143        @Override
144        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
145                if (uri == null && data == null && contentId == null) {
146                        warnings.add(new ValidationWarning(26));
147                }
148        }
149
150        @Override
151        protected Map<String, Object> toStringValues() {
152                Map<String, Object> values = super.toStringValues();
153                values.put("contentId", contentId);
154                return values;
155        }
156
157        @Override
158        public Attachment copy() {
159                return new Attachment(this);
160        }
161
162        @Override
163        public int hashCode() {
164                final int prime = 31;
165                int result = super.hashCode();
166                result = prime * result + ((contentId == null) ? 0 : contentId.hashCode());
167                return result;
168        }
169
170        @Override
171        public boolean equals(Object obj) {
172                if (this == obj) return true;
173                if (!super.equals(obj)) return false;
174                Attachment other = (Attachment) obj;
175                if (contentId == null) {
176                        if (other.contentId != null) return false;
177                } else if (!contentId.equals(other.contentId)) return false;
178                return true;
179        }
180}
