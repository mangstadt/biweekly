028/**
029 * <p>
030 * Defines a color that clients may use when displaying the component data.
031 * Clients may use this color in any way they wish. For example, they can use it
032 * as a background color.
033 * </p>
034 * <p>
035 * Acceptable values are defined in <a
036 * href="https://www.w3.org/TR/2011/REC-css3-color-20110607/#svg-color">Section
037 * 4.3 of the CSS Color Module Level 3 Recommendation</a>. For example,
038 * "aliceblue", "green", "navy".
039 * </p>
040 * <p>
041 * <b>Code sample:</b>
042 * </p>
043 * 
044 * <pre class="brush:java">
045 * VEvent event = new VEvent();
046 * 
047 * Color color = new Color("mistyrose");
048 * event.setColor(color);
049 * </pre>
050 * @author Michael Angstadt
051 * @see <a
052 * href="http://tools.ietf.org/html/draft-ietf-calext-extensions-01#page-9">draft-ietf-calext-extensions-01
053 * p.9</a>
054 */
055public class Color extends TextProperty {
056        /**
057         * Creates a color property.
058         * @param color the color name (case insensitive). Acceptable values are
059         * defined in <a
060         * href="https://www.w3.org/TR/2011/REC-css3-color-20110607/#svg-color"
061         * >Section 4.3 of the CSS Color Module Level 3 Recommendation</a>. For
062         * example, "aliceblue", "green", "navy".
063         */
064        public Color(String color) {
065                super(color);
066        }
067
068        /**
069         * Copy constructor.
070         * @param original the property to make a copy of
071         */
072        public Color(Color original) {
073                super(original);
074        }
075
076        /**
077         * Gets the value of this property.
078         * @return the value (case insensitive). Acceptable values are defined in <a
079         * href="https://www.w3.org/TR/2011/REC-css3-color-20110607/#svg-color"
080         * >Section 4.3 of the CSS Color Module Level 3 Recommendation</a>. For
081         * example, "aliceblue", "green", "navy".
082         */
083        @Override
084        public String getValue() {
085                return super.getValue();
086        }
087
088        /**
089         * Sets the value of this property.
090         * @param value the value (case insensitive). Acceptable values are defined
091         * in <a
092         * href="https://www.w3.org/TR/2011/REC-css3-color-20110607/#svg-color"
093         * >Section 4.3 of the CSS Color Module Level 3 Recommendation</a>. For
094         * example, "aliceblue", "green", "navy".
095         */
096        @Override
097        public void setValue(String value) {
098                super.setValue(value);
099        }
100
101        @Override
102        public Color copy() {
103                return new Color(this);
104        }
105
106        @Override
107        protected int valueHashCode() {
108                return value.toLowerCase().hashCode();
109        }
110
111        @Override
112        protected boolean valueEquals(String otherValue) {
113                return value.equalsIgnoreCase(otherValue);
114        }
115}
