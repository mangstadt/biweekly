030/**
031 * <p>
032 * The meaning of this property varies depending on whether the iCalendar object
033 * has a {@link Method} property.
034 * </p>
035 * <ul>
036 * <li><b>Has a {@link Method} property:</b> Defines the creation date of the
037 * iCalendar object itself (<u>not</u> the creation date of the actual calendar
038 * data from the originating data store). Use the {@link Created} property to
039 * define the creation date of the actual calendar data.</li>
040 * <li><b>Does not have a {@link Method} property:</b> Defines the date that the
041 * calendar data was last updated (the {@link LastModified} property also holds
042 * this information).</li>
043 * </ul>
044 * <p>
045 * <b>Code sample:</b>
046 * </p>
047 * 
048 * <pre class="brush:java">
049 * VEvent event = new VEvent();
050 * 
051 * Date datetime = ... 
052 * DateTimeStamp dtstamp = new DateTimeStamp(datetime);
053 * event.setDateTimeStamp(dtstamp);
054 * </pre>
055 * @author Michael Angstadt
056 * @see <a href="http://tools.ietf.org/html/rfc5545#page-137">RFC 5545
057 * p.137-8</a>
058 * @see <a href="http://tools.ietf.org/html/rfc2445#page-130">RFC 2445
059 * p.130-1</a>
060 */
061public class DateTimeStamp extends DateTimeProperty {
062        /**
063         * Creates a date time stamp property.
064         * @param date the date
065         */
066        public DateTimeStamp(Date date) {
067                super(date);
068        }
069
070        /**
071         * Copy constructor.
072         * @param original the property to make a copy of
073         */
074        public DateTimeStamp(DateTimeStamp original) {
075                super(original);
076        }
077
078        @Override
079        public DateTimeStamp copy() {
080                return new DateTimeStamp(this);
081        }
082}
