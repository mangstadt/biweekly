030/**
031 * <p>
032 * Defines a list of physical resources that are needed for an event or to-do
033 * task (for example, a "projector" or "DVD player").
034 * </p>
035 * <p>
036 * <b>Code sample:</b>
037 * </p>
038 * 
039 * <pre class="brush:java">
040 * VEvent event = new VEvent();
041 * 
042 * Resources resources = new Resources("projector", "DVD player");
043 * event.addResources(resources);
044 * </pre>
045 * @author Michael Angstadt
046 * @see <a href="http://tools.ietf.org/html/rfc5545#page-91">RFC 5545 p.91</a>
047 * @see <a href="http://tools.ietf.org/html/rfc2445#page-87">RFC 2445 p.87-8</a>
048 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.34-5</a>
049 */
050public class Resources extends ListProperty<String> {
051        /**
052         * Creates a new resources property.
053         */
054        public Resources() {
055                super();
056        }
057
058        /**
059         * Creates a new resources property.
060         * @param values the values to initialize the property with (e.g. "easel",
061         * "projector")
062         */
063        public Resources(String... values) {
064                super(values);
065        }
066
067        /**
068         * Creates a new resources property.
069         * @param values the values to initialize the property with (e.g. "easel",
070         * "projector")
071         */
072        public Resources(List<String> values) {
073                super(values);
074        }
075
076        /**
077         * Copy constructor.
078         * @param original the property to make a copy of
079         */
080        public Resources(Resources original) {
081                super(original);
082        }
083
084        @Override
085        public String getAltRepresentation() {
086                return super.getAltRepresentation();
087        }
088
089        @Override
090        public void setAltRepresentation(String uri) {
091                super.setAltRepresentation(uri);
092        }
093
094        @Override
095        public String getLanguage() {
096                return super.getLanguage();
097        }
098
099        @Override
100        public void setLanguage(String language) {
101                super.setLanguage(language);
102        }
103
104        @Override
105        public Resources copy() {
106                return new Resources(this);
107        }
108}
