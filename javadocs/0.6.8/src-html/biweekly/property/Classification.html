033/**
034 * <p>
035 * Defines the level of sensitivity of the iCalendar data. If not specified, the
036 * data should be considered "public".
037 * </p>
038 * <p>
039 * <b>Code sample (creating):</b>
040 * </p>
041 * 
042 * <pre class="brush:java">
043 * VEvent event = new VEvent();
044 * event.setClassification(Classification.public_());
045 * </pre>
046 * 
047 * <p>
048 * <b>Code sample (retrieving):</b>
049 * </p>
050 * 
051 * <pre class="brush:java">
052 * ICalendar ical = ...
053 * VEvent event = ical.getEvents().get(0);
054 * 
055 * Classification classification = event.getClassification();
056 * if (classification.isPublic()) {
057 *   //...
058 * } else if (classification.isPrivate()) {
059 *   //...
060 * } else if (classification.isConfidential()) {
061 *   //...
062 * }
063 * </pre>
064 * @author Michael Angstadt
065 * @see <a href="http://tools.ietf.org/html/rfc5545#page-82">RFC 5545 p.82-3</a>
066 * @see <a href="http://tools.ietf.org/html/rfc2445#page-79">RFC 2445
067 * p.79-80</a>
068 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.28-9</a>
069 */
070public class Classification extends EnumProperty {
071        public static final String PUBLIC = "PUBLIC";
072        public static final String PRIVATE = "PRIVATE";
073        public static final String CONFIDENTIAL = "CONFIDENTIAL";
074
075        /**
076         * Creates a new classification property. Use the static factory methods to
077         * create a property with a standard classification level.
078         * @param classification the classification level (e.g. "PUBLIC")
079         */
080        public Classification(String classification) {
081                super(classification);
082        }
083
084        /**
085         * Copy constructor.
086         * @param original the property to make a copy of
087         */
088        public Classification(Classification original) {
089                super(original);
090        }
091
092        /**
093         * Creates a "public" classification property.
094         * @return the property
095         */
096        public static Classification public_() {
097                return create(PUBLIC);
098        }
099
100        /**
101         * Determines if the classification level is "public".
102         * @return true if it's "public", false if not
103         */
104        public boolean isPublic() {
105                return is(PUBLIC);
106        }
107
108        /**
109         * Creates a "private" classification property.
110         * @return the property
111         */
112        public static Classification private_() {
113                return create(PRIVATE);
114        }
115
116        /**
117         * Determines if the classification level is "private".
118         * @return true if it's "private", false if not
119         */
120        public boolean isPrivate() {
121                return is(PRIVATE);
122        }
123
124        /**
125         * Creates a "confidential" classification property.
126         * @return the property
127         */
128        public static Classification confidential() {
129                return create(CONFIDENTIAL);
130        }
131
132        /**
133         * Determines if the classification level is "confidential".
134         * @return true if it's "confidential", false if not
135         */
136        public boolean isConfidential() {
137                return is(CONFIDENTIAL);
138        }
139
140        private static Classification create(String classification) {
141                return new Classification(classification);
142        }
143
144        @Override
145        protected Collection<String> getStandardValues(ICalVersion version) {
146                return Arrays.asList(PUBLIC, PRIVATE, CONFIDENTIAL);
147        }
148
149        @Override
150        public Classification copy() {
151                return new Classification(this);
152        }
153}
