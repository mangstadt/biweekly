034/**
035 * <p>
036 * Defines the type of action to invoke when an alarm is triggered.
037 * </p>
038 * 
039 * <p>
040 * <b>Code sample (creating):</b>
041 * </p>
042 * 
043 * <pre class="brush:java">
044 * Action action = Action.audio();
045 * </pre>
046 * 
047 * <p>
048 * <b>Code sample (retrieving):</b>
049 * </p>
050 * 
051 * <pre class="brush:java">
052 * ICalendar ical = ...
053 * for (VAlarm alarm : ical.getAlarms()) {
054 *   Action action = alarm.getAction();
055 *   if (action.isAudio()) {
056 *     //...
057 *   } else if (action.isEmail()) {
058 *     //...
059 *   } else if (action.isDisplay()) {
060 *     //...
061 *   }
062 * }
063 * </pre>
064 * @author Michael Angstadt
065 * @see <a href="http://tools.ietf.org/html/rfc5545#page-132">RFC 5545
066 * p.132-3</a>
067 * @see <a href="http://tools.ietf.org/html/rfc2445#page-126">RFC 2445 p.126</a>
068 */
069public class Action extends EnumProperty {
070        public static final String AUDIO = "AUDIO";
071        public static final String DISPLAY = "DISPLAY";
072        public static final String EMAIL = "EMAIL";
073        public static final String PROCEDURE = "PROCEDURE";
074
075        /**
076         * Creates an action property. Use of this constructor is discouraged and
077         * may put the property in an invalid state. Use one of the static factory
078         * methods instead.
079         * @param value the value (e.g. "AUDIO")
080         */
081        public Action(String value) {
082                super(value);
083        }
084
085        /**
086         * Copy constructor.
087         * @param original the property to make a copy of
088         */
089        public Action(Action original) {
090                super(original);
091        }
092
093        /**
094         * Creates an "audio" action property.
095         * @return the property
096         */
097        public static Action audio() {
098                return create(AUDIO);
099        }
100
101        /**
102         * Determines if this property is an "audio" action.
103         * @return true if it's an "audio" action, false if not
104         */
105        public boolean isAudio() {
106                return is(AUDIO);
107        }
108
109        /**
110         * Creates an "display" action property.
111         * @return the property
112         */
113        public static Action display() {
114                return create(DISPLAY);
115        }
116
117        /**
118         * Determines if this property is an "display" action.
119         * @return true if it's an "display" action, false if not
120         */
121        public boolean isDisplay() {
122                return is(DISPLAY);
123        }
124
125        /**
126         * Creates an "email" action property.
127         * @return the property
128         */
129        public static Action email() {
130                return create(EMAIL);
131        }
132
133        /**
134         * Determines if this property is an "email" action.
135         * @return true if it's an "email" action, false if not
136         */
137        public boolean isEmail() {
138                return is(EMAIL);
139        }
140
141        /**
142         * Creates a "procedure" action property (vCal 1.0 only).
143         * @return the property
144         */
145        public static Action procedure() {
146                return create(PROCEDURE);
147        }
148
149        /**
150         * Determines if this property is a "procedure" action (vCal 1.0 only).
151         * @return true if it's a "procedure" action, false if not
152         */
153        public boolean isProcedure() {
154                return is(PROCEDURE);
155        }
156
157        private static Action create(String value) {
158                return new Action(value);
159        }
160
161        @Override
162        protected Collection<String> getStandardValues(ICalVersion version) {
163                switch (version) {
164                case V1_0:
165                        return Arrays.asList(AUDIO, DISPLAY, EMAIL, PROCEDURE);
166                default:
167                        return Arrays.asList(AUDIO, DISPLAY, EMAIL);
168                }
169        }
170
171        @Override
172        protected Collection<ICalVersion> getValueSupportedVersions() {
173                if (value == null) {
174                        return Collections.emptyList();
175                }
176
177                if (isAudio() || isDisplay() || isEmail()) {
178                        return Arrays.asList(ICalVersion.values());
179                }
180                if (isProcedure()) {
181                        return Collections.singletonList(ICalVersion.V1_0);
182                }
183
184                return Collections.emptyList();
185        }
186
187        @Override
188        public Action copy() {
189                return new Action(this);
190        }
191}
