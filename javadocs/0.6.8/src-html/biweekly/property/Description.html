030/**
031 * <p>
032 * Defines a detailed description of the component that this property belongs
033 * to. The description should be a more detailed version of the text provided by
034 * the {@link Summary} property.
035 * </p>
036 * <p>
037 * If defined in the top-level {@link ICalendar} component, it contains a
038 * detailed description of the calendar as a whole. An iCalendar object can only
039 * have one description, but multiple Description properties can exist in order
040 * to specify the description in multiple languages. In this case, each property
041 * instance must be assigned a LANGUAGE parameter.
042 * </p>
043 * <p>
044 * <b>Code sample:</b>
045 * </p>
046 * 
047 * <pre class="brush:java">
048 * VEvent event = new VEvent();
049 * 
050 * Description description = new Description("During this meeting, we will discuss...");
051 * event.setDescription(description);
052 * </pre>
053 * @author Michael Angstadt
054 * @see <a href="http://tools.ietf.org/html/rfc5545#page-84">RFC 5545 p.84-5</a>
055 * @see <a href="http://tools.ietf.org/html/rfc2445#page-81">RFC 2445 p.81-2</a>
056 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.30</a>
057 * @see <a
058 * href="http://tools.ietf.org/html/draft-ietf-calext-extensions-01#page-6">draft-ietf-calext-extensions-01
059 * p.6</a>
060 */
061public class Description extends TextProperty {
062        /**
063         * Creates a description property.
064         * @param description the description
065         */
066        public Description(String description) {
067                super(description);
068        }
069
070        /**
071         * Copy constructor.
072         * @param original the property to make a copy of
073         */
074        public Description(Description original) {
075                super(original);
076        }
077
078        @Override
079        public String getAltRepresentation() {
080                return super.getAltRepresentation();
081        }
082
083        @Override
084        public void setAltRepresentation(String uri) {
085                super.setAltRepresentation(uri);
086        }
087
088        @Override
089        public String getLanguage() {
090                return super.getLanguage();
091        }
092
093        @Override
094        public void setLanguage(String language) {
095                super.setLanguage(language);
096        }
097
098        @Override
099        public Description copy() {
100                return new Description(this);
101        }
102}
