042/**
043 * Stores a property that was parsed from an xCal document (XML-encoded
044 * iCalendar object) whose XML namespace was not part of the xCal XML namespace.
045 * @author Michael Angstadt
046 * @see <a href="http://tools.ietf.org/html/rfc6321#page-17">RFC 6321 p.17-8</a>
047 */
048public class Xml extends ValuedProperty<Document> {
049        /**
050         * Creates an XML property.
051         * @param xml the XML to use as the property's value
052         * @throws SAXException if the XML cannot be parsed
053         */
054        public Xml(String xml) throws SAXException {
055                this((xml == null) ? null : XmlUtils.toDocument(xml));
056        }
057
058        /**
059         * Creates an XML property.
060         * @param element the XML element to use as the property's value (the
061         * element is imported into an empty {@link Document} object)
062         */
063        public Xml(Element element) {
064                this((element == null) ? null : XmlUtils.createDocument());
065                if (element != null) {
066                        Node imported = value.importNode(element, true);
067                        value.appendChild(imported);
068                }
069        }
070
071        /**
072         * Creates an XML property.
073         * @param document the XML document to use as the property's value
074         */
075        public Xml(Document document) {
076                super(document);
077        }
078
079        /**
080         * Copy constructor.
081         * @param original the property to make a copy of
082         */
083        public Xml(Xml original) {
084                super(original);
085                if (original.value != null) {
086                        value = XmlUtils.createDocument();
087                        Element root = original.value.getDocumentElement();
088                        if (root != null) {
089                                Node node = value.importNode(root, true);
090                                value.appendChild(node);
091                        }
092                }
093        }
094
095        @Override
096        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
097                if (value == null) {
098                        warnings.add(new ValidationWarning(26));
099                }
100        }
101
102        @Override
103        protected Map<String, Object> toStringValues() {
104                Map<String, Object> values = new LinkedHashMap<String, Object>();
105                values.put("value", (value == null) ? "null" : XmlUtils.toString(value));
106                return values;
107        }
108
109        @Override
110        public Xml copy() {
111                return new Xml(this);
112        }
113
114        @Override
115        protected int valueHashCode() {
116                return XmlUtils.toString(value).hashCode();
117        }
118
119        @Override
120        protected boolean valueEquals(Document otherValue) {
121                if (otherValue == null) return false;
122                return XmlUtils.toString(value).equals(XmlUtils.toString(otherValue));
123        }
124}
