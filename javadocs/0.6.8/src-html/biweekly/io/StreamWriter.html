052/**
053 * Writes iCalendar objects to a data stream.
054 * @author Michael Angstadt
055 */
056public abstract class StreamWriter implements Closeable {
057        protected ScribeIndex index = new ScribeIndex();
058        protected WriteContext context;
059        protected TimezoneAssignment globalTimezone;
060        private TimezoneInfo tzinfo;
061
062        /**
063         * Writes an iCalendar object to the data stream.
064         * @param ical the iCalendar object to write
065         * @throws IllegalArgumentException if the scribe class for a component or
066         * property object cannot be found (only happens when an experimental
067         * property/component scribe is not registered with the
068         * {@code registerScribe} method.)
069         * @throws IOException if there's a problem writing to the data stream
070         */
071        public void write(ICalendar ical) throws IOException {
072                Collection<Class<?>> unregistered = findScribeless(ical);
073                if (!unregistered.isEmpty()) {
074                        List<String> classNames = new ArrayList<String>(unregistered.size());
075                        for (Class<?> clazz : unregistered) {
076                                classNames.add(clazz.getName());
077                        }
078                        throw Messages.INSTANCE.getIllegalArgumentException(13, classNames);
079                }
080
081                tzinfo = ical.getTimezoneInfo();
082                context = new WriteContext(getTargetVersion(), tzinfo, globalTimezone);
083                _write(ical);
084        }
085
086        /**
087         * Gets the {@link VTimezone} components that need to be written to the
088         * output stream.
089         * @return the components
090         */
091        protected Collection<VTimezone> getTimezoneComponents() {
092                if (globalTimezone != null) {
093                        VTimezone component = globalTimezone.getComponent();
094                        return (component == null) ? Collections.<VTimezone> emptyList() : Collections.singletonList(component);
095                }
096
097                return tzinfo.getComponents();
098        }
099
100        /**
101         * Gets the version that the next iCalendar object will be written as.
102         * @return the version
103         */
104        protected abstract ICalVersion getTargetVersion();
105
106        /**
107         * Writes an iCalendar object to the data stream.
108         * @param ical the iCalendar object to write
109         * @throws IOException if there's a problem writing to the data stream
110         */
111        protected abstract void _write(ICalendar ical) throws IOException;
112
113        /**
114         * Gets the timezone that all date/time property values will be formatted
115         * in. If set, this setting will override the timezone information
116         * associated with each {@link ICalendar} object.
117         * @return the global timezone or null if not set (defaults to null)
118         */
119        public TimezoneAssignment getGlobalTimezone() {
120                return globalTimezone;
121        }
122
123        /**
124         * Sets the timezone that all date/time property values will be formatted
125         * in. This is a convenience method that overrides the timezone information
126         * associated with each {@link ICalendar} object that is passed into this
127         * writer.
128         * @param globalTimezone the global timezone or null not to set a global
129         * timezone (defaults to null)
130         */
131        public void setGlobalTimezone(TimezoneAssignment globalTimezone) {
132                this.globalTimezone = globalTimezone;
133        }
134
135        /**
136         * <p>
137         * Registers an experimental property scribe. Can also be used to override
138         * the scribe of a standard property (such as DTSTART). Calling this method
139         * is the same as calling:
140         * </p>
141         * <p>
142         * {@code getScribeIndex().register(scribe)}.
143         * </p>
144         * @param scribe the scribe to register
145         */
146        public void registerScribe(ICalPropertyScribe<? extends ICalProperty> scribe) {
147                index.register(scribe);
148        }
149
150        /**
151         * <p>
152         * Registers an experimental component scribe. Can also be used to override
153         * the scribe of a standard component (such as VEVENT). Calling this method
154         * is the same as calling:
155         * </p>
156         * <p>
157         * {@code getScribeIndex().register(scribe)}.
158         * </p>
159         * @param scribe the scribe to register
160         */
161        public void registerScribe(ICalComponentScribe<? extends ICalComponent> scribe) {
162                index.register(scribe);
163        }
164
165        /**
166         * Gets the object that manages the component/property scribes.
167         * @return the scribe index
168         */
169        public ScribeIndex getScribeIndex() {
170                return index;
171        }
172
173        /**
174         * Sets the object that manages the component/property scribes.
175         * @param scribe the scribe index
176         */
177        public void setScribeIndex(ScribeIndex scribe) {
178                this.index = scribe;
179        }
180
181        /**
182         * Gets the component/property classes that don't have scribes associated
183         * with them.
184         * @param ical the iCalendar object
185         * @return the component/property classes
186         */
187        private Collection<Class<?>> findScribeless(ICalendar ical) {
188                Set<Class<?>> unregistered = new HashSet<Class<?>>();
189                LinkedList<ICalComponent> components = new LinkedList<ICalComponent>();
190                components.add(ical);
191
192                while (!components.isEmpty()) {
193                        ICalComponent component = components.removeLast();
194
195                        Class<? extends ICalComponent> componentClass = component.getClass();
196                        if (componentClass != RawComponent.class && index.getComponentScribe(componentClass) == null) {
197                                unregistered.add(componentClass);
198                        }
199
200                        for (Map.Entry<Class<? extends ICalProperty>, List<ICalProperty>> entry : component.getProperties()) {
201                                List<ICalProperty> properties = entry.getValue();
202                                if (properties.isEmpty()) {
203                                        continue;
204                                }
205
206                                Class<? extends ICalProperty> clazz = entry.getKey();
207                                if (clazz != RawProperty.class && index.getPropertyScribe(clazz) == null) {
208                                        unregistered.add(clazz);
209                                }
210                        }
211
212                        components.addAll(component.getComponents().values());
213                }
214
215                return unregistered;
216        }
217
218        /**
219         * Adds a VERSION property to the given component if the component is the
220         * root ICalendar component and the component does not have a VERSION
221         * property.
222         * @param component the component
223         * @param propertyObjs the component's properties (VERSION will be prepended
224         * to this list)
225         */
226        @SuppressWarnings({ "rawtypes", "unchecked" })
227        protected void addVersionPropertyIfMissing(ICalComponent component, List propertyObjs) {
228                if (!(component instanceof ICalendar)) {
229                        return;
230                }
231
232                if (component.getProperty(Version.class) != null) {
233                        return;
234                }
235
236                propertyObjs.add(0, new Version(getTargetVersion()));
237        }
238
239        /**
240         * Adds the VTIMEZONE components to the given component if the component is
241         * the root ICalendar component and the component does not already have the
242         * VTIMEZONE components.
243         * @param component the component
244         * @param subComponents the component's sub-components (the VTIMEZONEs will
245         * be prepended to this list)
246         */
247        @SuppressWarnings({ "unchecked", "rawtypes" })
248        protected void addTimezoneComponentsIfMissing(ICalComponent component, List subComponents) {
249                if (!(component instanceof ICalendar)) {
250                        return;
251                }
252
253                Collection<VTimezone> timezones = getTimezoneComponents();
254                for (VTimezone timezone : timezones) {
255                        if (!subComponents.contains(timezone)) {
256                                subComponents.add(0, timezone);
257                        }
258                }
259        }
