053/**
054 * Chainer class for writing xCal (XML-encoded iCalendar objects).
055 * @see Biweekly#writeXml(Collection)
056 * @see Biweekly#writeXml(ICalendar...)
057 * @author Michael Angstadt
058 */
059public class ChainingXmlWriter extends ChainingWriter<ChainingXmlWriter> {
060        private final XCalOutputProperties outputProperties = new XCalOutputProperties();
061        private final Map<String, ICalDataType> parameterDataTypes = new HashMap<String, ICalDataType>(0);
062
063        /**
064         * @param icals the iCValendar objects to write
065         */
066        public ChainingXmlWriter(Collection<ICalendar> icals) {
067                super(icals);
068        }
069
070        /**
071         * Sets the number of indent spaces to use for pretty-printing. If not set,
072         * then the XML will not be pretty-printed.
073         * @param indent the number of spaces in the indent string or "null" not to
074         * pretty-print (disabled by default)
075         * @return this
076         */
077        public ChainingXmlWriter indent(Integer indent) {
078                outputProperties.setIndent(indent);
079                return this;
080        }
081
082        /**
083         * Sets the XML version to use. Note that many JDKs only support 1.0
084         * natively. For XML 1.1 support, add a JAXP library like <a href=
085         * "http://search.maven.org/#search%7Cgav%7C1%7Cg%3A%22xalan%22%20AND%20a%3A%22xalan%22"
086         * >xalan</a> to your project.
087         * @param xmlVersion the XML version (defaults to "1.0")
088         * @return this
089         */
090        public ChainingXmlWriter xmlVersion(String xmlVersion) {
091                outputProperties.setXmlVersion(xmlVersion);
092                return this;
093        }
094
095        /**
096         * Assigns an output property to the JAXP transformer (see
097         * {@link Transformer#setOutputProperty}).
098         * @param name the property name
099         * @param value the property value
100         * @return this
101         */
102        public ChainingXmlWriter outputProperty(String name, String value) {
103                outputProperties.put(name, value);
104                return this;
105        }
106
107        /**
108         * Assigns all of the given output properties to the JAXP transformer (see
109         * {@link Transformer#setOutputProperty}).
110         * @param outputProperties the properties
111         * @return this
112         */
113        public ChainingXmlWriter outputProperties(Map<String, String> outputProperties) {
114                this.outputProperties.putAll(outputProperties);
115                return this;
116        }
117
118        @Override
119        public ChainingXmlWriter tz(TimeZone defaultTimeZone, boolean outlookCompatible) {
120                return super.tz(defaultTimeZone, outlookCompatible);
121        }
122
123        @Override
124        public ChainingXmlWriter register(ICalPropertyScribe<? extends ICalProperty> scribe) {
125                return super.register(scribe);
126        }
127
128        @Override
129        public ChainingXmlWriter register(ICalComponentScribe<? extends ICalComponent> scribe) {
130                return super.register(scribe);
131        }
132
133        /**
134         * Registers the data type of a non-standard parameter. Non-standard
135         * parameters use the "unknown" data type by default.
136         * @param parameterName the parameter name (e.g. "x-foo")
137         * @param dataType the data type
138         * @return this
139         */
140        public ChainingXmlWriter register(String parameterName, ICalDataType dataType) {
141                parameterDataTypes.put(parameterName, dataType);
142                return this;
143        }
144
145        /**
146         * Writes the iCalendar objects to a string.
147         * @return the XML document
148         */
149        public String go() {
150                return createXCalDocument().write(outputProperties);
151        }
152
153        /**
154         * Writes the iCalendar objects to an output stream.
155         * @param out the output stream to write to
156         * @throws TransformerException if there's a problem writing to the output
157         * stream
158         */
159        public void go(OutputStream out) throws TransformerException {
160                createXCalDocument().write(out, outputProperties);
161        }
162
163        /**
164         * Writes the iCalendar objects to a file.
165         * @param file the file to write to
166         * @throws IOException if the file can't be opened
167         * @throws TransformerException if there's a problem writing to the file
168         */
169        public void go(File file) throws IOException, TransformerException {
170                createXCalDocument().write(file, outputProperties);
171        }
172
173        /**
174         * Writes the iCalendar objects to a writer.
175         * @param writer the writer to write to
176         * @throws TransformerException if there's a problem writing to the writer
177         */
178        public void go(Writer writer) throws TransformerException {
179                createXCalDocument().write(writer, outputProperties);
180        }
181
182        /**
183         * Generates an XML document object model (DOM) containing the iCalendar
184         * objects.
185         * @return the DOM
186         */
187        public Document dom() {
188                return createXCalDocument().getDocument();
189        }
190
191        private XCalDocument createXCalDocument() {
192                XCalDocument document = new XCalDocument();
193
194                XCalDocumentStreamWriter writer = document.writer();
195                if (defaultTimeZone != null) {
196                        writer.setGlobalTimezone(defaultTimeZone);
197                }
198                for (Map.Entry<String, ICalDataType> entry : parameterDataTypes.entrySet()) {
199                        String parameterName = entry.getKey();
200                        ICalDataType dataType = entry.getValue();
201                        writer.registerParameterDataType(parameterName, dataType);
202                }
203                if (index != null) {
204                        writer.setScribeIndex(index);
205                }
206
207                for (ICalendar ical : icals) {
208                        writer.write(ical);
209                }
210
211                return document;
212        }
213}
