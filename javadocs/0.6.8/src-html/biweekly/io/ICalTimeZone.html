066/**
067 * A timezone that is based on an iCalendar {@link VTimezone} component. This
068 * class is not thread safe.
069 * @author Michael Angstadt
070 */
071@SuppressWarnings("serial")
072public class ICalTimeZone extends TimeZone {
073        private final VTimezone component;
074        private final Map<Observance, List<DateValue>> observanceDateCache;
075        final List<Observance> sortedObservances;
076        private final int rawOffset;
077        private final TimeZone utc = TimeZone.getTimeZone("UTC");
078        private final Calendar utcCalendar = Calendar.getInstance(utc);

080        /**
081         * Creates a new timezone based on an iCalendar VTIMEZONE component.
082         * @param component the VTIMEZONE component to wrap
083         */
084        public ICalTimeZone(VTimezone component) {
085                this.component = component;
086
087                int numObservances = component.getStandardTimes().size() + component.getDaylightSavingsTime().size();
088                observanceDateCache = new IdentityHashMap<Observance, List<DateValue>>(numObservances);
089
090                sortedObservances = calculateSortedObservances();
091
092                rawOffset = calculateRawOffset();
093
094                String id = getValue(component.getTimezoneId());
095                if (id != null) {
096                        setID(id);
097                }
098        }
099
100        /**
101         * Builds a list of all the observances in the VTIMEZONE component, sorted
102         * by DTSTART.
103         * @return the sorted observances
104         */
105        private List<Observance> calculateSortedObservances() {
106                List<DaylightSavingsTime> daylights = component.getDaylightSavingsTime();
107                List<StandardTime> standards = component.getStandardTimes();
108
109                int numObservances = standards.size() + daylights.size();
110                List<Observance> sortedObservances = new ArrayList<Observance>(numObservances);
111
112                sortedObservances.addAll(standards);
113                sortedObservances.addAll(daylights);
114
115                Collections.sort(sortedObservances, new Comparator<Observance>() {
116                        public int compare(Observance left, Observance right) {
117                                ICalDate startLeft = getValue(left.getDateStart());
118                                ICalDate startRight = getValue(right.getDateStart());
119                                if (startLeft == null && startRight == null) {
120                                        return 0;
121                                }
122                                if (startLeft == null) {
123                                        return -1;
124                                }
125                                if (startRight == null) {
126                                        return 1;
127                                }
128
129                                return startLeft.getRawComponents().compareTo(startRight.getRawComponents());
130                        }
131                });
132
133                return Collections.unmodifiableList(sortedObservances);
134        }

136        @Override
137        public String getDisplayName(boolean daylight, int style, Locale locale) {
138                ListIterator<Observance> it = sortedObservances.listIterator(sortedObservances.size());
139                while (it.hasPrevious()) {
140                        Observance observance = it.previous();
141
142                        if ((daylight && observance instanceof DaylightSavingsTime) || (!daylight && observance instanceof StandardTime)) {
143                                String name = getDisplayName(observance);
144                                if (name != null) {
145                                        return name;
146                                }
147                        }
148                }
149
150                return super.getDisplayName(daylight, style, locale);
151        }
152
153        private String getDisplayName(Observance observance) {
154                List<TimezoneName> names = observance.getTimezoneNames();
155                return names.isEmpty() ? null : names.get(0).getValue();
156        }

158        @Override
159        public int getOffset(int era, int year, int month, int day, int dayOfWeek, int millis) {
160                int hour = millis / 1000 / 60 / 60;
161                millis -= hour * 1000 * 60 * 60;
162                int minute = millis / 1000 / 60;
163                millis -= minute * 1000 * 60;
164                int second = millis / 1000;
165
166                Observance observance = getObservance(year, month + 1, day, hour, minute, second);
167                if (observance == null) {
168                        /*
169                         * Find the first observance that has a DTSTART property and a
170                         * TZOFFSETFROM property.
171                         */
172                        for (Observance obs : sortedObservances) {
173                                ICalDate dateStart = getValue(obs.getDateStart());
174                                if (dateStart == null) {
175                                        continue;
176                                }
177
178                                UtcOffset offsetFrom = getValue(obs.getTimezoneOffsetFrom());
179                                if (offsetFrom == null) {
180                                        continue;
181                                }
182
183                                return (int) offsetFrom.getMillis();
184                        }
185                        return 0;
186                }
187
188                UtcOffset offsetTo = getValue(observance.getTimezoneOffsetTo());
189                return (offsetTo == null) ? 0 : (int) offsetTo.getMillis();
190        }

194                return rawOffset;

197        private int calculateRawOffset() {
198                Observance observance = getObservance(new Date());
199                if (observance == null) {
200                        //return the offset of the first STANDARD component
201                        for (Observance obs : sortedObservances) {
202                                if (!(obs instanceof StandardTime)) {
203                                        continue;
204                                }
205
206                                UtcOffset offsetTo = getValue(obs.getTimezoneOffsetTo());
207                                if (offsetTo == null) {
208                                        continue;
209                                }
210
211                                return (int) offsetTo.getMillis();
212                        }
213                        return 0;
214                }
215
216                UtcOffset offset = getValue((observance instanceof StandardTime) ? observance.getTimezoneOffsetTo() : observance.getTimezoneOffsetFrom());
217                return (offset == null) ? 0 : (int) offset.getMillis();
218        }

220        @Override
221        public boolean inDaylightTime(Date date) {
222                if (!useDaylightTime()) {
223                        return false;
224                }
225
226                Observance observance = getObservance(date);
227                return (observance == null) ? false : (observance instanceof DaylightSavingsTime);
228        }

239        @Override
240        public boolean useDaylightTime() {
241                for (Observance observance : sortedObservances) {
242                        if (observance instanceof DaylightSavingsTime) {
243                                return true;
244                        }
245                }
246                return false;
247        }

250         * Gets the timezone information of a date.
251         * @param date the date
252         * @return the timezone information
253         */
254        public Boundary getObservanceBoundary(Date date) {
255                utcCalendar.setTime(date);
256                int year = utcCalendar.get(Calendar.YEAR);
257                int month = utcCalendar.get(Calendar.MONTH) + 1;
258                int day = utcCalendar.get(Calendar.DATE);
259                int hour = utcCalendar.get(Calendar.HOUR);
260                int minute = utcCalendar.get(Calendar.MINUTE);
261                int second = utcCalendar.get(Calendar.SECOND);
262
263                return getObservanceBoundary(year, month, day, hour, minute, second);
264        }

306        /**
307         * Gets the observance information of a date.
308         * @param year the year
309         * @param month the month (1-12)
310         * @param day the day of the month
311         * @param hour the hour
312         * @param minute the minute
313         * @param second the second
314         * @return the observance information or null if none was found
315         */
316        private Boundary getObservanceBoundary(int year, int month, int day, int hour, int minute, int second) {
317                if (sortedObservances.isEmpty()) {
318                        return null;
319                }
320
321                DateValue givenTime = new DateTimeValueImpl(year, month, day, hour, minute, second);
322                int closestIndex = -1;
323                Observance closest = null;
324                DateValue closestValue = null;
325                for (int i = 0; i < sortedObservances.size(); i++) {
326                        Observance observance = sortedObservances.get(i);
327
328                        //skip observances that start after the given time
329                        ICalDate dtstart = getValue(observance.getDateStart());
330                        if (dtstart != null) {
331                                DateValue dtstartValue = convertFromRawComponents(dtstart);
332                                if (dtstartValue.compareTo(givenTime) > 0) {
333                                        continue;
334                                }
335                        }
336
337                        DateValue dateValue = getObservanceDateClosestToTheGivenDate(observance, givenTime, false);
338                        if (dateValue != null && (closestValue == null || closestValue.compareTo(dateValue) < 0)) {
339                                closestValue = dateValue;
340                                closest = observance;
341                                closestIndex = i;
342                        }
343                }
344
345                Observance observanceIn = closest;
346                DateValue observanceInStart = closestValue;
347                Observance observanceAfter = null;
348                DateValue observanceAfterStart = null;
349                if (closestIndex < sortedObservances.size() - 1) {
350                        observanceAfter = sortedObservances.get(closestIndex + 1);
351                        observanceAfterStart = getObservanceDateClosestToTheGivenDate(observanceAfter, givenTime, true);
352                }
353
354                /*
355                 * If any of the DTSTART properties are missing their time components,
356                 * then observanceInStart/observanceAfterStart could be a DateValue
357                 * object. If so, convert it to a DateTimeValue object (see Issue 77).
358                 */
359                if (observanceInStart != null && !(observanceInStart instanceof DateTimeValue)) {
360                        observanceInStart = new DTBuilder(observanceInStart).toDateTime();
361                }
362                if (observanceAfterStart != null && !(observanceAfterStart instanceof DateTimeValue)) {
363                        observanceAfterStart = new DTBuilder(observanceAfterStart).toDateTime();
364                }
365
366                return new Boundary((DateTimeValue) observanceInStart, observanceIn, (DateTimeValue) observanceAfterStart, observanceAfter);
367        }

369        /**
370         * Iterates through each of the timezone boundary dates defined by the given
371         * observance and finds the date that comes closest to the given date.
372         * @param observance the observance
373         * @param givenDate the given date
374         * @param after true to return the closest date <b>greater than</b> the
375         * given date, false to return the closest date <b>less than or equal to</b>
376         * the given date.
377         * @return the closest date
378         */
379        private DateValue getObservanceDateClosestToTheGivenDate(Observance observance, DateValue givenDate, boolean after) {
380                List<DateValue> dateCache = observanceDateCache.get(observance);
381                if (dateCache == null) {
382                        dateCache = new ArrayList<DateValue>();
383                        observanceDateCache.put(observance, dateCache);
384                }
385
386                if (dateCache.isEmpty()) {
387                        DateValue prev = null, cur = null;
388                        boolean stopped = false;
389                        RecurrenceIterator it = createIterator(observance);
390                        while (it.hasNext()) {
391                                cur = it.next();
392                                dateCache.add(cur);
393
394                                if (givenDate.compareTo(cur) < 0) {
395                                        //stop if we have passed the givenTime
396                                        stopped = true;
397                                        break;
398                                }
399
400                                prev = cur;
401                        }
402                        return after ? (stopped ? cur : null) : prev;
403                }
404
405                DateValue last = dateCache.get(dateCache.size() - 1);
406                int comparison = last.compareTo(givenDate);
407                if ((after && comparison <= 0) || comparison < 0) {
408                        RecurrenceIterator it = createIterator(observance);
409
410                        /*
411                         * Calling "it.advanceTo()" here causes problems.
412                         * 
413                         * See: https://github.com/mangstadt/biweekly/issues/126
414                         */
415                        //it.advanceTo(last);
416
417                        DateValue prev = null, cur = null;
418                        boolean stopped = false;
419                        while (it.hasNext()) {
420                                cur = it.next();
421                                int curCompareToLast = cur.compareTo(last);
422                                if (curCompareToLast < 0) {
423                                        continue;
424                                }
425                                if (curCompareToLast > 0) {
426                                        dateCache.add(cur);
427                                }
428                                if (curCompareToLast == 0) {
429                                        //do nothing; don't add to dateCache
430                                }
431
432                                if (givenDate.compareTo(cur) < 0) {
433                                        //stop if we have passed the givenTime
434                                        stopped = true;
435                                        break;
436                                }
437
438                                prev = cur;
439                        }
440                        return after ? (stopped ? cur : null) : prev;
441                }
442
443                /*
444                 * The date is somewhere in the cached list, so find it.
445                 * 
446                 * Note: Read the "binarySearch" method Javadoc carefully for an
447                 * explanation of its return value.
448                 */
449                int index = Collections.binarySearch(dateCache, givenDate);
450
451                if (index < 0) {
452                        /*
453                         * The index where the date would be if it was inside the list.
454                         */
455                        index = (index * -1) - 1;
456
457                        if (after) {
458                                /*
459                                 * This is where the date would be if it was inside the list, so
460                                 * we want to return the date value that's currently at that
461                                 * position.
462                                 */
463                                int afterIndex = index;
464
465                                return (afterIndex < dateCache.size()) ? dateCache.get(afterIndex) : null;
466                        }
467
468                        int beforeIndex = index - 1;
469                        if (beforeIndex < 0) {
470                                return null;
471                        }
472                        if (beforeIndex >= dateCache.size()) {
473                                return dateCache.get(dateCache.size() - 1);
474                        }
475                        return dateCache.get(beforeIndex);
476                }
477
478                /*
479                 * An exact match was found.
480                 */
481                if (after) {
482                        int afterIndex = index + 1; //remember: the date must be >
483                        return (afterIndex < dateCache.size()) ? dateCache.get(afterIndex) : null;
484                }
485                return dateCache.get(index); //remember: the date must be <=
486        }

488        /**
489         * Creates an iterator which iterates over each of the dates in an
490         * observance.
491         * @param observance the observance
492         * @return the iterator
493         */
494        RecurrenceIterator createIterator(Observance observance) {
495                List<RecurrenceIterator> inclusions = new ArrayList<RecurrenceIterator>();
496                List<RecurrenceIterator> exclusions = new ArrayList<RecurrenceIterator>();
497
498                ICalDate dtstart = getValue(observance.getDateStart());
499                if (dtstart != null) {
500                        DateValue dtstartValue = convertFromRawComponents(dtstart);
501
502                        //add DTSTART property
503                        inclusions.add(new DateValueRecurrenceIterator(Collections.singletonList(dtstartValue)));
504
505                        //add RRULE properties
506                        for (RecurrenceRule rrule : observance.getProperties(RecurrenceRule.class)) {
507                                Recurrence recur = rrule.getValue();
508                                if (recur != null) {
509                                        inclusions.add(RecurrenceIteratorFactory.createRecurrenceIterator(recur, dtstartValue, utc));
510                                }
511                        }
512
513                        //add EXRULE properties
514                        for (ExceptionRule exrule : observance.getProperties(ExceptionRule.class)) {
515                                Recurrence recur = exrule.getValue();
516                                if (recur != null) {
517                                        exclusions.add(RecurrenceIteratorFactory.createRecurrenceIterator(recur, dtstartValue, utc));
518                                }
519                        }
520                }
521
522                //add RDATE properties
523                List<ICalDate> rdates = new ArrayList<ICalDate>();
524                for (RecurrenceDates rdate : observance.getRecurrenceDates()) {
525                        rdates.addAll(rdate.getDates());
526                }
527                Collections.sort(rdates);
528                inclusions.add(new DateRecurrenceIterator(rdates));
529
530                //add EXDATE properties
531                List<ICalDate> exdates = new ArrayList<ICalDate>();
532                for (ExceptionDates exdate : observance.getProperties(ExceptionDates.class)) {
533                        exdates.addAll(exdate.getValues());
534                }
535                Collections.sort(exdates);
536                exclusions.add(new DateRecurrenceIterator(exdates));
537
538                RecurrenceIterator included = join(inclusions);
539                if (exclusions.isEmpty()) {
540                        return included;
541                }
542
543                RecurrenceIterator excluded = join(exclusions);
544                return RecurrenceIteratorFactory.except(included, excluded);
545        }
546
547        private static RecurrenceIterator join(List<RecurrenceIterator> iterators) {
548                if (iterators.isEmpty()) {
549                        return new EmptyRecurrenceIterator();
550                }
551
552                RecurrenceIterator first = iterators.get(0);
553                if (iterators.size() == 1) {
554                        return first;
555                }
556
557                List<RecurrenceIterator> theRest = iterators.subList(1, iterators.size());
558                return RecurrenceIteratorFactory.join(first, theRest.toArray(new RecurrenceIterator[0]));
559        }

613        /**
614         * A recurrence iterator that wraps an {@link Iterator}.
615         */
616        private static abstract class IteratorWrapper<T> implements RecurrenceIterator {
617                protected final Iterator<T> it;
618                private DateValue next;
619
620                public IteratorWrapper(Iterator<T> it) {
621                        this.it = it;
622                }
623
624                public DateValue next() {
625                        if (next != null) {
626                                DateValue value = next;
627                                next = null;
628                                return value;
629                        }
630                        return toDateValue(it.next());
631                }
632
633                public boolean hasNext() {
634                        return next != null || it.hasNext();
635                }
636
637                public void advanceTo(DateValue newStartUtc) {
638                        if (this.next != null && this.next.compareTo(newStartUtc) >= 0) {
639                                return;
640                        }
641
642                        while (it.hasNext()) {
643                                DateValue next = toDateValue(it.next());
644                                if (next.compareTo(newStartUtc) >= 0) {
645                                        this.next = next;
646                                        break;
647                                }
648                        }
649                }
650
651                public void remove() {
652                        //RecurrenceIterator does not support this method
653                        throw new UnsupportedOperationException();
654                }
655
656                protected abstract DateValue toDateValue(T next);
657        }

659        /**
660         * Holds the timezone observance information of a particular date.
661         */
662        public static class Boundary {
663                private final DateTimeValue observanceInStart, observanceAfterStart;
664                private final Observance observanceIn, observanceAfter;
665
666                public Boundary(DateTimeValue observanceInStart, Observance observanceIn, DateTimeValue observanceAfterStart, Observance observanceAfter) {
667                        this.observanceInStart = observanceInStart;
668                        this.observanceAfterStart = observanceAfterStart;
669                        this.observanceIn = observanceIn;
670                        this.observanceAfter = observanceAfter;
671                }
672
673                /**
674                 * Gets start time of the observance that the date resides in.
675                 * @return the time
676                 */
677                public DateTimeValue getObservanceInStart() {
678                        return observanceInStart;
679                }
680
681                /**
682                 * Gets the start time the observance that comes after the observance
683                 * that the date resides in.
684                 * @return the time
685                 */
686                public DateTimeValue getObservanceAfterStart() {
687                        return observanceAfterStart;
688                }
689
690                /**
691                 * Gets the observance that the date resides in.
692                 * @return the observance
693                 */
694                public Observance getObservanceIn() {
695                        return observanceIn;
696                }
697
698                /**
699                 * Gets the observance that comes after the observance that the date
700                 * resides in.
701                 * @return the observance
702                 */
703                public Observance getObservanceAfter() {
704                        return observanceAfter;
705                }
706
707                @Override
708                public String toString() {
709                        return "Boundary [observanceInStart=" + observanceInStart + ", observanceAfterStart=" + observanceAfterStart + ", observanceIn=" + observanceIn + ", observanceAfter=" + observanceAfter + "]";
710                }
711        }
