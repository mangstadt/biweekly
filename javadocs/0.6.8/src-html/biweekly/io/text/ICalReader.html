063/**
064 * <p>
065 * Parses {@link ICalendar} objects from a plain-text iCalendar data stream.
066 * </p>
067 * <p>
068 * <b>Example:</b>
069 * </p>
070 * 
071 * <pre class="brush:java">
072 * File file = new File("icals.ics");
073 * ICalReader reader = null;
074 * try {
075 *   reader = new ICalReader(file);
076 *   ICalendar ical;
077 *   while ((ical = reader.readNext()) != null) {
078 *     //...
079 *   }
080 * } finally {
081 *   if (reader != null) reader.close();
082 * }
083 * </pre>
084 * @author Michael Angstadt
085 * @see <a href="http://www.imc.org/pdi/pdiproddev.html">1.0 specs</a>
086 * @see <a href="https://tools.ietf.org/html/rfc2445">RFC 2445</a>
087 * @see <a href="http://tools.ietf.org/html/rfc5545">RFC 5545</a>
088 */
089public class ICalReader extends StreamReader {
090        private static final String VCALENDAR_COMPONENT_NAME = ScribeIndex.getICalendarScribe().getComponentName(); //"VCALENDAR"
091
092        private final VObjectReader reader;
093        private final ICalVersion defaultVersion;

096         * Creates a new iCalendar reader.

159        /**
160         * Creates a new iCalendar reader.
161         * @param reader the reader to read from
162         * @param defaultVersion the version to assume the iCalendar object is in
163         * until a VERSION property is encountered (defaults to 2.0)
164         */
165        public ICalReader(Reader reader, ICalVersion defaultVersion) {
166                SyntaxRules rules = SyntaxRules.iCalendar();
167                rules.setDefaultSyntaxStyle(defaultVersion.getSyntaxStyle());
168                this.reader = new VObjectReader(reader, rules);
169                this.defaultVersion = defaultVersion;
170        }

229        /**
230         * <p>
231         * Gets the iCalendar version that this reader will assume each iCalendar
232         * object is formatted in up until a VERSION property is encountered.
233         * </p>
234         * <p>
235         * All standards-compliant iCalendar objects contain a VERSION property at
236         * the very beginning of the object, so for the vast majority of iCalendar
237         * objects, this setting does nothing. This setting is needed for when the
238         * iCalendar object does not have a VERSION property or for when the VERSION
239         * property is not located at the beginning of the object.
240         * </p>
241         * @return the default version (defaults to "2.0")
242         */
243        public ICalVersion getDefaultVersion() {
244                return defaultVersion;
245        }

247        @Override
248        protected ICalendar _readNext() throws IOException {
249                VObjectDataListenerImpl listener = new VObjectDataListenerImpl();
250                reader.parse(listener);
251                return listener.ical;
252        }
253
254        private class VObjectDataListenerImpl implements VObjectDataListener {
255                private ICalendar ical = null;
256                private ICalVersion version = defaultVersion;
257                private ComponentStack stack = new ComponentStack();
258
259                public void onComponentBegin(String name, Context vobjectContext) {
260                        //ignore everything until a VCALENDAR component is read
261                        if (ical == null && !isVCalendarComponent(name)) {
262                                return;
263                        }
264
265                        ICalComponent parentComponent = stack.peek();
266
267                        ICalComponentScribe<? extends ICalComponent> scribe = index.getComponentScribe(name, version);
268                        ICalComponent component = scribe.emptyInstance();
269                        stack.push(component);
270
271                        if (parentComponent == null) {
272                                ical = (ICalendar) component;
273                                context.setVersion(version);
274                        } else {
275                                parentComponent.addComponent(component);
276                        }
277                }
278
279                public void onComponentEnd(String name, Context vobjectContext) {
280                        //VCALENDAR component not read yet, ignore
281                        if (ical == null) {
282                                return;
283                        }
284
285                        /*
286                         * VObjectDataListener guarantees correct ordering of component
287                         * begin/end callback invocations (see javadocs), so we can pop
288                         * blindly without checking if the component name matches.
289                         */
290                        stack.pop();
291
292                        //stop reading when "END:VCALENDAR" is reached
293                        if (stack.isEmpty()) {
294                                vobjectContext.stop();
295                        }
296                }
297
298                public void onProperty(VObjectProperty vobjectProperty, Context vobjectContext) {
299                        //VCALENDAR component not read yet, ignore
300                        if (ical == null) {
301                                return;
302                        }
303
304                        String propertyName = vobjectProperty.getName();
305                        ICalParameters parameters = new ICalParameters(vobjectProperty.getParameters().getMap());
306                        String value = vobjectProperty.getValue();
307                        
308                        context.getWarnings().clear();
309                        context.setLineNumber(vobjectContext.getLineNumber());
310                        context.setPropertyName(propertyName);
311
312                        ICalPropertyScribe<? extends ICalProperty> scribe = index.getPropertyScribe(propertyName, version);
313
314                        //process nameless parameters
315                        processNamelessParameters(parameters, version);
316
317                        //get the data type (VALUE parameter)
318                        ICalDataType dataType = parameters.getValue();
319                        parameters.setValue(null);
320                        if (dataType == null) {
321                                //use the property's default data type if there is no VALUE parameter
322                                dataType = scribe.defaultDataType(version);
323                        }
324
325                        ICalComponent parentComponent = stack.peek();
326                        try {
327                                ICalProperty property = scribe.parseText(value, dataType, parameters, context);
328                                parentComponent.addProperty(property);
329                        } catch (SkipMeException e) {
330                                //@formatter:off
331                                warnings.add(new ParseWarning.Builder(context)
332                                        .message(0, e.getMessage())
333                                        .build()
334                                );
335                                //@formatter:on
336                        } catch (CannotParseException e) {
337                                //@formatter:off
338                                warnings.add(new ParseWarning.Builder(context)
339                                        .message(e)
340                                        .build()
341                                );
342                                //@formatter:on
343                                ICalProperty property = new RawPropertyScribe(propertyName).parseText(value, dataType, parameters, context);
344                                parentComponent.addProperty(property);
345                        } catch (DataModelConversionException e) {
346                                for (ICalProperty property : e.getProperties()) {
347                                        parentComponent.addProperty(property);
348                                }
349                                for (ICalComponent component : e.getComponents()) {
350                                        parentComponent.addComponent(component);
351                                }
352                        }
353
354                        warnings.addAll(context.getWarnings());
355                }
356
357                public void onVersion(String value, Context vobjectContext) {
358                        //ignore if we are not directly under the root VCALENDAR component
359                        if (stack.size() != 1) {
360                                return;
361                        }
362
363                        version = ICalVersion.get(value);
364                        context.setVersion(version);
365                }
366
367                public void onWarning(com.github.mangstadt.vinnie.io.Warning warning, VObjectProperty property, Exception thrown, Context vobjectContext) {
368                        //VCALENDAR component not read yet, ignore
369                        if (ical == null) {
370                                return;
371                        }
372
373                        //@formatter:off
374                        warnings.add(new ParseWarning.Builder()
375                                .lineNumber(vobjectContext.getLineNumber())
376                                .propertyName((property == null) ? null : property.getName())
377                                .message(warning.getMessage())
378                                .build()
379                        );
380                        //@formatter:on
381                }
382
383                private boolean isVCalendarComponent(String componentName) {
384                        return VCALENDAR_COMPONENT_NAME.equals(componentName);
385                }
386
387                /**
388                 * Assigns names to all nameless parameters. v2.0 requires all
389                 * parameters to have names, but v1.0 does not.
390                 * @param parameters the parameters
391                 * @param version the iCal version
392                 */
393                private void processNamelessParameters(ICalParameters parameters, ICalVersion version) {
394                        List<String> namelessParamValues = parameters.removeAll(null);
395                        if (namelessParamValues.isEmpty()) {
396                                return;
397                        }
398
399                        if (version != ICalVersion.V1_0) {
400                                //@formatter:off
401                                warnings.add(new ParseWarning.Builder(context)
402                                        .message(4, namelessParamValues)
403                                        .build()
404                                );
405                                //@formatter:on
406                        }
407
408                        for (String paramValue : namelessParamValues) {
409                                String paramName = guessParameterName(paramValue);
410                                parameters.put(paramName, paramValue);
411                        }
412                }
413
414                /**
415                 * Makes a guess as to what a parameter value's name should be.
416                 * @param value the parameter value
417                 * @return the guessed name
418                 */
419                private String guessParameterName(String value) {
420                        if (ICalDataType.find(value) != null) {
421                                return ICalParameters.VALUE;
422                        }
423
424                        if (Encoding.find(value) != null) {
425                                return ICalParameters.ENCODING;
426                        }
427
428                        //otherwise, assume it's a TYPE
429                        return ICalParameters.TYPE;
430                }
431        }

433        /**
434         * Keeps track of the hierarchy of nested components.
435         */
436        private static class ComponentStack {
437                private final List<ICalComponent> components = new ArrayList<ICalComponent>();
438
439                /**
440                 * Gets the top component from the stack.
441                 * @return the component or null if the stack is empty
442                 */
443                public ICalComponent peek() {
444                        return isEmpty() ? null : components.get(components.size() - 1);
445                }
446
447                /**
448                 * Adds a component to the stack
449                 * @param component the component
450                 */
451                public void push(ICalComponent component) {
452                        components.add(component);
453                }
454
455                /**
456                 * Removes the top component from the stack and returns it.
457                 * @return the top component or null if the stack is empty
458                 */
459                public ICalComponent pop() {
460                        return isEmpty() ? null : components.remove(components.size() - 1);
461                }
462
463                /**
464                 * Determines if the stack is empty.
465                 * @return true if it's empty, false if not
466                 */
467                public boolean isEmpty() {
468                        return components.isEmpty();
469                }
470
471                /**
472                 * Gets the size of the stack.
473                 * @return the size
474                 */
475                public int size() {
476                        return components.size();
477                }
478        }

480        /**
481         * Closes the input stream.
482         * @throws IOException if there's a problem closing the input stream
483         */
484        public void close() throws IOException {
485                reader.close();
486        }
