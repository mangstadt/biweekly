038/**
039 * Stores information used to write the properties in an iCalendar object.
040 * @author Michael Angstadt
041 */
042public class WriteContext {
043        private final ICalVersion version;
044        private final TimezoneInfo timezoneOptions;
045        private final TimezoneAssignment globalTimezone;
046        private final List<Date> dates = new ArrayList<Date>();
047        private ICalComponent parent;
048
049        public WriteContext(ICalVersion version, TimezoneInfo timezoneOptions, TimezoneAssignment globalTimezone) {
050                this.version = version;
051                this.timezoneOptions = timezoneOptions;
052                this.globalTimezone = globalTimezone;
053        }
054
055        /**
056         * Gets the version of the iCalendar object that is being written.
057         * @return the iCalendar version
058         */
059        public ICalVersion getVersion() {
060                return version;
061        }
062
063        /**
064         * Gets the timezone options for this iCalendar object.
065         * @return the timezone options
066         */
067        public TimezoneInfo getTimezoneInfo() {
068                return timezoneOptions;
069        }
070
071        /**
072         * Gets the global timezone to format all date/time values in, regardless of
073         * each individual {@link ICalendar}'s timezone settings.
074         * @return the global timezone or null if not set
075         */
076        public TimezoneAssignment getGlobalTimezone() {
077                return globalTimezone;
078        }
079
080        /**
081         * Gets the parent component of the property that is being written.
082         * @return the parent component
083         */
084        public ICalComponent getParent() {
085                return parent;
086        }
087
088        /**
089         * Sets the parent component of the property that is being written.
090         * @param parent the parent component
091         */
092        public void setParent(ICalComponent parent) {
093                this.parent = parent;
094        }
095
096        /**
097         * Gets the timezoned date-time property values that are in the iCalendar
098         * object.
099         * @return the timezoned date-time property values
100         */
101        public List<Date> getDates() {
102                return dates;
103        }
104
105        /**
106         * Records the timezoned date-time values that are being written. This is
107         * used to generate a DAYLIGHT property for vCalendar objects.
108         * @param floating true if the date is floating, false if not
109         * @param tz the timezone to format the date in or null for UTC
110         * @param date the date value
111         */
112        public void addDate(ICalDate date, boolean floating, TimeZone tz) {
113                if (date != null && date.hasTime() && !floating && tz != null) {
114                        dates.add(date);
115                }
116        }
117}
