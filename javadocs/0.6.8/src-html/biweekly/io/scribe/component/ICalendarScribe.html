036/**
037 * @author Michael Angstadt
038 */
039public class ICalendarScribe extends ICalComponentScribe<ICalendar> {
040        public ICalendarScribe() {
041                super(ICalendar.class, "VCALENDAR");
042        }
043
044        @Override
045        protected ICalendar _newInstance() {
046                return new ICalendar();
047        }
048
049        @Override
050        public List<ICalProperty> getProperties(ICalendar component) {
051                List<ICalProperty> properties = new ArrayList<ICalProperty>(component.getProperties().values());
052
053                /*
054                 * Move VERSION properties to the front (if any are present), followed
055                 * by PRODID properties. This is not required by the specs, but may help
056                 * with interoperability because all the examples in the specs put the
057                 * VERSION and PRODID at the very beginning of the iCalendar.
058                 */
059                moveToFront(ProductId.class, component, properties);
060                moveToFront(Version.class, component, properties);
061
062                return properties;
063        }
064
065        private <T extends ICalProperty> void moveToFront(Class<T> clazz, ICalendar component, List<ICalProperty> properties) {
066                List<T> toMove = component.getProperties(clazz);
067                properties.removeAll(toMove);
068                properties.addAll(0, toMove);
069        }
070}
