056/**
057 * @author Michael Angstadt
058 */
059public class VAlarmScribe extends ICalComponentScribe<VAlarm> {
060        public VAlarmScribe() {
061                super(VAlarm.class, "VALARM");
062        }
063
064        @Override
065        protected VAlarm _newInstance() {
066                return new VAlarm(null, null);
067        }
068
069        @Override
070        public void checkForDataModelConversions(VAlarm component, ICalComponent parent, ICalVersion version) {
071                if (version != ICalVersion.V1_0) {
072                        return;
073                }
074
075                VCalAlarmProperty vcalAlarm = convert(component, parent);
076                if (vcalAlarm == null) {
077                        return;
078                }
079
080                DataModelConversionException e = new DataModelConversionException(null);
081                e.getProperties().add(vcalAlarm);
082                throw e;
083        }
084
085        @Override
086        public Set<ICalVersion> getSupportedVersions() {
087                return EnumSet.of(ICalVersion.V2_0_DEPRECATED, ICalVersion.V2_0);
088        }
089
090        /**
091         * Converts a {@link VAlarm} component to a vCal alarm property.
092         * @param valarm the component
093         * @param parent the component's parent
094         * @return the vCal alarm property or null if it cannot be converted
095         */
096        private static VCalAlarmProperty convert(VAlarm valarm, ICalComponent parent) {
097                VCalAlarmProperty property = create(valarm);
098                if (property == null) {
099                        return null;
100                }
101
102                property.setStart(determineStartDate(valarm, parent));
103
104                DurationProperty duration = valarm.getDuration();
105                if (duration != null) {
106                        property.setSnooze(duration.getValue());
107                }
108
109                Repeat repeat = valarm.getRepeat();
110                if (repeat != null) {
111                        property.setRepeat(repeat.getValue());
112                }
113
114                return property;
115        }
116
117        /**
118         * Creates a new {@link VCalAlarmProperty} based on the given {@link VAlarm}
119         * component, setting fields that are common to all
120         * {@link VCalAlarmProperty} classes.
121         * @param valarm the source component
122         * @return the property or null if it cannot be created
123         */
124        private static VCalAlarmProperty create(VAlarm valarm) {
125                Action action = valarm.getAction();
126                if (action == null) {
127                        return null;
128                }
129
130                if (action.isAudio()) {
131                        return createAudioAlarm(valarm);
132                }
133
134                if (action.isDisplay()) {
135                        createDisplayAlarm(valarm);
136                }
137
138                if (action.isEmail()) {
139                        return createEmailAlarm(valarm);
140                }
141
142                if (action.isProcedure()) {
143                        return createProcedureAlarm(valarm);
144                }
145
146                return null;
147        }
148
149        private static AudioAlarm createAudioAlarm(VAlarm valarm) {
150                AudioAlarm aalarm = new AudioAlarm();
151
152                List<Attachment> attachments = valarm.getAttachments();
153                if (!attachments.isEmpty()) {
154                        Attachment firstAttachment = attachments.get(0);
155
156                        String formatType = firstAttachment.getFormatType();
157                        aalarm.setParameter("TYPE", formatType);
158
159                        byte[] data = firstAttachment.getData();
160                        if (data != null) {
161                                aalarm.setData(data);
162                        }
163
164                        String uri = firstAttachment.getUri();
165                        if (uri != null) {
166                                String contentId = StringUtils.afterPrefixIgnoreCase(uri, "cid:");
167                                if (contentId == null) {
168                                        aalarm.setUri(uri);
169                                } else {
170                                        aalarm.setContentId(contentId);
171                                }
172                        }
173                }
174
175                return aalarm;
176        }
177
178        private static DisplayAlarm createDisplayAlarm(VAlarm valarm) {
179                Description description = valarm.getDescription();
180                String text = ValuedProperty.getValue(description);
181                return new DisplayAlarm(text);
182        }
183
184        private static EmailAlarm createEmailAlarm(VAlarm valarm) {
185                List<Attendee> attendees = valarm.getAttendees();
186                String email = attendees.isEmpty() ? null : attendees.get(0).getEmail();
187                EmailAlarm malarm = new EmailAlarm(email);
188
189                Description description = valarm.getDescription();
190                String note = ValuedProperty.getValue(description);
191                malarm.setNote(note);
192
193                return malarm;
194        }
195
196        private static ProcedureAlarm createProcedureAlarm(VAlarm valarm) {
197                Description description = valarm.getDescription();
198                String path = ValuedProperty.getValue(description);
199                return new ProcedureAlarm(path);
200        }
201
202        /**
203         * Determines what the alarm property's start date should be.
204         * @param valarm the component that is being converted to a vCal alarm
205         * property
206         * @param parent the component's parent
207         * @return the start date or null if it cannot be determined
208         */
209        private static Date determineStartDate(VAlarm valarm, ICalComponent parent) {
210                Trigger trigger = valarm.getTrigger();
211                if (trigger == null) {
212                        return null;
213                }
214
215                Date triggerStart = trigger.getDate();
216                if (triggerStart != null) {
217                        return triggerStart;
218                }
219
220                Duration triggerDuration = trigger.getDuration();
221                if (triggerDuration == null) {
222                        return null;
223                }
224
225                if (parent == null) {
226                        return null;
227                }
228
229                Related related = trigger.getRelated();
230                Date date = null;
231                if (related == Related.START) {
232                        date = ValuedProperty.getValue(parent.getProperty(DateStart.class));
233                } else if (related == Related.END) {
234                        date = ValuedProperty.getValue(parent.getProperty(DateEnd.class));
235                        if (date == null) {
236                                Date dateStart = ValuedProperty.getValue(parent.getProperty(DateStart.class));
237                                Duration duration = ValuedProperty.getValue(parent.getProperty(DurationProperty.class));
238                                if (duration != null && dateStart != null) {
239                                        date = duration.add(dateStart);
240                                }
241                        }
242                }
243
244                return (date == null) ? null : triggerDuration.add(date);
245        }
