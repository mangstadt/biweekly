035/**
036 * Marshals {@link Attachment} properties.
037 * @author Michael Angstadt
038 */
039public class AttachmentScribe extends BinaryPropertyScribe<Attachment> {
040        public AttachmentScribe() {
041                super(Attachment.class, "ATTACH");
042        }
043
044        @Override
045        protected ICalDataType _dataType(Attachment property, ICalVersion version) {
046                if (property.getContentId() != null) {
047                        return (version == ICalVersion.V1_0) ? ICalDataType.CONTENT_ID : ICalDataType.URI;
048                }
049                return super._dataType(property, version);
050        }
051
052        @Override
053        protected Attachment newInstance(byte[] data) {
054                /*
055                 * Note: "formatType" will be set when the parameters are assigned to
056                 * the property object.
057                 */
058                return new Attachment(null, data);
059        }
060
061        @Override
062        protected Attachment newInstance(String value, ICalDataType dataType) {
063                /*
064                 * Note: "formatType" will be set when the parameters are assigned to
065                 * the property object.
066                 */
067
068                if (dataType == ICalDataType.CONTENT_ID) {
069                        String contentId = getCidUriValue(value);
070                        if (contentId == null) {
071                                contentId = value;
072                        }
073                        Attachment attach = new Attachment(null, (String) null);
074                        attach.setContentId(contentId);
075                        return attach;
076                }
077
078                String contentId = getCidUriValue(value);
079                if (contentId != null) {
080                        Attachment attach = new Attachment(null, (String) null);
081                        attach.setContentId(contentId);
082                        return attach;
083                }
084
085                return new Attachment(null, value);
086        }
087
088        @Override
089        protected String _writeText(Attachment property, WriteContext context) {
090                String contentId = property.getContentId();
091                if (contentId != null) {
092                        return (context.getVersion() == ICalVersion.V1_0) ? '<' + contentId + '>' : "cid:" + contentId;
093                }
094
095                return super._writeText(property, context);
096        }
097
098        @Override
099        protected void _writeXml(Attachment property, XCalElement element, WriteContext context) {
100                String contentId = property.getContentId();
101                if (contentId != null) {
102                        element.append(ICalDataType.URI, "cid:" + contentId);
103                        return;
104                }
105
106                super._writeXml(property, element, context);
107        }
108
109        @Override
110        protected JCalValue _writeJson(Attachment property, WriteContext context) {
111                String contentId = property.getContentId();
112                if (contentId != null) {
113                        return JCalValue.single("cid:" + contentId);
114                }
115
116                return super._writeJson(property, context);
117        }
118
119        /**
120         * Gets the value of the given "cid" URI.
121         * @param uri the "cid" URI
122         * @return the URI value or null if the given string is not a "cid" URI
123         */
124        private static String getCidUriValue(String uri) {
125                int colon = uri.indexOf(':');
126                if (colon == 3) {
127                        String scheme = uri.substring(0, colon);
128                        return "cid".equalsIgnoreCase(scheme) ? uri.substring(colon + 1) : null;
129                }
130
131                if (uri.length() > 0 && uri.charAt(0) == '<' && uri.charAt(uri.length() - 1) == '>') {
132                        return uri.substring(1, uri.length() - 1);
133                }
134
135                return null;
136        }
137}
