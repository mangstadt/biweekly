046/**
047 * Marshals {@link Daylight} properties.
048 * @author Michael Angstadt
049 */
050public class DaylightScribe extends ICalPropertyScribe<Daylight> {
051        public DaylightScribe() {
052                super(Daylight.class, "DAYLIGHT");
053        }
054
055        @Override
056        protected String _writeText(Daylight property, WriteContext context) {
057                if (!property.isDaylight()) {
058                        return "FALSE";
059                }
060
061                List<String> values = new ArrayList<String>();
062                values.add("TRUE");
063
064                UtcOffset offset = property.getOffset();
065                values.add((offset == null) ? "" : offset.toString());
066
067                ICalDate start = property.getStart();
068                values.add((start == null || start.getRawComponents() == null) ? "" : start.getRawComponents().toString(true, false));
069
070                ICalDate end = property.getEnd();
071                values.add((end == null || end.getRawComponents() == null) ? "" : end.getRawComponents().toString(true, false));
072
073                String standardName = property.getStandardName();
074                values.add((standardName == null) ? "" : standardName);
075
076                String daylightName = property.getDaylightName();
077                values.add((daylightName == null) ? "" : daylightName);
078
079                return VObjectPropertyValues.writeSemiStructured(values, false, true);
080        }
081
082        @Override
083        protected Daylight _parseText(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
084                SemiStructuredValueIterator it = new SemiStructuredValueIterator(value);
085
086                String next = it.next();
087                boolean flag = (next == null) ? false : Boolean.parseBoolean(next);
088
089                UtcOffset offset = null;
090                next = it.next();
091                if (next != null) {
092                        try {
093                                offset = UtcOffset.parse(next);
094                        } catch (IllegalArgumentException e) {
095                                throw new CannotParseException(33, next);
096                        }
097                }
098
099                ICalDate start = null;
100                next = it.next();
101                if (next != null) {
102                        try {
103                                start = date(next).parse();
104                        } catch (IllegalArgumentException e) {
105                                throw new CannotParseException(34, next);
106                        }
107                }
108
109                ICalDate end = null;
110                next = it.next();
111                if (next != null) {
112                        try {
113                                end = date(next).parse();
114                        } catch (IllegalArgumentException e) {
115                                throw new CannotParseException(35, next);
116                        }
117                }
118
119                String standardName = it.next();
120                String daylightName = it.next();
121
122                return new Daylight(flag, offset, start, end, standardName, daylightName);
123        }
124
125        @Override
126        public Set<ICalVersion> getSupportedVersions() {
127                return EnumSet.of(ICalVersion.V1_0);
128        }
129}
