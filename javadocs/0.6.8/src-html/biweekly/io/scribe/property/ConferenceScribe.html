039/**
040 * Marshals {@link Conference} properties.
041 * @author Michael Angstadt
042 */
043public class ConferenceScribe extends ICalPropertyScribe<Conference> {
044        public ConferenceScribe() {
045                super(Conference.class, "CONFERENCE", ICalDataType.URI);
046        }
047
048        @Override
049        protected String _writeText(Conference property, WriteContext context) {
050                return write(property);
051        }
052
053        @Override
054        protected Conference _parseText(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
055                value = VObjectPropertyValues.unescape(value);
056                return parse(value);
057        }
058
059        @Override
060        protected void _writeXml(Conference property, XCalElement element, WriteContext context) {
061                element.append(ICalDataType.URI, write(property));
062        }
063
064        @Override
065        protected Conference _parseXml(XCalElement element, ICalParameters parameters, ParseContext context) {
066                String uri = element.first(ICalDataType.URI);
067                if (uri != null) {
068                        return parse(uri);
069                }
070
071                throw missingXmlElements(ICalDataType.URI);
072        }
073
074        @Override
075        protected JCalValue _writeJson(Conference property, WriteContext context) {
076                return JCalValue.single(write(property));
077        }
078
079        @Override
080        protected Conference _parseJson(JCalValue value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
081                String uri = value.asSingle();
082                return parse(uri);
083        }
084
085        private static String write(Conference property) {
086                String uri = property.getUri();
087                if (uri != null) {
088                        return uri;
089                }
090
091                String text = property.getText();
092                if (text != null) {
093                        return new DataUri(text).toString();
094                }
095
096                return "";
097        }
098
099        private static Conference parse(String value) {
100                try {
101                        DataUri uri = DataUri.parse(value);
102                        String text = uri.getText();
103                        if (text != null) {
104                                Conference property = new Conference((String) null);
105                                property.setText(text);
106                                return property;
107                        }
108                } catch (IllegalArgumentException e) {
109                        //not a data URI
110                }
111
112                return new Conference(value);
113        }
114}
