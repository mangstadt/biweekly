092        @Override
093        protected String _writeText(T property, WriteContext context) {
094                //null value
095                Recurrence recur = property.getValue();
096                if (recur == null) {
097                        return "";
098                }
099
100                switch (context.getVersion()) {
101                case V1_0:
102                        return writeTextV1(property, context);
103                default:
104                        return writeTextV2(property, context);
105                }
106        }

108        private String writeTextV1(T property, WriteContext context) {
109                return new RecurrenceWriterV1(property, context).write();
110        }

112        private String writeTextV2(T property, WriteContext context) {
113                ListMultimap<String, Object> components = buildComponents(property, context, false);
114                return VObjectPropertyValues.writeMultimap(components.getMap());
115        }

117        @Override
118        protected T _parseText(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
119                if (value.isEmpty()) {
120                        return newInstance(new Recurrence.Builder((Frequency) null).build());
121                }
122
123                switch (context.getVersion()) {
124                case V1_0:
125                        return parseTextV1(value, dataType, parameters, context);
126                default:
127                        return parseTextV2(value, dataType, parameters, context);
128                }
129        }
130
131        /**
132         * @param value the property value
133         * @param dataType the property data type
134         * @param parameters the property parameters
135         * @param context the parse context
136         * @return the parsed property
137         * @throws DataModelConversionException if the property contains multiple
138         * RRULE values
139         */
140        private T parseTextV1(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
141                RecurrenceParserV1 parser = new RecurrenceParserV1(context);
142
143                List<String> rrules = RecurrenceParserV1.splitPropertyValue(value);
144                if (rrules.size() == 1) {
145                        Recurrence recur = parser.parse(value);
146                        return newInstance(recur, context, parameters);
147                }
148
149                /*
150                 * Version 1.0 allows multiple RRULE values to be defined inside of the
151                 * same property. If there are multiple values, parse them and throw a
152                 * DataModelConversionException.
153                 */
154                DataModelConversionException conversionException = new DataModelConversionException(null);
155                for (String rrule : rrules) {
156                        ICalParameters parametersCopy = new ICalParameters(parameters);
157
158                        ICalProperty property;
159                        try {
160                                Recurrence recur = parser.parse(rrule);
161                                property = newInstance(recur, context, parameters);
162                        } catch (CannotParseException e) {
163                                //@formatter:off
164                                context.getWarnings().add(new ParseWarning.Builder(context)
165                                        .message(e)
166                                        .build()
167                                );
168                                //@formatter:on
169                                property = new RawProperty(getPropertyName(context.getVersion()), dataType, rrule);
170                                property.setParameters(parametersCopy);
171                        }
172                        conversionException.getProperties().add(property);
173                }
174
175                throw conversionException;
176        }

178        private T parseTextV2(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
179                Recurrence.Builder builder = new Recurrence.Builder((Frequency) null);
180                ListMultimap<String, String> rules = new ListMultimap<String, String>(VObjectPropertyValues.parseMultimap(value));
181
182                parseFreq(rules, builder, context);
183                parseUntil(rules, builder, context);
184                parseCount(rules, builder, context);
185                parseInterval(rules, builder, context);
186                parseBySecond(rules, builder, context);
187                parseByMinute(rules, builder, context);
188                parseByHour(rules, builder, context);
189                parseByDay(rules, builder, context);
190                parseByMonthDay(rules, builder, context);
191                parseByYearDay(rules, builder, context);
192                parseByWeekNo(rules, builder, context);
193                parseByMonth(rules, builder, context);
194                parseBySetPos(rules, builder, context);
195                parseWkst(rules, builder, context);
196                parseXRules(rules, builder); //must be called last
197
198                return newInstance(builder.build(), context, parameters);
199        }

201        @Override
202        protected void _writeXml(T property, XCalElement element, WriteContext context) {
203                XCalElement recurElement = element.append(dataType(property, null));
204
205                Recurrence recur = property.getValue();
206                if (recur == null) {
207                        return;
208                }
209
210                ListMultimap<String, Object> components = buildComponents(property, context, true);
211                for (Map.Entry<String, List<Object>> component : components) {
212                        String name = component.getKey().toLowerCase();
213                        for (Object value : component.getValue()) {
214                                recurElement.append(name, value.toString());
215                        }
216                }
217        }

219        @Override
220        protected T _parseXml(XCalElement element, ICalParameters parameters, ParseContext context) {
221                ICalDataType dataType = defaultDataType(context.getVersion());
222                XCalElement value = element.child(dataType);
223                if (value == null) {
224                        throw missingXmlElements(dataType);
225                }
226
227                ListMultimap<String, String> rules = new ListMultimap<String, String>();
228                for (Element child : XmlUtils.toElementList(value.getElement().getChildNodes())) {
229                        if (!XCalNamespaceContext.XCAL_NS.equals(child.getNamespaceURI())) {
230                                continue;
231                        }
232
233                        String name = child.getLocalName().toUpperCase();
234                        String text = child.getTextContent();
235                        rules.put(name, text);
236                }
237
238                Recurrence.Builder builder = new Recurrence.Builder((Frequency) null);
239
240                parseFreq(rules, builder, context);
241                parseUntil(rules, builder, context);
242                parseCount(rules, builder, context);
243                parseInterval(rules, builder, context);
244                parseBySecond(rules, builder, context);
245                parseByMinute(rules, builder, context);
246                parseByHour(rules, builder, context);
247                parseByDay(rules, builder, context);
248                parseByMonthDay(rules, builder, context);
249                parseByYearDay(rules, builder, context);
250                parseByWeekNo(rules, builder, context);
251                parseByMonth(rules, builder, context);
252                parseBySetPos(rules, builder, context);
253                parseWkst(rules, builder, context);
254                parseXRules(rules, builder); //must be called last
255
256                T property = newInstance(builder.build());
257
258                ICalDate until = property.getValue().getUntil();
259                if (until != null) {
260                        context.addDate(until, property, parameters);
261                }
262
263                return property;
264        }

266        @Override
267        protected JCalValue _writeJson(T property, WriteContext context) {
268                Recurrence recur = property.getValue();
269                if (recur == null) {
270                        return JCalValue.object(new ListMultimap<String, Object>(0));
271                }
272
273                ListMultimap<String, Object> components = buildComponents(property, context, true);
274
275                //lower-case all the keys
276                ListMultimap<String, Object> object = new ListMultimap<String, Object>(components.keySet().size());
277                for (Map.Entry<String, List<Object>> entry : components) {
278                        String key = entry.getKey().toLowerCase();
279                        object.putAll(key, entry.getValue());
280                }
281
282                return JCalValue.object(object);
283        }

285        @Override
286        protected T _parseJson(JCalValue value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
287                Recurrence.Builder builder = new Recurrence.Builder((Frequency) null);
288
289                //upper-case the keys
290                ListMultimap<String, String> object = value.asObject();
291                ListMultimap<String, String> rules = new ListMultimap<String, String>(object.keySet().size());
292                for (Map.Entry<String, List<String>> entry : object) {
293                        String key = entry.getKey().toUpperCase();
294                        rules.putAll(key, entry.getValue());
295                }
296
297                parseFreq(rules, builder, context);
298                parseUntil(rules, builder, context);
299                parseCount(rules, builder, context);
300                parseInterval(rules, builder, context);
301                parseBySecond(rules, builder, context);
302                parseByMinute(rules, builder, context);
303                parseByHour(rules, builder, context);
304                parseByDay(rules, builder, context);
305                parseByMonthDay(rules, builder, context);
306                parseByYearDay(rules, builder, context);
307                parseByWeekNo(rules, builder, context);
308                parseByMonth(rules, builder, context);
309                parseBySetPos(rules, builder, context);
310                parseWkst(rules, builder, context);
311                parseXRules(rules, builder); //must be called last
312
313                T property = newInstance(builder.build());
314
315                ICalDate until = property.getValue().getUntil();
316                if (until != null) {
317                        context.addDate(until, property, parameters);
318                }
319
320                return property;
321        }

330        private T newInstance(Recurrence recur, ParseContext context, ICalParameters parameters) {
331                T property = newInstance(recur);
332
333                if (recur.getUntil() != null) {
334                        context.addDate(recur.getUntil(), property, parameters);
335                }
336
337                return property;
338        }

340        private void parseFreq(ListMultimap<String, String> rules, final Recurrence.Builder builder, final ParseContext context) {
341                parseFirst(rules, FREQ, new Handler<String>() {
342                        public void handle(String value) {
343                                value = value.toUpperCase();
344                                try {
345                                        builder.frequency(Frequency.valueOf(value));
346                                } catch (IllegalArgumentException e) {
347                                        context.addWarning(7, FREQ, value);
348                                }
349                        }
350                });
351        }

354                parseFirst(rules, UNTIL, new Handler<String>() {

356                                try {
357                                        builder.until(date(value).parse());
358                                } catch (IllegalArgumentException e) {
359                                        context.addWarning(7, UNTIL, value);
360                                }
361                        }
362                });
363        }

413        private void parseByDay(ListMultimap<String, String> rules, Recurrence.Builder builder, ParseContext context) {
414                Pattern p = Pattern.compile("^([-+]?\\d+)?(.*)$");
415                for (String value : rules.removeAll(BYDAY)) {
416                        Matcher m = p.matcher(value);
417                        if (!m.find()) {
418                                //this should never happen
419                                //the regex contains a "match-all" pattern and should never not find anything
420                                context.addWarning(7, BYDAY, value);
421                                continue;
422                        }
423
424                        String dayStr = m.group(2);
425                        DayOfWeek day = DayOfWeek.valueOfAbbr(dayStr);
426                        if (day == null) {
427                                context.addWarning(7, BYDAY, value);
428                                continue;
429                        }
430
431                        String prefixStr = m.group(1);
432                        Integer prefix = (prefixStr == null) ? null : Integer.valueOf(prefixStr); //no need to catch NumberFormatException because the regex guarantees that it will be a number
433
434                        builder.byDay(prefix, day);
435                }
436        }

501        private ListMultimap<String, Object> buildComponents(T property, WriteContext context, boolean extended) {
502                ListMultimap<String, Object> components = new ListMultimap<String, Object>();
503                Recurrence recur = property.getValue();
504
505                //FREQ must come first
506                if (recur.getFrequency() != null) {
507                        components.put(FREQ, recur.getFrequency().name());
508                }
509
510                ICalDate until = recur.getUntil();
511                if (until != null) {
512                        components.put(UNTIL, writeUntil(until, context, extended));
513                }
514
515                if (recur.getCount() != null) {
516                        components.put(COUNT, recur.getCount());
517                }
518
519                if (recur.getInterval() != null) {
520                        components.put(INTERVAL, recur.getInterval());
521                }
522
523                components.putAll(BYSECOND, recur.getBySecond());
524                components.putAll(BYMINUTE, recur.getByMinute());
525                components.putAll(BYHOUR, recur.getByHour());
526
527                for (ByDay byDay : recur.getByDay()) {
528                        Integer prefix = byDay.getNum();
529                        DayOfWeek day = byDay.getDay();
530
531                        String value = day.getAbbr();
532                        if (prefix != null) {
533                                value = prefix + value;
534                        }
535                        components.put(BYDAY, value);
536                }
537
538                components.putAll(BYMONTHDAY, recur.getByMonthDay());
539                components.putAll(BYYEARDAY, recur.getByYearDay());
540                components.putAll(BYWEEKNO, recur.getByWeekNo());
541                components.putAll(BYMONTH, recur.getByMonth());
542                components.putAll(BYSETPOS, recur.getBySetPos());
543
544                if (recur.getWorkweekStarts() != null) {
545                        components.put(WKST, recur.getWorkweekStarts().getAbbr());
546                }
547
548                for (Map.Entry<String, List<String>> entry : recur.getXRules().entrySet()) {
549                        String name = entry.getKey();
550                        List<String> values = entry.getValue();
551                        components.putAll(name, values);
552                }
553
554                return components;
555        }

557        private String writeUntil(ICalDate until, WriteContext context, boolean extended) {
558                if (!until.hasTime()) {
559                        return date(until).extended(extended).write();
560                }
561
562                /*
563                 * RFC 5545 p.41
564                 * 
565                 * In the case of the "STANDARD" and "DAYLIGHT" sub-components the UNTIL
566                 * rule part MUST always be specified as a date with UTC time. If
567                 * specified as a DATE-TIME value, then it MUST be specified in a UTC
568                 * time format.
569                 */
570
571                if (isInObservance(context)) {
572                        return date(until).utc(true).extended(extended).write();
573                }
574
575                /*
576                 * RFC 2445 p.42
577                 * 
578                 * If specified as a date-time value, then it MUST be specified in an
579                 * UTC time format.
580                 */
581                if (context.getVersion() == ICalVersion.V2_0_DEPRECATED) {
582                        return date(until).extended(extended).utc(true).write();
583                }
584
585                /*
586                 * RFC 5545 p.41
587                 * 
588                 * Furthermore, if the "DTSTART" property is specified as a date with
589                 * local time, then the UNTIL rule part MUST also be specified as a date
590                 * with local time. If the "DTSTART" property is specified as a date
591                 * with UTC time or a date with local time and time zone reference, then
592                 * the UNTIL rule part MUST be specified as a date with UTC time.
593                 */
594
595                ICalComponent parent = context.getParent();
596                if (parent == null) {
597                        return date(until).extended(extended).utc(true).write();
598                }
599
600                DateStart dtstart = parent.getProperty(DateStart.class);
601                if (dtstart == null) {
602                        return date(until).extended(extended).utc(true).write();
603                }
604
605                /*
606                 * If DTSTART is floating, then UNTIL should be floating.
607                 */
608                TimezoneInfo tzinfo = context.getTimezoneInfo();
609                boolean dtstartFloating = tzinfo.isFloating(dtstart);
610                if (dtstartFloating) {
611                        return date(until).extended(extended).tz(true, null).write();
612                }
613
614                /*
615                 * Otherwise, UNTIL should be UTC.
616                 */
617                return date(until).extended(extended).utc(true).write();
618        }

620        private void parseFirst(ListMultimap<String, String> rules, String name, Handler<String> handler) {
621                List<String> values = rules.removeAll(name);
622                if (values.isEmpty()) {
623                        return;
624                }
625
626                String value = values.get(0);
627                handler.handle(value);
628        }

630        private void parseIntegerList(String name, ListMultimap<String, String> rules, ParseContext context, Handler<Integer> handler) {
631                List<String> values = rules.removeAll(name);
632                for (String value : values) {
633                        try {
634                                handler.handle(Integer.valueOf(value));
635                        } catch (NumberFormatException e) {
636                                context.addWarning(8, name, value);
637                        }
638                }
639        }
