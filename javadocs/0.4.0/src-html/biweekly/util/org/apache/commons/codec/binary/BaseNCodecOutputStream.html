018package biweekly.util.org.apache.commons.codec.binary;
019
020import java.io.FilterOutputStream;
021import java.io.IOException;
022import java.io.OutputStream;
023
024/**
025 * Abstract superclass for Base-N output streams.
026 * 
027 * @since 1.5
028 */
029public class BaseNCodecOutputStream extends FilterOutputStream {
030
031    private final boolean doEncode;
032
033    private final BaseNCodec baseNCodec;
034
035    private final byte[] singleByte = new byte[1];
036
037    public BaseNCodecOutputStream(OutputStream out, BaseNCodec basedCodec, boolean doEncode) {
038        super(out);
039        this.baseNCodec = basedCodec;
040        this.doEncode = doEncode;
041    }
042
043    /**
044     * Writes the specified <code>byte</code> to this output stream.
045     * 
046     * @param i
047     *            source byte
048     * @throws IOException
049     *             if an I/O error occurs.
050     */
051    @Override
052    public void write(int i) throws IOException {
053        singleByte[0] = (byte) i;
054        write(singleByte, 0, 1);
055    }
056
057    /**
058     * Writes <code>len</code> bytes from the specified <code>b</code> array starting at <code>offset</code> to this
059     * output stream.
060     * 
061     * @param b
062     *            source byte array
063     * @param offset
064     *            where to start reading the bytes
065     * @param len
066     *            maximum number of bytes to write
067     * 
068     * @throws IOException
069     *             if an I/O error occurs.
070     * @throws NullPointerException
071     *             if the byte array parameter is null
072     * @throws IndexOutOfBoundsException
073     *             if offset, len or buffer size are invalid
074     */
075    @Override
076    public void write(byte b[], int offset, int len) throws IOException {
077        if (b == null) {
078            throw new NullPointerException();
079        } else if (offset < 0 || len < 0) {
080            throw new IndexOutOfBoundsException();
081        } else if (offset > b.length || offset + len > b.length) {
082            throw new IndexOutOfBoundsException();
083        } else if (len > 0) {
084            if (doEncode) {
085                baseNCodec.encode(b, offset, len);
086            } else {
087                baseNCodec.decode(b, offset, len);
088            }
089            flush(false);
090        }
091    }
092
093    /**
094     * Flushes this output stream and forces any buffered output bytes to be written out to the stream. If propogate is
095     * true, the wrapped stream will also be flushed.
096     * 
097     * @param propogate
098     *            boolean flag to indicate whether the wrapped OutputStream should also be flushed.
099     * @throws IOException
100     *             if an I/O error occurs.
101     */
102    private void flush(boolean propogate) throws IOException {
103        int avail = baseNCodec.available();
104        if (avail > 0) {
105            byte[] buf = new byte[avail];
106            int c = baseNCodec.readResults(buf, 0, avail);
107            if (c > 0) {
108                out.write(buf, 0, c);
109            }
110        }
111        if (propogate) {
112            out.flush();
113        }
114    }
115
116    /**
117     * Flushes this output stream and forces any buffered output bytes to be written out to the stream.
118     * 
119     * @throws IOException
120     *             if an I/O error occurs.
121     */
122    @Override
123    public void flush() throws IOException {
124        flush(true);
125    }
126
127    /**
128     * Closes this output stream and releases any system resources associated with the stream.
129     * 
130     * @throws IOException
131     *             if an I/O error occurs.
132     */
133    @Override
134    public void close() throws IOException {
135        // Notify encoder of EOF (-1).
136        if (doEncode) {
137            baseNCodec.encode(singleByte, 0, -1);
138        } else {
139            baseNCodec.decode(singleByte, 0, -1);
140        }
141        flush();
142        out.close();
143    }
144
145}
