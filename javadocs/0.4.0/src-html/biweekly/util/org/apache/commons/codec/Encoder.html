018package biweekly.util.org.apache.commons.codec;
019
020/**
021 * <p>Provides the highest level of abstraction for Encoders.
022 * This is the sister interface of {@link Decoder}.  Every implementation of
023 * Encoder provides this common generic interface which allows a user to pass a 
024 * generic Object to any Encoder implementation in the codec package.</p>
025 *
026 * @author Apache Software Foundation
027 * @version $Id: Encoder.java 1170351 2011-09-13 21:09:09Z ggregory $
028 */
029public interface Encoder {
030    
031    /**
032     * Encodes an "Object" and returns the encoded content 
033     * as an Object.  The Objects here may just be <code>byte[]</code>
034     * or <code>String</code>s depending on the implementation used.
035     *   
036     * @param source An object to encode
037     * 
038     * @return An "encoded" Object
039     * 
040     * @throws EncoderException an encoder exception is
041     *  thrown if the encoder experiences a failure
042     *  condition during the encoding process.
043     */
044    Object encode(Object source) throws EncoderException;
045}
