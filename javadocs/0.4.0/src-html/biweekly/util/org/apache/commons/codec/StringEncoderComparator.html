018package biweekly.util.org.apache.commons.codec;
019
020import java.util.Comparator;
021
022/**
023 * Compares Strings using a {@link StringEncoder}. This comparator is used to sort Strings by an encoding scheme such as
024 * Soundex, Metaphone, etc. This class can come in handy if one need to sort Strings by an encoded form of a name such
025 * as Soundex.
026 * 
027 * @author Apache Software Foundation
028 * @version $Id: StringEncoderComparator.java 1201520 2011-11-13 21:36:18Z ggregory $
029 */
030@SuppressWarnings("rawtypes")
031public class StringEncoderComparator implements Comparator {
032
033    /**
034     * Internal encoder instance.
035     */
036    private final StringEncoder stringEncoder;
037
038    /**
039     * Constructs a new instance.
040     * 
041     * @deprecated Creating an instance without a {@link StringEncoder} leads to a {@link NullPointerException}. Will be
042     *             removed in 2.0.
043     */
044    public StringEncoderComparator() {
045        this.stringEncoder = null; // Trying to use this will cause things to break
046    }
047
048    /**
049     * Constructs a new instance with the given algorithm.
050     * 
051     * @param stringEncoder
052     *            the StringEncoder used for comparisons.
053     */
054    public StringEncoderComparator(StringEncoder stringEncoder) {
055        this.stringEncoder = stringEncoder;
056    }
057
058    /**
059     * Compares two strings based not on the strings themselves, but on an encoding of the two strings using the
060     * StringEncoder this Comparator was created with.
061     * 
062     * If an {@link EncoderException} is encountered, return <code>0</code>.
063     * 
064     * @param o1
065     *            the object to compare
066     * @param o2
067     *            the object to compare to
068     * @return the Comparable.compareTo() return code or 0 if an encoding error was caught.
069     * @see Comparable
070     */
071    @SuppressWarnings("unchecked")
072    public int compare(Object o1, Object o2) {
073
074        int compareCode = 0;
075
076        try {
077            Comparable s1 = (Comparable) this.stringEncoder.encode(o1);
078            Comparable s2 = (Comparable) this.stringEncoder.encode(o2);
079            compareCode = s1.compareTo(s2);
080        } catch (EncoderException ee) {
081            compareCode = 0;
082        }
083        return compareCode;
084    }
085
086}
