064/**
065 * <p>
066 * Parses {@link ICalendar} objects from a jCal data stream (JSON).
067 * </p>
068 * <p>
069 * <b>Example:</b>
070 * </p>
071 * 
072 * <pre class="brush:java">
073 * File file = new File("icals.json");
074 * JCalReader reader = null;
075 * try {
076 *   reader = new JCalReader(file);
077 *   ICalendar ical;
078 *   while ((ical = reader.readNext()) != null) {
079 *     //...
080 *   }
081 * } finally {
082 *   if (reader != null) reader.close();
083 * }
084 * </pre>
085 * @author Michael Angstadt
086 * @see <a href="http://tools.ietf.org/html/rfc7265">RFC 7265</a>
087 */
088public class JCalReader extends StreamReader {
089        private static final ICalendarScribe icalScribe = ScribeIndex.getICalendarScribe();
090        private final JCalRawReader reader;
091
092        /**
093         * @param json the JSON string to read from
094         */
095        public JCalReader(String json) {
096                this(new StringReader(json));
097        }
098
099        /**
100         * @param in the input stream to read from
101         */
102        public JCalReader(InputStream in) {
103                this(new Utf8Reader(in));
104        }
105
106        /**
107         * @param file the file to read from
108         * @throws FileNotFoundException if the file doesn't exist
109         */
110        public JCalReader(File file) throws FileNotFoundException {
111                this(new BufferedReader(new Utf8Reader(file)));
112        }
113
114        /**
115         * @param reader the reader to read from
116         */
117        public JCalReader(Reader reader) {
118                this.reader = new JCalRawReader(reader);
119        }
120
121        /**
122         * @param parser the parser to read from
123         */
124        public JCalReader(JsonParser parser) {
125                this.reader = new JCalRawReader(parser, true);
126        }
127
128        /**
129         * Reads the next iCalendar object from the JSON data stream.
130         * @return the iCalendar object or null if there are no more
131         * @throws JCalParseException if the jCal syntax is incorrect (the JSON
132         * syntax may be valid, but it is not in the correct jCal format).
133         * @throws JsonParseException if the JSON syntax is incorrect
134         * @throws IOException if there is a problem reading from the data stream
135         */
136        @Override
137        public ICalendar _readNext() throws IOException {
138                if (reader.eof()) {
139                        return null;
140                }
141
142                context.setVersion(ICalVersion.V2_0);
143
144                JCalDataStreamListenerImpl listener = new JCalDataStreamListenerImpl();
145                reader.readNext(listener);
146
147                return listener.getICalendar();
148        }
149
150        //@Override
151        public void close() throws IOException {
152                reader.close();
153        }
154
155        private class JCalDataStreamListenerImpl implements JCalDataStreamListener {
156                private final Map<List<String>, ICalComponent> components = new HashMap<List<String>, ICalComponent>();
157
158                public void readProperty(List<String> componentHierarchy, String propertyName, ICalParameters parameters, ICalDataType dataType, JCalValue value) {
159                        context.getWarnings().clear();
160                        context.setLineNumber(reader.getLineNum());
161                        context.setPropertyName(propertyName);
162
163                        //get the component that the property belongs to
164                        ICalComponent parent = components.get(componentHierarchy);
165
166                        //unmarshal the property
167                        ICalPropertyScribe<? extends ICalProperty> scribe = index.getPropertyScribe(propertyName, ICalVersion.V2_0);
168                        try {
169                                ICalProperty property = scribe.parseJson(value, dataType, parameters, context);
170                                warnings.addAll(context.getWarnings());
171
172                                //set "ICalendar.version" if the value of the VERSION property is recognized
173                                //otherwise, unmarshal VERSION like a normal property
174                                if (parent instanceof ICalendar && property instanceof Version) {
175                                        Version version = (Version) property;
176                                        ICalVersion icalVersion = version.toICalVersion();
177                                        if (icalVersion != null) {
178                                                context.setVersion(icalVersion);
179                                                return;
180                                        }
181                                }
182
183                                parent.addProperty(property);
184                        } catch (SkipMeException e) {
185                                //@formatter:off
186                                warnings.add(new ParseWarning.Builder(context)
187                                        .message(0, e.getMessage())
188                                        .build()
189                                );
190                                //@formatter:on
191                        } catch (CannotParseException e) {
192                                RawProperty property = new RawPropertyScribe(propertyName).parseJson(value, dataType, parameters, context);
193                                parent.addProperty(property);
194
195                                //@formatter:off
196                                warnings.add(new ParseWarning.Builder(context)
197                                        .message(e)
198                                        .build()
199                                );
200                                //@formatter:on
201                        }
202                }
203
204                public void readComponent(List<String> parentHierarchy, String componentName) {
205                        ICalComponentScribe<? extends ICalComponent> scribe = index.getComponentScribe(componentName, ICalVersion.V2_0);
206                        ICalComponent component = scribe.emptyInstance();
207
208                        ICalComponent parent = components.get(parentHierarchy);
209                        if (parent != null) {
210                                parent.addComponent(component);
211                        }
212
213                        List<String> hierarchy = new ArrayList<String>(parentHierarchy);
214                        hierarchy.add(componentName);
215                        components.put(hierarchy, component);
216                }
217
218                public ICalendar getICalendar() {
219                        if (components.isEmpty()) {
220                                //EOF
221                                return null;
222                        }
223
224                        ICalComponent component = components.get(Collections.singletonList(icalScribe.getComponentName().toLowerCase()));
225                        if (component == null) {
226                                //should never happen because the parser always looks for a "vcalendar" component
227                                return null;
228                        }
229
230                        if (component instanceof ICalendar) {
231                                //should happen every time
232                                return (ICalendar) component;
233                        }
234
235                        //this will only happen if the user decides to override the ICalendarScribe for some reason
236                        ICalendar ical = icalScribe.emptyInstance();
237                        ical.addComponent(component);
238                        return ical;
239                }
240        }
