045/**
046 * Writes data to an iCalendar JSON data stream (jCal).
047 * @author Michael Angstadt
048 * @see <a href="http://tools.ietf.org/html/rfc7265">RFC 7265</a>
049 */
050public class JCalRawWriter implements Closeable, Flushable {
051        private final Writer writer;
052        private final boolean wrapInArray;
053        private final LinkedList<Info> stack = new LinkedList<Info>();
054        private JsonGenerator generator;
055        private boolean prettyPrint = false;
056        private boolean componentEnded = false;
057        private boolean closeGenerator = true;
058        private PrettyPrinter prettyPrinter;
059
060        /**
061         * @param writer the writer to wrap
062         * @param wrapInArray true to wrap everything in an array, false not to
063         * (useful when writing more than one iCalendar object)
064         */
065        public JCalRawWriter(Writer writer, boolean wrapInArray) {
066                this.writer = writer;
067                this.wrapInArray = wrapInArray;
068        }
069
070        /**
071         * @param generator the generator to write to
072         */
073        public JCalRawWriter(JsonGenerator generator) {
074                this.writer = null;
075                this.generator = generator;
076                this.closeGenerator = false;
077                this.wrapInArray = false;
078        }
079
080        /**
081         * Gets whether or not the JSON will be pretty-printed.
082         * @return true if it will be pretty-printed, false if not (defaults to
083         * false)
084         */
085        public boolean isPrettyPrint() {
086                return prettyPrint;
087        }
088
089        /**
090         * Sets whether or not to pretty-print the JSON.
091         * @param prettyPrint true to pretty-print it, false not to (defaults to
092         * false)
093         */
094        public void setPrettyPrint(boolean prettyPrint) {
095                this.prettyPrint = prettyPrint;
096        }
097
098        /**
099         * Sets the pretty printer to pretty-print the JSON with. Note that this
100         * method implicitly enables indenting, so {@code setPrettyPrint(true)} does
101         * not also need to be called.
102         * @param prettyPrinter the custom pretty printer (defaults to an instance
103         * of {@link JCalPrettyPrinter}, if {@code setPrettyPrint(true)} has been
104         * called)
105         */
106        public void setPrettyPrinter(PrettyPrinter prettyPrinter) {
107                prettyPrint = true;
108                this.prettyPrinter = prettyPrinter;
109        }
110
111        /**
112         * Writes the beginning of a new component array.
113         * @param componentName the component name (e.g. "vevent")
114         * @throws IOException if there's an I/O problem
115         */
116        public void writeStartComponent(String componentName) throws IOException {
117                if (generator == null) {
118                        init();
119                }
120
121                componentEnded = false;
122
123                if (!stack.isEmpty()) {
124                        Info parent = stack.getLast();
125                        if (!parent.wroteEndPropertiesArray) {
126                                generator.writeEndArray();
127                                parent.wroteEndPropertiesArray = true;
128                        }
129                        if (!parent.wroteStartSubComponentsArray) {
130                                generator.writeStartArray();
131                                parent.wroteStartSubComponentsArray = true;
132                        }
133                }
134
135                generator.writeStartArray();
136                generator.writeString(componentName);
137                generator.writeStartArray(); //start properties array
138
139                stack.add(new Info());
140        }
141
142        /**
143         * Closes the current component array.
144         * @throws IllegalStateException if there are no open components (
145         * {@link #writeStartComponent(String)} must be called first)
146         * @throws IOException if there's an I/O problem
147         */
148        public void writeEndComponent() throws IOException {
149                if (stack.isEmpty()) {
150                        throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(2));
151                }
152                Info cur = stack.removeLast();
153
154                if (!cur.wroteEndPropertiesArray) {
155                        generator.writeEndArray();
156                }
157                if (!cur.wroteStartSubComponentsArray) {
158                        generator.writeStartArray();
159                }
160
161                generator.writeEndArray(); //end sub-components array
162                generator.writeEndArray(); //end the array of this component
163
164                componentEnded = true;
165        }
166
167        /**
168         * Writes a property to the current component.
169         * @param propertyName the property name (e.g. "version")
170         * @param dataType the property's data type (e.g. "text")
171         * @param value the property value
172         * @throws IllegalStateException if there are no open components (
173         * {@link #writeStartComponent(String)} must be called first) or if the last
174         * method called was {@link #writeEndComponent()}.
175         * @throws IOException if there's an I/O problem
176         */
177        public void writeProperty(String propertyName, ICalDataType dataType, JCalValue value) throws IOException {
178                writeProperty(propertyName, new ICalParameters(), dataType, value);
179        }
180
181        /**
182         * Writes a property to the current component.
183         * @param propertyName the property name (e.g. "version")
184         * @param parameters the parameters
185         * @param dataType the property's data type (e.g. "text")
186         * @param value the property value
187         * @throws IllegalStateException if there are no open components (
188         * {@link #writeStartComponent(String)} must be called first) or if the last
189         * method called was {@link #writeEndComponent()}.
190         * @throws IOException if there's an I/O problem
191         */
192        public void writeProperty(String propertyName, ICalParameters parameters, ICalDataType dataType, JCalValue value) throws IOException {
193                if (stack.isEmpty()) {
194                        throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(2));
195                }
196                if (componentEnded) {
197                        throw new IllegalStateException(Messages.INSTANCE.getExceptionMessage(3));
198                }
199
200                generator.setCurrentValue(JCalPrettyPrinter.PROPERTY_VALUE);
201
202                generator.writeStartArray();
203
204                //write the property name
205                generator.writeString(propertyName);
206
207                //write parameters
208                generator.writeStartObject();
209                for (Map.Entry<String, List<String>> entry : parameters) {
210                        String name = entry.getKey().toLowerCase();
211                        List<String> values = entry.getValue();
212                        if (values.isEmpty()) {
213                                continue;
214                        }
215
216                        if (values.size() == 1) {
217                                generator.writeStringField(name, values.get(0));
218                        } else {
219                                generator.writeArrayFieldStart(name);
220                                for (String paramValue : values) {
221                                        generator.writeString(paramValue);
222                                }
223                                generator.writeEndArray();
224                        }
225                }
226                generator.writeEndObject();
227
228                //write data type
229                generator.writeString((dataType == null) ? "unknown" : dataType.getName().toLowerCase());
230
231                //write value
232                for (JsonValue jsonValue : value.getValues()) {
233                        writeValue(jsonValue);
234                }
235
236                generator.writeEndArray();
237
238                generator.setCurrentValue(null);
239        }
240
241        private void writeValue(JsonValue jsonValue) throws IOException {
242                if (jsonValue.isNull()) {
243                        generator.writeNull();
244                        return;
245                }
246
247                Object val = jsonValue.getValue();
248                if (val != null) {
249                        if (val instanceof Byte) {
250                                generator.writeNumber((Byte) val);
251                        } else if (val instanceof Short) {
252                                generator.writeNumber((Short) val);
253                        } else if (val instanceof Integer) {
254                                generator.writeNumber((Integer) val);
255                        } else if (val instanceof Long) {
256                                generator.writeNumber((Long) val);
257                        } else if (val instanceof Float) {
258                                generator.writeNumber((Float) val);
259                        } else if (val instanceof Double) {
260                                generator.writeNumber((Double) val);
261                        } else if (val instanceof Boolean) {
262                                generator.writeBoolean((Boolean) val);
263                        } else {
264                                generator.writeString(val.toString());
265                        }
266                        return;
267                }
268
269                List<JsonValue> array = jsonValue.getArray();
270                if (array != null) {
271                        generator.writeStartArray();
272                        for (JsonValue element : array) {
273                                writeValue(element);
274                        }
275                        generator.writeEndArray();
276                        return;
277                }
278
279                Map<String, JsonValue> object = jsonValue.getObject();
280                if (object != null) {
281                        generator.writeStartObject();
282                        for (Map.Entry<String, JsonValue> entry : object.entrySet()) {
283                                generator.writeFieldName(entry.getKey());
284                                writeValue(entry.getValue());
285                        }
286                        generator.writeEndObject();
287                        return;
288                }
289        }
290
291        /**
292         * Flushes the JSON stream.
293         * @throws IOException if there's a problem flushing the stream
294         */
295        public void flush() throws IOException {
296                if (generator == null) {
297                        return;
298                }
299
300                generator.flush();
301        }
302
303        /**
304         * Finishes writing the JSON document so that it is syntactically correct.
305         * No more data can be written once this method is called.
306         * @throws IOException if there's a problem closing the stream
307         */
308        public void closeJsonStream() throws IOException {
309                if (generator == null) {
310                        return;
311                }
312
313                while (!stack.isEmpty()) {
314                        writeEndComponent();
315                }
316
317                if (wrapInArray) {
318                        generator.writeEndArray();
319                }
320
321                if (closeGenerator) {
322                        generator.close();
323                }
324        }
325
326        /**
327         * Finishes writing the JSON document and closes the underlying
328         * {@link Writer}.
329         * @throws IOException if there's a problem closing the stream
330         */
331        public void close() throws IOException {
332                if (generator == null) {
333                        return;
334                }
335
336                closeJsonStream();
337
338                if (writer != null) {
339                        writer.close();
340                }
341        }
342
343        private void init() throws IOException {
344                JsonFactory factory = new JsonFactory();
345                factory.configure(Feature.AUTO_CLOSE_TARGET, false);
346                generator = factory.createGenerator(writer);
347
348                if (prettyPrint) {
349                        if (prettyPrinter == null) {
350                                prettyPrinter = new JCalPrettyPrinter();
351                        }
352                        generator.setPrettyPrinter(prettyPrinter);
353                }
354
355                if (wrapInArray) {
356                        generator.writeStartArray();
357                }
358        }
359
360        private static class Info {
361                public boolean wroteEndPropertiesArray = false;
362                public boolean wroteStartSubComponentsArray = false;
363        }
