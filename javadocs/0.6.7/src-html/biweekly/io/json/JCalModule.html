038/**
039 * <p>
040 * Module for jackson-databind that serializes and deserializes jCals.
041 * </p>
042 * <p>
043 * <b>Example:</b>
044 * </p>
045 * 
046 * <pre class="brush:java">
047 * ObjectMapper mapper = new ObjectMapper();
048 * mapper.registerModule(new JCalModule());
049 * ICalendar result = mapper.readValue(..., ICalendar.class);
050 * </pre>
051 * @author Buddy Gorven
052 * @author Michael Angstadt
053 */
054public class JCalModule extends SimpleModule {
055        private static final long serialVersionUID = 8022429868572303471L;
056        private static final String MODULE_NAME = "biweekly-jcal";
057        private static final Version MODULE_VERSION = moduleVersion();
058
059        private final JCalDeserializer deserializer = new JCalDeserializer();
060        private final JCalSerializer serializer = new JCalSerializer();
061
062        private ScribeIndex index;
063
064        /**
065         * Creates the module.
066         */
067        public JCalModule() {
068                super(MODULE_NAME, MODULE_VERSION);
069
070                setScribeIndex(new ScribeIndex());
071                addSerializer(serializer);
072                addDeserializer(ICalendar.class, deserializer);
073        }
074
075        private static Version moduleVersion() {
076                String[] split = Biweekly.VERSION.split("[.-]");
077                if (split.length < 3) {
078                        /*
079                         * This can happen during development if the "biweekly.properties"
080                         * file has not been filtered by Maven.
081                         */
082                        return new Version(0, 0, 0, "", Biweekly.GROUP_ID, Biweekly.ARTIFACT_ID);
083                }
084
085                int major = Integer.parseInt(split[0]);
086                int minor = Integer.parseInt(split[1]);
087                int patch = Integer.parseInt(split[2]);
088                String snapshot = (split.length > 3) ? split[3] : "RELEASE";
089
090                return new Version(major, minor, patch, snapshot, Biweekly.GROUP_ID, Biweekly.ARTIFACT_ID);
091        }
092
093        /**
094         * <p>
095         * Registers a property scribe. This is the same as calling:
096         * </p>
097         * <p>
098         * {@code getScribeIndex().register(scribe)}
099         * </p>
100         * @param scribe the scribe to register
101         */
102        public void registerScribe(ICalPropertyScribe<? extends ICalProperty> scribe) {
103                index.register(scribe);
104        }
105
106        /**
107         * Gets the scribe index used by the serializer and deserializer.
108         * @return the scribe index
109         */
110        public ScribeIndex getScribeIndex() {
111                return index;
112        }
113
114        /**
115         * Sets the scribe index for the serializer and deserializer to use.
116         * @param index the scribe index
117         */
118        public void setScribeIndex(ScribeIndex index) {
119                this.index = index;
120                serializer.setScribeIndex(index);
121                deserializer.setScribeIndex(index);
122        }
123
124        /**
125         * Gets the timezone that all date/time property values will be formatted
126         * in. If set, this setting will override the timezone information
127         * associated with each {@link ICalendar} object.
128         * @return the global timezone or null if not set (defaults to null)
129         */
130        public TimezoneAssignment getGlobalTimezone() {
131                return serializer.getGlobalTimezone();
132        }
133
134        /**
135         * Sets the timezone that all date/time property values will be formatted
136         * in. This is a convenience method that overrides the timezone information
137         * associated with each {@link ICalendar} object that is passed into this
138         * writer.
139         * @param globalTimezone the global timezone or null not to set a global
140         * timezone (defaults to null)
141         */
142        public void setGlobalTimezone(TimezoneAssignment globalTimezone) {
143                serializer.setGlobalTimezone(globalTimezone);
144        }
145}
