044/**
045 * Chainer class for writing jCals (JSON-encoded iCalendar objects).
046 * @see Biweekly#writeJson(Collection)
047 * @see Biweekly#writeJson(ICalendar...)
048 * @author Michael Angstadt
049 */
050public class ChainingJsonWriter extends ChainingWriter<ChainingJsonWriter> {
051        private boolean prettyPrint = false;
052
053        /**
054         * @param icals the iCalendar objects to write
055         */
056        public ChainingJsonWriter(Collection<ICalendar> icals) {
057                super(icals);
058        }
059
060        /**
061         * Sets whether or not to pretty-print the JSON.
062         * @param prettyPrint true to pretty-print it, false not to (defaults to
063         * false)
064         * @return this
065         */
066        public ChainingJsonWriter prettyPrint(boolean prettyPrint) {
067                this.prettyPrint = prettyPrint;
068                return this;
069        }
070
071        @Override
072        public ChainingJsonWriter tz(TimeZone defaultTimeZone, boolean outlookCompatible) {
073                return super.tz(defaultTimeZone, outlookCompatible);
074        }
075
076        @Override
077        public ChainingJsonWriter register(ICalPropertyScribe<? extends ICalProperty> scribe) {
078                return super.register(scribe);
079        }
080
081        @Override
082        public ChainingJsonWriter register(ICalComponentScribe<? extends ICalComponent> scribe) {
083                return super.register(scribe);
084        }
085
086        /**
087         * Writes the iCalendar objects to a string.
088         * @return the JSON string
089         */
090        public String go() {
091                StringWriter sw = new StringWriter();
092                try {
093                        go(sw);
094                } catch (IOException e) {
095                        //should never be thrown because we're writing to a string
096                        throw new RuntimeException(e);
097                }
098                return sw.toString();
099        }
100
101        /**
102         * Writes the iCalendar objects to an output stream.
103         * @param out the output stream to write to
104         * @throws IOException if there's a problem writing to the output stream
105         */
106        public void go(OutputStream out) throws IOException {
107                go(new JCalWriter(out, wrapInArray()));
108        }
109
110        /**
111         * Writes the iCalendar objects to a file.
112         * @param file the file to write to
113         * @throws IOException if there's a problem writing to the file
114         */
115        public void go(File file) throws IOException {
116                JCalWriter writer = new JCalWriter(file, wrapInArray());
117                try {
118                        go(writer);
119                } finally {
120                        writer.close();
121                }
122        }
123
124        /**
125         * Writes the iCalendar objects to a writer.
126         * @param writer the writer to write to
127         * @throws IOException if there's a problem writing to the writer
128         */
129        public void go(Writer writer) throws IOException {
130                go(new JCalWriter(writer, wrapInArray()));
131        }
132
133        private void go(JCalWriter writer) throws IOException {
134                if (defaultTimeZone != null) {
135                        writer.setGlobalTimezone(defaultTimeZone);
136                }
137                writer.setPrettyPrint(prettyPrint);
138                if (index != null) {
139                        writer.setScribeIndex(index);
140                }
141                try {
142                        for (ICalendar ical : icals) {
143                                writer.write(ical);
144                                writer.flush();
145                        }
146                } finally {
147                        writer.closeJsonStream();
148                }
149        }
150
151        private boolean wrapInArray() {
152                return icals.size() > 1;
153        }
154}
