035/**
036 * Represents a timezone definition that is used within an iCalendar object.
037 * @author Michael Angstadt
038 */
039public class TimezoneAssignment {
040        private final TimeZone timezone;
041        private final VTimezone component;
042        private final String globalId;
043
044        /**
045         * Creates a timezone that will be inserted into the iCalendar object as a
046         * VTIMEZONE component. This what most iCalendar files use.
047         * @param timezone the Java timezone object
048         * @param component the iCalendar component
049         * @throws IllegalArgumentException if the given {@link VTimezone} component
050         * doesn't have a {@link TimezoneId} property
051         */
052        public TimezoneAssignment(TimeZone timezone, VTimezone component) {
053                String id = ValuedProperty.getValue(component.getTimezoneId());
054                if (id == null || id.trim().isEmpty()) {
055                        throw Messages.INSTANCE.getIllegalArgumentException(14);
056                }
057
058                this.timezone = timezone;
059                this.component = component;
060                this.globalId = null;
061        }
062
063        /**
064         * <p>
065         * Creates a timezone that will be inserted into the iCalendar object as a
066         * "global ID". This means that a {@link VTimezone} component containing the
067         * timezone definition will NOT be inserted into the iCalendar object.
068         * </p>
069         * <p>
070         * Because the timezone definition is not included inside of the iCalendar
071         * object, the client consuming the iCalendar object must know how to
072         * interpret such an ID. The iCalendar specification does not specify a list
073         * of such IDs, but suggests using the naming convention of an existing
074         * timezone specification, such as the
075         * <a href="http://www.twinsun.com/tz/tz-link.htm">public-domain TZ
076         * database</a>.
077         * </p>
078         * @param timezone the Java timezone object
079         * @param globalId the global ID (e.g. "America/New_York")
080         */
081        public TimezoneAssignment(TimeZone timezone, String globalId) {
082                this.timezone = timezone;
083                this.component = null;
084                this.globalId = globalId;
085        }
086
087        /**
088         * Creates a timezone whose VTIMEZONE component is downloaded from
089         * <a href="http://www.tzurl.org">tzurl.org</a>.
090         * @param timezone the Java timezone object
091         * @param outlookCompatible true to download a {@link VTimezone} component
092         * that is tailored for Microsoft Outlook email clients, false to download a
093         * standards-based one.
094         * @return the timezone assignment
095         * @throws IllegalArgumentException if an appropriate VTIMEZONE component
096         * cannot be found on the website
097         */
098        public static TimezoneAssignment download(TimeZone timezone, boolean outlookCompatible) {
099                TzUrlDotOrgGenerator generator = new TzUrlDotOrgGenerator(outlookCompatible);
100                VTimezone component = generator.generate(timezone);
101                return new TimezoneAssignment(timezone, component);
102        }
103
104        /**
105         * Gets the Java object associated with the timezone.
106         * @return the Java object
107         */
108        public TimeZone getTimeZone() {
109                return timezone;
110        }
111
112        /**
113         * Gets the iCalendar component associated with the timezone.
114         * @return the component or null if the timezone uses a global ID
115         */
116        public VTimezone getComponent() {
117                return component;
118        }
119
120        /**
121         * Gets the global ID associated with the timezone.
122         * @return the global ID or null if the timezone uses an iCalendar component
123         */
124        public String getGlobalId() {
125                return globalId;
126        }
127}
