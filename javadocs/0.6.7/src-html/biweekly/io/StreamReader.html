053/**
054 * Parses iCalendar objects from a data stream.
055 * @author Michael Angstadt
056 */
057public abstract class StreamReader implements Closeable {
058        protected final List<ParseWarning> warnings = new ArrayList<ParseWarning>();
059        protected ScribeIndex index = new ScribeIndex();
060        protected ParseContext context;
061        private TimeZone defaultTimezone = TimeZone.getDefault();

156        /**
157         * Reads all iCalendar objects from the data stream.
158         * @return the iCalendar objects
159         * @throws IOException if there's a problem reading from the stream
160         */
161        public List<ICalendar> readAll() throws IOException {
162                List<ICalendar> icals = new ArrayList<ICalendar>();
163                ICalendar ical;
164                while ((ical = readNext()) != null) {
165                        icals.add(ical);
166                }
167                return icals;
168        }
169
170        /**
171         * Reads the next iCalendar object from the data stream.
172         * @return the next iCalendar object or null if there are no more
173         * @throws IOException if there's a problem reading from the stream
174         */
175        public ICalendar readNext() throws IOException {
176                warnings.clear();
177                context = new ParseContext();
178                ICalendar ical = _readNext();
179                if (ical == null) {
180                        return null;
181                }
182
183                ical.setVersion(context.getVersion());
184                handleTimezones(ical);
185                return ical;
186        }

188        /**
189         * Reads the next iCalendar object from the data stream.
190         * @return the next iCalendar object or null if there are no more
191         * @throws IOException if there's a problem reading from the stream
192         */
193        protected abstract ICalendar _readNext() throws IOException;

195        private void handleTimezones(ICalendar ical) {
196                TimezoneInfo tzinfo = ical.getTimezoneInfo();
197
198                //convert vCalendar DAYLIGHT and TZ properties to a VTIMEZONE component
199                TimezoneAssignment vcalTimezone = extractVCalTimezone(ical);
200
201                //assign a TimeZone object to each VTIMEZONE component.
202                Iterator<VTimezone> it = ical.getComponents(VTimezone.class).iterator();
203                while (it.hasNext()) {
204                        VTimezone component = it.next();
205
206                        //make sure the component has an ID
207                        String id = ValuedProperty.getValue(component.getTimezoneId());
208                        if (id == null || id.trim().isEmpty()) {
209                                //note: do not remove invalid VTIMEZONE components from the ICalendar object
210                                warnings.add(new ParseWarning.Builder().message(39).build());
211                                continue;
212                        }
213
214                        TimeZone timezone = new ICalTimeZone(component);
215                        tzinfo.getTimezones().add(new TimezoneAssignment(timezone, component));
216
217                        //remove the component from the ICalendar object
218                        it.remove();
219                }
220
221                boolean userChangedTheDefaultTimezone = !defaultTimezone.equals(TimeZone.getDefault());
222
223                if (vcalTimezone != null) {
224                        //vCal: parse floating dates according to the DAYLIGHT and TZ properties (which were converted to a VTIMEZONE component)
225                        Calendar cal = Calendar.getInstance(vcalTimezone.getTimeZone());
226                        for (TimezonedDate timezonedDate : context.getFloatingDates()) {
227                                reparseDateUnderDifferentTimezone(timezonedDate, cal);
228                        }
229                } else {
230                        //iCal: treat floating dates as floating dates
231                        for (TimezonedDate timezonedDate : context.getFloatingDates()) {
232                                tzinfo.setFloating(timezonedDate.getProperty(), true);
233                        }
234
235                        //convert all floating dates to the default timezone
236                        if (userChangedTheDefaultTimezone) {
237                                Calendar cal = Calendar.getInstance(defaultTimezone);
238                                for (TimezonedDate timezonedDate : context.getFloatingDates()) {
239                                        reparseDateUnderDifferentTimezone(timezonedDate, cal);
240                                }
241                        }
242                }
243
244                //convert all date values to their appropriate timezone
245                for (Map.Entry<String, List<TimezonedDate>> entry : context.getTimezonedDates()) {
246                        String tzid = entry.getKey();
247
248                        //determine which timezone is associated with the given TZID
249                        TimezoneAssignment assignment = determineTimezoneAssignment(tzid, tzinfo);
250
251                        /*
252                         * If a timezone assignment could not be found for the given TZID
253                         * and the user did not change the default timezone, then there is
254                         * no need to further process the properties that are assigned to
255                         * this TZID--the date value should remain unchanged (parsed under
256                         * the local machine's default timezone), and its TZID parameter
257                         * should also remain.
258                         */
259                        if (assignment == null && !userChangedTheDefaultTimezone) {
260                                continue;
261                        }
262
263                        //convert each property to the timezone
264                        TimeZone tz = (assignment == null) ? defaultTimezone : assignment.getTimeZone();
265                        Calendar cal = Calendar.getInstance(tz);
266                        for (TimezonedDate timezonedDate : entry.getValue()) {
267                                ICalProperty property = timezonedDate.getProperty();
268
269                                if (assignment != null) {
270                                        tzinfo.setTimezone(property, assignment);
271
272                                        /*
273                                         * Only remove the TZID parameter if the TZID is *valid*.
274                                         * Invalid TZID parameters should remain so that user can
275                                         * inspect the invalid information.
276                                         */
277                                        property.getParameters().setTimezoneId(null);
278                                }
279
280                                reparseDateUnderDifferentTimezone(timezonedDate, cal);
281                        }
282                }
283        }
284
285        private void reparseDateUnderDifferentTimezone(TimezonedDate timezonedDate, Calendar cal) {
286                ICalDate date = timezonedDate.getDate();
287
288                //parse its raw date components under its real timezone
289                Date realDate = date.getRawComponents().toDate(cal);
290
291                //update the Date object with the new timestamp
292                date.setTime(realDate.getTime());
293        }

295        /**
296         * Determines the timezone definition that is associated with the given ID.
297         * @param tzid the timezone ID
298         * @param tzinfo the timezone settings of the iCalendar object
299         * @return the timezone definition or null to use the default timezone
300         */
301        private TimezoneAssignment determineTimezoneAssignment(String tzid, TimezoneInfo tzinfo) {
302                boolean isOlsenId = tzid.startsWith("/");
303
304                //HANDLE OLSEN IDS======================================================
305
306                if (isOlsenId) {
307                        String globalId = tzid.substring(1);
308                        TimeZone timezone = globalTimezoneIdResolver.resolve(globalId);
309                        if (timezone != null) {
310                                /*
311                                 * Olsen ID is valid. Everything is Ok.
312                                 */
313                                TimezoneAssignment assignment = new TimezoneAssignment(timezone, globalId);
314                                tzinfo.getTimezones().add(assignment);
315                                return assignment;
316                        }
317
318                        /*
319                         * Even though the TZID is marked as an Olsen ID, and the timezone
320                         * isn't recognized by Java, try looking for a VTIMEZONE component
321                         * that matches it.
322                         *
323                         * This is done as a courtesy and is not required by the specs.
324                         */
325                        TimezoneAssignment assignment = tzinfo.getTimezoneById(tzid);
326                        int warning;
327                        if (assignment == null) {
328                                /*
329                                 * TZID does not match any VTIMEZONE components, use the default
330                                 * timezone.
331                                 */
332                                warning = 38;
333                        } else {
334                                warning = 43;
335                        }
336
337                        warnings.add(new ParseWarning.Builder().message(warning, tzid).build());
338                        return assignment;
339                }
340
341                //HANDLE VTIMEZONE COMPONENT IDS========================================
342
343                TimezoneAssignment assignment = tzinfo.getTimezoneById(tzid);
344                if (assignment != null) {
345                        /*
346                         * VTIMEZONE component with the given TZID was found.
347                         * Everything is Ok.
348                         */
349                        return assignment;
350                }
351
352                /*
353                 * Try treating the TZID as an Olsen timezone ID, even though it does
354                 * not start with a forward slash.
355                 *
356                 * This is done as a courtesy for users who do not know they must prefix
357                 * Olsen IDs with a forward slash. It is not required by the specs.
358                 */
359                String globalId = tzid;
360                TimeZone timezone = globalTimezoneIdResolver.resolve(globalId);
361                int warning;
362                if (timezone == null) {
363                        /*
364                         * TZID is not a valid Olsen ID, use the default timezone.
365                         */
366                        warning = 38;
367                        assignment = null;
368                } else {
369                        /*
370                         * TZID was successfully parsed as an Olsen ID.
371                         */
372                        warning = 37;
373                        assignment = new TimezoneAssignment(timezone, globalId);
374                        tzinfo.getTimezones().add(assignment);
375                }
376
377                warnings.add(new ParseWarning.Builder().message(warning, globalId).build());
378                return assignment;
379        }
380
381        private TimezoneAssignment extractVCalTimezone(ICalendar ical) {
382                List<Daylight> daylights = ical.removeProperties(Daylight.class);
383                List<Timezone> timezones = ical.removeProperties(Timezone.class);
384
385                Timezone timezone = timezones.isEmpty() ? null : timezones.get(0);
386                VTimezone vcalComponent = convert(daylights, timezone);
387                if (vcalComponent == null) {
388                        return null;
389                }
390
391                TimeZone icalTimezone = new ICalTimeZone(vcalComponent);
392                TimezoneInfo tzinfo = ical.getTimezoneInfo();
393                TimezoneAssignment assignment = new TimezoneAssignment(icalTimezone, vcalComponent);
394                tzinfo.setDefaultTimezone(assignment);
395
396                return assignment;
397        }
