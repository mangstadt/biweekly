056/**
057 * @author Michael Angstadt
058 */
059public class VAlarmScribe extends ICalComponentScribe<VAlarm> {
060        public VAlarmScribe() {
061                super(VAlarm.class, "VALARM");
062        }
063
064        @Override
065        protected VAlarm _newInstance() {
066                return new VAlarm(null, null);
067        }
068
069        @Override
070        public void checkForDataModelConversions(VAlarm component, ICalComponent parent, ICalVersion version) {
071                if (version != ICalVersion.V1_0) {
072                        return;
073                }
074
075                VCalAlarmProperty vcalAlarm = convert(component, parent);
076                if (vcalAlarm == null) {
077                        return;
078                }
079
080                DataModelConversionException e = new DataModelConversionException(null);
081                e.getProperties().add(vcalAlarm);
082                throw e;
083        }
084
085        @Override
086        public Set<ICalVersion> getSupportedVersions() {
087                return EnumSet.of(ICalVersion.V2_0_DEPRECATED, ICalVersion.V2_0);
088        }
089
090        /**
091         * Converts a {@link VAlarm} component to a vCal alarm property.
092         * @param valarm the component
093         * @param parent the component's parent
094         * @return the vCal alarm property or null if it cannot be converted
095         */
096        private static VCalAlarmProperty convert(VAlarm valarm, ICalComponent parent) {
097                VCalAlarmProperty property = create(valarm);
098                if (property == null) {
099                        return null;
100                }
101
102                property.setStart(determineStartDate(valarm, parent));
103
104                DurationProperty duration = valarm.getDuration();
105                if (duration != null) {
106                        property.setSnooze(duration.getValue());
107                }
108
109                Repeat repeat = valarm.getRepeat();
110                if (repeat != null) {
111                        property.setRepeat(repeat.getValue());
112                }
113
114                return property;
115        }
116
117        /**
118         * Creates a new {@link VCalAlarmProperty} based on the given {@link VAlarm}
119         * component, setting fields that are common to all
120         * {@link VCalAlarmProperty} classes.
121         * @param valarm the source component
122         * @return the property or null if it cannot be created
123         */
124        private static VCalAlarmProperty create(VAlarm valarm) {
125                Action action = valarm.getAction();
126                if (action == null) {
127                        return null;
128                }
129
130                if (action.isAudio()) {
131                        AudioAlarm aalarm = new AudioAlarm();
132
133                        List<Attachment> attaches = valarm.getAttachments();
134                        if (!attaches.isEmpty()) {
135                                Attachment attach = attaches.get(0);
136
137                                String formatType = attach.getFormatType();
138                                aalarm.setParameter("TYPE", formatType);
139
140                                byte[] data = attach.getData();
141                                if (data != null) {
142                                        aalarm.setData(data);
143                                }
144
145                                String uri = attach.getUri();
146                                if (uri != null) {
147                                        String contentId = StringUtils.afterPrefixIgnoreCase(uri, "cid:");
148                                        if (contentId == null) {
149                                                aalarm.setUri(uri);
150                                        } else {
151                                                aalarm.setContentId(contentId);
152                                        }
153                                }
154                        }
155
156                        return aalarm;
157                }
158
159                if (action.isDisplay()) {
160                        Description description = valarm.getDescription();
161                        String text = ValuedProperty.getValue(description);
162                        return new DisplayAlarm(text);
163                }
164
165                if (action.isEmail()) {
166                        List<Attendee> attendees = valarm.getAttendees();
167                        String email = attendees.isEmpty() ? null : attendees.get(0).getEmail();
168                        EmailAlarm malarm = new EmailAlarm(email);
169
170                        Description description = valarm.getDescription();
171                        String note = ValuedProperty.getValue(description);
172                        malarm.setNote(note);
173
174                        return malarm;
175                }
176
177                if (action.isProcedure()) {
178                        Description description = valarm.getDescription();
179                        String path = ValuedProperty.getValue(description);
180                        return new ProcedureAlarm(path);
181                }
182
183                return null;
184        }
185
186        /**
187         * Determines what the alarm property's start date should be.
188         * @param valarm the component that is being converted to a vCal alarm
189         * property
190         * @param parent the component's parent
191         * @return the start date or null if it cannot be determined
192         */
193        private static Date determineStartDate(VAlarm valarm, ICalComponent parent) {
194                Trigger trigger = valarm.getTrigger();
195                if (trigger == null) {
196                        return null;
197                }
198
199                Date triggerStart = trigger.getDate();
200                if (triggerStart != null) {
201                        return triggerStart;
202                }
203
204                Duration triggerDuration = trigger.getDuration();
205                if (triggerDuration == null) {
206                        return null;
207                }
208
209                if (parent == null) {
210                        return null;
211                }
212
213                Related related = trigger.getRelated();
214                Date date = null;
215                if (related == Related.START) {
216                        date = ValuedProperty.getValue(parent.getProperty(DateStart.class));
217                } else if (related == Related.END) {
218                        date = ValuedProperty.getValue(parent.getProperty(DateEnd.class));
219                        if (date == null) {
220                                Date dateStart = ValuedProperty.getValue(parent.getProperty(DateStart.class));
221                                Duration duration = ValuedProperty.getValue(parent.getProperty(DurationProperty.class));
222                                if (duration != null && dateStart != null) {
223                                        date = duration.add(dateStart);
224                                }
225                        }
226                }
227
228                return (date == null) ? null : triggerDuration.add(date);
229        }
