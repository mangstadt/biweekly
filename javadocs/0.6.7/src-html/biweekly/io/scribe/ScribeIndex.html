115/**
116 * <p>
117 * Manages a listing of component and property scribes. This is useful for
118 * injecting the scribes of any experimental components or properties you have
119 * defined into a reader or writer object. The same ScribeIndex instance can be
120 * reused and injected into multiple reader/writer classes.
121 * </p>
122 * <p>
123 * <b>Example:</b>
124 * </p>
125 * 
126 * <pre class="brush:java">
127 * //init the index
128 * ScribeIndex index = new ScribeIndex();
129 * index.register(new CustomPropertyScribe());
130 * index.register(new AnotherCustomPropertyScribe());
131 * index.register(new CustomComponentScribe());
132 * 
133 * //inject into a reader class
134 * ICalReader reader = new ICalReader(...);
135 * reader.setScribeIndex(index);
136 * List&lt;ICalendar&gt; icals = new ArrayList&lt;ICalendar&gt;();
137 * ICalendar ical;
138 * while ((ical = reader.readNext()) != null) {
139 *   icals.add(ical);
140 * }
141 * 
142 * //inject the same instance in another reader/writer class
143 * JCalWriter writer = new JCalWriter(...);
144 * writer.setScribeIndex(index);
145 * for (ICalendar ical : icals) {
146 *   writer.write(ical);
147 * }
148 * </pre>
149 * @author Michael Angstadt
150 */
151public class ScribeIndex {

250        /**
251         * Gets a component scribe by name.
252         * @param componentName the component name (e.g. "VEVENT")
253         * @param version the version of the iCalendar object being parsed
254         * @return the component scribe or a {@link RawComponentScribe} if not found
255         */
256        public ICalComponentScribe<? extends ICalComponent> getComponentScribe(String componentName, ICalVersion version) {
257                componentName = componentName.toUpperCase();
258
259                ICalComponentScribe<? extends ICalComponent> scribe = experimentalCompByName.get(componentName);
260                if (scribe == null) {
261                        scribe = standardCompByName.get(componentName);
262                }
263
264                if (scribe == null) {
265                        return new RawComponentScribe(componentName);
266                }
267
268                if (version != null && !scribe.getSupportedVersions().contains(version)) {
269                        //treat the component as a raw component if the current iCal version doesn't support it
270                        return new RawComponentScribe(componentName);
271                }
272
273                return scribe;
274        }
275
276        /**
277         * Gets a property scribe by name.
278         * @param propertyName the property name (e.g. "UID")
279         * @param version the version of the iCalendar object being parsed
280         * @return the property scribe or a {@link RawPropertyScribe} if not found
281         */
282        public ICalPropertyScribe<? extends ICalProperty> getPropertyScribe(String propertyName, ICalVersion version) {
283                propertyName = propertyName.toUpperCase();
284
285                String key = propertyNameKey(propertyName, version);
286                ICalPropertyScribe<? extends ICalProperty> scribe = experimentalPropByName.get(key);
287                if (scribe == null) {
288                        scribe = standardPropByName.get(key);
289                }
290
291                if (scribe == null) {
292                        return new RawPropertyScribe(propertyName);
293                }
294
295                if (version != null && !scribe.getSupportedVersions().contains(version)) {
296                        //treat the property as a raw property if the current iCal version doesn't support it
297                        return new RawPropertyScribe(propertyName);
298                }
299
300                return scribe;
301        }

331        /**
332         * Gets the appropriate component scribe for a given component instance.
333         * @param component the component instance
334         * @return the component scribe or null if not found
335         */
336        public ICalComponentScribe<? extends ICalComponent> getComponentScribe(ICalComponent component) {
337                if (component instanceof RawComponent) {
338                        RawComponent raw = (RawComponent) component;
339                        return new RawComponentScribe(raw.getName());
340                }
341
342                return getComponentScribe(component.getClass());
343        }

351                if (property instanceof RawProperty) {

355
356                return getPropertyScribe(property.getClass());
357        }

359        /**
360         * Gets a property scribe by XML local name and namespace.
361         * @param qname the XML local name and namespace
362         * @return the property scribe or a {@link XmlScribe} if not found
363         */
364        public ICalPropertyScribe<? extends ICalProperty> getPropertyScribe(QName qname) {
365                ICalPropertyScribe<? extends ICalProperty> scribe = experimentalPropByQName.get(qname);
366                if (scribe == null) {
367                        scribe = standardPropByQName.get(qname);
368                }
369
370                if (scribe == null || !scribe.getSupportedVersions().contains(ICalVersion.V2_0)) {
371                        if (XCalNamespaceContext.XCAL_NS.equals(qname.getNamespaceURI())) {
372                                return new RawPropertyScribe(qname.getLocalPart().toUpperCase());
373                        }
374                        return getPropertyScribe(Xml.class);
375                }
376
377                return scribe;
378        }

380        /**
381         * Registers a component scribe.
382         * @param scribe the scribe to register
383         */
384        public void register(ICalComponentScribe<? extends ICalComponent> scribe) {
385                experimentalCompByName.put(scribe.getComponentName().toUpperCase(), scribe);
386                experimentalCompByClass.put(scribe.getComponentClass(), scribe);
387        }
388
389        /**
390         * Registers a property scribe.
391         * @param scribe the scribe to register
392         */
393        public void register(ICalPropertyScribe<? extends ICalProperty> scribe) {
394                for (ICalVersion version : ICalVersion.values()) {
395                        experimentalPropByName.put(propertyNameKey(scribe, version), scribe);
396                }
397                experimentalPropByClass.put(scribe.getPropertyClass(), scribe);
398                experimentalPropByQName.put(scribe.getQName(), scribe);
399        }

402         * Unregisters a component scribe.
403         * @param scribe the scribe to unregister
404         */
405        public void unregister(ICalComponentScribe<? extends ICalComponent> scribe) {
406                experimentalCompByName.remove(scribe.getComponentName().toUpperCase());
407                experimentalCompByClass.remove(scribe.getComponentClass());
408        }

414        public void unregister(ICalPropertyScribe<? extends ICalProperty> scribe) {
415                for (ICalVersion version : ICalVersion.values()) {
416                        experimentalPropByName.remove(propertyNameKey(scribe, version));
417                }
418                experimentalPropByClass.remove(scribe.getPropertyClass());
419                experimentalPropByQName.remove(scribe.getQName());

444        private static String propertyNameKey(ICalPropertyScribe<? extends ICalProperty> scribe, ICalVersion version) {
445                return propertyNameKey(scribe.getPropertyName(version), version);
446        }
447
448        private static String propertyNameKey(String propertyName, ICalVersion version) {
449                return version.ordinal() + propertyName.toUpperCase();
450        }
