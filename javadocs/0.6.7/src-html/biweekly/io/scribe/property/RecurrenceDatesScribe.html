051/**
052 * Marshals {@link RecurrenceDates} properties.
053 * @author Michael Angstadt
054 */
055public class RecurrenceDatesScribe extends ICalPropertyScribe<RecurrenceDates> {
056        public RecurrenceDatesScribe() {
057                super(RecurrenceDates.class, "RDATE", DATE_TIME);
058        }
059
060        @Override
061        protected ICalParameters _prepareParameters(RecurrenceDates property, WriteContext context) {
062                if (isInObservance(context)) {
063                        return property.getParameters();
064                }
065
066                List<Period> periods = property.getPeriods();
067                List<ICalDate> dates = property.getDates();
068                boolean hasTime;
069                if (periods.isEmpty() && dates.isEmpty()) {
070                        hasTime = false;
071                } else {
072                        ICalDataType dataType = dataType(property, context.getVersion());
073                        hasTime = (dataType == DATE_TIME || dataType == PERIOD);
074                }
075                return handleTzidParameter(property, hasTime, context);
076        }
077
078        @Override
079        protected ICalDataType _dataType(RecurrenceDates property, ICalVersion version) {
080                List<ICalDate> dates = property.getDates();
081                if (!dates.isEmpty()) {
082                        return dates.get(0).hasTime() ? DATE_TIME : DATE;
083                }
084
085                if (!property.getPeriods().isEmpty()) {
086                        return PERIOD;
087                }
088
089                return defaultDataType(version);
090        }
091
092        @Override
093        protected String _writeText(final RecurrenceDates property, final WriteContext context) {
094                List<ICalDate> dates = property.getDates();
095                if (!dates.isEmpty()) {
096                        boolean inObservance = isInObservance(context);
097                        List<String> values = new ArrayList<String>(dates.size());
098                        for (ICalDate date : dates) {
099                                String value;
100                                if (inObservance) {
101                                        value = date(date).observance(true).extended(false).write();
102                                } else {
103                                        value = date(date, property, context).extended(false).write();
104                                }
105                                values.add(value);
106                        }
107                        return VObjectPropertyValues.writeList(values);
108                }
109
110                //TODO vCal does not support periods
111                List<Period> periods = property.getPeriods();
112                if (!periods.isEmpty()) {
113                        List<String> values = new ArrayList<String>(periods.size());
114                        for (Period period : periods) {
115                                StringBuilder sb = new StringBuilder();
116
117                                Date start = period.getStartDate();
118                                if (start != null) {
119                                        String date = date(start, property, context).extended(false).write();
120                                        sb.append(date);
121                                }
122
123                                sb.append('/');
124
125                                Date end = period.getEndDate();
126                                Duration duration = period.getDuration();
127                                if (end != null) {
128                                        String date = date(end, property, context).extended(false).write();
129                                        sb.append(date);
130                                } else if (duration != null) {
131                                        sb.append(duration);
132                                }
133
134                                values.add(sb.toString());
135                        }
136                        return VObjectPropertyValues.writeList(values);
137                }
138
139                return "";
140        }
141
142        @Override
143        protected RecurrenceDates _parseText(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
144                return parse(VObjectPropertyValues.parseList(value), dataType, parameters, context);
145        }

147        @Override
148        protected void _writeXml(RecurrenceDates property, XCalElement element, WriteContext context) {
149                ICalDataType dataType = dataType(property, context.getVersion());
150                List<ICalDate> dates = property.getDates();
151                if (!dates.isEmpty()) {
152                        boolean inObservance = isInObservance(context);
153                        for (ICalDate date : dates) {
154                                String dateStr;
155                                if (inObservance) {
156                                        dateStr = date(date).observance(true).extended(true).write();
157                                } else {
158                                        dateStr = date(date, property, context).extended(true).write();
159                                }
160
161                                element.append(dataType, dateStr);
162                        }
163                        return;
164                }
165
166                List<Period> periods = property.getPeriods();
167                if (!periods.isEmpty()) {
168                        for (Period period : periods) {
169                                XCalElement periodElement = element.append(dataType);
170
171                                Date start = period.getStartDate();
172                                if (start != null) {
173                                        String dateStr = date(start, property, context).extended(true).write();
174                                        periodElement.append("start", dateStr);
175                                }
176
177                                Date end = period.getEndDate();
178                                if (end != null) {
179                                        String dateStr = date(end, property, context).extended(true).write();
180                                        periodElement.append("end", dateStr);
181                                }
182
183                                Duration duration = period.getDuration();
184                                if (duration != null) {
185                                        periodElement.append("duration", duration.toString());
186                                }
187                        }
188                        return;
189                }
190
191                element.append(defaultDataType(context.getVersion()), "");
192        }
193
194        @Override
195        protected RecurrenceDates _parseXml(XCalElement element, ICalParameters parameters, ParseContext context) {
196                List<XCalElement> periodElements = element.children(PERIOD);
197                List<String> dateTimeElements = element.all(DATE_TIME);
198                List<String> dateElements = element.all(DATE);
199                if (periodElements.isEmpty() && dateTimeElements.isEmpty() && dateElements.isEmpty()) {
200                        throw missingXmlElements(PERIOD, DATE_TIME, DATE);
201                }
202
203                RecurrenceDates property = new RecurrenceDates();
204
205                //parse periods
206                for (XCalElement periodElement : periodElements) {
207                        String startStr = periodElement.first("start");
208                        if (startStr == null) {
209                                throw new CannotParseException(9);
210                        }
211
212                        ICalDate start;
213                        try {
214                                start = date(startStr).parse();
215                        } catch (IllegalArgumentException e) {
216                                throw new CannotParseException(10, startStr);
217                        }
218
219                        String endStr = periodElement.first("end");
220                        if (endStr != null) {
221                                try {
222                                        ICalDate end = date(endStr).parse();
223                                        property.getPeriods().add(new Period(start, end));
224                                        context.addDate(start, property, parameters);
225                                        context.addDate(end, property, parameters);
226                                } catch (IllegalArgumentException e) {
227                                        throw new CannotParseException(11, endStr);
228                                }
229                                continue;
230                        }
231
232                        String durationStr = periodElement.first("duration");
233                        if (durationStr != null) {
234                                try {
235                                        Duration duration = Duration.parse(durationStr);
236                                        property.getPeriods().add(new Period(start, duration));
237                                        context.addDate(start, property, parameters);
238                                } catch (IllegalArgumentException e) {
239                                        throw new CannotParseException(12, durationStr);
240                                }
241                                continue;
242                        }
243
244                        throw new CannotParseException(13);
245                }
246
247                //parse date-times
248                for (String dateTimeStr : dateTimeElements) {
249                        try {
250                                ICalDate date = date(dateTimeStr).hasTime(true).parse();
251                                property.getDates().add(date);
252                                context.addDate(date, property, parameters);
253                        } catch (IllegalArgumentException e) {
254                                throw new CannotParseException(15, dateTimeStr);
255                        }
256                }
257
258                //parse dates
259                for (String dateStr : dateElements) {
260                        try {
261                                ICalDate date = date(dateStr).hasTime(false).parse();
262                                property.getDates().add(date);
263                        } catch (IllegalArgumentException e) {
264                                throw new CannotParseException(15, dateStr);
265                        }
266                }
267
268                return property;
269        }

271        @Override
272        protected JCalValue _writeJson(RecurrenceDates property, WriteContext context) {
273                List<String> values = new ArrayList<String>();
274                List<ICalDate> dates = property.getDates();
275                List<Period> periods = property.getPeriods();
276                if (!dates.isEmpty()) {
277                        boolean inObservance = isInObservance(context);
278                        for (ICalDate date : dates) {
279                                String dateStr;
280                                if (inObservance) {
281                                        dateStr = date(date).observance(true).extended(true).write();
282                                } else {
283                                        dateStr = date(date, property, context).extended(true).write();
284                                }
285
286                                values.add(dateStr);
287                        }
288                } else if (!periods.isEmpty()) {
289                        for (Period period : property.getPeriods()) {
290                                StringBuilder sb = new StringBuilder();
291                                Date start = period.getStartDate();
292                                if (start != null) {
293                                        String dateStr = date(start, property, context).extended(true).write();
294                                        sb.append(dateStr);
295                                }
296
297                                sb.append('/');
298
299                                Date end = period.getEndDate();
300                                Duration duration = period.getDuration();
301                                if (end != null) {
302                                        String dateStr = date(end, property, context).extended(true).write();
303                                        sb.append(dateStr);
304                                } else if (duration != null) {
305                                        sb.append(duration);
306                                }
307
308                                values.add(sb.toString());
309                        }
310                }
311
312                if (values.isEmpty()) {
313                        values.add("");
314                }
315                return JCalValue.multi(values);
316        }

318        @Override
319        protected RecurrenceDates _parseJson(JCalValue value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
320                return parse(value.asMulti(), dataType, parameters, context);
321        }

323        private RecurrenceDates parse(List<String> valueStrs, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
324                RecurrenceDates property = new RecurrenceDates();
325
326                if (dataType == PERIOD) {
327                        //parse as periods
328                        for (String timePeriodStr : valueStrs) {
329                                int slash = timePeriodStr.indexOf('/');
330                                if (slash < 0) {
331                                        throw new CannotParseException(13);
332                                }
333
334                                String startStr = timePeriodStr.substring(0, slash);
335                                ICalDate start;
336                                try {
337                                        start = date(startStr).parse();
338                                } catch (IllegalArgumentException e) {
339                                        throw new CannotParseException(10, startStr);
340                                }
341
342                                String endStr = timePeriodStr.substring(slash + 1);
343                                ICalDate end;
344                                try {
345                                        end = date(endStr).parse();
346                                        property.getPeriods().add(new Period(start, end));
347                                        context.addDate(start, property, parameters);
348                                        context.addDate(end, property, parameters);
349                                } catch (IllegalArgumentException e) {
350                                        //must be a duration
351                                        try {
352                                                Duration duration = Duration.parse(endStr);
353                                                property.getPeriods().add(new Period(start, duration));
354                                                context.addDate(start, property, parameters);
355                                        } catch (IllegalArgumentException e2) {
356                                                throw new CannotParseException(14, endStr);
357                                        }
358                                }
359                        }
360                        return property;
361                }
362
363                //parse as dates
364                boolean hasTime = (dataType == DATE_TIME);
365                for (String valueStr : valueStrs) {
366                        ICalDate date;
367                        try {
368                                date = date(valueStr).hasTime(hasTime).parse();
369                        } catch (IllegalArgumentException e) {
370                                throw new CannotParseException(15, valueStr);
371                        }
372                        property.getDates().add(date);
373                        context.addDate(date, property, parameters);
374                }
375                return property;
376        }
