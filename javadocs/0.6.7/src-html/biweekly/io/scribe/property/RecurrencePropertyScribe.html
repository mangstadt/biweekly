096                //null value
097                Recurrence recur = property.getValue();

099                        return "";
100                }

105                default:
106                        return writeTextV2(property, context);

110        private String writeTextV1(T property, WriteContext context) {
111                Recurrence recur = property.getValue();
112                Frequency frequency = recur.getFrequency();
113                if (frequency == null) {
114                        return "";
115                }
116
117                StringBuilder sb = new StringBuilder();
118
119                Integer interval = recur.getInterval();
120                if (interval == null) {
121                        interval = 1;
122                }
123
124                switch (frequency) {
125                case YEARLY:
126                        if (!recur.getByMonth().isEmpty()) {
127                                sb.append("YM").append(interval);
128                                for (Integer month : recur.getByMonth()) {
129                                        sb.append(' ').append(month);
130                                }
131                        } else {
132                                sb.append("YD").append(interval);
133                                for (Integer day : recur.getByYearDay()) {
134                                        sb.append(' ').append(day);
135                                }
136                        }
137                        break;
138
139                case MONTHLY:
140                        if (!recur.getByMonthDay().isEmpty()) {
141                                sb.append("MD").append(interval);
142                                for (Integer day : recur.getByMonthDay()) {
143                                        sb.append(' ').append(writeVCalInt(day));
144                                }
145                        } else {
146                                sb.append("MP").append(interval);
147                                for (ByDay byDay : recur.getByDay()) {
148                                        DayOfWeek day = byDay.getDay();
149                                        Integer prefix = byDay.getNum();
150                                        if (prefix == null) {
151                                                prefix = 1;
152                                        }
153
154                                        sb.append(' ').append(writeVCalInt(prefix)).append(' ').append(day.getAbbr());
155                                }
156                        }
157                        break;
158
159                case WEEKLY:
160                        sb.append("W").append(interval);
161                        for (ByDay byDay : recur.getByDay()) {
162                                sb.append(' ').append(byDay.getDay().getAbbr());
163                        }
164                        break;
165
166                case DAILY:
167                        sb.append("D").append(interval);
168                        break;
169
170                case HOURLY:
171                        sb.append("M").append(interval * 60);
172                        break;
173
174                case MINUTELY:
175                        sb.append("M").append(interval);
176                        break;
177
178                default:
179                        return "";
180                }
181
182                Integer count = recur.getCount();
183                ICalDate until = recur.getUntil();
184                sb.append(' ');
185
186                if (count != null) {
187                        sb.append('#').append(count);
188                } else if (until != null) {
189                        String dateStr = date(until, property, context).extended(false).write();
190                        sb.append(dateStr);
191                } else {
192                        sb.append("#0");
193                }
194
195                return sb.toString();
196        }

203        @Override
204        protected T _parseText(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
205                if (value.isEmpty()) {
206                        return newInstance(new Recurrence.Builder((Frequency) null).build());
207                }
208
209                switch (context.getVersion()) {
210                case V1_0:
211                        handleVersion1Multivalued(value, dataType, parameters, context);
212                        return parseTextV1(value, dataType, parameters, context);
213                default:
214                        return parseTextV2(value, dataType, parameters, context);
215                }
216        }

218        /**
219         * Version 1.0 allows multiple RRULE values to be defined inside of the same
220         * property. This method checks for this and, if multiple values are found,
221         * parses them and throws a {@link DataModelConversionException}.
222         * @param value the property value
223         * @param dataType the property data type
224         * @param parameters the property parameters
225         * @param context the parse context
226         * @throws DataModelConversionException if the property contains multiple
227         * RRULE values
228         */
229        private void handleVersion1Multivalued(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
230                List<String> rrules = splitRRULEValues(value);
231                if (rrules.size() == 1) {
232                        return;
233                }
234
235                DataModelConversionException conversionException = new DataModelConversionException(null);
236                for (String rrule : rrules) {
237                        ICalParameters parametersCopy = new ICalParameters(parameters);
238
239                        ICalProperty property;
240                        try {
241                                property = parseTextV1(rrule, dataType, parametersCopy, context);
242                        } catch (CannotParseException e) {
243                                //@formatter:off
244                                context.getWarnings().add(new ParseWarning.Builder(context)
245                                        .message(e)
246                                        .build()
247                                );
248                                //@formatter:on
249                                property = new RawProperty(getPropertyName(context.getVersion()), dataType, rrule);
250                                property.setParameters(parametersCopy);
251                        }
252                        conversionException.getProperties().add(property);
253                }
254
255                throw conversionException;
256        }

258        /**
259         * Version 1.0 allows multiple RRULE values to be defined inside of the same
260         * property. This method extracts each RRULE value from the property value.
261         * @param value the property value
262         * @return the RRULE values
263         */
264        private List<String> splitRRULEValues(String value) {
265                List<String> values = new ArrayList<String>();
266                Pattern p = Pattern.compile("#\\d+|\\d{8}T\\d{6}Z?");
267                Matcher m = p.matcher(value);
268
269                int prevIndex = 0;
270                while (m.find()) {
271                        int end = m.end();
272                        String subValue = value.substring(prevIndex, end).trim();
273                        values.add(subValue);
274                        prevIndex = end;
275                }
276                String subValue = value.substring(prevIndex).trim();
277                if (subValue.length() > 0) {
278                        values.add(subValue);
279                }
280
281                return values;
282        }

284        private T parseTextV1(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
285                final Recurrence.Builder builder = new Recurrence.Builder((Frequency) null);
286
287                List<String> splitValues = Arrays.asList(value.toUpperCase().split("\\s+"));
288
289                //parse the frequency and interval from the first token (e.g. "W2")
290                String frequencyStr;
291                Integer interval;
292                {
293                        String firstToken = splitValues.get(0);
294                        Pattern p = Pattern.compile("^([A-Z]+)(\\d+)$");
295                        Matcher m = p.matcher(firstToken);
296                        if (!m.find()) {
297                                throw new CannotParseException(40, firstToken);
298                        }
299
300                        frequencyStr = m.group(1);
301                        interval = integerValueOf(m.group(2));
302
303                        splitValues = splitValues.subList(1, splitValues.size());
304                }
305                builder.interval(interval);
306
307                Integer count = null;
308                ICalDate until = null;
309                if (splitValues.isEmpty()) {
310                        count = 2;
311                } else {
312                        String lastToken = splitValues.get(splitValues.size() - 1);
313                        if (lastToken.startsWith("#")) {
314                                String countStr = lastToken.substring(1);
315                                count = integerValueOf(countStr);
316                                if (count == 0) {
317                                        //infinite
318                                        count = null;
319                                }
320
321                                splitValues = splitValues.subList(0, splitValues.size() - 1);
322                        } else {
323                                try {
324                                        //see if the value is an "until" date
325                                        until = date(lastToken).parse();
326                                        splitValues = splitValues.subList(0, splitValues.size() - 1);
327                                } catch (IllegalArgumentException e) {
328                                        //last token is a regular value
329                                        count = 2;
330                                }
331                        }
332                }
333                builder.count(count);
334                builder.until(until);
335
336                //determine what frequency enum to use and how to treat each tokenized value
337                Frequency frequency;
338                Handler<String> handler;
339                if ("YD".equals(frequencyStr)) {
340                        frequency = Frequency.YEARLY;
341                        handler = new Handler<String>() {
342                                public void handle(String value) {
343                                        if (value == null) {
344                                                return;
345                                        }
346
347                                        Integer dayOfYear = integerValueOf(value);
348                                        builder.byYearDay(dayOfYear);
349                                }
350                        };
351                } else if ("YM".equals(frequencyStr)) {
352                        frequency = Frequency.YEARLY;
353                        handler = new Handler<String>() {
354                                public void handle(String value) {
355                                        if (value == null) {
356                                                return;
357                                        }
358
359                                        Integer month = integerValueOf(value);
360                                        builder.byMonth(month);
361                                }
362                        };
363                } else if ("MD".equals(frequencyStr)) {
364                        frequency = Frequency.MONTHLY;
365                        handler = new Handler<String>() {
366                                public void handle(String value) {
367                                        if (value == null) {
368                                                return;
369                                        }
370
371                                        try {
372                                                Integer date = "LD".equals(value) ? -1 : parseVCalInt(value);
373                                                builder.byMonthDay(date);
374                                        } catch (NumberFormatException e) {
375                                                throw new CannotParseException(40, value);
376                                        }
377                                }
378                        };
379                } else if ("MP".equals(frequencyStr)) {
380                        frequency = Frequency.MONTHLY;
381                        handler = new Handler<String>() {
382                                private final List<Integer> nums = new ArrayList<Integer>();
383                                private final List<DayOfWeek> days = new ArrayList<DayOfWeek>();
384                                private boolean readNum = false;
385
386                                public void handle(String value) {
387                                        if (value == null) {
388                                                //end of list
389                                                for (Integer num : nums) {
390                                                        for (DayOfWeek day : days) {
391                                                                builder.byDay(num, day);
392                                                        }
393                                                }
394                                                return;
395                                        }
396
397                                        if (value.matches("\\d{4}")) {
398                                                readNum = false;
399
400                                                Integer hour = integerValueOf(value.substring(0, 2));
401                                                builder.byHour(hour);
402
403                                                Integer minute = integerValueOf(value.substring(2, 4));
404                                                builder.byMinute(minute);
405                                                return;
406                                        }
407
408                                        try {
409                                                Integer curNum = parseVCalInt(value);
410
411                                                if (!readNum) {
412                                                        //reset lists, new segment
413                                                        for (Integer num : nums) {
414                                                                for (DayOfWeek day : days) {
415                                                                        builder.byDay(num, day);
416                                                                }
417                                                        }
418                                                        nums.clear();
419                                                        days.clear();
420
421                                                        readNum = true;
422                                                }
423
424                                                nums.add(curNum);
425                                        } catch (NumberFormatException e) {
426                                                readNum = false;
427
428                                                DayOfWeek day = parseDay(value);
429                                                days.add(day);
430                                        }
431                                }
432                        };
433                } else if ("W".equals(frequencyStr)) {
434                        frequency = Frequency.WEEKLY;
435                        handler = new Handler<String>() {
436                                public void handle(String value) {
437                                        if (value == null) {
438                                                return;
439                                        }
440
441                                        DayOfWeek day = parseDay(value);
442                                        builder.byDay(day);
443                                }
444                        };
445                } else if ("D".equals(frequencyStr)) {
446                        frequency = Frequency.DAILY;
447                        handler = new Handler<String>() {
448                                public void handle(String value) {
449                                        if (value == null) {
450                                                return;
451                                        }
452
453                                        Integer hour = integerValueOf(value.substring(0, 2));
454                                        builder.byHour(hour);
455
456                                        Integer minute = integerValueOf(value.substring(2, 4));
457                                        builder.byMinute(minute);
458                                }
459                        };
460                } else if ("M".equals(frequencyStr)) {
461                        frequency = Frequency.MINUTELY;
462                        handler = new Handler<String>() {
463                                public void handle(String value) {
464                                        //TODO can this ever have values?
465                                }
466                        };
467                } else {
468                        throw new CannotParseException(41, frequencyStr);
469                }
470
471                builder.frequency(frequency);
472
473                //parse the rest of the tokens
474                for (String splitValue : splitValues) {
475                        //TODO not sure how to handle the "$" symbol, ignore it
476                        if (splitValue.endsWith("$")) {
477                                context.addWarning(36, splitValue);
478                                splitValue = splitValue.substring(0, splitValue.length() - 1);
479                        }
480
481                        handler.handle(splitValue);
482                }
483                handler.handle(null);
484
485                T property = newInstance(builder.build());
486                if (until != null) {
487                        context.addDate(until, property, parameters);
488                }
489
490                return property;
491        }

493        private T parseTextV2(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
494                Recurrence.Builder builder = new Recurrence.Builder((Frequency) null);
495                ListMultimap<String, String> rules = new ListMultimap<String, String>(VObjectPropertyValues.parseMultimap(value));
496
497                parseFreq(rules, builder, context);
498                parseUntil(rules, builder, context);
499                parseCount(rules, builder, context);
500                parseInterval(rules, builder, context);
501                parseBySecond(rules, builder, context);
502                parseByMinute(rules, builder, context);
503                parseByHour(rules, builder, context);
504                parseByDay(rules, builder, context);
505                parseByMonthDay(rules, builder, context);
506                parseByYearDay(rules, builder, context);
507                parseByWeekNo(rules, builder, context);
508                parseByMonth(rules, builder, context);
509                parseBySetPos(rules, builder, context);
510                parseWkst(rules, builder, context);
511                parseXRules(rules, builder); //must be called last
512
513                T property = newInstance(builder.build());
514
515                ICalDate until = property.getValue().getUntil();
516                if (until != null) {
517                        context.addDate(until, property, parameters);
518                }
519
520                return property;
521        }

523        /**
524         * Parses an integer string, where the sign is at the end of the string
525         * instead of at the beginning (for example, "5-").
526         * @param value the string
527         * @return the value
528         * @throws NumberFormatException if the string cannot be parsed as an
529         * integer
530         */
531        private static int parseVCalInt(String value) {
532                int negate = 1;
533                if (value.endsWith("+")) {
534                        value = value.substring(0, value.length() - 1);
535                } else if (value.endsWith("-")) {
536                        value = value.substring(0, value.length() - 1);
537                        negate = -1;
538                }
539
540                return Integer.parseInt(value) * negate;
541        }

543        /**
544         * Same as {@link Integer#valueOf(String)}, but throws a
545         * {@link CannotParseException} when it fails.
546         * @param value the string to parse
547         * @return the parse integer
548         * @throws CannotParseException if the string cannot be parsed
549         */
550        private static Integer integerValueOf(String value) {
551                try {
552                        return Integer.valueOf(value);
553                } catch (NumberFormatException e) {
554                        throw new CannotParseException(40, value);
555                }
556        }

558        private static String writeVCalInt(Integer value) {
559                if (value > 0) {
560                        return value + "+";
561                }
562
563                if (value < 0) {
564                        return Math.abs(value) + "-";
565                }
566
567                return value.toString();
568        }

570        private DayOfWeek parseDay(String value) {
571                DayOfWeek day = DayOfWeek.valueOfAbbr(value);
572                if (day == null) {
573                        throw new CannotParseException(42, value);
574                }
575
576                return day;
577        }

581                XCalElement recurElement = element.append(dataType(property, null));
582
583                Recurrence recur = property.getValue();
584                if (recur == null) {
585                        return;
586                }
587
588                ListMultimap<String, Object> components = buildComponents(property, context, true);
589                for (Map.Entry<String, List<Object>> component : components) {
590                        String name = component.getKey().toLowerCase();
591                        for (Object value : component.getValue()) {
592                                recurElement.append(name, value.toString());
593                        }
594                }

597        @Override
598        protected T _parseXml(XCalElement element, ICalParameters parameters, ParseContext context) {
599                ICalDataType dataType = defaultDataType(context.getVersion());
600                XCalElement value = element.child(dataType);
601                if (value == null) {
602                        throw missingXmlElements(dataType);
603                }
604
605                ListMultimap<String, String> rules = new ListMultimap<String, String>();
606                for (Element child : XmlUtils.toElementList(value.getElement().getChildNodes())) {
607                        if (!XCalNamespaceContext.XCAL_NS.equals(child.getNamespaceURI())) {
608                                continue;
609                        }
610
611                        String name = child.getLocalName().toUpperCase();
612                        String text = child.getTextContent();
613                        rules.put(name, text);
614                }
615
616                Recurrence.Builder builder = new Recurrence.Builder((Frequency) null);
617
618                parseFreq(rules, builder, context);
619                parseUntil(rules, builder, context);
620                parseCount(rules, builder, context);
621                parseInterval(rules, builder, context);
622                parseBySecond(rules, builder, context);
623                parseByMinute(rules, builder, context);
624                parseByHour(rules, builder, context);
625                parseByDay(rules, builder, context);
626                parseByMonthDay(rules, builder, context);
627                parseByYearDay(rules, builder, context);
628                parseByWeekNo(rules, builder, context);
629                parseByMonth(rules, builder, context);
630                parseBySetPos(rules, builder, context);
631                parseWkst(rules, builder, context);
632                parseXRules(rules, builder); //must be called last
633
634                T property = newInstance(builder.build());
635
636                ICalDate until = property.getValue().getUntil();
637                if (until != null) {
638                        context.addDate(until, property, parameters);
639                }
640
641                return property;
642        }

645        protected JCalValue _writeJson(T property, WriteContext context) {
646                Recurrence recur = property.getValue();
647                if (recur == null) {
648                        return JCalValue.object(new ListMultimap<String, Object>(0));
649                }
650
651                ListMultimap<String, Object> components = buildComponents(property, context, true);
652
653                //lower-case all the keys
654                ListMultimap<String, Object> object = new ListMultimap<String, Object>(components.keySet().size());
655                for (Map.Entry<String, List<Object>> entry : components) {
656                        String key = entry.getKey().toLowerCase();
657                        object.putAll(key, entry.getValue());
658                }
659
660                return JCalValue.object(object);
661        }

708        private void parseFreq(ListMultimap<String, String> rules, final Recurrence.Builder builder, final ParseContext context) {
709                parseFirst(rules, FREQ, new Handler<String>() {
710                        public void handle(String value) {
711                                value = value.toUpperCase();
712                                try {
713                                        builder.frequency(Frequency.valueOf(value));
714                                } catch (IllegalArgumentException e) {
715                                        context.addWarning(7, FREQ, value);
716                                }
717                        }
718                });
719        }

721        private void parseUntil(ListMultimap<String, String> rules, final Recurrence.Builder builder, final ParseContext context) {
722                parseFirst(rules, UNTIL, new Handler<String>() {
723                        public void handle(String value) {
724                                try {

781        private void parseByDay(ListMultimap<String, String> rules, Recurrence.Builder builder, ParseContext context) {
782                Pattern p = Pattern.compile("^([-+]?\\d+)?(.*)$");
783                for (String value : rules.removeAll(BYDAY)) {
784                        Matcher m = p.matcher(value);
785                        if (!m.find()) {
786                                //this should never happen
787                                //the regex contains a "match-all" pattern and should never not find anything
788                                context.addWarning(7, BYDAY, value);
789                                continue;
790                        }
791
792                        String dayStr = m.group(2);
793                        DayOfWeek day = DayOfWeek.valueOfAbbr(dayStr);
794                        if (day == null) {
795                                context.addWarning(7, BYDAY, value);
796                                continue;
797                        }
798
799                        String prefixStr = m.group(1);
800                        Integer prefix = (prefixStr == null) ? null : Integer.valueOf(prefixStr); //no need to catch NumberFormatException because the regex guarantees that it will be a number
801
802                        builder.byDay(prefix, day);
803                }
804        }

869        private ListMultimap<String, Object> buildComponents(T property, WriteContext context, boolean extended) {
870                ListMultimap<String, Object> components = new ListMultimap<String, Object>();
871                Recurrence recur = property.getValue();
872
873                //FREQ must come first
874                if (recur.getFrequency() != null) {
875                        components.put(FREQ, recur.getFrequency().name());
876                }
877
878                ICalDate until = recur.getUntil();
879                if (until != null) {
880                        components.put(UNTIL, writeUntil(until, context, extended));
881                }
882
883                if (recur.getCount() != null) {
884                        components.put(COUNT, recur.getCount());
885                }
886
887                if (recur.getInterval() != null) {
888                        components.put(INTERVAL, recur.getInterval());
889                }
890
891                components.putAll(BYSECOND, recur.getBySecond());
892                components.putAll(BYMINUTE, recur.getByMinute());
893                components.putAll(BYHOUR, recur.getByHour());
894
895                for (ByDay byDay : recur.getByDay()) {
896                        Integer prefix = byDay.getNum();
897                        DayOfWeek day = byDay.getDay();
898
899                        String value = day.getAbbr();
900                        if (prefix != null) {
901                                value = prefix + value;
902                        }
903                        components.put(BYDAY, value);
904                }
905
906                components.putAll(BYMONTHDAY, recur.getByMonthDay());
907                components.putAll(BYYEARDAY, recur.getByYearDay());
908                components.putAll(BYWEEKNO, recur.getByWeekNo());
909                components.putAll(BYMONTH, recur.getByMonth());
910                components.putAll(BYSETPOS, recur.getBySetPos());
911
912                if (recur.getWorkweekStarts() != null) {
913                        components.put(WKST, recur.getWorkweekStarts().getAbbr());
914                }
915
916                for (Map.Entry<String, List<String>> entry : recur.getXRules().entrySet()) {
917                        String name = entry.getKey();
918                        List<String> values = entry.getValue();
919                        components.putAll(name, values);
920                }
921
922                return components;
923        }

925        private String writeUntil(ICalDate until, WriteContext context, boolean extended) {
926                if (!until.hasTime()) {
927                        return date(until).extended(extended).write();
928                }
929
930                /*
931                 * RFC 5545 p.41
932                 * 
933                 * In the case of the "STANDARD" and "DAYLIGHT" sub-components the UNTIL
934                 * rule part MUST always be specified as a date with UTC time. If
935                 * specified as a DATE-TIME value, then it MUST be specified in a UTC
936                 * time format.
937                 */
938
939                if (isInObservance(context)) {
940                        return date(until).utc(true).extended(extended).write();
941                }
942
943                /*
944                 * RFC 2445 p.42
945                 * 
946                 * If specified as a date-time value, then it MUST be specified in an
947                 * UTC time format.
948                 */
949                if (context.getVersion() == ICalVersion.V2_0_DEPRECATED) {
950                        return date(until).extended(extended).utc(true).write();
951                }
952
953                /*
954                 * RFC 5545 p.41
955                 * 
956                 * Furthermore, if the "DTSTART" property is specified as a date with
957                 * local time, then the UNTIL rule part MUST also be specified as a date
958                 * with local time. If the "DTSTART" property is specified as a date
959                 * with UTC time or a date with local time and time zone reference, then
960                 * the UNTIL rule part MUST be specified as a date with UTC time.
961                 */
962
963                ICalComponent parent = context.getParent();
964                if (parent == null) {
965                        return date(until).extended(extended).utc(true).write();
966                }
967
968                DateStart dtstart = parent.getProperty(DateStart.class);
969                if (dtstart == null) {
970                        return date(until).extended(extended).utc(true).write();
971                }
972
973                /*
974                 * If DTSTART is floating, then UNTIL should be floating.
975                 */
976                TimezoneInfo tzinfo = context.getTimezoneInfo();
977                boolean dtstartFloating = tzinfo.isFloating(dtstart);
978                if (dtstartFloating) {
979                        return date(until).extended(extended).tz(true, null).write();
980                }
981
982                /*
983                 * Otherwise, UNTIL should be UTC.
984                 */
985                return date(until).extended(extended).utc(true).write();
986        }

988        private void parseFirst(ListMultimap<String, String> rules, String name, Handler<String> handler) {
989                List<String> values = rules.removeAll(name);
990                if (values.isEmpty()) {
991                        return;
992                }
993
994                String value = values.get(0);
995                handler.handle(value);
996        }
