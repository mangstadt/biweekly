038/**
039 * Marshals {@link DisplayAlarm} properties.
040 * @author Michael Angstadt
041 */
042public class DisplayAlarmScribe extends VCalAlarmPropertyScribe<DisplayAlarm> {
043        public DisplayAlarmScribe() {
044                super(DisplayAlarm.class, "DALARM", ICalDataType.TEXT);
045        }
046
047        @Override
048        protected List<String> writeData(DisplayAlarm property) {
049                String text = property.getText();
050                return (text == null) ? Collections.<String>emptyList() : Collections.singletonList(text);
051        }
052
053        @Override
054        protected DisplayAlarm create(ICalDataType dataType, SemiStructuredValueIterator it) {
055                return new DisplayAlarm(it.next());
056        }
057
058        @Override
059        protected void toVAlarm(VAlarm valarm, DisplayAlarm property) {
060                valarm.setDescription(property.getText());
061        }
062
063        @Override
064        protected Action action() {
065                return Action.display();
066        }
067}
