044/**
045 * Marshals {@link Attendee} properties.
046 * @author Michael Angstadt
047 */
048public class AttendeeScribe extends ICalPropertyScribe<Attendee> {
049        public AttendeeScribe() {
050                super(Attendee.class, "ATTENDEE");
051        }
052
053        @Override
054        protected ICalDataType _defaultDataType(ICalVersion version) {
055                switch (version) {
056                case V1_0:
057                        return null;
058                default:
059                        return ICalDataType.CAL_ADDRESS;
060                }
061        }
062
063        @Override
064        protected ICalDataType _dataType(Attendee property, ICalVersion version) {
065                if (version == ICalVersion.V1_0 && property.getUri() != null) {
066                        return ICalDataType.URL;
067                }
068                return defaultDataType(version);
069        }
070
071        @Override
072        protected ICalParameters _prepareParameters(Attendee property, WriteContext context) {
073                /*
074                 * Note: Parameter values are assigned using "put()" instead of the
075                 * appropriate "setter" methods so that any existing parameter values
076                 * are not overwritten.
077                 */
078
079                ICalParameters copy = new ICalParameters(property.getParameters());
080
081                //RSVP parameter
082                //1.0 - Uses the values "YES" and "NO"
083                //2.0 - Uses the values "TRUE" and "FALSE"
084                Boolean rsvp = property.getRsvp();
085                if (rsvp != null) {
086                        String value;
087                        switch (context.getVersion()) {
088                        case V1_0:
089                                value = rsvp ? "YES" : "NO";
090                                break;
091
092                        default:
093                                value = rsvp ? "TRUE" : "FALSE";
094                                break;
095                        }
096
097                        copy.put(ICalParameters.RSVP, value);
098                }
099
100                //ROLE and EXPECT parameters
101                //1.0 - Uses ROLE and EXPECT
102                //2.0 - Uses only ROLE
103                Role role = property.getRole();
104                ParticipationLevel level = property.getParticipationLevel();
105                switch (context.getVersion()) {
106                case V1_0:
107                        if (role != null) {
108                                copy.put(ICalParameters.ROLE, role.getValue());
109                        }
110                        if (level != null) {
111                                copy.put(ICalParameters.EXPECT, level.getValue(context.getVersion()));
112                        }
113                        break;
114
115                default:
116                        String value = null;
117                        if (role == Role.CHAIR) {
118                                value = role.getValue();
119                        } else if (level != null) {
120                                value = level.getValue(context.getVersion());
121                        } else if (role != null) {
122                                value = role.getValue();
123                        }
124
125                        if (value != null) {
126                                copy.put(ICalParameters.ROLE, value);
127                        }
128                        break;
129                }
130
131                //PARTSTAT vs STATUS
132                //1.0 - Calls the parameter "STATUS"
133                //2.0 - Calls the parameter "PARTSTAT"
134                ParticipationStatus partStat = property.getParticipationStatus();
135                if (partStat != null) {
136                        String paramName;
137                        String paramValue;
138
139                        switch (context.getVersion()) {
140                        case V1_0:
141                                paramName = ICalParameters.STATUS;
142                                paramValue = (partStat == ParticipationStatus.NEEDS_ACTION) ? "NEEDS ACTION" : partStat.getValue();
143                                break;
144
145                        default:
146                                paramName = ICalParameters.PARTSTAT;
147                                paramValue = partStat.getValue();
148                                break;
149                        }
150
151                        copy.put(paramName, paramValue);
152                }
153
154                //CN parameter
155                String name = property.getCommonName();
156                if (name != null && context.getVersion() != ICalVersion.V1_0) {
157                        copy.put(ICalParameters.CN, name);
158                }
159
160                //EMAIL parameter
161                String uri = property.getUri();
162                String email = property.getEmail();
163                if (uri != null && email != null && context.getVersion() != ICalVersion.V1_0) {
164                        copy.put(ICalParameters.EMAIL, email);
165                }
166
167                return copy;
168        }
169
170        @Override
171        protected Attendee _parseText(String value, ICalDataType dataType, ICalParameters parameters, ParseContext context) {
172                String uri = null, name = null, email = null;
173                Boolean rsvp = null;
174                Role role = null;
175                ParticipationLevel participationLevel = null;
176                ParticipationStatus participationStatus = null;
177
178                switch (context.getVersion()) {
179                case V1_0:
180                        Iterator<String> it = parameters.get(ICalParameters.RSVP).iterator();
181                        while (it.hasNext()) {
182                                String rsvpStr = it.next();
183
184                                if ("YES".equalsIgnoreCase(rsvpStr)) {
185                                        rsvp = Boolean.TRUE;
186                                        it.remove();
187                                        break;
188                                }
189
190                                if ("NO".equalsIgnoreCase(rsvpStr)) {
191                                        rsvp = Boolean.FALSE;
192                                        it.remove();
193                                        break;
194                                }
195                        }
196
197                        String roleStr = parameters.first(ICalParameters.ROLE);
198                        if (roleStr != null) {
199                                role = Role.get(roleStr);
200                                parameters.remove(ICalParameters.ROLE, roleStr);
201                        }
202
203                        String expectStr = parameters.getExpect();
204                        if (expectStr != null) {
205                                participationLevel = ParticipationLevel.get(expectStr);
206                                parameters.remove(ICalParameters.EXPECT, expectStr);
207                        }
208
209                        String statusStr = parameters.getStatus();
210                        if (statusStr != null) {
211                                participationStatus = ParticipationStatus.get(statusStr);
212                                parameters.remove(ICalParameters.STATUS, statusStr);
213                        }
214
215                        int bracketStart = value.lastIndexOf('<');
216                        int bracketEnd = value.lastIndexOf('>');
217                        if (bracketStart >= 0 && bracketEnd >= 0 && bracketStart < bracketEnd) {
218                                name = value.substring(0, bracketStart).trim();
219                                email = value.substring(bracketStart + 1, bracketEnd).trim();
220                        } else if (dataType == ICalDataType.URL) {
221                                uri = value;
222                        } else {
223                                email = value;
224                        }
225
226                        break;
227
228                default:
229                        it = parameters.get(ICalParameters.RSVP).iterator();
230                        while (it.hasNext()) {
231                                String rsvpStr = it.next();
232
233                                if ("TRUE".equalsIgnoreCase(rsvpStr)) {
234                                        rsvp = Boolean.TRUE;
235                                        it.remove();
236                                        break;
237                                }
238
239                                if ("FALSE".equalsIgnoreCase(rsvpStr)) {
240                                        rsvp = Boolean.FALSE;
241                                        it.remove();
242                                        break;
243                                }
244                        }
245
246                        roleStr = parameters.first(ICalParameters.ROLE);
247                        if (roleStr != null) {
248                                if (roleStr.equalsIgnoreCase(Role.CHAIR.getValue())) {
249                                        role = Role.CHAIR;
250                                } else {
251                                        ParticipationLevel l = ParticipationLevel.find(roleStr);
252                                        if (l == null) {
253                                                role = Role.get(roleStr);
254                                        } else {
255                                                participationLevel = l;
256                                        }
257                                }
258                                parameters.remove(ICalParameters.ROLE, roleStr);
259                        }
260
261                        String participationStatusStr = parameters.getParticipationStatus();
262                        if (participationStatusStr != null) {
263                                participationStatus = ParticipationStatus.get(participationStatusStr);
264                                parameters.remove(ICalParameters.PARTSTAT, participationStatusStr);
265                        }
266
267                        name = parameters.getCommonName();
268                        if (name != null) {
269                                parameters.remove(ICalParameters.CN, name);
270                        }
271
272                        email = parameters.getEmail();
273                        if (email == null) {
274                                int colon = value.indexOf(':');
275                                if (colon == 6) {
276                                        String scheme = value.substring(0, colon);
277                                        if (scheme.equalsIgnoreCase("mailto")) {
278                                                email = value.substring(colon + 1);
279                                        } else {
280                                                uri = value;
281                                        }
282                                } else {
283                                        uri = value;
284                                }
285                        } else {
286                                uri = value;
287                                parameters.remove(ICalParameters.EMAIL, email);
288                        }
289
290                        break;
291                }
292
293                Attendee attendee = new Attendee(name, email, uri);
294                attendee.setParticipationStatus(participationStatus);
295                attendee.setParticipationLevel(participationLevel);
296                attendee.setRole(role);
297                attendee.setRsvp(rsvp);
298
299                if (context.getVersion() == ICalVersion.V1_0 && attendee.getRole() == Role.ORGANIZER) {
300                        Organizer organizer = new Organizer(attendee.getCommonName(), attendee.getEmail());
301                        organizer.setUri(attendee.getUri());
302                        organizer.setParameters(parameters);
303
304                        attendee.setParameters(parameters);
305                        DataModelConversionException conversionException = new DataModelConversionException(attendee);
306                        conversionException.getProperties().add(organizer);
307                        throw conversionException;
308                }
309
310                return attendee;
311        }
312
313        @Override
314        protected String _writeText(Attendee property, WriteContext context) {
315                String uri = property.getUri();
316                if (uri != null) {
317                        return uri;
318                }
319
320                String name = property.getCommonName();
321                String email = property.getEmail();
322                switch (context.getVersion()) {
323                case V1_0:
324                        if (email != null) {
325                                String value = (name == null) ? email : name + " <" + email + ">";
326                                return VObjectPropertyValues.escape(value);
327                        }
328
329                        break;
330
331                default:
332                        if (email != null) {
333                                return "mailto:" + email;
334                        }
335                        break;
336                }
337
338                return "";
339        }
