042/**
043 * Holds the timezone-related settings of an iCalendar object.
044 * @author Michael Angstadt
045 */
046public class TimezoneInfo {
047        @SuppressWarnings("serial")
048        private final Collection<TimezoneAssignment> assignments = new HashSet<TimezoneAssignment>() {
049                @Override
050                public boolean remove(Object assignment) {
051                        //remove all property assignments
052                        Collection<TimezoneAssignment> values = propertyTimezones.values();
053                        while (values.remove(assignment)) {
054                                //empty
055                        }
056
057                        return super.remove(assignment);
058                }
059        };
060        private final Map<ICalProperty, TimezoneAssignment> propertyTimezones = new IdentityHashMap<ICalProperty, TimezoneAssignment>();
061        private final List<ICalProperty> floatingProperties = new ArrayList<ICalProperty>();
062
063        private TimezoneAssignment defaultTimezone;
064        private boolean globalFloatingTime = false;
065
066        /**
067         * Gets all the timezones assigned to this object.
068         * @return the timezones (collection is mutable)
069         */
070        public Collection<TimezoneAssignment> getTimezones() {
071                return assignments;
072        }
073
074        /**
075         * <p>
076         * Gets the timezone to format all date/time values in (by default, all
077         * dates are formatted in UTC).
078         * </p>
079         * <p>
080         * The default timezone is not used for properties that are configured to
081         * use their own timezone (see {@link #setTimezone}).
082         * </p>
083         * @return the timezone or null if using UTC
084         */
085        public TimezoneAssignment getDefaultTimezone() {
086                return defaultTimezone;
087        }
088
089        /**
090         * <p>
091         * Sets the timezone to format all date/time values in (by default, all
092         * dates are formatted in UTC).
093         * </p>
094         * <p>
095         * The default timezone is not used for properties that are configured to
096         * use their own timezone (see {@link #setTimezone}).
097         * </p>
098         * @param timezone the timezone or null to use UTC
099         */
100        public void setDefaultTimezone(TimezoneAssignment timezone) {
101                if (timezone == null) {
102                        if (defaultTimezone != null && !propertyTimezones.containsValue(defaultTimezone)) {
103                                assignments.remove(defaultTimezone);
104                        }
105                } else {
106                        assignments.add(timezone);
107                }
108
109                defaultTimezone = timezone;
110        }
111
112        /**
113         * Assigns a timezone to a specific property.
114         * @param property the property
115         * @param timezone the timezone or null to format the property according to
116         * the default timezone (see {@link #setDefaultTimezone}).
117         */
118        public void setTimezone(ICalProperty property, TimezoneAssignment timezone) {
119                if (timezone == null) {
120                        TimezoneAssignment existing = propertyTimezones.remove(property);
121                        if (existing != null && existing != defaultTimezone && !propertyTimezones.containsValue(existing)) {
122                                assignments.remove(existing);
123                        }
124                        return;
125                }
126
127                assignments.add(timezone);
128                propertyTimezones.put(property, timezone);
129        }
130
131        /**
132         * Gets the timezone that is assigned to a property.
133         * @param property the property
134         * @return the timezone or null if no timezone is assigned to the property
135         */
136        public TimezoneAssignment getTimezone(ICalProperty property) {
137                return propertyTimezones.get(property);
138        }
139
140        /**
141         * <p>
142         * Determines the timezone that a particular property should be formatted in
143         * when written to an output stream.
144         * </p>
145         * <p>
146         * Note: You should call {@link #isFloating} first, to determine if the
147         * property's value is floating (without a timezone).
148         * </p>
149         * @param property the property
150         * @return the timezone or null for UTC
151         */
152        public TimezoneAssignment getTimezoneToWriteIn(ICalProperty property) {
153                TimezoneAssignment assignment = getTimezone(property);
154                return (assignment == null) ? defaultTimezone : assignment;
155        }
156
157        /**
158         * Gets the timezone whose {@link VTimezone} component contains a
159         * {@link TimezoneId} property with the given value.
160         * @param tzid the value of the {@link TimezoneId} property
161         * @return the timezone or null if not found
162         */
163        public TimezoneAssignment getTimezoneById(String tzid) {
164                for (TimezoneAssignment assignment : assignments) {
165                        VTimezone component = assignment.getComponent();
166                        if (component == null) {
167                                continue;
168                        }
169
170                        String componentId = ValuedProperty.getValue(component.getTimezoneId());
171                        if (tzid.equals(componentId)) {
172                                return assignment;
173                        }
174                }
175                return null;
176        }
177
178        /**
179         * <p>
180         * Gets whether to format all date/time values as floating times (defaults
181         * to false).
182         * </p>
183         * <p>
184         * This setting does not apply to properties whose floating time settings
185         * are configured individually (see: {@link #setFloating}) or that are
186         * configured to use their own timezone (see {@link #setTimezone}).
187         * </p>
188         * <p>
189         * A floating time value does not have a timezone associated with it, and is
190         * to be interpreted as being in the local timezone of the computer that is
191         * consuming the iCalendar object.
192         * </p>
193         * @return true if enabled, false if disabled
194         */
195        public boolean isGlobalFloatingTime() {
196                return globalFloatingTime;
197        }
198
199        /**
200         * <p>
201         * Sets whether to format all date/time values as floating times (defaults
202         * to false).
203         * </p>
204         * <p>
205         * This setting does not apply to properties whose floating time settings
206         * are configured individually (see: {@link #setFloating}) or that are
207         * configured to use their own timezone (see {@link #setTimezone}).
208         * </p>
209         * <p>
210         * A floating time value does not have a timezone associated with it, and is
211         * to be interpreted as being in the local timezone of the computer that is
212         * consuming the iCalendar object.
213         * </p>
214         * @param enable true to enable, false to disable
215         */
216        public void setGlobalFloatingTime(boolean enable) {
217                globalFloatingTime = enable;
218        }
219
220        /**
221         * Determines if a property value should be formatted in floating time when
222         * written to an output stream.
223         * @param property the property
224         * @return true to format in floating time, false not to
225         */
226        public boolean isFloating(ICalProperty property) {
227                if (containsIdentity(floatingProperties, property)) {
228                        return true;
229                }
230
231                if (propertyTimezones.containsKey(property)) {
232                        return false;
233                }
234
235                return globalFloatingTime;
236        }
237
238        /**
239         * <p>
240         * Sets whether a property value should be formatted in floating time when
241         * written to an output stream (by default, floating time is disabled for
242         * all properties).
243         * </p>
244         * <p>
245         * A floating time value does not have a timezone associated with it, and is
246         * to be interpreted as being in the local timezone of the computer that is
247         * consuming the iCalendar object.
248         * </p>
249         * @param property the property
250         * @param enable true to enable floating time for this property, false to
251         * disable
252         */
253        public void setFloating(ICalProperty property, boolean enable) {
254                if (enable) {
255                        floatingProperties.add(property);
256                } else {
257                        removeIdentity(floatingProperties, property);
258                }
259        }
260
261        /**
262         * Gets all of the iCalendar {@link VTimezone} components that have been
263         * registered with this object.
264         * @return the components (this collection is immutable)
265         */
266        public Collection<VTimezone> getComponents() {
267                List<VTimezone> components = new ArrayList<VTimezone>(assignments.size());
268                for (TimezoneAssignment assignment : assignments) {
269                        VTimezone component = assignment.getComponent();
270                        if (component != null) {
271                                components.add(component);
272                        }
273                }
274                return Collections.unmodifiableList(components);
275        }
276
277        /**
278         * Removes an object from a list using reference equality.
279         * @param list the list
280         * @param object the object to remove
281         */
282        private static <T> void removeIdentity(List<T> list, T object) {
283                Iterator<T> it = list.iterator();
284                while (it.hasNext()) {
285                        if (object == it.next()) {
286                                it.remove();
287                        }
288                }
289        }
290
291        /**
292         * Searches for an item in a list using reference equality.
293         * @param list the list
294         * @param object the object to search for
295         * @return true if the object was found, false if not
296         */
297        private static <T> boolean containsIdentity(List<T> list, T object) {
298                for (T item : list) {
299                        if (item == object) {
300                                return true;
301                        }
302                }
303                return false;
304        }
305}
