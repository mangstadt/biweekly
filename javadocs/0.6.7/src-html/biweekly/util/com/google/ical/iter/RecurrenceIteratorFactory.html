065/**
066 * <p>
067 * Calculates the occurrences of an individual RRULE definition or groups of
068 * RRULEs, RDATEs, EXRULEs, and EXDATEs.
069 * </p>
070 * <p>
071 * <b>Glossary</b>
072 * </p>
073 * <ul>
074 * <li>Period - year|month|day|...</li>
075 * <li>Day of the week - an int in [0,6]</li>
076 * <li>Day of the year - zero indexed in [0,365]</li>
077 * <li>Day of the month - 1 indexed in [1,31]</li>
078 * <li>Month - 1 indexed integer in [1,12]</li>
079 * </ul>
080 * <p>
081 * <b>Abstractions</b>
082 * </p>
083 * <ul>
084 * <li>Generator - a function corresponding to an RRULE part that takes a date
085 * and returns a later (year or month or day depending on its period) within the
086 * next larger period. A generator ignores all periods in its input smaller than
087 * its period.</li>
088 * <li>Filter - a function that returns true iff the given date matches the
089 * subrule.</li>
090 * <li>Condition - returns true if the given date is past the end of the
091 * recurrence.</li>
092 * </ul>
093 * <p>
094 * All the functions that represent rule parts are stateful.
095 * </p>
096 * @author mikesamuel+svn@gmail.com (Mike Samuel)
097 * @author Michael Angstadt
098 */
099public class RecurrenceIteratorFactory {
100        /**
101         * Creates a recurrence iterator from an RDATE or EXDATE list.
102         * @param dates the list of dates
103         * @return the iterator
104         */
105        public static RecurrenceIterator createRecurrenceIterator(Collection<? extends DateValue> dates) {
106                DateValue[] datesArray = dates.toArray(new DateValue[0]);
107                return new RDateIteratorImpl(datesArray);
108        }

110        /**
111         * Creates a recurrence iterable from an RRULE.
112         * @param rrule the recurrence rule
113         * @param dtStart the start date of the series
114         * @param tzid the timezone that the start date is in, as well as the
115         * timezone to iterate in
116         * @return the iterable
117         */
118        public static RecurrenceIterable createRecurrenceIterable(final Recurrence rrule, final DateValue dtStart, final TimeZone tzid) {
119                return new RecurrenceIterable() {
120                        public RecurrenceIterator iterator() {
121                                return createRecurrenceIterator(rrule, dtStart, tzid);
122                        }
123                };
124        }

126        /**
127         * Creates a recurrence iterator from an RRULE.
128         * @param rrule the recurrence rule
129         * @param dtStart the start date of the series
130         * @param tzid the timezone that the start date is in, as well as the
131         * timezone to iterate in
132         * @return the iterator
133         */
134        public static RecurrenceIterator createRecurrenceIterator(Recurrence rrule, DateValue dtStart, TimeZone tzid) {
135                Frequency freq = rrule.getFrequency();
136
137                /*
138                 * If the given RRULE is malformed and does not have a frequency
139                 * specified, default to "yearly".
140                 */
141                if (freq == null) {
142                        freq = Frequency.YEARLY;
143                }
144
145                DayOfWeek wkst = rrule.getWorkweekStarts();
146
147                ICalDate until = rrule.getUntil();
148                int count = toInt(rrule.getCount());
149                int interval = toInt(rrule.getInterval());
150                ByDay[] byDay = rrule.getByDay().toArray(new ByDay[0]);
151                int[] byMonth = toIntArray(rrule.getByMonth());
152                int[] byMonthDay = toIntArray(rrule.getByMonthDay());
153                int[] byWeekNo = toIntArray(rrule.getByWeekNo());
154                int[] byYearDay = toIntArray(rrule.getByYearDay());
155                int[] bySetPos = toIntArray(rrule.getBySetPos());
156                int[] byHour = toIntArray(rrule.getByHour());
157                int[] byMinute = toIntArray(rrule.getByMinute());
158                int[] bySecond = toIntArray(rrule.getBySecond());
159                boolean canShortcutAdvance = true;
160
161                if (interval <= 0) {
162                        interval = 1;
163                }
164
165                if (wkst == null) {
166                        wkst = DayOfWeek.MONDAY;
167                }
168
169                //optimize out BYSETPOS where possible
170                if (bySetPos.length > 0) {
171                        switch (freq) {
172                        case HOURLY:
173                                if (byHour.length > 0 && byMinute.length <= 1 && bySecond.length <= 1) {
174                                        byHour = filterBySetPos(byHour, bySetPos);
175                                }
176
177                                /*
178                                 * Handling bySetPos for rules that are more frequent than daily
179                                 * tends to lead to large amounts of processor being used before
180                                 * other work limiting features can kick in since there many
181                                 * seconds between dtStart and where the year limit kicks in.
182                                 * There are no known use cases for the use of bySetPos with
183                                 * hourly minutely and secondly rules so we just ignore it.
184                                 */
185                                bySetPos = NO_INTS;
186                                break;
187                        case MINUTELY:
188                                if (byMinute.length > 0 && bySecond.length <= 1) {
189                                        byMinute = filterBySetPos(byMinute, bySetPos);
190                                }
191                                //see bySetPos handling comment above
192                                bySetPos = NO_INTS;
193                                break;
194                        case SECONDLY:
195                                if (bySecond.length > 0) {
196                                        bySecond = filterBySetPos(bySecond, bySetPos);
197                                }
198                                //see bySetPos handling comment above
199                                bySetPos = NO_INTS;
200                                break;
201                        default:
202                        }
203
204                        canShortcutAdvance = false;
205                }
206
207                DateValue start = dtStart;
208                if (bySetPos.length > 0) {
209                        /*
210                         * Roll back until the beginning of the period to make sure that any
211                         * positive indices are indexed properly. The actual iterator
212                         * implementation is responsible for anything < dtStart.
213                         */
214                        switch (freq) {
215                        case YEARLY:
216                                if (dtStart instanceof TimeValue) {
217                                        TimeValue tv = (TimeValue) dtStart;
218                                        start = new DateTimeValueImpl(start.year(), 1, 1, tv.hour(), tv.minute(), tv.second());
219                                } else {
220                                        start = new DateValueImpl(start.year(), 1, 1);
221                                }
222                                break;
223                        case MONTHLY:
224                                if (dtStart instanceof TimeValue) {
225                                        TimeValue tv = (TimeValue) dtStart;
226                                        start = new DateTimeValueImpl(start.year(), start.month(), 1, tv.hour(), tv.minute(), tv.second());
227                                } else {
228                                        start = new DateValueImpl(start.year(), start.month(), 1);
229                                }
230                                break;
231                        case WEEKLY:
232                                int d = (7 + wkst.ordinal() - TimeUtils.dayOfWeek(dtStart).getCalendarConstant()) % 7;
233                                start = TimeUtils.add(dtStart, new DateValueImpl(0, 0, -d));
234                                break;
235                        default:
236                                break;
237                        }
238                }
239
240                /*
241                 * Recurrences are implemented as a sequence of periodic generators.
242                 * First a year is generated, and then months, and within months, days.
243                 */
244                ThrottledGenerator yearGenerator = Generators.serialYearGenerator(freq == Frequency.YEARLY ? interval : 1, dtStart);
245                Generator monthGenerator = null;
246                Generator dayGenerator = null;
247                Generator secondGenerator = null;
248                Generator minuteGenerator = null;
249                Generator hourGenerator = null;
250
251                /*
252                 * When multiple generators are specified for a period, they act as a
253                 * union operator. We could have multiple generators (say, for day) and
254                 * then run each and merge the results, but some generators are more
255                 * efficient than others. So to avoid generating 53 Sundays and throwing
256                 * away all but 1 for RRULE:FREQ=YEARLY;BYDAY=TU;BYWEEKNO=1, we
257                 * reimplement some of the more prolific generators as filters.
258                 */
259                // TODO(msamuel): don't need a list here
260                List<Predicate<? super DateValue>> filters = new ArrayList<Predicate<? super DateValue>>();
261
262                switch (freq) {
263                case SECONDLY:
264                        if (bySecond.length == 0 || interval != 1) {
265                                secondGenerator = Generators.serialSecondGenerator(interval, dtStart);
266                                if (bySecond.length > 0) {
267                                        filters.add(Filters.bySecondFilter(bySecond));
268                                }
269                        }
270                        break;
271                case MINUTELY:
272                        if (byMinute.length == 0 || interval != 1) {
273                                minuteGenerator = Generators.serialMinuteGenerator(interval, dtStart);
274                                if (byMinute.length > 0) {
275                                        filters.add(Filters.byMinuteFilter(byMinute));
276                                }
277                        }
278                        break;
279                case HOURLY:
280                        if (byHour.length == 0 || interval != 1) {
281                                hourGenerator = Generators.serialHourGenerator(interval, dtStart);
282                                if (byHour.length > 0) {
283                                        filters.add(Filters.byHourFilter(bySecond));
284                                }
285                        }
286                        break;
287                case DAILY:
288                        break;
289                case WEEKLY:
290                        /*
291                         * Week is not considered a period because a week may span multiple
292                         * months and/or years. There are no week generators, so a filter is
293                         * used to make sure that FREQ=WEEKLY;INTERVAL=2 only generates
294                         * dates within the proper week.
295                         */
296                        if (byDay.length > 0) {
297                                dayGenerator = Generators.byDayGenerator(byDay, false, start);
298                                byDay = NO_DAYS;
299                                if (interval > 1) {
300                                        filters.add(Filters.weekIntervalFilter(interval, wkst, dtStart));
301                                }
302                        } else {
303                                dayGenerator = Generators.serialDayGenerator(interval * 7, dtStart);
304                        }
305                        break;
306                case YEARLY:
307                        if (byYearDay.length > 0) {
308                                /*
309                                 * The BYYEARDAY rule part specifies a COMMA separated list of
310                                 * days of the year. Valid values are 1 to 366 or -366 to -1.
311                                 * For example, -1 represents the last day of the year (December
312                                 * 31st) and -306 represents the 306th to the last day of the
313                                 * year (March 1st).
314                                 */
315                                dayGenerator = Generators.byYearDayGenerator(byYearDay, start);
316                                break;
317                        }
318                        // $FALL-THROUGH$
319                case MONTHLY:
320                        if (byMonthDay.length > 0) {
321                                /*
322                                 * The BYMONTHDAY rule part specifies a COMMA separated list of
323                                 * days of the month. Valid values are 1 to 31 or -31 to -1. For
324                                 * example, -10 represents the tenth to the last day of the
325                                 * month.
326                                 */
327                                dayGenerator = Generators.byMonthDayGenerator(byMonthDay, start);
328                                byMonthDay = NO_INTS;
329                        } else if (byWeekNo.length > 0 && Frequency.YEARLY == freq) {
330                                /*
331                                 * The BYWEEKNO rule part specifies a COMMA separated list of
332                                 * ordinals specifying weeks of the year. This rule part is only
333                                 * valid for YEARLY rules.
334                                 */
335                                dayGenerator = Generators.byWeekNoGenerator(byWeekNo, wkst, start);
336                                byWeekNo = NO_INTS;
337                        } else if (byDay.length > 0) {
338                                /*
339                                 * Each BYDAY value can also be preceded by a positive (n) or
340                                 * negative (-n) integer. If present, this indicates the nth
341                                 * occurrence of the specific day within the MONTHLY or YEARLY
342                                 * RRULE. For example, within a MONTHLY rule, +1MO (or simply
343                                 * 1MO) represents the first Monday within the month, whereas
344                                 * -1MO represents the last Monday of the month. If an integer
345                                 * modifier is not present, it means all days of this type
346                                 * within the specified frequency. For example, within a MONTHLY
347                                 * rule, MO represents all Mondays within the month.
348                                 */
349                                dayGenerator = Generators.byDayGenerator(byDay, Frequency.YEARLY == freq && byMonth.length == 0, start);
350                                byDay = NO_DAYS;
351                        } else {
352                                if (Frequency.YEARLY == freq) {
353                                        monthGenerator = Generators.byMonthGenerator(new int[] { dtStart.month() }, start);
354                                }
355                                dayGenerator = Generators.byMonthDayGenerator(new int[] { dtStart.day() }, start);
356                        }
357                        break;
358                }
359
360                if (secondGenerator == null) {
361                        secondGenerator = Generators.bySecondGenerator(bySecond, start);
362                }
363                if (minuteGenerator == null) {
364                        if (byMinute.length == 0 && freq.compareTo(Frequency.MINUTELY) < 0) {
365                                minuteGenerator = Generators.serialMinuteGenerator(1, dtStart);
366                        } else {
367                                minuteGenerator = Generators.byMinuteGenerator(byMinute, start);
368                        }
369                }
370                if (hourGenerator == null) {
371                        if (byHour.length == 0 && freq.compareTo(Frequency.HOURLY) < 0) {
372                                hourGenerator = Generators.serialHourGenerator(1, dtStart);
373                        } else {
374                                hourGenerator = Generators.byHourGenerator(byHour, start);
375                        }
376                }
377
378                if (dayGenerator == null) {
379                        boolean dailyOrMoreOften = freq.compareTo(Frequency.DAILY) <= 0;
380                        if (byMonthDay.length > 0) {
381                                dayGenerator = Generators.byMonthDayGenerator(byMonthDay, start);
382                                byMonthDay = NO_INTS;
383                        } else if (byDay.length > 0) {
384                                dayGenerator = Generators.byDayGenerator(byDay, Frequency.YEARLY == freq, start);
385                                byDay = NO_DAYS;
386                        } else if (dailyOrMoreOften) {
387                                dayGenerator = Generators.serialDayGenerator(Frequency.DAILY == freq ? interval : 1, dtStart);
388                        } else {
389                                dayGenerator = Generators.byMonthDayGenerator(new int[] { dtStart.day() }, start);
390                        }
391                }
392
393                if (byDay.length > 0) {
394                        filters.add(Filters.byDayFilter(byDay, Frequency.YEARLY == freq, wkst));
395                        byDay = NO_DAYS;
396                }
397
398                if (byMonthDay.length > 0) {
399                        filters.add(Filters.byMonthDayFilter(byMonthDay));
400                }
401
402                //generator inference common to all periods
403                if (byMonth.length > 0) {
404                        monthGenerator = Generators.byMonthGenerator(byMonth, start);
405                } else if (monthGenerator == null) {
406                        monthGenerator = Generators.serialMonthGenerator(freq == Frequency.MONTHLY ? interval : 1, dtStart);
407                }
408
409                /*
410                 * The condition tells the iterator when to halt. The condition is
411                 * exclusive, so the date that triggers it will not be included.
412                 */
413                Predicate<DateValue> condition;
414                if (count != 0) {
415                        condition = Conditions.countCondition(count);
416
417                        /*
418                         * We can't shortcut because the countCondition must see every
419                         * generated instance.
420                         * 
421                         * TODO(msamuel): If count is large, we might try predicting the end
422                         * date so that we can convert the COUNT condition to an UNTIL
423                         * condition.
424                         */
425                        canShortcutAdvance = false;
426                } else if (until != null) {
427                        DateValue untilUtc;
428                        if (until.hasTime()) {
429                                TimeZone utc = TimeZone.getTimeZone("UTC");
430                                untilUtc = Google2445Utils.convert(until, utc);
431                        } else {
432                                //treat the ICalDate object as a timezone-less, calendar date
433                                Calendar c = Calendar.getInstance();
434                                c.setTime(until);
435                                untilUtc = new DateValueImpl( //@formatter:off
436                                        c.get(Calendar.YEAR),
437                                        c.get(Calendar.MONTH) + 1,
438                                        c.get(Calendar.DAY_OF_MONTH)
439                                ); //@formatter:on
440                        }
441
442                        if ((untilUtc instanceof TimeValue) != (dtStart instanceof TimeValue)) {
443                                // TODO(msamuel): warn
444                                if (dtStart instanceof TimeValue) {
445                                        untilUtc = TimeUtils.dayStart(untilUtc);
446                                } else {
447                                        untilUtc = TimeUtils.toDateValue(untilUtc);
448                                }
449                        }
450                        condition = Conditions.untilCondition(untilUtc);
451                } else {
452                        condition = Predicates.alwaysTrue();
453                }
454
455                //combine filters into a single function
456                Predicate<? super DateValue> filter;
457                switch (filters.size()) {
458                case 0:
459                        filter = Predicates.<DateValue> alwaysTrue();
460                        break;
461                case 1:
462                        filter = filters.get(0);
463                        break;
464                default:
465                        filter = Predicates.and(filters);
466                        break;
467                }
468
469                Generator instanceGenerator;
470                if (bySetPos.length > 0) {
471                        instanceGenerator = InstanceGenerators.bySetPosInstanceGenerator(bySetPos, freq, wkst, filter, yearGenerator, monthGenerator, dayGenerator, hourGenerator, minuteGenerator, secondGenerator);
472                } else {
473                        instanceGenerator = InstanceGenerators.serialInstanceGenerator(filter, yearGenerator, monthGenerator, dayGenerator, hourGenerator, minuteGenerator, secondGenerator);
474                }
475
476                return new RRuleIteratorImpl(dtStart, tzid, condition, instanceGenerator, yearGenerator, monthGenerator, dayGenerator, hourGenerator, minuteGenerator, secondGenerator, canShortcutAdvance);
477        }

479        /**
480         * Generates a recurrence iterator that iterates over the union of the given
481         * recurrence iterators.
482         * @param first the first recurrence iterator
483         * @param rest the other recurrence iterators
484         * @return the union iterator
485         */
486        public static RecurrenceIterator join(RecurrenceIterator first, RecurrenceIterator... rest) {
487                List<RecurrenceIterator> all = new ArrayList<RecurrenceIterator>();
488                all.add(first);
489                all.addAll(Arrays.asList(rest));
490                return new CompoundIteratorImpl(all, Collections.<RecurrenceIterator> emptyList());
491        }

493        /**
494         * <p>
495         * Generates a recurrence iterator that iterates over all the dates in a
496         * {@link RecurrenceIterator}, excluding those dates found in another
497         * {@link RecurrenceIterator}.
498         * </p>
499         * <p>
500         * Exclusions trump inclusions, and {@link DateValue dates} and
501         * {@link DateTimeValue date-times} never match one another.
502         * </p>
503         * @param included the dates to include
504         * @param excluded the dates to exclude
505         * @return the resultant iterator
506         */
507        public static RecurrenceIterator except(RecurrenceIterator included, RecurrenceIterator excluded) {
508                return new CompoundIteratorImpl(Collections.singleton(included), Collections.singleton(excluded));
509        }

511        /**
512         * <p>
513         * Creates an optimized version of an array based on the given BYSETPOS
514         * array.
515         * </p>
516         * <p>
517         * For example, given the array <code>BYMONTH=2,3,4,5</code> and a BYSETPOS
518         * of <code>BYSETPOS=1,-1</code>, this method will return
519         * <code>BYMONTH=2,5</code>.
520         * </p>
521         * @param members the array to optimize
522         * @param bySetPos the BYSETPOS array
523         * @return the optimized array
524         */
525        private static int[] filterBySetPos(int[] members, int[] bySetPos) {
526                members = Util.uniquify(members);
527                IntSet iset = new IntSet();
528                for (int pos : bySetPos) {
529                        if (pos == 0) {
530                                continue;
531                        }
532                        if (pos < 0) {
533                                pos += members.length;
534                        } else {
535                                --pos; // Zero-index.
536                        }
537                        if (pos >= 0 && pos < members.length) {
538                                iset.add(members[pos]);
539                        }
540                }
541                return iset.toIntArray();
542        }

544        /**
545         * Converts an {@link Integer} list to an int array. Null values are
546         * converted to zero.
547         * @param list the {@link Integer} list
548         * @return the int array
549         */
550        private static int[] toIntArray(List<Integer> list) {
551                int[] array = new int[list.size()];
552                int i = 0;
553                for (Integer intObj : list) {
554                        array[i++] = toInt(intObj);
555                }
556                return array;
557        }
558
559        private static int toInt(Integer integer) {
560                return (integer == null) ? 0 : integer;
561        }

564        private static final ByDay[] NO_DAYS = new ByDay[0];
