040package biweekly.util.com.google.ical.compat.javautil;
041
042import java.util.Calendar;
043import java.util.Date;
044import java.util.GregorianCalendar;
045
046import biweekly.util.com.google.ical.iter.RecurrenceIterable;
047import biweekly.util.com.google.ical.iter.RecurrenceIterator;
048import biweekly.util.com.google.ical.iter.RecurrenceIteratorFactory;
049import biweekly.util.com.google.ical.util.TimeUtils;
050import biweekly.util.com.google.ical.values.DateTimeValueImpl;
051import biweekly.util.com.google.ical.values.DateValue;
052import biweekly.util.com.google.ical.values.DateValueImpl;
053import biweekly.util.com.google.ical.values.TimeValue;
054
055/**
056 * A factory for converting RRULEs and RDATEs into
057 * <code>Iterator&lt;Date&gt;</code> and <code>Iterable&lt;Date&gt;</code>.
058 * @see RecurrenceIteratorFactory
059 * @author mikesamuel+svn@gmail.com (Mike Samuel)
060 * @author Michael Angstadt
061 */
062public class DateIteratorFactory {
063        /**
064         * Creates a date iterator from a recurrence iterator.
065         * @param rit the recurrence iterator
066         * @return the date iterator
067         */
068        public static DateIterator createDateIterator(RecurrenceIterator rit) {
069                return new RecurrenceIteratorWrapper(rit);
070        }
071
072        /**
073         * Creates a date iterable from a recurrence iterable.
074         * @param rit the recurrence iterable
075         * @return the date iterable
076         */
077        public static DateIterable createDateIterable(RecurrenceIterable rit) {
078                return new RecurrenceIterableWrapper(rit);
079        }
080
081        private static final class RecurrenceIterableWrapper implements DateIterable {
082                private final RecurrenceIterable it;
083
084                public RecurrenceIterableWrapper(RecurrenceIterable it) {
085                        this.it = it;
086                }
087
088                public DateIterator iterator() {
089                        return new RecurrenceIteratorWrapper(it.iterator());
090                }
091        }
092
093        private static final class RecurrenceIteratorWrapper implements DateIterator {
094                private final RecurrenceIterator it;
095                private final Calendar utcCalendar = new GregorianCalendar(TimeUtils.utcTimezone());
096
097                public RecurrenceIteratorWrapper(RecurrenceIterator it) {
098                        this.it = it;
099                }
100
101                public boolean hasNext() {
102                        return it.hasNext();
103                }
104
105                public Date next() {
106                        return toDate(it.next());
107                }
108
109                public void advanceTo(Date d) {
110                        it.advanceTo(toDateValue(d));
111                }
112
113                public void remove() {
114                        throw new UnsupportedOperationException();
115                }
116
117                /**
118                 * Converts a {@link DateValue} object into a Java {@link Date} object.
119                 * @param dateValue the date value object (assumed to be in UTC time)
120                 * @return the Java date object
121                 */
122                private Date toDate(DateValue dateValue) {
123                        TimeValue time = TimeUtils.timeOf(dateValue);
124                        utcCalendar.clear();
125                        //@formatter:off
126                        utcCalendar.set(
127                                dateValue.year(),
128                                dateValue.month() - 1, //java.util's dates are zero-indexed
129                                dateValue.day(),
130                                time.hour(),
131                                time.minute(),
132                                time.second()
133                        );
134                        //@formatter:on
135                        return utcCalendar.getTime();
136                }
137
138                /**
139                 * Converts a Java {@link Date} object into a {@link DateValue} object.
140                 * The {@link DateValue} object will be in UTC time.
141                 * @param date the Java date object
142                 * @return the date value object (in UTC time)
143                 */
144                private DateValue toDateValue(Date date) {
145                        utcCalendar.setTime(date);
146
147                        int year = utcCalendar.get(Calendar.YEAR);
148                        int month = utcCalendar.get(Calendar.MONTH) + 1; //java.util's dates are zero-indexed
149                        int day = utcCalendar.get(Calendar.DAY_OF_MONTH);
150                        int hour = utcCalendar.get(Calendar.HOUR_OF_DAY);
151                        int minute = utcCalendar.get(Calendar.MINUTE);
152                        int second = utcCalendar.get(Calendar.SECOND);
153
154                        /*
155                         * We need to treat midnight as a date value so that passing in
156                         * dateValueToDate(<some-date-value>) will not advance past any
157                         * occurrences of some-date-value in the iterator.
158                         */
159                        if ((hour | minute | second) == 0) {
160                                return new DateValueImpl(year, month, day);
161                        }
162                        return new DateTimeValueImpl(year, month, day, hour, minute, second);
163                }
164        }
165
166        private DateIteratorFactory() {
167                // uninstantiable
168        }
169}
