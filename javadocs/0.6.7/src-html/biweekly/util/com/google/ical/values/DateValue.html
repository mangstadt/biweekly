043package biweekly.util.com.google.ical.values;
044
045/**
046 * A calendar date.
047 * @author Neal Gafter
048 * @author Michael Angstadt
049 */
050public interface DateValue extends Comparable<DateValue> {
051        /**
052         * Gets the Gregorian year (for example, 2004).
053         * @return the year
054         */
055        int year();
056
057        /**
058         * Gets the Gregorian month (in the range 1-12).
059         * @return the month
060         */
061        int month();
062
063        /**
064         * Gets the Gregorian day of the month (in the range 1-31).
065         * @return the day of the month
066         */
067        int day();
068}
