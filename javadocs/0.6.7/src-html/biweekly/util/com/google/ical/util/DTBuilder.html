040package biweekly.util.com.google.ical.util;
041
042import biweekly.util.com.google.ical.values.DateTimeValue;
043import biweekly.util.com.google.ical.values.DateTimeValueImpl;
044import biweekly.util.com.google.ical.values.DateValue;
045import biweekly.util.com.google.ical.values.DateValueImpl;
046import biweekly.util.com.google.ical.values.TimeValue;
047
048/**
049 * A mutable buffer used to build {@link DateValue}s and {@link DateTimeValue}s.
050 * @author mikesamuel+svn@gmail.com (Mike Samuel)
051 * @author Michael Angstadt
052 */
053public class DTBuilder {
054        /**
055         * The year.
056         */
057        public int year;
058
059        /**
060         * The month. This value is one-indexed, so "1" represents January.
061         */
062        public int month;
063
064        /**
065         * The day of the month.
066         */
067        public int day;
068
069        /**
070         * The hour.
071         */
072        public int hour;
073
074        /**
075         * The minute.
076         */
077        public int minute;
078
079        /**
080         * The second.
081         */
082        public int second;
083
084        /**
085         * Creates a new date builder.
086         * @param year the initial year
087         * @param month the initial month (this value is one-indexed, so "1"
088         * represents January)
089         * @param day the initial day
090         * @param hour the initial hour
091         * @param minute the initial minute
092         * @param second the initial second
093         */
094        public DTBuilder(int year, int month, int day, int hour, int minute, int second) {
095                this.year = year;
096                this.month = month;
097                this.day = day;
098                this.hour = hour;
099                this.minute = minute;
100                this.second = second;
101        }
102
103        /**
104         * Creates a new date builder. This constructor sets the time components to
105         * zero.
106         * @param year the initial year
107         * @param month the initial month (this value is one-indexed, so "1"
108         * represents January)
109         * @param day the initial day
110         */
111        public DTBuilder(int year, int month, int day) {
112                this(year, month, day, 0, 0, 0);
113        }
114
115        /**
116         * Creates a new date builder, initializing it to the given date value.
117         * @param date the date value to initialize the builder with
118         */
119        public DTBuilder(DateValue date) {
120                this.year = date.year();
121                this.month = date.month();
122                this.day = date.day();
123                if (date instanceof TimeValue) {
124                        TimeValue tv = (TimeValue) date;
125                        this.hour = tv.hour();
126                        this.minute = tv.minute();
127                        this.second = tv.second();
128                }
129        }
130
131        /**
132         * Produces a normalized date-time value, using zero for the time fields if
133         * none were provided.
134         * @return the date-time value
135         */
136        public DateTimeValue toDateTime() {
137                normalize();
138                return new DateTimeValueImpl(year, month, day, hour, minute, second);
139        }
140
141        /**
142         * Produces a normalized date value.
143         * @return the date value
144         */
145        public DateValue toDate() {
146                normalize();
147                return new DateValueImpl(year, month, day);
148        }
149
150        /**
151         * <p>
152         * Compares the value of this builder to a given {@link DateValue}. Note
153         * that this method's behavior is undefined unless {@link #normalize} is
154         * called first.
155         * </p>
156         * <p>
157         * If you're not sure whether it's appropriate to use this method, use
158         * <code>toDateValue().compareTo(dv)</code> instead.
159         * </p>
160         * @param date the date value to compare against
161         * @return a negative value if this date builder is less than the given date
162         * value, a positive value if this date builder is greater than the given
163         * date value, or zero if they are equal
164         */
165        public int compareTo(DateValue date) {
166                long dvComparable = (((((long) date.year()) << 4) + date.month()) << 5) + date.day();
167                long dtbComparable = ((((long) year << 4) + month << 5)) + day;
168                if (date instanceof TimeValue) {
169                        TimeValue tv = (TimeValue) date;
170                        dvComparable = (((((dvComparable << 5) + tv.hour()) << 6) + tv.minute()) << 6) + tv.second();
171                        dtbComparable = (((((dtbComparable << 5) + hour) << 6) + minute) << 6) + second;
172                }
173                long delta = dtbComparable - dvComparable;
174                return delta < 0 ? -1 : delta == 0 ? 0 : 1;
175        }
176
177        /**
178         * Makes sure that the fields are in the proper ranges (for example,
179         * converts 32 January to 1 February, and 25:00:00 to 1:00:00 of the next
180         * day).
181         */
182        public void normalize() {
183                normalizeTime();
184                normalizeDate();
185        }
186
187        @Override
188        public String toString() {
189                return year + "-" + month + "-" + day + " " + hour + ":" + minute + ":" + second;
190        }
191
192        @Override
193        public boolean equals(Object o) {
194                if (!(o instanceof DTBuilder)) {
195                        return false;
196                }
197                DTBuilder that = (DTBuilder) o;
198                return this.year == that.year && this.month == that.month && this.day == that.day && this.hour == that.hour && this.minute == that.minute && this.second == that.second;
199        }
200
201        @Override
202        public int hashCode() {
203                return ((((((((year << 4) + month << 5) + day) << 5) + hour) << 6) + minute) << 6) + second;
204        }
205
206        /**
207         * Makes sure that the time fields are in the proper ranges (for example,
208         * converts 25:00:00 to 1:00:00 of the next day).
209         */
210        private void normalizeTime() {
211                int addMinutes = ((second < 0) ? (second - 59) : second) / 60;
212                second -= addMinutes * 60;
213                minute += addMinutes;
214                int addHours = ((minute < 0) ? (minute - 59) : minute) / 60;
215                minute -= addHours * 60;
216                hour += addHours;
217                int addDays = ((hour < 0) ? (hour - 23) : hour) / 24;
218                hour -= addDays * 24;
219                day += addDays;
220        }
221
222        /**
223         * Makes sure that the date fields are in the proper ranges (for example,
224         * converts 32 January to 1 February).
225         */
226        private void normalizeDate() {
227                while (day <= 0) {
228                        int days = TimeUtils.yearLength(month > 2 ? year : year - 1);
229                        day += days;
230                        --year;
231                }
232
233                if (month <= 0) {
234                        int years = month / 12 - 1;
235                        year += years;
236                        month -= 12 * years;
237                } else if (month > 12) {
238                        int years = (month - 1) / 12;
239                        year += years;
240                        month -= 12 * years;
241                }
242
243                while (true) {
244                        if (month == 1) {
245                                int yearLength = TimeUtils.yearLength(year);
246                                if (day > yearLength) {
247                                        ++year;
248                                        day -= yearLength;
249                                }
250                        }
251
252                        int monthLength = TimeUtils.monthLength(year, month);
253                        if (day <= monthLength) {
254                                break;
255                        }
256
257                        day -= monthLength;
258                        if (++month > 12) {
259                                month -= 12;
260                                ++year;
261                        }
262                }
263        }
264}
