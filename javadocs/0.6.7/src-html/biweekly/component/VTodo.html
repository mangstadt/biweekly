1682        @SuppressWarnings("unchecked")
1683        @Override
1684        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
1685                if (version != ICalVersion.V1_0) {
1686                        checkRequiredCardinality(warnings, Uid.class, DateTimeStamp.class);
1687                        checkOptionalCardinality(warnings, Classification.class, Completed.class, Created.class, Description.class, DateStart.class, Geo.class, LastModified.class, Location.class, Organizer.class, PercentComplete.class, Priority.class, RecurrenceId.class, Sequence.class, Status.class, Summary.class, Url.class);
1688                }
1689
1690                checkOptionalCardinality(warnings, Color.class);
1691
1692                Status[] validStatuses;
1693                switch (version) {
1694                case V1_0:
1695                        validStatuses = new Status[] { Status.needsAction(), Status.completed(), Status.accepted(), Status.declined(), Status.delegated(), Status.sent() };
1696                        break;
1697                default:
1698                        validStatuses = new Status[] { Status.needsAction(), Status.completed(), Status.inProgress(), Status.cancelled() };
1699                        break;
1700                }
1701                checkStatus(warnings, validStatuses);
1702
1703                ICalDate dateStart = getValue(getDateStart());
1704                ICalDate dateDue = getValue(getDateDue());
1705                if (dateStart != null && dateDue != null) {
1706                        //DTSTART must come before DUE
1707                        if (dateStart.compareTo(dateDue) > 0) {
1708                                warnings.add(new ValidationWarning(22));
1709                        }
1710
1711                        //DTSTART and DUE must have the same data type
1712                        if (dateStart.hasTime() != dateDue.hasTime()) {
1713                                warnings.add(new ValidationWarning(23));
1714                        }
1715                }
1716
1717                //DUE and DURATION cannot both exist
1718                DurationProperty duration = getDuration();
1719                if (dateDue != null && duration != null) {
1720                        warnings.add(new ValidationWarning(24));
1721                }
1722
1723                //DTSTART is required if DURATION exists
1724                if (dateStart == null && duration != null) {
1725                        warnings.add(new ValidationWarning(25));
1726                }
1727
1728                //DTSTART and RECURRENCE-ID must have the same data type
1729                ICalDate recurrenceId = getValue(getRecurrenceId());
1730                if (recurrenceId != null && dateStart != null && dateStart.hasTime() != recurrenceId.hasTime()) {
1731                        warnings.add(new ValidationWarning(19));
1732                }
1733
1734                //BYHOUR, BYMINUTE, and BYSECOND cannot be specified in RRULE if DTSTART's data type is "date"
1735                //RFC 5545 p. 167
1736                Recurrence rrule = getValue(getRecurrenceRule());
1737                if (dateStart != null && rrule != null) {
1738                        if (!dateStart.hasTime() && (!rrule.getByHour().isEmpty() || !rrule.getByMinute().isEmpty() || !rrule.getBySecond().isEmpty())) {
1739                                warnings.add(new ValidationWarning(5));
1740                        }
1741                }
1742
1743                //there *should* be only 1 instance of RRULE
1744                //RFC 5545 p. 167
1745                if (getProperties(RecurrenceRule.class).size() > 1) {
1746                        warnings.add(new ValidationWarning(6));
1747                }
1748        }
