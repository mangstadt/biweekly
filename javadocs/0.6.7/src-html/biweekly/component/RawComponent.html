028/**
029 * Represents a component that does not have a scribe associated with it.
030 * @author Michael Angstadt
031 */
032public class RawComponent extends ICalComponent {
033        private final String name;
034
035        public RawComponent(String name) {
036                this.name = name;
037        }
038
039        /**
040         * Copy constructor.
041         * @param original the component to make a copy of
042         */
043        public RawComponent(RawComponent original) {
044                super(original);
045                this.name = original.name;
046        }
047
048        public String getName() {
049                return name;
050        }
051
052        @Override
053        public RawComponent copy() {
054                return new RawComponent(this);
055        }
056
057        @Override
058        public int hashCode() {
059                final int prime = 31;
060                int result = super.hashCode();
061                result = prime * result + ((name == null) ? 0 : name.hashCode());
062                return result;
063        }
064
065        @Override
066        public boolean equals(Object obj) {
067                if (!super.equals(obj)) return false;
068                RawComponent other = (RawComponent) obj;
069                if (name == null) {
070                        if (other.name != null) return false;
071                } else if (!name.equals(other.name)) return false;
072                return true;
073        }
074}
