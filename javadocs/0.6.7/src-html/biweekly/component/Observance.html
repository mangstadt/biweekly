045/**

375        @SuppressWarnings("unchecked")
376        @Override
377        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
378                if (version == ICalVersion.V1_0) {
379                        warnings.add(new ValidationWarning(48, version));
380                }
381
382                checkRequiredCardinality(warnings, DateStart.class, TimezoneOffsetTo.class, TimezoneOffsetFrom.class);
383
384                //BYHOUR, BYMINUTE, and BYSECOND cannot be specified in RRULE if DTSTART's data type is "date"
385                //RFC 5545 p. 167
386                DateStart dateStart = getDateStart();
387                RecurrenceRule rrule = getRecurrenceRule();
388                if (dateStart != null && rrule != null) {
389                        ICalDate start = dateStart.getValue();
390                        Recurrence recur = rrule.getValue();
391                        if (start != null && recur != null) {
392                                if (!start.hasTime() && (!recur.getByHour().isEmpty() || !recur.getByMinute().isEmpty() || !recur.getBySecond().isEmpty())) {
393                                        warnings.add(new ValidationWarning(5));
394                                }
395                        }
396                }
397
398                //there *should* be only 1 instance of RRULE
399                //RFC 5545 p. 167
400                if (getProperties(RecurrenceRule.class).size() > 1) {
401                        warnings.add(new ValidationWarning(6));
402                }
403        }
