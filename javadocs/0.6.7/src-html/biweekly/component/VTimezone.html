037/**
038 * <p>
039 * Defines a timezone's UTC offsets throughout the year.
040 * </p>
041 * 
042 * <p>
043 * <b>Examples:</b>
044 * </p>
045 * 
046 * <pre class="brush:java">
047 * VTimezone timezone = new VTimezone("Eastern Standard Time");
048 * 
049 * StandardTime standard = new StandardTime();
050 * DateTimeComponents componentsStandard = new DateTimeComponents(1998, 10, 25, 2, 0, 0, false);
051 * standard.setDateStart(componentsStandard);
052 * standard.setTimezoneOffsetFrom(-4, 0);
053 * standard.setTimezoneOffsetTo(-5, 0);
054 * timezone.addStandardTime(standard);
055 * 
056 * DaylightSavingsTime daylight = new DaylightSavingsTime();
057 * DateTimeComponents componentsDaylight = new DateTimeComponents(1999, 4, 4, 2, 0, 0, false);
058 * daylight.setDateStart(componentsDaylight);
059 * daylight.setTimezoneOffsetFrom(-5, 0);
060 * daylight.setTimezoneOffsetTo(-4, 0);
061 * timezone.addDaylightSavingsTime(daylight);
062 * </pre>
063 * @author Michael Angstadt
064 * @see <a href="http://tools.ietf.org/html/rfc5545#page-62">RFC 5545
065 * p.62-71</a>
066 * @see <a href="http://tools.ietf.org/html/rfc2445#page-60">RFC 2445 p.60-7</a>
067 */
068/*
069 * Note: References to the vCal 1.0 spec are omitted from the property
070 * getter/setter method Javadocs because vCal does not use the VTIMEZONE
071 * component.
072 */
073public class VTimezone extends ICalComponent {
074        /**
075         * Creates a new timezone component.
076         * @param identifier a unique identifier for this timezone (allows it to be
077         * referenced by date-time properties that support timezones).
078         */
079        public VTimezone(String identifier) {
080                setTimezoneId(identifier);
081        }
082
083        /**
084         * Copy constructor.
085         * @param original the component to make a copy of
086         */
087        public VTimezone(VTimezone original) {
088                super(original);
089        }
090
091        /**
092         * Gets the ID for this timezone. This is a <b>required</b> property.
093         * @return the timezone ID or null if not set
094         * @see <a href="http://tools.ietf.org/html/rfc5545#page-102">RFC 5545
095         * p.102-3</a>
096         * @see <a href="http://tools.ietf.org/html/rfc2445#page-97">RFC 2445
097         * p.97-8</a>
098         */
099        public TimezoneId getTimezoneId() {
100                return getProperty(TimezoneId.class);
101        }
102
103        /**
104         * Sets an ID for this timezone. This is a <b>required</b> property.
105         * @param timezoneId the timezone ID or null to remove
106         * @see <a href="http://tools.ietf.org/html/rfc5545#page-102">RFC 5545
107         * p.102-3</a>
108         * @see <a href="http://tools.ietf.org/html/rfc2445#page-97">RFC 2445
109         * p.97-8</a>
110         */
111        public void setTimezoneId(TimezoneId timezoneId) {
112                setProperty(TimezoneId.class, timezoneId);
113        }
114
115        /**
116         * Sets an ID for this timezone. This is a <b>required</b> property.
117         * @param timezoneId the timezone ID or null to remove
118         * @return the property that was created
119         * @see <a href="http://tools.ietf.org/html/rfc5545#page-102">RFC 5545
120         * p.102-3</a>
121         * @see <a href="http://tools.ietf.org/html/rfc2445#page-97">RFC 2445
122         * p.97-8</a>
123         */
124        public TimezoneId setTimezoneId(String timezoneId) {
125                TimezoneId prop = (timezoneId == null) ? null : new TimezoneId(timezoneId);
126                setTimezoneId(prop);
127                return prop;
128        }
129
130        /**
131         * Gets the date-time that the timezone data was last changed.
132         * @return the last modified date or null if not set
133         * @see <a href="http://tools.ietf.org/html/rfc5545#page-138">RFC 5545
134         * p.138</a>
135         * @see <a href="http://tools.ietf.org/html/rfc2445#page-131">RFC 2445
136         * p.131</a>
137         */
138        public LastModified getLastModified() {
139                return getProperty(LastModified.class);
140        }
141
142        /**
143         * Sets the date-time that the timezone data was last changed.
144         * @param lastModified the last modified date or null to remove
145         * @see <a href="http://tools.ietf.org/html/rfc5545#page-138">RFC 5545
146         * p.138</a>
147         * @see <a href="http://tools.ietf.org/html/rfc2445#page-131">RFC 2445
148         * p.131</a>
149         */
150        public void setLastModified(LastModified lastModified) {
151                setProperty(LastModified.class, lastModified);
152        }
153
154        /**
155         * Sets the date-time that the timezone data was last changed.
156         * @param lastModified the last modified date or null to remove
157         * @return the property that was created
158         * @see <a href="http://tools.ietf.org/html/rfc5545#page-138">RFC 5545
159         * p.138</a>
160         * @see <a href="http://tools.ietf.org/html/rfc2445#page-131">RFC 2445
161         * p.131</a>
162         */
163        public LastModified setLastModified(Date lastModified) {
164                LastModified prop = (lastModified == null) ? null : new LastModified(lastModified);
165                setLastModified(prop);
166                return prop;
167        }
168
169        /**
170         * Gets the timezone URL, which points to an iCalendar object that contains
171         * further information on the timezone.
172         * @return the URL or null if not set
173         * @see <a href="http://tools.ietf.org/html/rfc5545#page-106">RFC 5545
174         * p.106</a>
175         * @see <a href="http://tools.ietf.org/html/rfc2445#page-101">RFC 2445
176         * p.101</a>
177         */
178        public TimezoneUrl getTimezoneUrl() {
179                return getProperty(TimezoneUrl.class);
180        }
181
182        /**
183         * Sets the timezone URL, which points to an iCalendar object that contains
184         * further information on the timezone.
185         * @param url the URL or null to remove
186         * @see <a href="http://tools.ietf.org/html/rfc5545#page-106">RFC 5545
187         * p.106</a>
188         * @see <a href="http://tools.ietf.org/html/rfc2445#page-101">RFC 2445
189         * p.101</a>
190         */
191        public void setTimezoneUrl(TimezoneUrl url) {
192                setProperty(TimezoneUrl.class, url);
193        }
194
195        /**
196         * Sets the timezone URL, which points to an iCalendar object that contains
197         * further information on the timezone.
198         * @param url the timezone URL (e.g.
199         * "http://example.com/America-New_York.ics") or null to remove
200         * @return the property that was created
201         * @see <a href="http://tools.ietf.org/html/rfc5545#page-106">RFC 5545
202         * p.106</a>
203         * @see <a href="http://tools.ietf.org/html/rfc2445#page-101">RFC 2445
204         * p.101</a>
205         */
206        public TimezoneUrl setTimezoneUrl(String url) {
207                TimezoneUrl prop = (url == null) ? null : new TimezoneUrl(url);
208                setTimezoneUrl(prop);
209                return prop;
210        }
211
212        /**
213         * Gets the timezone's "standard" observance time ranges.
214         * @return the "standard" observance time ranges (any changes made this list
215         * will affect the parent component object and vice versa)
216         * @see <a href="http://tools.ietf.org/html/rfc5545#page-62">RFC 5545
217         * p.62-71</a>
218         * @see <a href="http://tools.ietf.org/html/rfc2445#page-60">RFC 2445
219         * p.60-7</a>
220         */
221        public List<StandardTime> getStandardTimes() {
222                return getComponents(StandardTime.class);
223        }
224
225        /**
226         * Adds a "standard" observance time range.
227         * @param standardTime the "standard" observance time
228         * @see <a href="http://tools.ietf.org/html/rfc5545#page-62">RFC 5545
229         * p.62-71</a>
230         * @see <a href="http://tools.ietf.org/html/rfc2445#page-60">RFC 2445
231         * p.60-7</a>
232         */
233        public void addStandardTime(StandardTime standardTime) {
234                addComponent(standardTime);
235        }
236
237        /**
238         * Gets the timezone's "daylight savings" observance time ranges.
239         * @return the "daylight savings" observance time ranges (any changes made
240         * this list will affect the parent component object and vice versa)
241         * @see <a href="http://tools.ietf.org/html/rfc5545#page-62">RFC 5545
242         * p.62-71</a>
243         * @see <a href="http://tools.ietf.org/html/rfc2445#page-60">RFC 2445
244         * p.60-7</a>
245         */
246        public List<DaylightSavingsTime> getDaylightSavingsTime() {
247                return getComponents(DaylightSavingsTime.class);
248        }
249
250        /**
251         * Adds a "daylight savings" observance time range.
252         * @param daylightSavingsTime the "daylight savings" observance time
253         * @see <a href="http://tools.ietf.org/html/rfc5545#page-62">RFC 5545
254         * p.62-71</a>
255         * @see <a href="http://tools.ietf.org/html/rfc2445#page-60">RFC 2445
256         * p.60-7</a>
257         */
258        public void addDaylightSavingsTime(DaylightSavingsTime daylightSavingsTime) {
259                addComponent(daylightSavingsTime);
260        }
261
262        @SuppressWarnings("unchecked")
263        @Override
264        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
265                if (version == ICalVersion.V1_0) {
266                        warnings.add(new ValidationWarning(48, version));
267                }
268
269                checkRequiredCardinality(warnings, TimezoneId.class);
270                checkOptionalCardinality(warnings, LastModified.class, TimezoneUrl.class);
271
272                //STANDARD or DAYLIGHT must be defined
273                if (getStandardTimes().isEmpty() && getDaylightSavingsTime().isEmpty()) {
274                        warnings.add(new ValidationWarning(21));
275                }
276        }
277
278        @Override
279        public VTimezone copy() {
280                return new VTimezone(this);
281        }
