030/**
031 * <p>
032 * Defines a relationship between the component that this property belongs to
033 * and another component.
034 * </p>
035 * <p>
036 * <b>Code samples:</b>
037 * </p>
038 * 
039 * <pre class="brush:java">
040 * VEvent event = new VEvent();
041 * 
042 * RelatedTo relatedTo = new RelatedTo("uid-value");
043 * event.addRelatedTo(relatedTo);
044 * </pre>
045 * @author Michael Angstadt
046 * @see <a href="http://tools.ietf.org/html/rfc5545#page-115">RFC 5545
047 * p.115-6</a>
048 * @see <a href="http://tools.ietf.org/html/rfc2445#page-109-10">RFC 2445
049 * p.109-10</a>
050 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.33-4</a>
051 */
052public class RelatedTo extends TextProperty {
053        /**
054         * Creates a related-to property.
055         * @param uid the value of the {@link Uid} property of the component that
056         * this property is referencing
057         */
058        public RelatedTo(String uid) {
059                super(uid);
060        }
061
062        /**
063         * Copy constructor.
064         * @param original the property to make a copy of
065         */
066        public RelatedTo(RelatedTo original) {
067                super(original);
068        }
069
070        /**
071         * Gets the relationship type.
072         * @return the relationship type (e.g. "child") or null if not set
073         * @see <a href="http://tools.ietf.org/html/rfc5545#page-25">RFC 5545
074         * p.25</a>
075         */
076        public RelationshipType getRelationshipType() {
077                return parameters.getRelationshipType();
078        }
079
080        /**
081         * Sets the relationship type.
082         * @param relationshipType the relationship type (e.g. "child") or null to
083         * remove
084         * @see <a href="http://tools.ietf.org/html/rfc5545#page-25">RFC 5545
085         * p.25</a>
086         */
087        public void setRelationshipType(RelationshipType relationshipType) {
088                parameters.setRelationshipType(relationshipType);
089        }
090
091        @Override
092        public RelatedTo copy() {
093                return new RelatedTo(this);
094        }
095}
