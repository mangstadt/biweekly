033/**
034 * Defines an alarm that will play an audio file when triggered. It is
035 * recommended that the {@link VAlarm} component be used to create alarms.
036 * @author Michael Angstadt
037 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.27-8</a>
038 * @see VAlarm#audio
039 */
040public class AudioAlarm extends VCalAlarmProperty {
041        private String contentId, uri;
042        private byte[] data;
043
044        public AudioAlarm() {
045                //empty
046        }
047
048        /**
049         * Copy constructor.
050         * @param original the property to make a copy of
051         */
052        public AudioAlarm(AudioAlarm original) {
053                super(original);
054                data = (original.data == null) ? null : original.data.clone();
055                uri = original.uri;
056                contentId = original.contentId;
057        }
058
059        public String getContentId() {
060                return contentId;
061        }
062
063        public void setContentId(String contentId) {
064                this.contentId = contentId;
065                this.uri = null;
066                this.data = null;
067        }
068
069        public String getUri() {
070                return uri;
071        }
072
073        public void setUri(String uri) {
074                this.uri = uri;
075                this.contentId = null;
076                this.data = null;
077        }
078
079        public byte[] getData() {
080                return data;
081        }
082
083        public void setData(byte[] data) {
084                this.data = data;
085                this.uri = null;
086                this.contentId = null;
087        }
088
089        public String getType() {
090                return parameters.getType();
091        }
092
093        public void setType(String type) {
094                parameters.setType(type);
095        }
096
097        @Override
098        protected Map<String, Object> toStringValues() {
099                Map<String, Object> values = super.toStringValues();
100                values.put("data", (data == null) ? "null" : "length: " + data.length);
101                values.put("uri", uri);
102                values.put("contentId", contentId);
103                return values;
104        }
105
106        @Override
107        public AudioAlarm copy() {
108                return new AudioAlarm(this);
109        }
110
111        @Override
112        public int hashCode() {
113                final int prime = 31;
114                int result = super.hashCode();
115                result = prime * result + ((contentId == null) ? 0 : contentId.hashCode());
116                result = prime * result + Arrays.hashCode(data);
117                result = prime * result + ((uri == null) ? 0 : uri.hashCode());
118                return result;
119        }
120
121        @Override
122        public boolean equals(Object obj) {
123                if (this == obj) return true;
124                if (!super.equals(obj)) return false;
125                AudioAlarm other = (AudioAlarm) obj;
126                if (contentId == null) {
127                        if (other.contentId != null) return false;
128                } else if (!contentId.equals(other.contentId)) return false;
129                if (uri == null) {
130                        if (other.uri != null) return false;
131                } else if (!uri.equals(other.uri)) return false;
132                if (!Arrays.equals(data, other.data)) return false;
133                return true;
134        }
135}
