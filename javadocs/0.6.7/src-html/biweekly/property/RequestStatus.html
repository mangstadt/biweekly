036/**
037 * <p>
038 * Represents a response to a scheduling request.
039 * </p>
040 * <p>
041 * This property must have a status code defined. The iCalendar specification
042 * defines the following status code families. However, these can have different
043 * meanings depending upon the type of scheduling request system being used
044 * (such as <a href="http://tools.ietf.org/html/rfc5546">iTIP</a>).
045 * </p>
046 * <ul>
047 * <li><b>1.x</b> - The request has been received, but is still being processed.
048 * </li>
049 * <li><b>2.x</b> - The request was processed successfully.</li>
050 * <li><b>3.x</b> - There is a client-side problem with the request (such as
051 * some incorrect syntax).</li>
052 * <li><b>4.x</b> - A scheduling error occurred on the server that prevented the
053 * request from being processed.</li>
054 * </ul>
055 * <p>
056 * <b>Code sample:</b>
057 * </p>
058 * 
059 * <pre class="brush:java">
060 * VEvent event = new VEvent();
061 * 
062 * RequestStatus requestStatus = new RequestStatus("2.0");
063 * requestStatus.setDescription("Success");
064 * event.setRequestStatus(requestStatus);
065 * </pre>
066 * @author Michael Angstadt
067 * @see <a href="http://tools.ietf.org/html/rfc5546#section-3.6">RFC 5546
068 * Section 3.6</a>
069 * @see <a href="http://tools.ietf.org/html/rfc5545#page-141">RFC 5545
070 * p.141-3</a>
071 * @see <a href="http://tools.ietf.org/html/rfc2445#page-134">RFC 2445
072 * p.134-6</a>
073 */
074public class RequestStatus extends ICalProperty {
075        private String statusCode, description, exceptionText;
076
077        /**
078         * Creates a request status property.
079         * @param statusCode the status code (e.g. "1.1.3")
080         */
081        public RequestStatus(String statusCode) {
082                setStatusCode(statusCode);
083        }
084
085        /**
086         * Copy constructor.
087         * @param original the property to make a copy of
088         */
089        public RequestStatus(RequestStatus original) {
090                super(original);
091                statusCode = original.statusCode;
092                description = original.description;
093                exceptionText = original.exceptionText;
094        }
095
096        /**
097         * Gets the status code. The following status code families are defined:
098         * <ul>
099         * <li><b>1.x</b> - The request has been received, but is still being
100         * processed.</li>
101         * <li><b>2.x</b> - The request was processed successfully.</li>
102         * <li><b>3.x</b> - There is a client-side problem with the request (such as
103         * some incorrect syntax).</li>
104         * <li><b>4.x</b> - A server-side error occurred.</li>
105         * </ul>
106         * @return the status code (e.g. "1.1.3")
107         */
108        public String getStatusCode() {
109                return statusCode;
110        }
111
112        /**
113         * Sets a status code. The following status code families are defined:
114         * <ul>
115         * <li><b>1.x</b> - The request has been received, but is still being
116         * processed.</li>
117         * <li><b>2.x</b> - The request was processed successfully.</li>
118         * <li><b>3.x</b> - There is a client-side problem with the request (such as
119         * some incorrect syntax).</li>
120         * <li><b>4.x</b> - A server-side error occurred.</li>
121         * </ul>
122         * @param statusCode the status code (e.g. "1.1.3")
123         */
124        public void setStatusCode(String statusCode) {
125                this.statusCode = statusCode;
126        }
127
128        /**
129         * Gets the human-readable description of the status.
130         * @return the description (e.g. "Success") or null if not set
131         */
132        public String getDescription() {
133                return description;
134        }
135
136        /**
137         * Sets a human-readable description of the status.
138         * @param description the description (e.g. "Success") or null to remove
139         */
140        public void setDescription(String description) {
141                this.description = description;
142        }
143
144        /**
145         * Gets any additional data related to the response.
146         * @return the additional data or null if not set
147         */
148        public String getExceptionText() {
149                return exceptionText;
150        }
151
152        /**
153         * Sets any additional data related to the response.
154         * @param exceptionText the additional data or null to remove
155         */
156        public void setExceptionText(String exceptionText) {
157                this.exceptionText = exceptionText;
158        }
159
160        @Override
161        public String getLanguage() {
162                return super.getLanguage();
163        }
164
165        @Override
166        public void setLanguage(String language) {
167                super.setLanguage(language);
168        }
169
170        @Override
171        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
172                if (statusCode == null) {
173                        warnings.add(new ValidationWarning(36));
174                }
175        }
176
177        @Override
178        protected Map<String, Object> toStringValues() {
179                Map<String, Object> values = new LinkedHashMap<String, Object>();
180                values.put("statusCode", statusCode);
181                values.put("description", description);
182                values.put("exceptionText", exceptionText);
183                return values;
184        }
185
186        @Override
187        public RequestStatus copy() {
188                return new RequestStatus(this);
189        }
190
191        @Override
192        public int hashCode() {
193                final int prime = 31;
194                int result = super.hashCode();
195                result = prime * result + ((description == null) ? 0 : description.hashCode());
196                result = prime * result + ((exceptionText == null) ? 0 : exceptionText.hashCode());
197                result = prime * result + ((statusCode == null) ? 0 : statusCode.hashCode());
198                return result;
199        }
200
201        @Override
202        public boolean equals(Object obj) {
203                if (this == obj) return true;
204                if (!super.equals(obj)) return false;
205                RequestStatus other = (RequestStatus) obj;
206                if (description == null) {
207                        if (other.description != null) return false;
208                } else if (!description.equals(other.description)) return false;
209                if (exceptionText == null) {
210                        if (other.exceptionText != null) return false;
211                } else if (!exceptionText.equals(other.exceptionText)) return false;
212                if (statusCode == null) {
213                        if (other.statusCode != null) return false;
214                } else if (!statusCode.equals(other.statusCode)) return false;
215                return true;
216        }
217}
