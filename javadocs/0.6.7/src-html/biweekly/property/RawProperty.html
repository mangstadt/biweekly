041/**
042 * Represents a property that does not have a scribe associated with it.
043 * @author Michael Angstadt
044 */
045public class RawProperty extends ICalProperty {
046        private String name;
047        private ICalDataType dataType;
048        private String value;
049
050        /**
051         * Creates a raw property.
052         * @param name the property name (e.g. "X-MS-ANNIVERSARY")
053         * @param value the property value
054         */
055        public RawProperty(String name, String value) {
056                this(name, null, value);
057        }
058
059        /**
060         * Creates a raw property.
061         * @param name the property name (e.g. "X-MS-ANNIVERSARY")
062         * @param dataType the property value's data type
063         * @param value the property value
064         */
065        public RawProperty(String name, ICalDataType dataType, String value) {
066                this.name = name;
067                this.dataType = dataType;
068                this.value = value;
069        }
070
071        /**
072         * Copy constructor.
073         * @param original the property to make a copy of
074         */
075        public RawProperty(RawProperty original) {
076                super(original);
077                name = original.name;
078                dataType = original.dataType;
079                value = original.value;
080        }
081
082        /**
083         * Gets the property value.
084         * @return the property value
085         */
086        public String getValue() {
087                return value;
088        }
089
090        /**
091         * Sets the property value.
092         * @param value the property value
093         */
094        public void setValue(String value) {
095                this.value = value;
096        }
097
098        /**
099         * Gets the property value's data type.
100         * @return the data type
101         */
102        public ICalDataType getDataType() {
103                return dataType;
104        }
105
106        /**
107         * Sets the property value's data type.
108         * @param dataType the data type
109         */
110        public void setDataType(ICalDataType dataType) {
111                this.dataType = dataType;
112        }
113
114        /**
115         * Gets the property name.
116         * @return the property name (e.g. "X-MS-ANNIVERSARY")
117         */
118        public String getName() {
119                return name;
120        }
121
122        /**
123         * Sets the property name.
124         * @param name the property name (e.g. "X-MS-ANNIVERSARY")
125         */
126        public void setName(String name) {
127                this.name = name;
128        }
129
130        @Override
131        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
132                SyntaxStyle syntax = version.getSyntaxStyle();
133                AllowedCharacters allowed = VObjectValidator.allowedCharactersParameterName(syntax, true);
134                if (!allowed.check(name)) {
135                        if (syntax == SyntaxStyle.OLD) {
136                                AllowedCharacters notAllowed = allowed.flip();
137                                warnings.add(new ValidationWarning(59, name, notAllowed.toString(true)));
138                        } else {
139                                warnings.add(new ValidationWarning(52, name));
140                        }
141                }
142        }
143
144        @Override
145        protected Map<String, Object> toStringValues() {
146                Map<String, Object> values = new LinkedHashMap<String, Object>();
147                values.put("name", name);
148                values.put("value", value);
149                values.put("dataType", dataType);
150                return values;
151        }
152
153        @Override
154        public RawProperty copy() {
155                return new RawProperty(this);
156        }
157
158        @Override
159        public int hashCode() {
160                final int prime = 31;
161                int result = super.hashCode();
162                result = prime * result + ((dataType == null) ? 0 : dataType.hashCode());
163                result = prime * result + ((name == null) ? 0 : name.toLowerCase().hashCode());
164                result = prime * result + ((value == null) ? 0 : value.hashCode());
165                return result;
166        }
167
168        @Override
169        public boolean equals(Object obj) {
170                if (this == obj) return true;
171                if (!super.equals(obj)) return false;
172                RawProperty other = (RawProperty) obj;
173                if (dataType != other.dataType) return false;
174                if (name == null) {
175                        if (other.name != null) return false;
176                } else if (!name.equalsIgnoreCase(other.name)) return false;
177                if (value == null) {
178                        if (other.value != null) return false;
179                } else if (!value.equals(other.value)) return false;
180                return true;
181        }
182
183}
