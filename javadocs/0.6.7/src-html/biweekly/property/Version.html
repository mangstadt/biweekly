037/**
038 * <p>
039 * Defines the min/max iCalendar versions a consumer must support in order to
040 * successfully parse the iCalendar object.
041 * </p>
042 * <p>
043 * <b>Code sample:</b>
044 * </p>
045 * 
046 * <pre class="brush:java">
047 * ICalendar ical = new ICalendar();
048 * 
049 * //all ICalendar objects are given a VERSION property on creation
050 * ical.getVersion(); //"2.0"
051 * 
052 * //get the default iCal version
053 * Version version = Version.v2_0();
054 * ical.setVersion(version);
055 * </pre>
056 * @author Michael Angstadt
057 * @see <a href="http://tools.ietf.org/html/rfc5545#page-79">RFC 5545
058 * p.79-80</a>
059 * @see <a href="http://tools.ietf.org/html/rfc2445#page-76">RFC 2445 p.76-7</a>
060 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.24</a>
061 */
062public class Version extends ICalProperty {
063        public static final VersionNumber VCAL = new VersionNumber(ICalVersion.V1_0.getVersion());
064        public static final VersionNumber ICAL = new VersionNumber(ICalVersion.V2_0.getVersion());
065
066        private VersionNumber minVersion, maxVersion;
067
068        /**
069         * Creates a new version property.
070         * @param version the version that a consumer must support in order to
071         * successfully parse the iCalendar object
072         */
073        public Version(ICalVersion version) {
074                this((version == null) ? null : version.getVersion());
075        }
076
077        /**
078         * Creates a new version property.
079         * @param version the version that a consumer must support in order to
080         * successfully parse the iCalendar object
081         * @throws IllegalArgumentException if the version string is invalid
082         */
083        public Version(String version) {
084                this(null, version);
085        }
086
087        /**
088         * Creates a new version property.
089         * @param minVersion the minimum version that a consumer must support in
090         * order to successfully parse the iCalendar object
091         * @param maxVersion the maximum version that a consumer must support in
092         * order to successfully parse the iCalendar object
093         * @throws IllegalArgumentException if one of the versions strings are
094         * invalid
095         */
096        public Version(String minVersion, String maxVersion) {
097                this((minVersion == null) ? null : new VersionNumber(minVersion), (maxVersion == null) ? null : new VersionNumber(maxVersion));
098        }
099
100        private Version(VersionNumber minVersion, VersionNumber maxVersion) {
101                this.minVersion = minVersion;
102                this.maxVersion = maxVersion;
103        }
104
105        /**
106         * Copy constructor.
107         * @param original the property to make a copy of
108         */
109        public Version(Version original) {
110                super(original);
111                minVersion = original.minVersion;
112                maxVersion = original.maxVersion;
113        }
114
115        /**
116         * Creates a version property that is set to the older vCalendar version
117         * (1.0).
118         * @return the property instance
119         */
120        public static Version v1_0() {
121                return new Version(ICalVersion.V1_0);
122        }
123
124        /**
125         * Creates a version property that is set to the latest iCalendar version
126         * (2.0).
127         * @return the property instance
128         */
129        public static Version v2_0() {
130                return new Version(ICalVersion.V2_0);
131        }
132
133        /**
134         * Determines if this property is set to the older vCalendar version.
135         * @return true if the version is "1.0", false if not
136         */
137        public boolean isV1_0() {
138                return VCAL.equals(maxVersion);
139        }
140
141        /**
142         * Determines if this property is set to the latest iCalendar version.
143         * @return true if the version is "2.0", false if not
144         */
145        public boolean isV2_0() {
146                return ICAL.equals(maxVersion);
147        }
148
149        /**
150         * Gets the minimum version that a consumer must support in order to
151         * successfully parse the iCalendar object.
152         * @return the minimum version or null if not set
153         */
154        public VersionNumber getMinVersion() {
155                return minVersion;
156        }
157
158        /**
159         * Sets the minimum version that a consumer must support in order to
160         * successfully parse the iCalendar object.
161         * @param minVersion the minimum version or null to remove
162         */
163        public void setMinVersion(VersionNumber minVersion) {
164                this.minVersion = minVersion;
165        }
166
167        /**
168         * Gets the maximum version that a consumer must support in order to
169         * successfully parse the iCalendar object.
170         * @return the maximum version or null if not set
171         */
172        public VersionNumber getMaxVersion() {
173                return maxVersion;
174        }
175
176        /**
177         * Sets the maximum version that a consumer must support in order to
178         * successfully parse the iCalendar object.
179         * @param maxVersion the maximum version (this field is <b>required</b>)
180         */
181        public void setMaxVersion(VersionNumber maxVersion) {
182                this.maxVersion = maxVersion;
183        }
184
185        /**
186         * Converts this property's value to an {@link ICalVersion} enum.
187         * @return the {@link ICalVersion} enum or null if it couldn't be converted
188         */
189        public ICalVersion toICalVersion() {
190                if (minVersion == null && maxVersion != null) {
191                        return ICalVersion.get(maxVersion.toString());
192                }
193                return null;
194        }
195
196        @Override
197        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
198                if (maxVersion == null) {
199                        warnings.add(new ValidationWarning(35));
200                }
201        }
202
203        @Override
204        protected Map<String, Object> toStringValues() {
205                Map<String, Object> values = new LinkedHashMap<String, Object>();
206                values.put("minVersion", minVersion);
207                values.put("maxVersion", maxVersion);
208                return values;
209        }
210
211        @Override
212        public Version copy() {
213                return new Version(this);
214        }
215
216        @Override
217        public int hashCode() {
218                final int prime = 31;
219                int result = super.hashCode();
220                result = prime * result + ((maxVersion == null) ? 0 : maxVersion.hashCode());
221                result = prime * result + ((minVersion == null) ? 0 : minVersion.hashCode());
222                return result;
223        }
224
225        @Override
226        public boolean equals(Object obj) {
227                if (this == obj) return true;
228                if (!super.equals(obj)) return false;
229                Version other = (Version) obj;
230                if (maxVersion == null) {
231                        if (other.maxVersion != null) return false;
232                } else if (!maxVersion.equals(other.maxVersion)) return false;
233                if (minVersion == null) {
234                        if (other.minVersion != null) return false;
235                } else if (!minVersion.equals(other.minVersion)) return false;
236                return true;
237        }
238}
