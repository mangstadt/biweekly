036/**
037 * <p>
038 * This property has different meanings depending on the component it belongs
039 * to:
040 * </p>
041 * <ul>
042 * <li>{@link VEvent} - The organizer of the event.</li>
043 * <li>{@link VTodo} - The creator of the to-do task.</li>
044 * <li>{@link VJournal} - The owner of the journal entry.</li>
045 * <li>{@link VFreeBusy} - The person requesting the free/busy time.</li>
046 * </ul>
047 * <p>
048 * <b>Code sample:</b>
049 * </p>
050 * 
051 * <pre class="brush:java">
052 * VEvent event = new VEvent();
053 * 
054 * Organizer organizer = Organizer.email("johndoe@example.com");
055 * organizer.setCommonName("John Doe");
056 * event.setOrganizer(organizer);
057 * </pre>
058 * @author Michael Angstadt
059 * @see <a href="http://tools.ietf.org/html/rfc5545#page-111">RFC 5545
060 * p.111-2</a>
061 * @see <a href="http://tools.ietf.org/html/rfc2445#page-106">RFC 2445
062 * p.106-7</a>
063 */
064public class Organizer extends ICalProperty {
065        private String uri, email, name;
066
067        /**
068         * Creates an organizer property
069         * @param name the organizer's name (e.g. "John Doe")
070         * @param email the organizer's email address (e.g. "jdoe@example.com")
071         */
072        public Organizer(String name, String email) {
073                this.name = name;
074                this.email = email;
075        }
076
077        /**
078         * Copy constructor.
079         * @param original the property to make a copy of
080         */
081        public Organizer(Organizer original) {
082                super(original);
083                name = original.name;
084                email = original.email;
085                uri = original.uri;
086        }
087
088        /**
089         * Gets the organizer's email
090         * @return the email (e.g. "jdoe@company.com")
091         */
092        public String getEmail() {
093                return email;
094        }
095
096        /**
097         * Sets the organizer's email
098         * @param email the email (e.g. "jdoe@company.com")
099         */
100        public void setEmail(String email) {
101                this.email = email;
102        }
103
104        /**
105         * Gets a URI representing the organizer.
106         * @return the URI (e.g. "mailto:jdoe@company.com")
107         */
108        public String getUri() {
109                return uri;
110        }
111
112        /**
113         * Sets a URI representing the organizer.
114         * @param uri the URI (e.g. "mailto:jdoe@company.com")
115         */
116        public void setUri(String uri) {
117                this.uri = uri;
118        }
119
120        @Override
121        public String getSentBy() {
122                return super.getSentBy();
123        }
124
125        @Override
126        public void setSentBy(String sentBy) {
127                super.setSentBy(sentBy);
128        }
129
130        @Override
131        public String getCommonName() {
132                return name;
133        }
134
135        @Override
136        public void setCommonName(String commonName) {
137                this.name = commonName;
138        }
139
140        @Override
141        public String getDirectoryEntry() {
142                return super.getDirectoryEntry();
143        }
144
145        @Override
146        public void setDirectoryEntry(String directoryEntry) {
147                super.setDirectoryEntry(directoryEntry);
148        }
149
150        /**
151         * Gets the language that the common name parameter is written in.
152         */
153        @Override
154        public String getLanguage() {
155                return super.getLanguage();
156        }
157
158        /**
159         * Sets the language that the common name parameter is written in.
160         */
161        @Override
162        public void setLanguage(String language) {
163                super.setLanguage(language);
164        }
165
166        @Override
167        protected Map<String, Object> toStringValues() {
168                Map<String, Object> values = new LinkedHashMap<String, Object>();
169                values.put("name", name);
170                values.put("email", email);
171                values.put("uri", uri);
172                return values;
173        }
174
175        @Override
176        public Organizer copy() {
177                return new Organizer(this);
178        }
179
180        @Override
181        public int hashCode() {
182                final int prime = 31;
183                int result = super.hashCode();
184                result = prime * result + ((email == null) ? 0 : email.hashCode());
185                result = prime * result + ((name == null) ? 0 : name.hashCode());
186                result = prime * result + ((uri == null) ? 0 : uri.hashCode());
187                return result;
188        }
189
190        @Override
191        public boolean equals(Object obj) {
192                if (this == obj) return true;
193                if (!super.equals(obj)) return false;
194                Organizer other = (Organizer) obj;
195                if (email == null) {
196                        if (other.email != null) return false;
197                } else if (!email.equals(other.email)) return false;
198                if (name == null) {
199                        if (other.name != null) return false;
200                } else if (!name.equals(other.name)) return false;
201                if (uri == null) {
202                        if (other.uri != null) return false;
203                } else if (!uri.equals(other.uri)) return false;
204                return true;
205        }
206}
