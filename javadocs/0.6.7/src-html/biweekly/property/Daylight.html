038/**
039 * Represents daylight savings time information.
040 * @author Michael Angstadt
041 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.23</a>
042 */
043public class Daylight extends ICalProperty {
044        private boolean daylight;
045        private UtcOffset offset;
046        private ICalDate start, end;
047        private String standardName, daylightName;
048
049        /**
050         * Creates a daylight savings property which states that the timezone does
051         * not observe daylight savings time.
052         */
053        public Daylight() {
054                this.daylight = false;
055        }
056
057        /**
058         * Creates a daylight savings property.
059         * @param daylight true if the timezone observes daylight savings time,
060         * false if not
061         * @param offset the UTC offset of daylight savings time
062         * @param start the start date of daylight savings time
063         * @param end the end date of daylight savings time
064         * @param standardName the timezone's name for standard time (e.g. "EST")
065         * @param daylightName the timezone's name for daylight savings time (e.g.
066         * "EDT")
067         */
068        public Daylight(boolean daylight, UtcOffset offset, ICalDate start, ICalDate end, String standardName, String daylightName) {
069                this.daylight = daylight;
070                this.offset = offset;
071                this.start = start;
072                this.end = end;
073                this.standardName = standardName;
074                this.daylightName = daylightName;
075        }
076
077        /**
078         * Copy constructor.
079         * @param original the property to make a copy of
080         */
081        public Daylight(Daylight original) {
082                super(original);
083                daylight = original.daylight;
084                offset = original.offset;
085                start = (original.start == null) ? null : new ICalDate(original.start);
086                end = (original.end == null) ? null : new ICalDate(original.end);
087                standardName = original.standardName;
088                daylightName = original.daylightName;
089        }
090
091        /**
092         * Gets whether this timezone observes daylight savings time.
093         * @return true if it observes daylight savings time, false if not
094         */
095        public boolean isDaylight() {
096                return daylight;
097        }
098
099        /**
100         * Sets whether this timezone observes daylight savings time.
101         * @param daylight true if it observes daylight savings time, false if not
102         */
103        public void setDaylight(boolean daylight) {
104                this.daylight = daylight;
105        }
106
107        /**
108         * Gets the UTC offset of daylight savings time.
109         * @return the UTC offset
110         */
111        public UtcOffset getOffset() {
112                return offset;
113        }
114
115        /**
116         * Sets the UTC offset of daylight savings time.
117         * @param offset the UTC offset
118         */
119        public void setOffset(UtcOffset offset) {
120                this.offset = offset;
121        }
122
123        /**
124         * Gets the start date of dayight savings time.
125         * @return the start date
126         */
127        public ICalDate getStart() {
128                return start;
129        }
130
131        /**
132         * Sets the start date of dayight savings time.
133         * @param start the start date
134         */
135        public void setStart(ICalDate start) {
136                this.start = start;
137        }
138
139        /**
140         * Gets the end date of daylight savings time.
141         * @return the end date
142         */
143        public ICalDate getEnd() {
144                return end;
145        }
146
147        /**
148         * Sets the end date of daylight savings time.
149         * @param end the end date
150         */
151        public void setEnd(ICalDate end) {
152                this.end = end;
153        }
154
155        /**
156         * Gets the name for standard time.
157         * @return the name (e.g. "EST")
158         */
159        public String getStandardName() {
160                return standardName;
161        }
162
163        /**
164         * Sets the name for standard time.
165         * @param name the name (e.g. "EST")
166         */
167        public void setStandardName(String name) {
168                this.standardName = name;
169        }
170
171        /**
172         * Gets the name of daylight savings time.
173         * @return the name (e.g. "EDT")
174         */
175        public String getDaylightName() {
176                return daylightName;
177        }
178
179        /**
180         * Sets the name of daylight savings time.
181         * @param name the name (e.g. "EDT")
182         */
183        public void setDaylightName(String name) {
184                this.daylightName = name;
185        }
186
187        @Override
188        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
189                if (daylight && (offset == null || start == null || end == null || standardName == null || daylightName == null)) {
190                        warnings.add(new ValidationWarning(43));
191                }
192        }
193
194        @Override
195        protected Map<String, Object> toStringValues() {
196                Map<String, Object> values = new LinkedHashMap<String, Object>();
197                values.put("daylight", daylight);
198                values.put("offset", offset);
199                values.put("start", start);
200                values.put("end", end);
201                values.put("standardName", standardName);
202                values.put("daylightName", daylightName);
203                return values;
204        }
205
206        @Override
207        public Daylight copy() {
208                return new Daylight(this);
209        }
210
211        @Override
212        public int hashCode() {
213                final int prime = 31;
214                int result = super.hashCode();
215                result = prime * result + (daylight ? 1231 : 1237);
216                result = prime * result + ((daylightName == null) ? 0 : daylightName.hashCode());
217                result = prime * result + ((end == null) ? 0 : end.hashCode());
218                result = prime * result + ((offset == null) ? 0 : offset.hashCode());
219                result = prime * result + ((standardName == null) ? 0 : standardName.hashCode());
220                result = prime * result + ((start == null) ? 0 : start.hashCode());
221                return result;
222        }
223
224        @Override
225        public boolean equals(Object obj) {
226                if (this == obj) return true;
227                if (!super.equals(obj)) return false;
228                Daylight other = (Daylight) obj;
229                if (daylight != other.daylight) return false;
230                if (daylightName == null) {
231                        if (other.daylightName != null) return false;
232                } else if (!daylightName.equals(other.daylightName)) return false;
233                if (end == null) {
234                        if (other.end != null) return false;
235                } else if (!end.equals(other.end)) return false;
236                if (offset == null) {
237                        if (other.offset != null) return false;
238                } else if (!offset.equals(other.offset)) return false;
239                if (standardName == null) {
240                        if (other.standardName != null) return false;
241                } else if (!standardName.equals(other.standardName)) return false;
242                if (start == null) {
243                        if (other.start != null) return false;
244                } else if (!start.equals(other.start)) return false;
245                return true;
246        }
247}
