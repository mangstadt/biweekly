032/**
033 * <p>
034 * Defines the due date of a to-do task.
035 * </p>
036 * <p>
037 * <b>Code sample (creating):</b>
038 * </p>
039 * 
040 * <pre class="brush:java">
041 * VTodo todo = new VTodo();
042 * 
043 * //date and time
044 * Date datetime = ...
045 * DateDue due = new DateDue(datetime);
046 * todo.setDateDue(due);
047 * 
048 * //date (without time component)
049 * Date date = ...
050 * due = new DateDue(date, false);
051 * todo.setDateDue(due);
052 * </pre>
053 * 
054 * <p>
055 * <b>Code sample (reading):</b>
056 * </p>
057 * 
058 * <pre class="brush:java">
059 * ICalendar ical = ...
060 * TimezoneInfo tzinfo = ical.getTimezoneInfo();
061 * 
062 * for (VTodo todo : ical.getTodos()) {
063 *   DateDue due = todo.getDateDue();
064 *   
065 *   //get property value (ICalDate extends java.util.Date)
066 *   ICalDate value = due.getValue();
067 *   
068 *   if (value.hasTime()) {
069 *     //the value includes a time component
070 *   } else {
071 *     //the value is just a date
072 *     //date object's time is set to "00:00:00" under local computer's default timezone
073 *   }
074 *   
075 *   //gets the timezone that the property value was parsed under if you are curious about that
076 *   TimeZone tz = tzinfo.getTimeZone(due);
077 * }
078 * </pre>
079 * 
080 * <p>
081 * <b>Code sample (using timezones):</b>
082 * </p>
083 * 
084 * <pre class="brush:java">
085 * ICalendar ical = new ICalendar();
086 * VTodo todo = new VTodo();
087 * Date datetime = ...
088 * DateDue due = new DateDue(datetime);
089 * todo.setDateDue(due);
090 * ical.addTodo(todo);
091 * 
092 * TimezoneAssignment tz = ...
093 * 
094 * //set the timezone of all date-time property values
095 * //date-time property values are written in UTC by default
096 * ical.getTimezoneInfo().setDefaultTimezone(tz);
097 * 
098 * //or set the timezone just for this property
099 * ical.getTimezoneInfo().setTimezone(due, tz);
100 * 
101 * //finally, write the iCalendar object
102 * ICalWriter writer = ...
103 * writer.write(ical);
104 * </pre>
105 * @author Michael Angstadt
106 * @see <a href="http://tools.ietf.org/html/rfc5545#page-96">RFC 5545 p.96-7</a>
107 * @see <a href="http://tools.ietf.org/html/rfc2445#page-92">RFC 2445 p.92-3</a>
108 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.30</a>
109 */
110public class DateDue extends DateOrDateTimeProperty {
111        /**
112         * Creates a due date property.
113         * @param dueDate the due date
114         */
115        public DateDue(Date dueDate) {
116                super(dueDate);
117        }
118
119        /**
120         * Creates a due date property.
121         * @param dueDate the due date
122         * @param hasTime true if the value has a time component, false if it is
123         * strictly a date
124         */
125        public DateDue(Date dueDate, boolean hasTime) {
126                super(dueDate, hasTime);
127        }
128
129        /**
130         * Creates a due date property.
131         * @param dueDate the due date
132         */
133        public DateDue(ICalDate dueDate) {
134                super(dueDate);
135        }
136
137        /**
138         * Copy constructor.
139         * @param original the property to make a copy of
140         */
141        public DateDue(DateDue original) {
142                super(original);
143        }
144
145        @Override
146        public DateDue copy() {
147                return new DateDue(this);
148        }
149}
