032/**
033 * <p>
034 * Defines the start date of an event, free/busy component, or timezone
035 * component.
036 * </p>
037 * <p>
038 * <b>Code sample (creating):</b>
039 * </p>
040 * 
041 * <pre class="brush:java">
042 * VEvent event = new VEvent();
043 * 
044 * //date and time
045 * Date datetime = ...
046 * DateStart dtstart = new DateStart(datetime);
047 * event.setDateStart(dtstart);
048 * 
049 * //date (without time component)
050 * Date date = ...
051 * dtstart = new DateStart(date, false);
052 * event.setDateStart(dtstart);
053 * </pre>
054 * 
055 * <p>
056 * <b>Code sample (reading):</b>
057 * </p>
058 * 
059 * <pre class="brush:java">
060 * ICalendar ical = ...
061 * for (VEvent event : ical.getEvents()) {
062 *   DateStart dtstart = event.getDateStart();
063 *   
064 *   //get property value (ICalDate extends java.util.Date)
065 *   ICalDate value = dtstart.getValue();
066 *   
067 *   if (value.hasTime()) {
068 *     //the value includes a time component
069 *   } else {
070 *     //the value is just a date
071 *     //date object's time is set to "00:00:00" under local computer's default timezone
072 *   }
073 *   
074 *   //gets the timezone that the property value was parsed under if you are curious about that
075 *   TimeZone tz = tzinfo.getTimeZone(dtstart);
076 * }
077 * </pre>
078 * 
079 * <p>
080 * <b>Code sample (using timezones):</b>
081 * </p>
082 * 
083 * <pre class="brush:java">
084 * ICalendar ical = new ICalendar();
085 * VEvent event = new VEvent();
086 * Date datetime = ...
087 * DateStart dtstart = new DateStart(datetime);
088 * event.setDateStart(dtstart);
089 * ical.addEvent(event);
090 * 
091 * TimezoneAssignment tz = ...
092 * 
093 * //set the timezone of all date-time property values
094 * //date-time property values are written in UTC by default
095 * ical.getTimezoneInfo().setDefaultTimezone(tz);
096 * 
097 * //or set the timezone just for this property
098 * ical.getTimezoneInfo().setTimezone(dtstart, tz);
099 * 
100 * //finally, write the iCalendar object
101 * ICalWriter writer = ...
102 * writer.write(ical);
103 * </pre>
104 * @author Michael Angstadt
105 * @see <a href="http://tools.ietf.org/html/rfc5545#page-97">RFC 5545 p.97-8</a>
106 * @see <a href="http://tools.ietf.org/html/rfc2445#page-93">RFC 2445 p.93-4</a>
107 * @see <a href="http://www.imc.org/pdi/vcal-10.doc">vCal 1.0 p.35</a>
108 */
109public class DateStart extends DateOrDateTimeProperty {
110        /**
111         * Creates a start date property.
112         * @param startDate the start date
113         */
114        public DateStart(Date startDate) {
115                super(startDate);
116        }
117
118        /**
119         * Creates a start date property.
120         * @param startDate the start date
121         * @param hasTime true if the value has a time component, false if it is
122         * strictly a date
123         */
124        public DateStart(Date startDate, boolean hasTime) {
125                super(startDate, hasTime);
126        }
127
128        /**
129         * Creates a start date property.
130         * @param startDate the start date
131         */
132        public DateStart(ICalDate startDate) {
133                super(startDate);
134        }
135
136        /**
137         * Copy constructor.
138         * @param original the property to make a copy of
139         */
140        public DateStart(DateStart original) {
141                super(original);
142        }
143
144        @Override
145        public DateStart copy() {
146                return new DateStart(this);
147        }
148}
