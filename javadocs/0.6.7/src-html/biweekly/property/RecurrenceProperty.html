041/**
042 * Represents a property whose value is a recurrence rule.
043 * @author Michael Angstadt
044 */
045public class RecurrenceProperty extends ValuedProperty<Recurrence> {
046        /**
047         * Creates a new recurrence property.
048         * @param recur the recurrence value
049         */
050        public RecurrenceProperty(Recurrence recur) {
051                super(recur);
052        }
053
054        /**
055         * Copy constructor.
056         * @param original the property to make a copy of
057         */
058        public RecurrenceProperty(RecurrenceProperty original) {
059                super(original);
060        }
061
062        /**
063         * Creates an iterator that computes the dates defined by this property.
064         * @param startDate the date that the recurrence starts (typically, the
065         * value of the accompanying {@link DateStart} property)
066         * @param timezone the timezone to iterate in (typically, the timezone
067         * associated with the {@link DateStart} property). This is needed in order
068         * to adjust for when the iterator passes over a daylight savings boundary.
069         * @return the iterator
070         * @see <a
071         * href="https://code.google.com/p/google-rfc-2445/">google-rfc-2445</a>
072         */
073        public DateIterator getDateIterator(Date startDate, TimeZone timezone) {
074                return getDateIterator(new ICalDate(startDate), timezone);
075        }
076
077        /**
078         * Creates an iterator that computes the dates defined by this property.
079         * @param startDate the date that the recurrence starts (typically, the
080         * value of the accompanying {@link DateStart} property)
081         * @param timezone the timezone to iterate in (typically, the timezone
082         * associated with the {@link DateStart} property). This is needed in order
083         * to adjust for when the iterator passes over a daylight savings boundary.
084         * @return the iterator
085         * @see <a
086         * href="https://code.google.com/p/google-rfc-2445/">google-rfc-2445</a>
087         */
088        public DateIterator getDateIterator(ICalDate startDate, TimeZone timezone) {
089                Recurrence recur = getValue();
090                return (recur == null) ? new Google2445Utils.EmptyDateIterator() : recur.getDateIterator(startDate, timezone);
091        }
092
093        @Override
094        protected void validate(List<ICalComponent> components, ICalVersion version, List<ValidationWarning> warnings) {
095                super.validate(components, version, warnings);
096                if (value == null) {
097                        return;
098                }
099
100                if (value.getFrequency() == null) {
101                        warnings.add(new ValidationWarning(30));
102                }
103
104                if (value.getUntil() != null && value.getCount() != null) {
105                        warnings.add(new ValidationWarning(31));
106                }
107
108                switch (version) {
109                case V1_0:
110                        if (!value.getXRules().isEmpty()) {
111                                warnings.add(new ValidationWarning("X-Rules are not supported by vCal."));
112                        }
113                        if (!value.getBySetPos().isEmpty()) {
114                                warnings.add(new ValidationWarning("BYSETPOS is not supported by vCal."));
115                        }
116                        if (value.getFrequency() == Frequency.SECONDLY) {
117                                warnings.add(new ValidationWarning(Frequency.SECONDLY.name() + " frequency is not supported by vCal."));
118                        }
119                        break;
120
121                case V2_0_DEPRECATED:
122                        //empty
123                        break;
124
125                case V2_0:
126                        if (!value.getXRules().isEmpty()) {
127                                warnings.add(new ValidationWarning(32));
128                        }
129
130                        break;
131                }
132        }
133}
